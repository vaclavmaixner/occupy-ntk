29class BaseRequest(object): 

30 """Very basic request object. This does not implement advanced stuff like 

31 entity tag parsing or cache controls. The request object is created with 

32 the WSGI environment as first argument and will add itself to the WSGI 

33 environment as ``'werkzeug.request'`` unless it's created with 

34 `populate_request` set to False. 

35 

36 There are a couple of mixins available that add additional functionality 

37 to the request object, there is also a class called `Request` which 

38 subclasses `BaseRequest` and all the important mixins. 

39 

40 It's a good idea to create a custom subclass of the :class:`BaseRequest` 

41 and add missing functionality either via mixins or direct implementation. 

42 Here an example for such subclasses:: 

43 

44 from werkzeug.wrappers import BaseRequest, ETagRequestMixin 

45 

46 class Request(BaseRequest, ETagRequestMixin): 

47 pass 

48 

49 Request objects are **read only**. As of 0.5 modifications are not 

50 allowed in any place. Unlike the lower level parsing functions the 

51 request object will use immutable objects everywhere possible. 

52 

53 Per default the request object will assume all the text data is `utf-8` 

54 encoded. Please refer to :doc:`the unicode chapter </unicode>` for more 

55 details about customizing the behavior. 

56 

57 Per default the request object will be added to the WSGI 

58 environment as `werkzeug.request` to support the debugging system. 

59 If you don't want that, set `populate_request` to `False`. 

60 

61 If `shallow` is `True` the environment is initialized as shallow 

62 object around the environ. Every operation that would modify the 

63 environ in any way (such as consuming form data) raises an exception 

64 unless the `shallow` attribute is explicitly set to `False`. This 

65 is useful for middlewares where you don't want to consume the form 

66 data by accident. A shallow request is not populated to the WSGI 

67 environment. 

68 

69 .. versionchanged:: 0.5 

70 read-only mode was enforced by using immutables classes for all 

71 data. 

72 """

207 @classmethod 

208 def application(cls, f): 

209 """Decorate a function as responder that accepts the request as 

210 the last argument. This works like the :func:`responder` 

211 decorator but the function is passed the request object as the 

212 last argument and the request object will be closed 

213 automatically:: 

214 

215 @Request.application 

216 def my_wsgi_app(request): 

217 return Response('Hello World!') 

218 

219 As of Werkzeug 0.14 HTTP exceptions are automatically caught and 

220 converted to responses instead of failing. 

221 

222 :param f: the WSGI callable to decorate 

223 :return: a new WSGI callable 

224 """ 

225 #: return a callable that wraps the -2nd argument with the request 

226 #: and calls the function with all the arguments up to that one and 

227 #: the request. The return value is then called with the latest 

228 #: two arguments. This makes it possible to use this decorator for 

229 #: both standalone WSGI functions as well as bound methods and 

230 #: partially applied functions. 

231 from ..exceptions import HTTPException 

232 

233 def application(*args): 

234 request = cls(args[-2]) 

235 with request: 

236 try: 

237 resp = f(*args[:-2] + (request,)) 

238 except HTTPException as e: 

239 resp = e.get_response(args[-2]) 

240 return resp(*args[-2:]) 

241 

242 return update_wrapper(application, f)

298 """Method used internally to retrieve submitted data. After calling

300 filled with the incoming form data. As a matter of fact the input 

301 stream will be empty afterwards. You can also call this method to 

302 force the parsing of the form data. 

303 

304 .. versionadded:: 0.8 

305 """ 

306 # abort early if we have already consumed the stream 

307 if "form" in self.__dict__: 

308 return

310 _assert_not_shallow(self)

316 parser = self.make_form_data_parser() 

317 data = parser.parse( 

318 self._get_stream_for_parsing(), mimetype, content_length, options 

319 ) 

320 else: 

321 data = ( 

322 self.stream, 

323 self.parameter_storage_class(), 

324 self.parameter_storage_class(), 

325 )

427 def get_data(self, cache=True, as_text=False, parse_form_data=False): 

428 """This reads the buffered incoming data from the client into one 

429 bytestring. By default this is cached but that behavior can be 

430 changed by setting `cache` to `False`. 

431 

432 Usually it's a bad idea to call this method without checking the 

433 content length first as a client could send dozens of megabytes or more 

434 to cause memory problems on the server. 

435 

436 Note that if the form data was already parsed this method will not 

437 return anything as form data parsing does not cache the data like 

438 this method does. To implicitly invoke form data parsing function 

439 set `parse_form_data` to `True`. When this is done the return value 

440 of this method will be an empty string if the form parser handles 

441 the data. This generally is not necessary as if the whole data is 

442 cached (which is the default) the form parser will used the cached 

443 data to parse the form data. Please be generally aware of checking 

444 the content length first in any case before calling this method 

445 to avoid exhausting server memory. 

446 

447 If `as_text` is set to `True` the return value will be a decoded 

448 unicode string. 

449 

450 .. versionadded:: 0.9 

451 """ 

452 rv = getattr(self, "_cached_data", None) 

453 if rv is None: 

454 if parse_form_data: 

455 self._load_form_data() 

456 rv = self.stream.read() 

457 if cache: 

458 self._cached_data = rv 

459 if as_text: 

460 rv = rv.decode(self.charset, self.encoding_errors) 

461 return rv

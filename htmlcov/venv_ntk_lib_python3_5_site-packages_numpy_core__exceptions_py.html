17def _display_as_base(cls): 

18 """ 

19 A decorator that makes an exception class look like its base. 

20 

21 We use this to hide subclasses that are implementation details - the user 

22 should catch the base type, which is what the traceback will show them. 

23 

24 Classes decorated with this decorator are subject to removal without a 

25 deprecation warning. 

26 """ 

27 assert issubclass(cls, Exception) 

28 cls.__name__ = cls.__base__.__name__ 

29 cls.__qualname__ = cls.__base__.__qualname__ 

30 set_module(cls.__base__.__module__)(cls) 

31 return cls

144class _ArrayMemoryError(MemoryError): 

145 """ Thrown when an array cannot be allocated""" 

146 def __init__(self, shape, dtype): 

147 self.shape = shape 

148 self.dtype = dtype

150 @property 

151 def _total_size(self): 

152 num_bytes = self.dtype.itemsize 

153 for dim in self.shape: 

154 num_bytes *= dim 

155 return num_bytes

157 @staticmethod 

158 def _size_to_string(num_bytes): 

159 """ Convert a number of bytes into a binary size string """ 

160 import math 

161 

162 # https://en.wikipedia.org/wiki/Binary_prefix 

163 LOG2_STEP = 10 

164 STEP = 1024 

165 units = ['bytes', 'KiB', 'MiB', 'GiB', 'TiB', 'PiB', 'EiB'] 

166 

167 unit_i = max(num_bytes.bit_length() - 1, 1) // LOG2_STEP 

168 unit_val = 1 << (unit_i * LOG2_STEP) 

169 n_units = num_bytes / unit_val 

170 del unit_val 

171 

172 # ensure we pick a unit that is correct after rounding 

173 if round(n_units) == STEP: 

174 unit_i += 1 

175 n_units /= STEP 

176 

177 # deal with sizes so large that we don't have units for them 

178 if unit_i >= len(units): 

179 new_unit_i = len(units) - 1 

180 n_units *= 1 << ((unit_i - new_unit_i) * LOG2_STEP) 

181 unit_i = new_unit_i 

182 

183 unit_name = units[unit_i] 

184 # format with a sensible number of digits 

185 if unit_i == 0: 

186 # no decimal point on bytes 

187 return '{:.0f} {}'.format(n_units, unit_name) 

188 elif round(n_units) < 1000: 

189 # 3 significant figures, if none are dropped to the left of the . 

190 return '{:#.3g} {}'.format(n_units, unit_name) 

191 else: 

192 # just give all the digits otherwise 

193 return '{:#.0f} {}'.format(n_units, unit_name) 

194 

195 def __str__(self): 

196 size_str = self._size_to_string(self._total_size) 

197 return ( 

198 "Unable to allocate {} for an array with shape {} and data type {}" 

199 .format(size_str, self.shape, self.dtype)

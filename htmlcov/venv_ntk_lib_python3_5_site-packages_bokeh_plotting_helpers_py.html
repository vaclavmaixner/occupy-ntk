100def _single_stack(stackers, spec, **kw): 

101 if spec in kw: 

102 raise ValueError("Stack property '%s' cannot appear in keyword args" % spec) 

103 

104 lengths = { len(x) for x in kw.values() if isinstance(x, (list, tuple)) } 

105 

106 # lengths will be empty if there are no kwargs supplied at all 

107 if len(lengths) > 0: 

108 if len(lengths) != 1: 

109 raise ValueError("Keyword argument sequences for broadcasting must all be the same lengths. Got lengths: %r" % sorted(list(lengths))) 

110 if lengths.pop() != len(stackers): 

111 raise ValueError("Keyword argument sequences for broadcasting must be the same length as stackers") 

112 

113 s = [] 

114 

115 _kw = [] 

116 

117 for i, val in enumerate(stackers): 

118 d = {'name': val} 

119 s.append(val) 

120 

121 d[spec] = stack(*s) 

122 

123 for k, v in kw.items(): 

124 if isinstance(v, (list, tuple)): 

125 d[k] = v[i] 

126 else: 

127 d[k] = v 

128 

129 _kw.append(d)

287def _pop_colors_and_alpha(glyphclass, kwargs, prefix="", default_alpha=1.0): 

288 """ 

289 Given a kwargs dict, a prefix, and a default value, looks for different 

290 color and alpha fields of the given prefix, and fills in the default value 

291 if it doesn't exist. 

292 """ 

293 result = dict() 

294 

295 # TODO: The need to do this and the complexity of managing this kind of 

296 # thing throughout the codebase really suggests that we need to have 

297 # a real stylesheet class, where defaults and Types can declaratively 

298 # substitute for this kind of imperative logic. 

299 color = kwargs.pop(prefix + "color", get_default_color()) 

300 for argname in ("fill_color", "line_color"): 

301 if argname not in glyphclass.properties(): 

302 continue 

303 result[argname] = kwargs.pop(prefix + argname, color) 

304 

305 # NOTE: text fill color should really always default to black, hard coding 

306 # this here now until the stylesheet solution exists 

307 if "text_color" in glyphclass.properties(): 

308 result["text_color"] = kwargs.pop(prefix + "text_color", "black") 

309 

310 alpha = kwargs.pop(prefix + "alpha", default_alpha) 

311 for argname in ("fill_alpha", "line_alpha", "text_alpha"): 

312 if argname not in glyphclass.properties(): 

313 continue 

314 result[argname] = kwargs.pop(prefix + argname, alpha) 

315 

316 return result

349def _process_sequence_literals(glyphclass, kwargs, source, is_user_source): 

350 incompatible_literal_spec_values = [] 

351 dataspecs = glyphclass.dataspecs_with_props() 

352 for var, val in kwargs.items(): 

353 

354 # ignore things that are not iterable 

355 if not isinstance(val, Iterable): 

356 continue 

357 

358 # pass dicts (i.e., values or fields) on as-is 

359 if isinstance(val, dict): 

360 continue 

361 

362 # let any non-dataspecs do their own validation (e.g., line_dash properties) 

363 if var not in dataspecs: 

364 continue 

365 

366 # strings sequences are handled by the dataspec as-is 

367 if isinstance(val, string_types): 

368 continue 

369 

370 # similarly colorspecs handle color tuple sequences as-is 

371 if (isinstance(dataspecs[var].property, ColorSpec) and isinstance(val, tuple)): 

372 continue 

373 

374 if isinstance(val, np.ndarray) and val.ndim != 1: 

375 raise RuntimeError("Columns need to be 1D (%s is not)" % var) 

376 

377 if is_user_source: 

378 incompatible_literal_spec_values.append(var) 

379 else: 

380 source.add(val, name=var) 

381 kwargs[var] = var 

382 

383 return incompatible_literal_spec_values

436def _handle_legend_field(label, legend, glyph_renderer): 

437 if not isinstance(label, string_types): 

438 raise ValueError("legend_field value must be a string") 

439 label = field(label) 

440 item = _find_legend_item(label, legend) 

441 if item: 

442 item.renderers.append(glyph_renderer) 

443 else: 

444 new_item = LegendItem(label=label, renderers=[glyph_renderer]) 

445 legend.items.append(new_item)

448def _handle_legend_group(label, legend, glyph_renderer): 

449 if not isinstance(label, string_types): 

450 raise ValueError("legend_group value must be a string") 

451 

452 source = glyph_renderer.data_source 

453 if source is None: 

454 raise ValueError("Cannot use 'legend_group' on a glyph without a data source already configured") 

455 if not (hasattr(source, 'column_names') and label in source.column_names): 

456 raise ValueError("Column to be grouped does not exist in glyph data source") 

457 

458 column = source.data[label] 

459 vals, inds = np.unique(column, return_index=1) 

460 for val, ind in zip(vals, inds): 

461 label = value(str(val)) 

462 new_item = LegendItem(label=label, renderers=[glyph_renderer], index=ind) 

463 legend.items.append(new_item)

605def _tool_from_string(name): 

606 """ Takes a string and returns a corresponding `Tool` instance. """ 

607 known_tools = sorted(_known_tools.keys()) 

608 

609 if name in known_tools: 

610 tool_fn = _known_tools[name] 

611 

612 if isinstance(tool_fn, string_types): 

613 tool_fn = _known_tools[tool_fn] 

614 

615 return tool_fn() 

616 else: 

617 matches, text = difflib.get_close_matches(name.lower(), known_tools), "similar" 

618 

619 if not matches: 

620 matches, text = known_tools, "possible"

645def _process_tools_arg(plot, tools, tooltips=None): 

646 """ Adds tools to the plot object 

647 

648 Args: 

649 plot (Plot): instance of a plot object 

650 tools (seq[Tool or str]|str): list of tool types or string listing the 

651 tool names. Those are converted using the _tool_from_string 

652 function. I.e.: `wheel_zoom,box_zoom,reset`. 

653 tooltips (string or seq[tuple[str, str]], optional): 

654 tooltips to use to configure a HoverTool 

655 

656 Returns: 

657 list of Tools objects added to plot, map of supplied string names to tools 

658 """ 

659 tool_objs = [] 

660 tool_map = {} 

661 temp_tool_str = "" 

662 repeated_tools = [] 

663 

664 if isinstance(tools, (list, tuple)): 

665 for tool in tools: 

666 if isinstance(tool, Tool): 

667 tool_objs.append(tool) 

668 elif isinstance(tool, string_types): 

669 temp_tool_str += tool + ',' 

670 else: 

671 raise ValueError("tool should be a string or an instance of Tool class") 

672 tools = temp_tool_str 

673 

674 for tool in re.split(r"\s*,\s*", tools.strip()): 

675 # re.split will return empty strings; ignore them. 

676 if tool == "": 

677 continue 

678 

679 tool_obj = _tool_from_string(tool) 

680 tool_objs.append(tool_obj) 

681 tool_map[tool] = tool_obj 

682 

683 for typename, group in itertools.groupby( 

684 sorted(tool.__class__.__name__ for tool in tool_objs)): 

685 if len(list(group)) > 1: 

686 repeated_tools.append(typename) 

687 

688 if repeated_tools: 

689 warnings.warn("%s are being repeated" % ",".join(repeated_tools)) 

690 

691 if tooltips is not None: 

692 for tool_obj in tool_objs: 

693 if isinstance(tool_obj, HoverTool): 

694 tool_obj.tooltips = tooltips 

695 break 

696 else: 

697 tool_objs.append(HoverTool(tooltips=tooltips)) 

698 

699 return tool_objs, tool_map

747def _get_argspecs(glyphclass): 

748 argspecs = OrderedDict() 

749 for arg in glyphclass._args: 

750 spec = {} 

751 descriptor = getattr(glyphclass, arg) 

752 

753 # running python with -OO will discard docstrings -> __doc__ is None 

754 if descriptor.__doc__: 

755 spec['desc'] = "\n ".join(textwrap.dedent(descriptor.__doc__).split("\n")) 

756 else: 

757 spec['desc'] = "" 

758 spec['default'] = descriptor.class_default(glyphclass) 

759 spec['type'] = descriptor.property._sphinx_type() 

760 argspecs[arg] = spec 

761 return argspecs 

762 

763# This template generates the following: 

764# 

765# def foo(self, x, y=10, kwargs): 

766# kwargs['x'] = x 

767# kwargs['y'] = y 

768# return func(self, **kwargs) 

769_sigfunc_template = """ 

770def %s(self, %s, **kwargs): 

771%s 

772 return func(self, **kwargs) 

773""" 

774 

775def _get_sigfunc(func_name, func, argspecs): 

776 # This code is to wrap the generic func(*args, **kw) glyph method so that 

777 # a much better signature is available to users. E.g., for ``square`` we have: 

778 # 

779 # Signature: p.square(x, y, size=4, angle=0.0, **kwargs) 

780 # 

781 # which provides descriptive names for positional args, as well as any defaults 

782 func_args_with_defaults = [] 

783 for arg, spec in argspecs.items(): 

784 if spec['default'] is None: 

785 func_args_with_defaults.append(arg) 

786 else: 

787 func_args_with_defaults.append("%s=%r" % (arg, spec['default'])) 

788 args_text = ", ".join(func_args_with_defaults) 

789 kwargs_assign_text = "\n".join(" kwargs[%r] = %s" % (x, x) for x in argspecs) 

790 func_text = _sigfunc_template % (func_name, args_text, kwargs_assign_text) 

791 func_code = compile(func_text, "fakesource", "exec") 

792 func_globals = {} 

793 eval(func_code, {"func": func}, func_globals) 

794 return func_globals[func_name] 

795 

796_arg_template = """ %s (%s) : %s 

797 (default: %r) 

798""" 

799_doc_template = """ Configure and add :class:`~bokeh.models.%s.%s` glyphs to this Figure. 

800 

801Args: 

802%s 

803 

804Keyword Args: 

805%s 

806 

807Other Parameters: 

808 alpha (float) : an alias to set all alpha keyword args at once 

809 color (Color) : an alias to set all color keyword args at once 

810 source (ColumnDataSource) : a user supplied data source 

811 legend (str) : a legend tag for this glyph 

812 x_range_name (str) : name an extra range to use for mapping x-coordinates 

813 y_range_name (str) : name an extra range to use for mapping y-coordinates 

814 level (Enum) : control the render level order for this glyph 

815 

816It is also possible to set the color and alpha parameters of a "nonselection" 

817glyph. To do so, prefix any visual parameter with ``'nonselection_'``. 

818For example, pass ``nonselection_alpha`` or ``nonselection_fill_alpha``. 

819 

820Returns: 

821 GlyphRenderer 

822"""

824def _add_sigfunc_info(func, argspecs, glyphclass, extra_docs): 

825 func.__name__ = glyphclass.__name__ 

826 

827 omissions = {'js_event_callbacks', 'js_property_callbacks', 'subscribed_events'} 

828 

829 kwlines = [] 

830 kws = glyphclass.properties() - set(argspecs) 

831 for kw in kws: 

832 # these are not really useful, and should also really be private, just skip them 

833 if kw in omissions: continue 

834 

835 descriptor = getattr(glyphclass, kw) 

836 typ = descriptor.property._sphinx_type() 

837 if descriptor.__doc__: 

838 desc = "\n ".join(textwrap.dedent(descriptor.__doc__).split("\n")) 

839 else: 

840 desc = "" 

841 kwlines.append(_arg_template % (kw, typ, desc, descriptor.class_default(glyphclass))) 

842 extra_kws = getattr(glyphclass, '_extra_kws', {}) 

843 for kw, (typ, desc) in extra_kws.items(): 

844 kwlines.append(" %s (%s) : %s" % (kw, typ, desc)) 

845 kwlines.sort() 

846 

847 arglines = [] 

848 for arg, spec in argspecs.items(): 

849 arglines.append(_arg_template % (arg, spec['type'], spec['desc'], spec['default'])) 

850 

851 mod = "markers" if issubclass(glyphclass, Marker) else "glyphs" 

852 func.__doc__ = _doc_template % (mod, func.__name__, "\n".join(arglines), "\n".join(kwlines)) 

853 if extra_docs: 

854 func.__doc__ += extra_docs

856def _glyph_function(glyphclass, extra_docs=None): 

857 

858 def func(self, **kwargs): 

859 

860 # Convert data source, if necessary 

861 is_user_source = kwargs.get('source', None) is not None 

862 if is_user_source: 

863 source = kwargs['source'] 

864 if not isinstance(source, ColumnarDataSource): 

865 try: 

866 # try converting the soruce to ColumnDataSource 

867 source = ColumnDataSource(source) 

868 except ValueError as err: 

869 msg = "Failed to auto-convert {curr_type} to ColumnDataSource.\n Original error: {err}".format( 

870 curr_type=str(type(source)), 

871 err=err.message 

872 ) 

873 reraise(ValueError, ValueError(msg), sys.exc_info()[2]) 

874 

875 # update reddered_kws so that others can use the new source 

876 kwargs['source'] = source 

877 

878 # Save off legend kwargs before we get going 

879 legend_kwarg = _pop_legend_kwarg(kwargs) 

880 

881 # Need to check if user source is present before _pop_renderer_args 

882 renderer_kws = _pop_renderer_args(kwargs) 

883 source = renderer_kws['data_source'] 

884 

885 # Assign global_alpha from alpha if glyph type is an image 

886 if 'alpha' in kwargs and glyphclass.__name__ in ('Image', 'ImageRGBA', 'ImageURL'): 

887 kwargs['global_alpha'] = kwargs['alpha'] 

888 

889 # handle the main glyph, need to process literals 

890 glyph_ca = _pop_colors_and_alpha(glyphclass, kwargs) 

891 incompatible_literal_spec_values = [] 

892 incompatible_literal_spec_values += _process_sequence_literals(glyphclass, kwargs, source, is_user_source) 

893 incompatible_literal_spec_values += _process_sequence_literals(glyphclass, glyph_ca, source, is_user_source) 

894 if incompatible_literal_spec_values: 

895 raise RuntimeError(_GLYPH_SOURCE_MSG % nice_join(incompatible_literal_spec_values, conjuction="and")) 

896 

897 # handle the nonselection glyph, we always set one 

898 nsglyph_ca = _pop_colors_and_alpha(glyphclass, kwargs, prefix='nonselection_', default_alpha=0.1) 

899 

900 # handle the selection glyph, if any properties were given 

901 if any(x.startswith('selection_') for x in kwargs): 

902 sglyph_ca = _pop_colors_and_alpha(glyphclass, kwargs, prefix='selection_') 

903 else: 

904 sglyph_ca = None 

905 

906 # handle the hover glyph, if any properties were given 

907 if any(x.startswith('hover_') for x in kwargs): 

908 hglyph_ca = _pop_colors_and_alpha(glyphclass, kwargs, prefix='hover_') 

909 else: 

910 hglyph_ca = None 

911 

912 # handle the mute glyph, if any properties were given 

913 if any(x.startswith('muted_') for x in kwargs): 

914 mglyph_ca = _pop_colors_and_alpha(glyphclass, kwargs, prefix='muted_') 

915 else: 

916 mglyph_ca = None 

917 

918 glyph = _make_glyph(glyphclass, kwargs, glyph_ca) 

919 nsglyph = _make_glyph(glyphclass, kwargs, nsglyph_ca) 

920 sglyph = _make_glyph(glyphclass, kwargs, sglyph_ca) 

921 hglyph = _make_glyph(glyphclass, kwargs, hglyph_ca) 

922 mglyph = _make_glyph(glyphclass, kwargs, mglyph_ca) 

923 

924 glyph_renderer = GlyphRenderer(glyph=glyph, 

925 nonselection_glyph=nsglyph, 

926 selection_glyph=sglyph, 

927 hover_glyph=hglyph, 

928 muted_glyph=mglyph, 

929 **renderer_kws) 

930 

931 if legend_kwarg: 

932 _update_legend(self, legend_kwarg, glyph_renderer) 

933 

934 self.renderers.append(glyph_renderer) 

935 

936 return glyph_renderer

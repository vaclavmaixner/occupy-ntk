1# -*- coding: utf-8 -*- 

2"""WorkController can be used to instantiate in-process workers. 

3 

4The command-line interface for the worker is in :mod:`celery.bin.worker`, 

5while the worker program is in :mod:`celery.apps.worker`. 

6 

7The worker program is responsible for adding signal handlers, 

8setting up logging, etc. This is a bare-bones worker without 

9global side-effects (i.e., except for the global state stored in 

10:mod:`celery.worker.state`). 

11 

12The worker consists of several components, all managed by bootsteps 

13(mod:`celery.bootsteps`). 

14"""

185 def setup_includes(self, includes): 

186 # Update celery_include to have all known task modules, so that we 

187 # ensure all task modules are imported in case an execv happens. 

188 prev = tuple(self.app.conf.include) 

189 if includes: 

190 prev += tuple(includes) 

191 [self.app.loader.import_task_module(m) for m in includes] 

192 self.include = includes 

193 task_modules = {task.__class__.__module__ 

194 for task in values(self.app.tasks)}

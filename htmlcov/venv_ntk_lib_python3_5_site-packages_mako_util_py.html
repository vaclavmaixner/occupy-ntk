170class LRUCache(dict): 

171 

172 """A dictionary-like object that stores a limited number of items, 

173 discarding lesser used items periodically. 

174 

175 this is a rewrite of LRUCache from Myghty to use a periodic timestamp-based 

176 paradigm so that synchronization is not really needed. the size management 

177 is inexact. 

178 """ 

179 

180 class _Item(object): 

181 def __init__(self, key, value): 

182 self.key = key 

183 self.value = value 

184 self.timestamp = timeit.default_timer() 

185 

186 def __repr__(self): 

187 return repr(self.value) 

188 

189 def __init__(self, capacity, threshold=0.5): 

190 self.capacity = capacity 

191 self.threshold = threshold 

192 

193 def __getitem__(self, key): 

194 item = dict.__getitem__(self, key) 

195 item.timestamp = timeit.default_timer() 

196 return item.value 

197 

198 def values(self): 

199 return [i.value for i in dict.values(self)] 

200 

201 def setdefault(self, key, value): 

202 if key in self: 

203 return self[key] 

204 else: 

205 self[key] = value 

206 return value 

207 

208 def __setitem__(self, key, value): 

209 item = dict.get(self, key) 

210 if item is None: 

211 item = self._Item(key, value) 

212 dict.__setitem__(self, key, item) 

213 else: 

214 item.value = value 

215 self._manage_size() 

216 

217 def _manage_size(self): 

218 while len(self) > self.capacity + self.capacity * self.threshold: 

219 bytime = sorted( 

220 dict.values(self), 

221 key=operator.attrgetter("timestamp"), 

222 reverse=True, 

223 ) 

224 for item in bytime[self.capacity :]: 

225 try: 

226 del self[item.key] 

227 except KeyError: 

228 # if we couldn't find a key, most likely some other thread 

229 # broke in on us. loop around and try again 

230 break

239def parse_encoding(fp): 

240 """Deduce the encoding of a Python source file (binary mode) from magic 

241 comment. 

242 

243 It does this in the same way as the `Python interpreter`__ 

244 

245 .. __: http://docs.python.org/ref/encodings.html 

246 

247 The ``fp`` argument should be a seekable file object in binary mode. 

248 """ 

249 pos = fp.tell() 

250 fp.seek(0) 

251 try: 

252 line1 = fp.readline() 

253 has_bom = line1.startswith(codecs.BOM_UTF8) 

254 if has_bom: 

255 line1 = line1[len(codecs.BOM_UTF8) :] 

256 

257 m = _PYTHON_MAGIC_COMMENT_re.match(line1.decode("ascii", "ignore")) 

258 if not m: 

259 try: 

260 import parser 

261 

262 parser.suite(line1.decode("ascii", "ignore")) 

263 except (ImportError, SyntaxError): 

264 # Either it's a real syntax error, in which case the source 

265 # is not valid python source, or line2 is a continuation of 

266 # line1, in which case we don't want to scan line2 for a magic 

267 # comment. 

268 pass 

269 else: 

270 line2 = fp.readline() 

271 m = _PYTHON_MAGIC_COMMENT_re.match( 

272 line2.decode("ascii", "ignore") 

273 ) 

274 

275 if has_bom: 

276 if m: 

277 raise SyntaxError( 

278 "python refuses to compile code with both a UTF8" 

279 " byte-order-mark and a magic encoding comment" 

280 ) 

281 return "utf_8" 

282 elif m: 

283 return m.group(1) 

284 else: 

285 return None 

286 finally: 

287 fp.seek(pos)

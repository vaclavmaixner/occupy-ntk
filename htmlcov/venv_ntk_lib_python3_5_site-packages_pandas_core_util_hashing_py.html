21def _combine_hash_arrays(arrays, num_items): 

22 """ 

23 Parameters 

24 ---------- 

25 arrays : generator 

26 num_items : int 

27 

28 Should be the same as CPython's tupleobject.c 

29 """ 

30 try: 

31 first = next(arrays) 

32 except StopIteration: 

33 return np.array([], dtype=np.uint64) 

34 

35 arrays = itertools.chain([first], arrays) 

36 

37 mult = np.uint64(1000003) 

38 out = np.zeros_like(first) + np.uint64(0x345678) 

39 for i, a in enumerate(arrays): 

40 inverse_i = num_items - i 

41 out ^= a 

42 out *= mult 

43 mult += np.uint64(82520 + inverse_i + inverse_i) 

44 assert i + 1 == num_items, 'Fed in wrong num_items' 

45 out += np.uint64(97531) 

46 return out

49def hash_pandas_object(obj, index=True, encoding='utf8', hash_key=None, 

50 categorize=True): 

51 """ 

52 Return a data hash of the Index/Series/DataFrame 

53 

54 .. versionadded:: 0.19.2 

55 

56 Parameters 

57 ---------- 

58 index : boolean, default True 

59 include the index in the hash (if Series/DataFrame) 

60 encoding : string, default 'utf8' 

61 encoding for data & key when strings 

62 hash_key : string key to encode, default to _default_hash_key 

63 categorize : bool, default True 

64 Whether to first categorize object arrays before hashing. This is more 

65 efficient when the array contains duplicate values. 

66 

67 .. versionadded:: 0.20.0 

68 

69 Returns 

70 ------- 

71 Series of uint64, same length as the object 

72 """ 

73 from pandas import Series 

74 if hash_key is None: 

75 hash_key = _default_hash_key 

76 

77 if isinstance(obj, ABCMultiIndex): 

78 return Series(hash_tuples(obj, encoding, hash_key), 

79 dtype='uint64', copy=False) 

80 

81 if isinstance(obj, ABCIndexClass): 

82 h = hash_array(obj.values, encoding, hash_key, 

83 categorize).astype('uint64', copy=False) 

84 h = Series(h, index=obj, dtype='uint64', copy=False) 

85 elif isinstance(obj, ABCSeries): 

86 h = hash_array(obj.values, encoding, hash_key, 

87 categorize).astype('uint64', copy=False) 

88 if index: 

89 index_iter = (hash_pandas_object(obj.index, 

90 index=False, 

91 encoding=encoding, 

92 hash_key=hash_key, 

93 categorize=categorize).values 

94 for _ in [None]) 

95 arrays = itertools.chain([h], index_iter) 

96 h = _combine_hash_arrays(arrays, 2) 

97 

98 h = Series(h, index=obj.index, dtype='uint64', copy=False) 

99 

100 elif isinstance(obj, ABCDataFrame): 

101 hashes = (hash_array(series.values) for _, series in obj.iteritems()) 

102 num_items = len(obj.columns) 

103 if index: 

104 index_hash_generator = (hash_pandas_object(obj.index, 

105 index=False, 

106 encoding=encoding, 

107 hash_key=hash_key, 

108 categorize=categorize).values # noqa 

109 for _ in [None]) 

110 num_items += 1 

111 hashes = itertools.chain(hashes, index_hash_generator) 

112 h = _combine_hash_arrays(hashes, num_items) 

113 

114 h = Series(h, index=obj.index, dtype='uint64', copy=False) 

115 else: 

116 raise TypeError("Unexpected type for hashing %s" % type(obj)) 

117 return h

120def hash_tuples(vals, encoding='utf8', hash_key=None): 

121 """ 

122 Hash an MultiIndex / list-of-tuples efficiently 

123 

124 .. versionadded:: 0.20.0 

125 

126 Parameters 

127 ---------- 

128 vals : MultiIndex, list-of-tuples, or single tuple 

129 encoding : string, default 'utf8' 

130 hash_key : string key to encode, default to _default_hash_key 

131 

132 Returns 

133 ------- 

134 ndarray of hashed values array 

135 """ 

136 is_tuple = False 

137 if isinstance(vals, tuple): 

138 vals = [vals] 

139 is_tuple = True 

140 elif not is_list_like(vals): 

141 raise TypeError("must be convertible to a list-of-tuples") 

142 

143 from pandas import Categorical, MultiIndex 

144 

145 if not isinstance(vals, ABCMultiIndex): 

146 vals = MultiIndex.from_tuples(vals) 

147 

148 # create a list-of-Categoricals 

149 vals = [Categorical(vals.codes[level], 

150 vals.levels[level], 

151 ordered=False, 

152 fastpath=True) 

153 for level in range(vals.nlevels)] 

154 

155 # hash the list-of-ndarrays 

156 hashes = (_hash_categorical(cat, 

157 encoding=encoding, 

158 hash_key=hash_key) 

159 for cat in vals) 

160 h = _combine_hash_arrays(hashes, len(vals)) 

161 if is_tuple: 

162 h = h[0] 

163 

164 return h

190def _hash_categorical(c, encoding, hash_key): 

191 """ 

192 Hash a Categorical by hashing its categories, and then mapping the codes 

193 to the hashes 

194 

195 Parameters 

196 ---------- 

197 c : Categorical 

198 encoding : string, default 'utf8' 

199 hash_key : string key to encode, default to _default_hash_key 

200 

201 Returns 

202 ------- 

203 ndarray of hashed values array, same size as len(c) 

204 """ 

205 # Convert ExtensionArrays to ndarrays 

206 values = np.asarray(c.categories.values) 

207 hashed = hash_array(values, encoding, hash_key, 

208 categorize=False) 

209 

210 # we have uint64, as we don't directly support missing values 

211 # we don't want to use take_nd which will coerce to float 

212 # instead, directly construct the result with a 

213 # max(np.uint64) as the missing value indicator 

214 # 

215 # TODO: GH 15362 

216 

217 mask = c.isna() 

218 if len(hashed): 

219 result = hashed.take(c.codes) 

220 else: 

221 result = np.zeros(len(mask), dtype='uint64') 

222 

223 if mask.any(): 

224 result[mask] = np.iinfo(np.uint64).max 

225 

226 return result

229def hash_array(vals, encoding='utf8', hash_key=None, categorize=True): 

230 """ 

231 Given a 1d array, return an array of deterministic integers. 

232 

233 .. versionadded:: 0.19.2 

234 

235 Parameters 

236 ---------- 

237 vals : ndarray, Categorical 

238 encoding : string, default 'utf8' 

239 encoding for data & key when strings 

240 hash_key : string key to encode, default to _default_hash_key 

241 categorize : bool, default True 

242 Whether to first categorize object arrays before hashing. This is more 

243 efficient when the array contains duplicate values. 

244 

245 .. versionadded:: 0.20.0 

246 

247 Returns 

248 ------- 

249 1d uint64 numpy array of hash values, same length as the vals 

250 """ 

251 

252 if not hasattr(vals, 'dtype'): 

253 raise TypeError("must pass a ndarray-like") 

254 dtype = vals.dtype 

255 

256 if hash_key is None: 

257 hash_key = _default_hash_key 

258 

259 # For categoricals, we hash the categories, then remap the codes to the 

260 # hash values. (This check is above the complex check so that we don't ask 

261 # numpy if categorical is a subdtype of complex, as it will choke). 

262 if is_categorical_dtype(dtype): 

263 return _hash_categorical(vals, encoding, hash_key) 

264 elif is_extension_array_dtype(dtype): 

265 vals, _ = vals._values_for_factorize() 

266 dtype = vals.dtype 

267 

268 # we'll be working with everything as 64-bit values, so handle this 

269 # 128-bit value early 

270 if np.issubdtype(dtype, np.complex128): 

271 return hash_array(vals.real) + 23 * hash_array(vals.imag) 

272 

273 # First, turn whatever array this is into unsigned 64-bit ints, if we can 

274 # manage it. 

275 elif isinstance(dtype, np.bool): 

276 vals = vals.astype('u8') 

277 elif issubclass(dtype.type, (np.datetime64, np.timedelta64)): 

278 vals = vals.view('i8').astype('u8', copy=False) 

279 elif issubclass(dtype.type, np.number) and dtype.itemsize <= 8: 

280 vals = vals.view('u{}'.format(vals.dtype.itemsize)).astype('u8') 

281 else: 

282 # With repeated values, its MUCH faster to categorize object dtypes, 

283 # then hash and rename categories. We allow skipping the categorization 

284 # when the values are known/likely to be unique. 

285 if categorize: 

286 from pandas import factorize, Categorical, Index 

287 codes, categories = factorize(vals, sort=False) 

288 cat = Categorical(codes, Index(categories), 

289 ordered=False, fastpath=True) 

290 return _hash_categorical(cat, encoding, hash_key) 

291 

292 try: 

293 vals = hashing.hash_object_array(vals, hash_key, encoding) 

294 except TypeError: 

295 # we have mixed types 

296 vals = hashing.hash_object_array(vals.astype(str).astype(object), 

297 hash_key, encoding) 

298 

299 # Then, redistribute these 64-bit ints within the space of 64-bit ints 

300 vals ^= vals >> 30 

301 vals *= np.uint64(0xbf58476d1ce4e5b9) 

302 vals ^= vals >> 27 

303 vals *= np.uint64(0x94d049bb133111eb) 

304 vals ^= vals >> 31 

305 return vals

308def _hash_scalar(val, encoding='utf8', hash_key=None): 

309 """ 

310 Hash scalar value 

311 

312 Returns 

313 ------- 

314 1d uint64 numpy array of hash value, of length 1 

315 """ 

316 

317 if isna(val): 

318 # this is to be consistent with the _hash_categorical implementation 

319 return np.array([np.iinfo(np.uint64).max], dtype='u8') 

320 

321 if getattr(val, 'tzinfo', None) is not None: 

322 # for tz-aware datetimes, we need the underlying naive UTC value and 

323 # not the tz aware object or pd extension type (as 

324 # infer_dtype_from_scalar would do) 

325 if not isinstance(val, tslibs.Timestamp): 

326 val = tslibs.Timestamp(val) 

327 val = val.tz_convert(None) 

328 

329 dtype, val = infer_dtype_from_scalar(val) 

330 vals = np.array([val], dtype=dtype) 

331 

332 return hash_array(vals, hash_key=hash_key, encoding=encoding, 

333 categorize=False)

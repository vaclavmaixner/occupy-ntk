62def collect_filtered_models(discard, *input_values): 

63 ''' Collect a duplicate-free list of all other Bokeh models referred to by 

64 this model, or by any of its references, etc, unless filtered-out by the 

65 provided callable. 

66 

67 Iterate over ``input_values`` and descend through their structure 

68 collecting all nested ``Models`` on the go. 

69 

70 Args: 

71 *discard (Callable[[Model], bool]) 

72 a callable which accepts a *Model* instance as its single argument 

73 and returns a boolean stating whether to discard the instance. The 

74 latter means that the instance will not be added to collected 

75 models nor will its references be explored. 

76 

77 *input_values (Model) 

78 Bokeh models to collect other models from 

79 

80 Returns: 

81 None 

82 

83 ''' 

84 

85 ids = set([]) 

86 collected = [] 

87 queued = [] 

88 

89 def queue_one(obj): 

90 if obj.id not in ids and not (callable(discard) and discard(obj)): 

91 queued.append(obj) 

92 

93 for value in input_values: 

94 _visit_value_and_its_immediate_references(value, queue_one) 

95 

96 while queued: 

97 obj = queued.pop(0) 

98 if obj.id not in ids: 

99 ids.add(obj.id) 

100 collected.append(obj) 

101 _visit_immediate_value_references(obj, queue_one) 

102 

103 return collected

162class MetaModel(MetaHasProps): 

163 ''' Specialize the construction of |Model| classes. 

164 

165 This class is a `metaclass`_ for |Model| that is responsible for 

166 automatically cataloging all Bokeh models that get defined, so that the 

167 serialization machinery between Bokeh and BokehJS can function properly. 

168 

169 .. note:: 

170 It is worth pointing out explicitly that this relies on the rules 

171 for Metaclass inheritance in Python. 

172 

173 Bokeh works by replicating Python model objects (e.g. plots, ranges, 

174 data sources, which are all |HasProps| subclasses) into BokehJS. In the 

175 case of using a Bokeh server, the Bokeh model objects can also be 

176 synchronized bidirectionally. This is accomplished by serializing the 

177 models to and from a JSON format, that includes the name of the model type 

178 as part of the payload, as well as a unique ID, and all the attributes: 

179 

180 .. code-block:: javascript 

181 

182 { 

183 type: "Plot", 

184 id: 100032, 

185 attributes: { ... } 

186 } 

187 

188 Typically the type name is inferred automatically from the Python class 

189 name, and is set as the ``__view_model__`` class attribute on the Model 

190 class that is create. But it is also possible to override this value 

191 explicitly: 

192 

193 .. code-block:: python 

194 

195 class Foo(Model): pass 

196 

197 class Bar(Model): 

198 __view_model__ == "Quux" 

199 

200 This metaclass will raise an error if two Bokeh models are created that 

201 attempt to have the same view model name. The only exception made is if 

202 one of the models has a custom ``__implementation__`` in its class 

203 definition. 

204 

205 This metaclass also handles subtype relationships between Bokeh models. 

206 Occasionally it may be necessary for multiple class types on the Python 

207 side to resolve to the same type on the BokehJS side. This is called 

208 subtyping, and is expressed through a ``__subtype__`` class attribute on 

209 a model: 

210 

211 .. code-block:: python 

212 

213 class Foo(Model): pass 

214 

215 class Bar(Foo): 

216 __view_model__ = "Foo" 

217 __subtype__ = "Bar" 

218 

219 In this case, python instances of ``Foo`` and ``Bar`` will both resolve to 

220 ``Foo`` models in BokehJS. In the context of a Bokeh server application, 

221 the original python types will be faithfully round-tripped. (Without the 

222 ``__subtype__`` specified, the above code would raise an error due to 

223 duplicate view model names.) 

224 

225 .. _metaclass: https://docs.python.org/3/reference/datamodel.html#metaclasses 

226 

227 ''' 

228

231 def __new__(meta_cls, class_name, bases, class_dict): 

232 ''' 

233 

234 Raises: 

235 Warning 

236 

237 ''' 

238 

239 # use an explicitly provided view model name if there is one 

240 if "__view_model__" not in class_dict: 

241 class_dict["__view_model__"] = class_name 

242 if "__view_module__" not in class_dict: 

243 class_dict["__view_module__"] = class_dict["__module__"] 

244 

245 module = class_dict["__view_module__"] 

246 model = class_dict.get("__subtype__", class_dict["__view_model__"]) 

247 impl = class_dict.get("__implementation__", None) 

248 

249 head = module.split(".")[0] 

250 if head == "bokeh" or head == "__main__" or impl is not None: 

251 qualified = model 

252 else: 

253 qualified = module + "." + model 

254 

255 class_dict["__qualified_model__"] = qualified 

256 

257 # call the parent metaclass to create the new model type 

258 newcls = super(MetaModel, meta_cls).__new__(meta_cls, class_name, bases, class_dict) 

259 

260 # update the mapping of view model names to classes, checking for any duplicates 

261 # and handling any subtype relationships or custom implementations 

262 if qualified in MetaModel.model_class_reverse_map and not hasattr(newcls, "__implementation__"): 

263 raise Warning("Duplicate qualified model declaration of '%s'. Previous definition: %s" % ( 

264 qualified, MetaModel.model_class_reverse_map[qualified])) 

265 MetaModel.model_class_reverse_map[qualified] = newcls 

266 

267 return newcls

749 def _to_json_like(self, include_defaults): 

750 ''' Returns a dictionary of the attributes of this object, in 

751 a layout corresponding to what BokehJS expects at unmarshalling time. 

752 

753 This method does not convert "Bokeh types" into "plain JSON types," 

754 for example each child Model will still be a Model, rather 

755 than turning into a reference, numpy isn't handled, etc. 

756 That's what "json like" means. 

757 

758 This method should be considered "private" or "protected", 

759 for use internal to Bokeh; use ``to_json()`` instead because 

760 it gives you only plain JSON-compatible types. 

761 

762 Args: 

763 include_defaults (bool) : whether to include attributes 

764 that haven't been changed from the default. 

765 

766 ''' 

767 all_attrs = self.properties_with_values(include_defaults=include_defaults) 

768 

769 # If __subtype__ is defined, then this model may introduce properties 

770 # that don't exist on __view_model__ in bokehjs. Don't serialize such 

771 # properties. 

772 subtype = getattr(self.__class__, "__subtype__", None) 

773 if subtype is not None and subtype != self.__class__.__view_model__: 

774 attrs = {} 

775 for attr, value in all_attrs.items(): 

776 if attr in self.__class__.__dict__: 

777 continue 

778 else: 

779 attrs[attr] = value 

780 else: 

781 attrs = all_attrs 

782 

783 for (k, v) in attrs.items(): 

784 # we can't serialize Infinity, we send it as None and 

785 # the other side has to fix it up. This transformation 

786 # can't be in our json_encoder because the json 

787 # module checks for inf before it calls the custom 

788 # encoder. 

789 if isinstance(v, float) and v == float('inf'): 

790 attrs[k] = None 

791 

792 return attrs

837def _visit_immediate_value_references(value, visitor): 

838 ''' Visit all references to another Model without recursing into any 

839 of the child Model; may visit the same Model more than once if 

840 it's referenced more than once. Does not visit the passed-in value. 

841 

842 ''' 

843 if isinstance(value, HasProps): 

844 for attr in value.properties_with_refs(): 

845 child = getattr(value, attr) 

846 _visit_value_and_its_immediate_references(child, visitor) 

847 else: 

848 _visit_value_and_its_immediate_references(value, visitor) 

849 

850 

851_common_types = {int, float, str} 

852 

853def _visit_value_and_its_immediate_references(obj, visitor): 

854 ''' Recurse down Models, HasProps, and Python containers 

855 

856 The ordering in this function is to optimize performance. We check the 

857 most comomn types (int, float, str) first so that we can quickly return in 

858 the common case. We avoid isinstance and issubclass checks in a couple 

859 places with `type` checks because isinstance checks can be slow. 

860 ''' 

861 typ = type(obj) 

862 if typ in _common_types: # short circuit on common base types 

863 return 

864 if typ is list or issubclass(typ, (list, tuple)): # check common containers 

865 for item in obj: 

866 _visit_value_and_its_immediate_references(item, visitor) 

867 elif issubclass(typ, dict): 

868 for key, value in iteritems(obj): 

869 _visit_value_and_its_immediate_references(key, visitor) 

870 _visit_value_and_its_immediate_references(value, visitor) 

871 elif issubclass(typ, HasProps): 

872 if issubclass(typ, Model): 

873 visitor(obj) 

874 else: 

875 # this isn't a Model, so recurse into it 

876 _visit_immediate_value_references(obj, visitor)

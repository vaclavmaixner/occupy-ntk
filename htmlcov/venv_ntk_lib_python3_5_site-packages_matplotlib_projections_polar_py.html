151 def transform_non_affine(self, xy): 

152 x = xy[:, 0:1] 

153 y = xy[:, 1:] 

154 r = np.sqrt(x*x + y*y) 

155 with np.errstate(invalid='ignore'): 

156 # At x=y=r=0 this will raise an 

157 # invalid value warning when doing 0/0 

158 # Divide by zero warnings are only raised when 

159 # the numerator is different from 0. That 

160 # should not happen here. 

161 theta = np.arccos(x / r) 

162 theta = np.where(y < 0, 2 * np.pi - theta, theta) 

163 

164 # PolarAxes does not use the theta transforms here, but apply them for 

165 # backwards-compatibility if not being used by it. 

166 if self._apply_theta_transforms and self._axis is not None: 

167 theta -= self._axis.get_theta_offset() 

168 theta *= self._axis.get_theta_direction() 

169 theta %= 2 * np.pi 

170 

171 if self._use_rmin and self._axis is not None: 

172 r += self._axis.get_rorigin() 

173 

174 return np.concatenate((theta, r), 1)

759class _WedgeBbox(mtransforms.Bbox): 

760 """ 

761 Transform (theta,r) wedge Bbox into axes bounding box. 

762 

763 Parameters 

764 ---------- 

765 center : (float, float) 

766 Center of the wedge 

767 viewLim : `~matplotlib.transforms.Bbox` 

768 Bbox determining the boundaries of the wedge 

769 originLim : `~matplotlib.transforms.Bbox` 

770 Bbox determining the origin for the wedge, if different from *viewLim* 

771 """ 

772 def __init__(self, center, viewLim, originLim, **kwargs): 

773 mtransforms.Bbox.__init__(self, 

774 np.array([[0.0, 0.0], [1.0, 1.0]], np.float), 

775 **kwargs) 

776 self._center = center 

777 self._viewLim = viewLim 

778 self._originLim = originLim 

779 self.set_children(viewLim, originLim) 

780 

781 def __str__(self): 

782 return ("{}(\n" 

783 "{},\n" 

784 "{},\n" 

785 "{})" 

786 .format(type(self).__name__, 

787 mtransforms._indent_str(self._center), 

788 mtransforms._indent_str(self._viewLim), 

789 mtransforms._indent_str(self._originLim)))

791 def get_points(self): 

792 if self._invalid: 

793 points = self._viewLim.get_points().copy() 

794 

795 # Scale angular limits to work with Wedge. 

796 points[:, 0] *= 180 / np.pi 

797 if points[0, 0] > points[1, 0]: 

798 points[:, 0] = points[::-1, 0] 

799 

800 # Scale radial limits based on origin radius. 

801 points[:, 1] -= self._originLim.y0 

802 

803 # Scale radial limits to match axes limits. 

804 rscale = 0.5 / points[1, 1] 

805 points[:, 1] *= rscale 

806 width = min(points[1, 1] - points[0, 1], 0.5) 

807 

808 # Generate bounding box for wedge. 

809 wedge = mpatches.Wedge(self._center, points[1, 1], 

810 points[0, 0], points[1, 0], 

811 width=width) 

812 self.update_from_path(wedge.get_path()) 

813 

814 # Ensure equal aspect ratio. 

815 w, h = self._points[1] - self._points[0] 

816 if h < w: 

817 deltah = (w - h) / 2.0 

818 deltaw = 0.0 

819 elif w < h: 

820 deltah = 0.0 

821 deltaw = (h - w) / 2.0 

822 else: 

823 deltah = 0.0 

824 deltaw = 0.0 

825 self._points += np.array([[-deltaw, -deltah], [deltaw, deltah]]) 

826 

827 self._invalid = 0 

828 

829 return self._points

889 def _set_lim_and_transforms(self): 

890 # A view limit where the minimum radius can be locked if the user 

891 # specifies an alternate origin. 

892 self._originViewLim = mtransforms.LockableBbox(self.viewLim) 

893 

894 # Handle angular offset and direction. 

895 self._direction = mtransforms.Affine2D() \ 

896 .scale(self._default_theta_direction, 1.0) 

897 self._theta_offset = mtransforms.Affine2D() \ 

898 .translate(self._default_theta_offset, 0.0) 

899 self.transShift = mtransforms.composite_transform_factory( 

900 self._direction, 

901 self._theta_offset) 

902 # A view limit shifted to the correct location after accounting for 

903 # orientation and offset. 

904 self._realViewLim = mtransforms.TransformedBbox(self.viewLim, 

905 self.transShift) 

906 

907 # Transforms the x and y axis separately by a scale factor 

908 # It is assumed that this part will have non-linear components 

909 self.transScale = mtransforms.TransformWrapper( 

910 mtransforms.IdentityTransform()) 

911 

912 # Scale view limit into a bbox around the selected wedge. This may be 

913 # smaller than the usual unit axes rectangle if not plotting the full 

914 # circle. 

915 self.axesLim = _WedgeBbox((0.5, 0.5), 

916 self._realViewLim, self._originViewLim) 

917 

918 # Scale the wedge to fill the axes. 

919 self.transWedge = mtransforms.BboxTransformFrom(self.axesLim) 

920 

921 # Scale the axes to fill the figure. 

922 self.transAxes = mtransforms.BboxTransformTo(self.bbox) 

923 

924 # A (possibly non-linear) projection on the (already scaled) 

925 # data. This one is aware of rmin 

926 self.transProjection = self.PolarTransform( 

927 self, 

928 _apply_theta_transforms=False) 

929 # Add dependency on rorigin. 

930 self.transProjection.set_children(self._originViewLim) 

931 

932 # An affine transformation on the data, generally to limit the 

933 # range of the axes 

934 self.transProjectionAffine = self.PolarAffine(self.transScale, 

935 self._originViewLim) 

936 

937 # The complete data transformation stack -- from data all the 

938 # way to display coordinates 

939 self.transData = ( 

940 self.transScale + self.transShift + self.transProjection + 

941 (self.transProjectionAffine + self.transWedge + self.transAxes)) 

942 

943 # This is the transform for theta-axis ticks. It is 

944 # equivalent to transData, except it always puts r == 0.0 and r == 1.0 

945 # at the edge of the axis circles. 

946 self._xaxis_transform = ( 

947 mtransforms.blended_transform_factory( 

948 mtransforms.IdentityTransform(), 

949 mtransforms.BboxTransformTo(self.viewLim)) + 

950 self.transData) 

951 # The theta labels are flipped along the radius, so that text 1 is on 

952 # the outside by default. This should work the same as before. 

953 flipr_transform = mtransforms.Affine2D() \ 

954 .translate(0.0, -0.5) \ 

955 .scale(1.0, -1.0) \ 

956 .translate(0.0, 0.5) 

957 self._xaxis_text_transform = flipr_transform + self._xaxis_transform 

958 

959 # This is the transform for r-axis ticks. It scales the theta 

960 # axis so the gridlines from 0.0 to 1.0, now go from thetamin to 

961 # thetamax. 

962 self._yaxis_transform = ( 

963 mtransforms.blended_transform_factory( 

964 mtransforms.BboxTransformTo(self.viewLim), 

965 mtransforms.IdentityTransform()) + 

966 self.transData) 

967 # The r-axis labels are put at an angle and padded in the r-direction 

968 self._r_label_position = mtransforms.Affine2D() \ 

969 .translate(self._default_rlabel_position, 0.0) 

970 self._yaxis_text_transform = mtransforms.TransformWrapper( 

971 self._r_label_position + self.transData)

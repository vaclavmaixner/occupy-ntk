113class CustomJSFilter(Filter): 

114 ''' Filter data sources with a custom defined JavaScript function. 

115 

116 .. warning:: 

117 The explicit purpose of this Bokeh Model is to embed *raw JavaScript 

118 code* for a browser to execute. If any part of the code is derived 

119 from untrusted user inputs, then you must take appropriate care to 

120 sanitize the user input prior to passing to Bokeh. 

121 

122 ''' 

123 

124 @classmethod 

125 def from_py_func(cls, func): 

126 ''' Create a ``CustomJSFilter`` instance from a Python function. The 

127 function is translated to JavaScript using PScript. 

128 

129 The ``func`` function namespace will contain the variable ``source`` 

130 at render time. This will be the data source associated with the ``CDSView`` 

131 that this filter is added to. 

132 ''' 

133 from bokeh.util.deprecation import deprecated 

134 deprecated("'from_py_func' is deprecated and will be removed in an eventual 2.0 release. " 

135 "Use CustomJSFilter directly instead.") 

136 

137 if not isinstance(func, FunctionType): 

138 raise ValueError('CustomJSFilter.from_py_func only accepts function objects.') 

139 

140 pscript = import_required( 

141 'pscript', 

142 dedent("""\ 

143 To use Python functions for CustomJSFilter, you need PScript 

144 '("conda install -c conda-forge pscript" or "pip install pscript")""") 

145 ) 

146 

147 argspec = inspect.getargspec(func) 

148 

149 default_names = argspec.args 

150 default_values = argspec.defaults or [] 

151 

152 if len(default_names) - len(default_values) != 0: 

153 raise ValueError("Function may only contain keyword arguments.") 

154 

155 # should the following be all of the values need to be Models? 

156 if default_values and not any(isinstance(value, Model) for value in default_values): 

157 raise ValueError("Default value must be a plot object.") 

158 

159 func_kwargs = dict(zip(default_names, default_values)) 

160 code = pscript.py2js(func, 'filter') + 'return filter(%s);\n' % ', '.join(default_names) 

161 

162 return cls(code=code, args=func_kwargs)

164 @classmethod 

165 def from_coffeescript(cls, code, args={}): 

166 ''' Create a ``CustomJSFilter`` instance from CoffeeScript snippets. 

167 The function bodies are translated to JavaScript functions using node 

168 and therefore require return statements. 

169 

170 The ``code`` function namespace will contain the variable ``source`` 

171 at render time. This will be the data source associated with the 

172 ``CDSView`` that this filter is added to. 

173 ''' 

174 

175 compiled = nodejs_compile(code, lang="coffeescript", file="???") 

176 if "error" in compiled: 

177 raise CompilationError(compiled.error) 

178 else: 

179 return cls(code=compiled.code, args=args) 

180 

181 args = Dict(String, AnyRef, help=""" 

182 A mapping of names to Python objects. In particular those can be bokeh's models. 

183 These objects are made available to the callback's code snippet as the values of 

184 named parameters to the callback. 

185 """) 

186 

187 code = String(default="", help=""" 

188 A snippet of JavaScript code to filter data contained in a columnar data source. 

189 The code is made into the body of a function, and all of of the named objects in 

190 ``args`` are available as parameters that the code can use. The variable 

191 ``source`` will contain the data source that is associated with the ``CDSView`` this 

192 filter is added to. 

193 

194 The code should either return the indices of the subset or an array of booleans 

195 to use to subset data source rows. 

196 

197 Example: 

198 

199 .. code-block:: javascript 

200 

201 code = ''' 

202 var indices = []; 

203 for (var i = 0; i <= source.data['some_column'].length; i++){ 

204 if (source.data['some_column'][i] == 'some_value') { 

205 indices.push(i) 

206 } 

207 } 

208 return indices; 

209 ''' 

210 

211 .. note:: Use ``CustomJS.from_coffeescript()`` for CoffeeScript source code. 

212 

213 """) 

214 

215 use_strict = Bool(default=False, help=""" 

216 Enables or disables automatic insertion of ``"use strict";`` into ``code``. 

217 """)

239 def get_label_coords(self, distances, XX, YY, ysize, lw): 

240 """ 

241 Return x, y, and the index of a label location. 

242 

243 Labels are plotted at a location with the smallest 

244 deviation of the contour from a straight line 

245 unless there is another label nearby, in which case 

246 the next best place on the contour is picked up. 

247 If all such candidates are rejected, the beginning 

248 of the contour is chosen. 

249 """ 

250 hysize = int(ysize / 2) 

251 adist = np.argsort(distances) 

252 

253 for ind in adist: 

254 x, y = XX[ind][hysize], YY[ind][hysize] 

255 if self.too_close(x, y, lw): 

256 continue 

257 return x, y, ind 

258 

259 ind = adist[0] 

260 x, y = XX[ind][hysize], YY[ind][hysize] 

261 return x, y, ind

370 def calc_label_rot_and_inline(self, slc, ind, lw, lc=None, spacing=5): 

371 """ 

372 This function calculates the appropriate label rotation given 

373 the linecontour coordinates in screen units, the index of the 

374 label location and the label width. 

375 

376 It will also break contour and calculate inlining if *lc* is 

377 not empty (lc defaults to the empty list if None). *spacing* 

378 is the space around the label in pixels to leave empty. 

379 

380 Do both of these tasks at once to avoid calculating path lengths 

381 multiple times, which is relatively costly. 

382 

383 The method used here involves calculating the path length 

384 along the contour in pixel coordinates and then looking 

385 approximately label width / 2 away from central point to 

386 determine rotation and then to break contour if desired. 

387 """ 

388 

389 if lc is None: 

390 lc = [] 

391 # Half the label width 

392 hlw = lw / 2.0 

393 

394 # Check if closed and, if so, rotate contour so label is at edge 

395 closed = _is_closed_polygon(slc) 

396 if closed: 

397 slc = np.r_[slc[ind:-1], slc[:ind + 1]] 

398 

399 if len(lc): # Rotate lc also if not empty 

400 lc = np.r_[lc[ind:-1], lc[:ind + 1]] 

401 

402 ind = 0 

403 

404 # Calculate path lengths 

405 pl = np.zeros(slc.shape[0], dtype=float) 

406 dx = np.diff(slc, axis=0) 

407 pl[1:] = np.cumsum(np.hypot(dx[:, 0], dx[:, 1])) 

408 pl = pl - pl[ind] 

409 

410 # Use linear interpolation to get points around label 

411 xi = np.array([-hlw, hlw]) 

412 if closed: # Look at end also for closed contours 

413 dp = np.array([pl[-1], 0]) 

414 else: 

415 dp = np.zeros_like(xi) 

416 

417 # Get angle of vector between the two ends of the label - must be 

418 # calculated in pixel space for text rotation to work correctly. 

419 (dx,), (dy,) = (np.diff(np.interp(dp + xi, pl, slc_col)) 

420 for slc_col in slc.T) 

421 rotation = np.rad2deg(np.arctan2(dy, dx)) 

422 

423 if self.rightside_up: 

424 # Fix angle so text is never upside-down 

425 rotation = (rotation + 90) % 180 - 90 

426 

427 # Break contour if desired 

428 nlc = [] 

429 if len(lc): 

430 # Expand range by spacing 

431 xi = dp + xi + np.array([-spacing, spacing]) 

432 

433 # Get (integer) indices near points of interest; use -1 as marker 

434 # for out of bounds. 

435 I = np.interp(xi, pl, np.arange(len(pl)), left=-1, right=-1) 

436 I = [np.floor(I[0]).astype(int), np.ceil(I[1]).astype(int)] 

437 if I[0] != -1: 

438 xy1 = [np.interp(xi[0], pl, lc_col) for lc_col in lc.T] 

439 if I[1] != -1: 

440 xy2 = [np.interp(xi[1], pl, lc_col) for lc_col in lc.T] 

441 

442 # Actually break contours 

443 if closed: 

444 # This will remove contour if shorter than label 

445 if all(i != -1 for i in I): 

446 nlc.append(np.row_stack([xy2, lc[I[1]:I[0]+1], xy1])) 

447 else: 

448 # These will remove pieces of contour if they have length zero 

449 if I[0] != -1: 

450 nlc.append(np.row_stack([lc[:I[0]+1], xy1])) 

451 if I[1] != -1: 

452 nlc.append(np.row_stack([xy2, lc[I[1]:]])) 

453 

454 # The current implementation removes contours completely 

455 # covered by labels. Uncomment line below to keep 

456 # original contour if this is the preferred behavior. 

457 # if not len(nlc): nlc = [ lc ] 

458 

459 return rotation, nlc

638 slc = slc0

671def _find_closest_point_on_leg(p1, p2, p0): 

672 """Find the closest point to p0 on line segment connecting p1 and p2.""" 

673 

674 # handle degenerate case 

675 if np.all(p2 == p1): 

676 d = np.sum((p0 - p1)**2) 

677 return d, p1 

678 

679 d21 = p2 - p1 

680 d01 = p0 - p1 

681 

682 # project on to line segment to find closest point 

683 proj = np.dot(d01, d21) / np.dot(d21, d21) 

684 if proj < 0: 

685 proj = 0 

686 if proj > 1: 

687 proj = 1 

688 pc = p1 + proj * d21 

689 

690 # find squared distance 

691 d = np.sum((pc-p0)**2) 

692 

693 return d, pc

696def _is_closed_polygon(X): 

697 """ 

698 Return whether first and last object in a sequence are the same. These are 

699 presumably coordinates on a polygonal curve, in which case this function 

700 tests if that curve is closed. 

701 """ 

702 return np.all(X[0] == X[-1])

705def _find_closest_point_on_path(lc, point): 

706 """ 

707 lc: coordinates of vertices 

708 point: coordinates of test point 

709 """ 

710 

711 # find index of closest vertex for this segment 

712 ds = np.sum((lc - point[None, :])**2, 1) 

713 imin = np.argmin(ds) 

714 

715 dmin = np.inf 

716 xcmin = None 

717 legmin = (None, None) 

718 

719 closed = _is_closed_polygon(lc) 

720 

721 # build list of legs before and after this vertex 

722 legs = [] 

723 if imin > 0 or closed: 

724 legs.append(((imin-1) % len(lc), imin)) 

725 if imin < len(lc) - 1 or closed: 

726 legs.append((imin, (imin+1) % len(lc))) 

727 

728 for leg in legs: 

729 d, xc = _find_closest_point_on_leg(lc[leg[0]], lc[leg[1]], point) 

730 if d < dmin: 

731 dmin = d 

732 xcmin = xc 

733 legmin = leg 

734 

735 return (dmin, xcmin, legmin)

1173 def _autolev(self, N): 

1174 """ 

1175 Select contour levels to span the data. 

1176 

1177 The target number of levels, *N*, is used only when the 

1178 scale is not log and default locator is used. 

1179 

1180 We need two more levels for filled contours than for 

1181 line contours, because for the latter we need to specify 

1182 the lower and upper boundary of each range. For example, 

1183 a single contour boundary, say at z = 0, requires only 

1184 one contour line, but two filled regions, and therefore 

1185 three levels to provide boundaries for both regions. 

1186 """ 

1187 self._auto = True 

1188 if self.locator is None: 

1189 if self.logscale: 

1190 self.locator = ticker.LogLocator() 

1191 else: 

1192 self.locator = ticker.MaxNLocator(N + 1, min_n_ticks=1) 

1193 

1194 lev = self.locator.tick_values(self.zmin, self.zmax) 

1195 

1196 try: 

1197 if self.locator._symmetric: 

1198 return lev 

1199 except AttributeError: 

1200 pass 

1201 

1202 # Trim excess levels the locator may have supplied. 

1203 under = np.nonzero(lev < self.zmin)[0] 

1204 i0 = under[-1] if len(under) else 0 

1205 over = np.nonzero(lev > self.zmax)[0] 

1206 i1 = over[0] + 1 if len(over) else len(lev) 

1207 if self.extend in ('min', 'both'): 

1208 i0 += 1 

1209 if self.extend in ('max', 'both'): 

1210 i1 -= 1 

1211 

1212 if i1 - i0 < 3: 

1213 i0, i1 = 0, len(lev) 

1214 

1215 return lev[i0:i1]

1252 def _process_levels(self): 

1253 """ 

1254 Assign values to :attr:`layers` based on :attr:`levels`, 

1255 adding extended layers as needed if contours are filled. 

1256 

1257 For line contours, layers simply coincide with levels; 

1258 a line is a thin layer. No extended levels are needed 

1259 with line contours. 

1260 """ 

1261 # Make a private _levels to include extended regions; we 

1262 # want to leave the original levels attribute unchanged. 

1263 # (Colorbar needs this even for line contours.) 

1264 self._levels = list(self.levels) 

1265 

1266 if self.logscale: 

1267 lower, upper = 1e-250, 1e250 

1268 else: 

1269 lower, upper = -1e250, 1e250 

1270 

1271 if self.extend in ('both', 'min'): 

1272 self._levels.insert(0, lower) 

1273 if self.extend in ('both', 'max'): 

1274 self._levels.append(upper) 

1275 self._levels = np.asarray(self._levels) 

1276 

1277 if not self.filled: 

1278 self.layers = self.levels 

1279 return 

1280 

1281 # Layer values are mid-way between levels in screen space. 

1282 if self.logscale: 

1283 # Avoid overflow by taking sqrt before multiplying. 

1284 self.layers = (np.sqrt(self._levels[:-1]) 

1285 * np.sqrt(self._levels[1:])) 

1286 else: 

1287 self.layers = 0.5 * (self._levels[:-1] + self._levels[1:])

1289 def _process_colors(self): 

1290 """ 

1291 Color argument processing for contouring. 

1292 

1293 Note that we base the color mapping on the contour levels 

1294 and layers, not on the actual range of the Z values. This 

1295 means we don't have to worry about bad values in Z, and we 

1296 always have the full dynamic range available for the selected 

1297 levels. 

1298 

1299 The color is based on the midpoint of the layer, except for 

1300 extended end layers. By default, the norm vmin and vmax 

1301 are the extreme values of the non-extended levels. Hence, 

1302 the layer color extremes are not the extreme values of 

1303 the colormap itself, but approach those values as the number 

1304 of levels increases. An advantage of this scheme is that 

1305 line contours, when added to filled contours, take on 

1306 colors that are consistent with those of the filled regions; 

1307 for example, a contour line on the boundary between two 

1308 regions will have a color intermediate between those 

1309 of the regions. 

1310 

1311 """ 

1312 self.monochrome = self.cmap.monochrome 

1313 if self.colors is not None: 

1314 # Generate integers for direct indexing.

1316 if self.filled: 

1317 i1 -= 1 

1318 # Out of range indices for over and under: 

1319 if self.extend in ('both', 'min'):

1322 i1 += 1

1324 self.set_norm(mcolors.NoNorm()) 

1325 else: 

1326 self.cvalues = self.layers 

1327 self.set_array(self.levels)

1330 self.norm.clip = False

1389 def find_nearest_contour(self, x, y, indices=None, pixel=True): 

1390 """ 

1391 Finds contour that is closest to a point. Defaults to 

1392 measuring distance in pixels (screen space - useful for manual 

1393 contour labeling), but this can be controlled via a keyword 

1394 argument. 

1395 

1396 Returns a tuple containing the contour, segment, index of 

1397 segment, x & y of segment point and distance to minimum point. 

1398 

1399 Optional keyword arguments: 

1400 

1401 *indices*: 

1402 Indexes of contour levels to consider when looking for 

1403 nearest point. Defaults to using all levels. 

1404 

1405 *pixel*: 

1406 If *True*, measure distance in pixel space, if not, measure 

1407 distance in axes space. Defaults to *True*. 

1408 

1409 """ 

1410 

1411 # This function uses a method that is probably quite 

1412 # inefficient based on converting each contour segment to 

1413 # pixel coordinates and then comparing the given point to 

1414 # those coordinates for each contour. This will probably be 

1415 # quite slow for complex contours, but for normal use it works 

1416 # sufficiently well that the time is not noticeable. 

1417 # Nonetheless, improvements could probably be made. 

1418 

1419 if indices is None: 

1420 indices = list(range(len(self.levels))) 

1421 

1422 dmin = np.inf 

1423 conmin = None 

1424 segmin = None 

1425 xmin = None 

1426 ymin = None 

1427 

1428 point = np.array([x, y]) 

1429 

1430 for icon in indices: 

1431 con = self.collections[icon] 

1432 trans = con.get_transform() 

1433 paths = con.get_paths() 

1434 

1435 for segNum, linepath in enumerate(paths): 

1436 lc = linepath.vertices 

1437 # transfer all data points to screen coordinates if desired 

1438 if pixel: 

1439 lc = trans.transform(lc) 

1440 

1441 d, xc, leg = _find_closest_point_on_path(lc, point) 

1442 if d < dmin: 

1443 dmin = d 

1444 conmin = icon 

1445 segmin = segNum 

1446 imin = leg[1] 

1447 xmin = xc[0] 

1448 ymin = xc[1] 

1449 

1450 return (conmin, segmin, imin, xmin, ymin, dmin)

1621 def _initialize_x_y(self, z): 

1622 """ 

1623 Return X, Y arrays such that contour(Z) will match imshow(Z) 

1624 if origin is not None. 

1625 The center of pixel Z[i,j] depends on origin: 

1626 if origin is None, x = j, y = i; 

1627 if origin is 'lower', x = j + 0.5, y = i + 0.5; 

1628 if origin is 'upper', x = j + 0.5, y = Nrows - i - 0.5 

1629 If extent is not None, x and y will be scaled to match, 

1630 as in imshow. 

1631 If origin is None and extent is not None, then extent 

1632 will give the minimum and maximum values of x and y. 

1633 """ 

1634 if z.ndim != 2: 

1635 raise TypeError("Input must be a 2D array.") 

1636 elif z.shape[0] < 2 or z.shape[1] < 2: 

1637 raise TypeError("Input z must be at least a 2x2 array.") 

1638 else: 

1639 Ny, Nx = z.shape 

1640 if self.origin is None: # Not for image-matching. 

1641 if self.extent is None: 

1642 return np.meshgrid(np.arange(Nx), np.arange(Ny)) 

1643 else: 

1644 x0, x1, y0, y1 = self.extent 

1645 x = np.linspace(x0, x1, Nx) 

1646 y = np.linspace(y0, y1, Ny) 

1647 return np.meshgrid(x, y) 

1648 # Match image behavior: 

1649 if self.extent is None: 

1650 x0, x1, y0, y1 = (0, Nx, 0, Ny) 

1651 else: 

1652 x0, x1, y0, y1 = self.extent 

1653 dx = (x1 - x0) / Nx 

1654 dy = (y1 - y0) / Ny 

1655 x = x0 + (np.arange(Nx) + 0.5) * dx 

1656 y = y0 + (np.arange(Ny) + 0.5) * dy 

1657 if self.origin == 'upper': 

1658 y = y[::-1] 

1659 return np.meshgrid(x, y)

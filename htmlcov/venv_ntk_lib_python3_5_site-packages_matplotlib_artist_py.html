238 def get_window_extent(self, renderer): 

239 """

241 Subclasses should override for inclusion in the bounding box 

242 "tight" calculation. Default is to return an empty bounding 

243 box at 0, 0.

418 def pick(self, mouseevent): 

419 """ 

420 Process pick event 

421 

422 each child artist will fire a pick event if *mouseevent* is over 

423 the artist and the artist has picker set 

424 """ 

425 # Pick self 

426 if self.pickable(): 

427 picker = self.get_picker() 

428 if callable(picker): 

429 inside, prop = picker(self, mouseevent) 

430 else: 

431 inside, prop = self.contains(mouseevent) 

432 if inside: 

433 self.figure.canvas.pick_event(mouseevent, self, **prop) 

434 

435 # Pick children 

436 for a in self.get_children(): 

437 # make sure the event happened in the same axes 

438 ax = getattr(a, 'axes', None) 

439 if (mouseevent.inaxes is None or ax is None 

440 or mouseevent.inaxes == ax): 

441 # we need to check if mouseevent.inaxes is None 

442 # because some objects associated with an axes (e.g., a 

443 # tick label) can be outside the bounding box of the 

444 # axes and inaxes will be None 

445 # also check that ax is None so that it traverse objects 

446 # which do no have an axes property but children might 

447 a.pick(mouseevent)

663 def set_clip_path(self, path, transform=None): 

664 """ 

665 Set the artist's clip path, which may be: 

666 

667 - a :class:`~matplotlib.patches.Patch` (or subclass) instance; or 

668 - a :class:`~matplotlib.path.Path` instance, in which case a 

669 :class:`~matplotlib.transforms.Transform` instance, which will be 

670 applied to the path before using it for clipping, must be provided; 

671 or 

672 - ``None``, to remove a previously set clipping path. 

673 

674 For efficiency, if the path happens to be an axis-aligned rectangle, 

675 this method will set the clipping box to the corresponding rectangle 

676 and set the clipping path to ``None``.

679 """

683 if transform is None: 

684 if isinstance(path, Rectangle): 

685 self.clipbox = TransformedBbox(Bbox.unit(), 

686 path.get_transform()) 

687 self._clippath = None 

688 success = True 

689 elif isinstance(path, Patch): 

690 self._clippath = TransformedPatchPath(path) 

691 success = True 

692 elif isinstance(path, tuple): 

693 path, transform = path 

694 

695 if path is None: 

696 self._clippath = None 

697 success = True 

698 elif isinstance(path, Path): 

699 self._clippath = TransformedPath(path, transform) 

700 success = True 

701 elif isinstance(path, TransformedPatchPath): 

702 self._clippath = path 

703 success = True 

704 elif isinstance(path, TransformedPath): 

705 self._clippath = path

1017 def findobj(self, match=None, include_self=True): 

1018 """

1088class ArtistInspector(object): 

1089 """ 

1090 A helper class to inspect an `~matplotlib.artist.Artist` and return 

1091 information about its settable properties and their current values. 

1092 """ 

1093 

1094 def __init__(self, o): 

1095 r""" 

1096 Initialize the artist inspector with an `Artist` or an iterable of 

1097 `Artist`\s. If an iterable is used, we assume it is a homogeneous 

1098 sequence (all `Artists` are of the same type) and it is your 

1099 responsibility to make sure this is so. 

1100 """ 

1101 if not isinstance(o, Artist): 

1102 if cbook.iterable(o): 

1103 o = list(o) 

1104 if len(o): 

1105 o = o[0] 

1106 

1107 self.oorig = o 

1108 if not inspect.isclass(o): 

1109 o = type(o) 

1110 self.o = o 

1111 

1112 self.aliasd = self.get_aliases() 

1113 

1114 def get_aliases(self): 

1115 """ 

1116 Get a dict mapping property fullnames to sets of aliases for each alias 

1117 in the :class:`~matplotlib.artist.ArtistInspector`. 

1118 

1119 e.g., for lines:: 

1120 

1121 {'markerfacecolor': {'mfc'}, 

1122 'linewidth' : {'lw'}, 

1123 } 

1124 """ 

1125 names = [name for name in dir(self.o) 

1126 if name.startswith(('set_', 'get_')) 

1127 and callable(getattr(self.o, name))] 

1128 aliases = {} 

1129 for name in names: 

1130 func = getattr(self.o, name) 

1131 if not self.is_alias(func): 

1132 continue 

1133 propname = re.search("`({}.*)`".format(name[:4]), # get_.*/set_.* 

1134 func.__doc__).group(1) 

1135 aliases.setdefault(propname, set()).add(name[4:]) 

1136 return aliases 

1137 

1138 _get_valid_values_regex = re.compile( 

1139 r"\n\s*(?:\.\.\s+)?ACCEPTS:\s*((?:.|\n)*?)(?:$|(?:\n\n))" 

1140 ) 

1141 

1142 def get_valid_values(self, attr): 

1143 """ 

1144 Get the legal arguments for the setter associated with *attr*. 

1145 

1146 This is done by querying the docstring of the function *set_attr* 

1147 for a line that begins with "ACCEPTS" or ".. ACCEPTS": 

1148 

1149 e.g., for a line linestyle, return 

1150 "[ ``'-'`` | ``'--'`` | ``'-.'`` | ``':'`` | ``'steps'`` | ``'None'`` 

1151 ]" 

1152 """ 

1153 

1154 name = 'set_%s' % attr 

1155 if not hasattr(self.o, name): 

1156 raise AttributeError('%s has no function %s' % (self.o, name)) 

1157 func = getattr(self.o, name) 

1158 

1159 docstring = func.__doc__ 

1160 if docstring is None: 

1161 return 'unknown' 

1162 

1163 if docstring.startswith('alias for '): 

1164 return None 

1165 

1166 match = self._get_valid_values_regex.search(docstring) 

1167 if match is not None: 

1168 return re.sub("\n *", " ", match.group(1)) 

1169 

1170 # Much faster than list(inspect.signature(func).parameters)[1], 

1171 # although barely relevant wrt. matplotlib's total import time. 

1172 param_name = func.__code__.co_varnames[1] 

1173 match = re.search("(?m)^ *{} : (.+)".format(param_name), docstring) 

1174 if match: 

1175 return match.group(1) 

1176 

1177 return 'unknown' 

1178 

1179 def _get_setters_and_targets(self): 

1180 """ 

1181 Get the attribute strings and a full path to where the setter 

1182 is defined for all setters in an object. 

1183 """ 

1184 

1185 setters = [] 

1186 for name in dir(self.o): 

1187 if not name.startswith('set_'): 

1188 continue 

1189 func = getattr(self.o, name) 

1190 if not callable(func): 

1191 continue 

1192 nargs = len(inspect.getfullargspec(func).args) 

1193 if nargs < 2 or self.is_alias(func): 

1194 continue 

1195 source_class = self.o.__module__ + "." + self.o.__name__ 

1196 for cls in self.o.mro(): 

1197 if name in cls.__dict__: 

1198 source_class = cls.__module__ + "." + cls.__name__ 

1199 break 

1200 source_class = self._replace_path(source_class) 

1201 setters.append((name[4:], source_class + "." + name)) 

1202 return setters 

1203 

1204 def _replace_path(self, source_class): 

1205 """ 

1206 Changes the full path to the public API path that is used 

1207 in sphinx. This is needed for links to work. 

1208 """ 

1209 

1210 replace_dict = {'_base._AxesBase': 'Axes', 

1211 '_axes.Axes': 'Axes'} 

1212 

1213 for key, value in replace_dict.items(): 

1214 source_class = source_class.replace(key, value) 

1215 return source_class 

1216 

1217 def get_setters(self): 

1218 """ 

1219 Get the attribute strings with setters for object. e.g., for a line, 

1220 return ``['markerfacecolor', 'linewidth', ....]``. 

1221 """ 

1222 

1223 return [prop for prop, target in self._get_setters_and_targets()] 

1224 

1225 def is_alias(self, o): 

1226 """ 

1227 Return *True* if method object *o* is an alias for another 

1228 function. 

1229 """ 

1230 ds = o.__doc__ 

1231 if ds is None: 

1232 return False 

1233 return ds.startswith('alias for ')

1235 def aliased_name(self, s): 

1236 """ 

1237 return 'PROPNAME or alias' if *s* has an alias, else return 

1238 PROPNAME. 

1239 

1240 e.g., for the line markerfacecolor property, which has an 

1241 alias, return 'markerfacecolor or mfc' and for the transform 

1242 property, which does not, return 'transform' 

1243 """ 

1244 

1245 if s in self.aliasd: 

1246 return s + ''.join([' or %s' % x 

1247 for x in sorted(self.aliasd[s])]) 

1248 else: 

1249 return s

1251 def aliased_name_rest(self, s, target): 

1252 """ 

1253 return 'PROPNAME or alias' if *s* has an alias, else return 

1254 PROPNAME formatted for ReST 

1255 

1256 e.g., for the line markerfacecolor property, which has an 

1257 alias, return 'markerfacecolor or mfc' and for the transform

1259 """ 

1260 

1261 if s in self.aliasd: 

1262 aliases = ''.join([' or %s' % x 

1263 for x in sorted(self.aliasd[s])]) 

1264 else: 

1265 aliases = ''

1274 property will be returned as a string of property : valid 

1275 values.

1278 pad = ' ' * leadingspace 

1279 else: 

1280 pad = ''

1283 return '%s%s: %s' % (pad, prop, accepts)

1288

1300

1307 else:

1313 attrs = self._get_setters_and_targets() 

1314 attrs.sort() 

1315 lines = [] 

1316 

1317 ######## 

1318 names = [self.aliased_name_rest(prop, target)

1345 def properties(self):

1348 """ 

1349 o = self.oorig

1351 if name.startswith('get_') and callable(getattr(o, name))] 

1352 getters.sort() 

1353 d = dict() 

1354 for name in getters: 

1355 func = getattr(o, name) 

1356 if self.is_alias(func): 

1357 continue

1359 try: 

1360 with warnings.catch_warnings(): 

1361 warnings.simplefilter('ignore') 

1362 val = func() 

1363 except: 

1364 continue 

1365 else: 

1366 d[name[4:]] = val 

1367 

1368 return d

1430def setp(obj, *args, **kwargs): 

1431 """

1433 

1434 matplotlib supports the use of :func:`setp` ("set property") and 

1435 :func:`getp` to set and get object properties, as well as to do 

1436 introspection on the object. For example, to set the linestyle of a 

1437 line to be dashed, you can do::

1443 the name of the property you want to set without a value:: 

1444 

1445 >>> setp(line, 'linestyle') 

1446 linestyle: [ '-' | '--' | '-.' | ':' | 'steps' | 'None' ]

1449 possible values, you can do::

1451 >>> setp(line) 

1452 ... long output listing omitted 

1453 

1454 You may specify another output file to `setp` if `sys.stdout` is not 

1455 acceptable for some reason using the `file` keyword-only argument::

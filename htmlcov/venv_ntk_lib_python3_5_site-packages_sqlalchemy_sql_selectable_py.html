513 def corresponding_column(self, column, require_embedded=False): 

514 """Given a :class:`.ColumnElement`, return the exported 

515 :class:`.ColumnElement` object from this :class:`.Selectable` 

516 which corresponds to that original 

517 :class:`~sqlalchemy.schema.Column` via a common ancestor 

518 column. 

519 

520 :param column: the target :class:`.ColumnElement` to be matched 

521 

522 :param require_embedded: only return corresponding columns for 

523 the given :class:`.ColumnElement`, if the given 

524 :class:`.ColumnElement` is actually present within a sub-element 

525 of this :class:`.FromClause`. Normally the column will match if 

526 it merely shares a common ancestor with one of the exported 

527 columns of this :class:`.FromClause`. 

528 

529 """ 

530 

531 def embedded(expanded_proxy_set, target_set): 

532 for t in target_set.difference(expanded_proxy_set): 

533 if not set(_expand_cloned([t])).intersection( 

534 expanded_proxy_set 

535 ): 

536 return False 

537 return True 

538 

539 # don't dig around if the column is locally present 

540 if self.c.contains_column(column): 

541 return column 

542 col, intersect = None, None 

543 target_set = column.proxy_set 

544 cols = self.c._all_columns 

545 for c in cols: 

546 expanded_proxy_set = set(_expand_cloned(c.proxy_set)) 

547 i = target_set.intersection(expanded_proxy_set) 

548 if i and ( 

549 not require_embedded 

550 or embedded(expanded_proxy_set, target_set) 

551 ): 

552 if col is None: 

553 

554 # no corresponding column yet, pick this one. 

555 

556 col, intersect = c, i 

557 elif len(i) > len(intersect): 

558 

559 # 'c' has a larger field of correspondence than 

560 # 'col'. i.e. selectable.c.a1_x->a1.c.x->table.c.x 

561 # matches a1.c.x->table.c.x better than 

562 # selectable.c.x->table.c.x does. 

563 

564 col, intersect = c, i 

565 elif i == intersect: 

566 

567 # they have the same field of correspondence. see 

568 # which proxy_set has fewer columns in it, which 

569 # indicates a closer relationship with the root 

570 # column. Also take into account the "weight" 

571 # attribute which CompoundSelect() uses to give 

572 # higher precedence to columns based on vertical 

573 # position in the compound statement, and discard 

574 # columns that have no reference to the target 

575 # column (also occurs with CompoundSelect) 

576 

577 col_distance = util.reduce( 

578 operator.add, 

579 [ 

580 sc._annotations.get("weight", 1) 

581 for sc in col._uncached_proxy_set() 

582 if sc.shares_lineage(column) 

583 ], 

584 ) 

585 c_distance = util.reduce( 

586 operator.add, 

587 [ 

588 sc._annotations.get("weight", 1) 

589 for sc in c._uncached_proxy_set() 

590 if sc.shares_lineage(column) 

591 ], 

592 ) 

593 if c_distance < col_distance: 

594 col, intersect = c, i 

595 return col

977 

978 for left in (a_subset, a):

982 b.foreign_keys, key=lambda fk: fk.parent._creation_order

1607 def cte(self, name=None, recursive=False): 

1608 r"""Return a new :class:`.CTE`, or Common Table Expression instance. 

1609 

1610 Common table expressions are a SQL standard whereby SELECT 

1611 statements can draw upon secondary statements specified along 

1612 with the primary statement, using a clause called "WITH". 

1613 Special semantics regarding UNION can also be employed to 

1614 allow "recursive" queries, where a SELECT statement can draw 

1615 upon the set of rows that have previously been selected. 

1616 

1617 CTEs can also be applied to DML constructs UPDATE, INSERT 

1618 and DELETE on some databases, both as a source of CTE rows 

1619 when combined with RETURNING, as well as a consumer of 

1620 CTE rows. 

1621 

1622 SQLAlchemy detects :class:`.CTE` objects, which are treated 

1623 similarly to :class:`.Alias` objects, as special elements 

1624 to be delivered to the FROM clause of the statement as well 

1625 as to a WITH clause at the top of the statement. 

1626 

1627 .. versionchanged:: 1.1 Added support for UPDATE/INSERT/DELETE as 

1628 CTE, CTEs added to UPDATE/INSERT/DELETE. 

1629 

1630 :param name: name given to the common table expression. Like 

1631 :meth:`._FromClause.alias`, the name can be left as ``None`` 

1632 in which case an anonymous symbol will be used at query 

1633 compile time. 

1634 :param recursive: if ``True``, will render ``WITH RECURSIVE``. 

1635 A recursive common table expression is intended to be used in 

1636 conjunction with UNION ALL in order to derive rows 

1637 from those already selected. 

1638 

1639 The following examples include two from PostgreSQL's documentation at 

1640 http://www.postgresql.org/docs/current/static/queries-with.html, 

1641 as well as additional examples. 

1642 

1643 Example 1, non recursive:: 

1644 

1645 from sqlalchemy import (Table, Column, String, Integer, 

1646 MetaData, select, func)

1651 Column('region', String), 

1652 Column('amount', Integer), 

1653 Column('product', String), 

1654 Column('quantity', Integer) 

1655 )

1657 regional_sales = select([ 

1658 orders.c.region, 

1659 func.sum(orders.c.amount).label('total_sales') 

1660 ]).group_by(orders.c.region).cte("regional_sales") 

1661 

1662 

1663 top_regions = select([regional_sales.c.region]).\ 

1664 where( 

1665 regional_sales.c.total_sales > 

1666 select([ 

1667 func.sum(regional_sales.c.total_sales)/10 

1668 ]) 

1669 ).cte("top_regions") 

1670 

1671 statement = select([ 

1672 orders.c.region, 

1673 orders.c.product, 

1674 func.sum(orders.c.quantity).label("product_units"), 

1675 func.sum(orders.c.amount).label("product_sales") 

1676 ]).where(orders.c.region.in_( 

1677 select([top_regions.c.region]) 

1678 )).group_by(orders.c.region, orders.c.product) 

1679 

1680 result = conn.execute(statement).fetchall()

1682 Example 2, WITH RECURSIVE::

1684 from sqlalchemy import (Table, Column, String, Integer, 

1685 MetaData, select, func)

1687 metadata = MetaData()

1690 Column('part', String), 

1691 Column('sub_part', String), 

1692 Column('quantity', Integer), 

1693 )

1695 included_parts = select([ 

1696 parts.c.sub_part, 

1697 parts.c.part, 

1698 parts.c.quantity]).\ 

1699 where(parts.c.part=='our part').\ 

1700 cte(recursive=True) 

1701 

1702 

1703 incl_alias = included_parts.alias() 

1704 parts_alias = parts.alias() 

1705 included_parts = included_parts.union_all( 

1706 select([ 

1707 parts_alias.c.sub_part, 

1708 parts_alias.c.part, 

1709 parts_alias.c.quantity 

1710 ]).

1718 ]).\ 

1719 group_by(included_parts.c.sub_part)

1726 from sqlalchemy import (MetaData, Table, Column, Integer,

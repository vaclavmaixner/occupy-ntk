499 where : array of timestamps

696 def _maybe_cast_slice_bound(self, label, side, kind): 

697 """ 

698 If label is a string or a datetime, cast it to Period.ordinal according 

699 to resolution. 

700 

701 Parameters 

702 ---------- 

703 label : object 

704 side : {'left', 'right'} 

705 kind : {'ix', 'loc', 'getitem'} 

706 

707 Returns 

708 ------- 

709 bound : Period or object 

710 

711 Notes 

712 ----- 

713 Value of `side` parameter should be validated in caller. 

714 

715 """ 

716 assert kind in ['ix', 'loc', 'getitem'] 

717 

718 if isinstance(label, datetime): 

719 return Period(label, freq=self.freq) 

720 elif isinstance(label, compat.string_types): 

721 try: 

722 _, parsed, reso = parse_time_string(label, self.freq) 

723 bounds = self._parsed_string_to_bounds(reso, parsed) 

724 return bounds[0 if side == 'left' else 1] 

725 except Exception: 

726 raise KeyError(label) 

727 elif is_integer(label) or is_float(label): 

728 self._invalid_indexer('slice', label) 

729 

730 return label

732 def _parsed_string_to_bounds(self, reso, parsed): 

733 if reso == 'year': 

734 t1 = Period(year=parsed.year, freq='A') 

735 elif reso == 'month': 

736 t1 = Period(year=parsed.year, month=parsed.month, freq='M') 

737 elif reso == 'quarter': 

738 q = (parsed.month - 1) // 3 + 1 

739 t1 = Period(year=parsed.year, quarter=q, freq='Q-DEC') 

740 elif reso == 'day': 

741 t1 = Period(year=parsed.year, month=parsed.month, day=parsed.day, 

742 freq='D') 

743 elif reso == 'hour': 

744 t1 = Period(year=parsed.year, month=parsed.month, day=parsed.day, 

745 hour=parsed.hour, freq='H') 

746 elif reso == 'minute': 

747 t1 = Period(year=parsed.year, month=parsed.month, day=parsed.day, 

748 hour=parsed.hour, minute=parsed.minute, freq='T') 

749 elif reso == 'second': 

750 t1 = Period(year=parsed.year, month=parsed.month, day=parsed.day, 

751 hour=parsed.hour, minute=parsed.minute, 

752 second=parsed.second, freq='S') 

753 else: 

754 raise KeyError(reso) 

755 return (t1.asfreq(self.freq, how='start'), 

756 t1.asfreq(self.freq, how='end'))

758 def _get_string_slice(self, key): 

759 if not self.is_monotonic: 

760 raise ValueError('Partial indexing only valid for ' 

761 'ordered time series') 

762 

763 key, parsed, reso = parse_time_string(key, self.freq) 

764 grp = resolution.Resolution.get_freq_group(reso) 

765 freqn = resolution.get_freq_group(self.freq) 

766 if reso in ['day', 'hour', 'minute', 'second'] and not grp < freqn: 

767 raise KeyError(key) 

768 

769 t1, t2 = self._parsed_string_to_bounds(reso, parsed) 

770 return slice(self.searchsorted(t1.ordinal, side='left'), 

771 self.searchsorted(t2.ordinal, side='right'))

905def period_range(start=None, end=None, periods=None, freq=None, name=None): 

906 """ 

907 Return a fixed frequency PeriodIndex, with day (calendar) as the default 

908 frequency 

909 

910 Parameters 

911 ---------- 

912 start : string or period-like, default None 

913 Left bound for generating periods 

914 end : string or period-like, default None 

915 Right bound for generating periods 

916 periods : integer, default None 

917 Number of periods to generate 

918 freq : string or DateOffset, optional 

919 Frequency alias. By default the freq is taken from `start` or `end` 

920 if those are Period objects. Otherwise, the default is ``"D"`` for 

921 daily frequency. 

922 

923 name : string, default None 

924 Name of the resulting PeriodIndex 

925 

926 Returns 

927 ------- 

928 prng : PeriodIndex 

929 

930 Notes 

931 ----- 

932 Of the three parameters: ``start``, ``end``, and ``periods``, exactly two 

933 must be specified. 

934 

935 To learn more about the frequency strings, please see `this link 

936 <http://pandas.pydata.org/pandas-docs/stable/timeseries.html#offset-aliases>`__. 

937 

938 Examples 

939 -------- 

940 

941 >>> pd.period_range(start='2017-01-01', end='2018-01-01', freq='M') 

942 PeriodIndex(['2017-01', '2017-02', '2017-03', '2017-04', '2017-05', 

943 '2017-06', '2017-06', '2017-07', '2017-08', '2017-09', 

944 '2017-10', '2017-11', '2017-12', '2018-01'], 

945 dtype='period[M]', freq='M') 

946 

947 If ``start`` or ``end`` are ``Period`` objects, they will be used as anchor 

948 endpoints for a ``PeriodIndex`` with frequency matching that of the 

949 ``period_range`` constructor. 

950 

951 >>> pd.period_range(start=pd.Period('2017Q1', freq='Q'), 

952 ... end=pd.Period('2017Q2', freq='Q'), freq='M') 

953 PeriodIndex(['2017-03', '2017-04', '2017-05', '2017-06'], 

954 dtype='period[M]', freq='M') 

955 """ 

956 if com.count_not_none(start, end, periods) != 2: 

957 raise ValueError('Of the three parameters: start, end, and periods, ' 

958 'exactly two must be specified') 

959 if freq is None and (not isinstance(start, Period) 

960 and not isinstance(end, Period)): 

961 freq = 'D' 

962 

963 data, freq = PeriodArray._generate_range(start, end, periods, freq, 

964 fields={}) 

965 data = PeriodArray(data, freq=freq) 

966 return PeriodIndex(data, name=name)

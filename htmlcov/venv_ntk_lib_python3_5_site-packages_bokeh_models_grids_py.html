46class Grid(GuideRenderer): 

47 ''' Display horizontal or vertical grid lines at locations 

48 given by a supplied ``Ticker``. 

49 

50 ''' 

51 dimension = Int(0, help=""" 

52 Which dimension the Axis Grid lines will intersect. The 

53 x-axis is dimension 0 (vertical Grid lines) and the y-axis 

54 is dimension 1 (horizontal Grid lines). 

55 """) 

56 

57 bounds = Either(Auto, Tuple(Float, Float), help=""" 

58 Bounds for the rendered grid lines. By default, a grid will look for a 

59 corresponding axis to ask for bounds. If one cannot be found, the grid 

60 will span the entire visible range. 

61 """) 

62 

63 # Note: we must allow the possibility of setting both 

64 # range names be cause if a grid line is "traced" along 

65 # a path, ranges in both dimensions will matter. 

66 

67 x_range_name = String('default', help=""" 

68 A particular (named) x-range to use for computing screen 

69 locations when rendering a grid on the plot. If unset, use the 

70 default x-range. 

71 """) 

72 

73 y_range_name = String('default', help=""" 

74 A particular (named) y-range to use for computing screen 

75 locations when rendering a grid on the plot. If unset, use the 

76 default y-range. 

77 """) 

78 

79 ticker = Instance(Ticker, help=""" 

80 The Ticker to use for computing locations for the Grid lines. 

81 """).accepts(Seq(Float), lambda ticks: FixedTicker(ticks=ticks)) 

82 

83 grid_props = Include(ScalarLineProps, help=""" 

84 The %s of the Grid lines. 

85 """) 

86 

87 grid_line_color = Override(default='#e5e5e5') 

88 

89 minor_grid_props = Include(ScalarLineProps, help=""" 

90 The %s of the minor Grid lines. 

91 """) 

92 

93 minor_grid_line_color = Override(default=None) 

94 

95 band_fill_props = Include(ScalarFillProps, use_prefix="band", help=""" 

96 The %s of alternating bands between Grid lines. 

97 """) 

98 

99 band_fill_alpha = Override(default=0) 

100 

101 band_fill_color = Override(default=None) 

102 

103 band_hatch_props = Include(ScalarHatchProps, use_prefix="band", help=""" 

104 The %s of alternating bands between Grid lines. 

105 """) 

106 

107 level = Override(default="underlay")

12def _check_is_partition(parts, whole): 

13 whole = set(whole) 

14 parts = [set(x) for x in parts]

16 raise ValueError(

91def _sparse_series_to_coo(ss, row_levels=(0, ), column_levels=(1, ), 

92 sort_labels=False): 

93 """ Convert a SparseSeries to a scipy.sparse.coo_matrix using index 

94 levels row_levels, column_levels as the row and column 

95 labels respectively. Returns the sparse_matrix, row and column labels. 

96 """ 

97 

98 import scipy.sparse 

99 

100 if ss.index.nlevels < 2: 

101 raise ValueError('to_coo requires MultiIndex with nlevels > 2') 

102 if not ss.index.is_unique: 

103 raise ValueError('Duplicate index entries are not allowed in to_coo ' 

104 'transformation.') 

105 

106 # to keep things simple, only rely on integer indexing (not labels) 

107 row_levels = [ss.index._get_level_number(x) for x in row_levels] 

108 column_levels = [ss.index._get_level_number(x) for x in column_levels] 

109 

110 v, i, j, rows, columns = _to_ijv(ss, row_levels=row_levels, 

111 column_levels=column_levels, 

112 sort_labels=sort_labels) 

113 sparse_matrix = scipy.sparse.coo_matrix( 

114 (v, (i, j)), shape=(len(rows), len(columns))) 

115 return sparse_matrix, rows, columns

71def _get_textbox(text, renderer): 

72 """ 

73 Calculate the bounding box of the text. Unlike 

74 :meth:`matplotlib.text.Text.get_extents` method, The bbox size of 

75 the text before the rotation is calculated. 

76 """ 

77 # TODO : This function may move into the Text class as a method. As a 

78 # matter of fact, The information from the _get_textbox function 

79 # should be available during the Text._get_layout() call, which is 

80 # called within the _get_textbox. So, it would better to move this 

81 # function as a method with some refactoring of _get_layout method. 

82 

83 projected_xs = [] 

84 projected_ys = [] 

85 

86 theta = np.deg2rad(text.get_rotation()) 

87 tr = Affine2D().rotate(-theta) 

88 

89 _, parts, d = text._get_layout(renderer) 

90 

91 for t, wh, x, y in parts: 

92 w, h = wh 

93 

94 xt1, yt1 = tr.transform_point((x, y)) 

95 yt1 -= d 

96 xt2, yt2 = xt1 + w, yt1 + h 

97 

98 projected_xs.extend([xt1, xt2]) 

99 projected_ys.extend([yt1, yt2]) 

100 

101 xt_box, yt_box = min(projected_xs), min(projected_ys) 

102 w_box, h_box = max(projected_xs) - xt_box, max(projected_ys) - yt_box 

103 

104 x_box, y_box = Affine2D().rotate(theta).transform_point((xt_box, yt_box)) 

105 

106 return x_box, y_box, w_box, h_box

280 def _get_layout(self, renderer): 

281 """ 

282 return the extent (bbox) of the text together with 

283 multiple-alignment information. Note that it returns an extent 

284 of a rotated text when necessary. 

285 """ 

286 key = self.get_prop_tup(renderer=renderer) 

287 if key in self._cached: 

288 return self._cached[key] 

289 

290 horizLayout = [] 

291 

292 thisx, thisy = 0.0, 0.0 

293 xmin, ymin = 0.0, 0.0 

294 width, height = 0.0, 0.0 

295 lines = self.get_text().split('\n') 

296 

297 whs = np.zeros((len(lines), 2)) 

298 horizLayout = np.zeros((len(lines), 4)) 

299 

300 # Find full vertical extent of font, 

301 # including ascenders and descenders: 

302 tmp, lp_h, lp_bl = renderer.get_text_width_height_descent('lp', 

303 self._fontproperties, 

304 ismath=False) 

305 offsety = (lp_h - lp_bl) * self._linespacing 

306 

307 baseline = 0 

308 for i, line in enumerate(lines): 

309 clean_line, ismath = self.is_math_text(line, self.get_usetex()) 

310 if clean_line: 

311 w, h, d = renderer.get_text_width_height_descent(clean_line, 

312 self._fontproperties, 

313 ismath=ismath) 

314 else: 

315 w, h, d = 0, 0, 0 

316 

317 # For multiline text, increase the line spacing when the 

318 # text net-height(excluding baseline) is larger than that 

319 # of a "l" (e.g., use of superscripts), which seems 

320 # what TeX does. 

321 h = max(h, lp_h) 

322 d = max(d, lp_bl) 

323 

324 whs[i] = w, h 

325 

326 baseline = (h - d) - thisy 

327 thisy -= max(offsety, (h - d) * self._linespacing) 

328 horizLayout[i] = thisx, thisy, w, h 

329 thisy -= d 

330 width = max(width, w) 

331 descent = d 

332 

333 ymin = horizLayout[-1][1] 

334 ymax = horizLayout[0][1] + horizLayout[0][3] 

335 height = ymax - ymin 

336 xmax = xmin + width 

337 

338 # get the rotation matrix 

339 M = Affine2D().rotate_deg(self.get_rotation()) 

340 

341 offsetLayout = np.zeros((len(lines), 2)) 

342 offsetLayout[:] = horizLayout[:, 0:2] 

343 # now offset the individual text lines within the box 

344 if len(lines) > 1: # do the multiline aligment 

345 malign = self._get_multialignment() 

346 if malign == 'center': 

347 offsetLayout[:, 0] += width / 2.0 - horizLayout[:, 2] / 2.0 

348 elif malign == 'right': 

349 offsetLayout[:, 0] += width - horizLayout[:, 2] 

350 

351 # the corners of the unrotated bounding box 

352 cornersHoriz = np.array( 

353 [(xmin, ymin), (xmin, ymax), (xmax, ymax), (xmax, ymin)], float) 

354 cornersHoriz[:, 1] -= descent 

355 

356 # now rotate the bbox 

357 cornersRotated = M.transform(cornersHoriz) 

358 

359 txs = cornersRotated[:, 0] 

360 tys = cornersRotated[:, 1] 

361 

362 # compute the bounds of the rotated box 

363 xmin, xmax = txs.min(), txs.max() 

364 ymin, ymax = tys.min(), tys.max() 

365 width = xmax - xmin 

366 height = ymax - ymin 

367 

368 # Now move the box to the target position offset the display 

369 # bbox by alignment 

370 halign = self._horizontalalignment 

371 valign = self._verticalalignment 

372 

373 rotation_mode = self.get_rotation_mode() 

374 if rotation_mode != "anchor": 

375 # compute the text location in display coords and the offsets 

376 # necessary to align the bbox with that location 

377 if halign == 'center': 

378 offsetx = (xmin + width / 2.0) 

379 elif halign == 'right': 

380 offsetx = (xmin + width) 

381 else: 

382 offsetx = xmin 

383 

384 if valign == 'center': 

385 offsety = (ymin + height / 2.0) 

386 elif valign == 'top': 

387 offsety = (ymin + height) 

388 elif valign == 'baseline': 

389 offsety = (ymin + height) - baseline 

390 elif valign == 'center_baseline': 

391 offsety = ymin + height - baseline / 2.0 

392 else: 

393 offsety = ymin 

394 else: 

395 xmin1, ymin1 = cornersHoriz[0] 

396 xmax1, ymax1 = cornersHoriz[2] 

397 

398 if halign == 'center': 

399 offsetx = (xmin1 + xmax1) / 2.0 

400 elif halign == 'right': 

401 offsetx = xmax1 

402 else: 

403 offsetx = xmin1 

404 

405 if valign == 'center': 

406 offsety = (ymin1 + ymax1) / 2.0 

407 elif valign == 'top': 

408 offsety = ymax1 

409 elif valign == 'baseline': 

410 offsety = ymax1 - baseline 

411 elif valign == 'center_baseline': 

412 offsety = (ymin1 + ymax1 - baseline) / 2.0 

413 else: 

414 offsety = ymin1 

415 

416 offsetx, offsety = M.transform_point((offsetx, offsety)) 

417 

418 xmin -= offsetx 

419 ymin -= offsety 

420 

421 bbox = Bbox.from_bounds(xmin, ymin, width, height) 

422 

423 # now rotate the positions around the first x,y position 

424 xys = M.transform(offsetLayout) 

425 xys -= (offsetx, offsety) 

426 

427 xs, ys = xys[:, 0], xys[:, 1] 

428 

429 ret = bbox, list(zip(lines, whs, xs, ys)), descent 

430 self._cached[key] = ret 

431 return ret

600 def _get_wrap_line_width(self): 

601 """ 

602 Return the maximum line width for wrapping text based on the current 

603 orientation. 

604 """ 

605 x0, y0 = self.get_transform().transform(self.get_position()) 

606 figure_box = self.get_figure().get_window_extent() 

607 

608 # Calculate available width based on text alignment 

609 alignment = self.get_horizontalalignment() 

610 self.set_rotation_mode('anchor') 

611 rotation = self.get_rotation() 

612 

613 left = self._get_dist_to_box(rotation, x0, y0, figure_box) 

614 right = self._get_dist_to_box( 

615 (180 + rotation) % 360, x0, y0, figure_box) 

616 

617 if alignment == 'left': 

618 line_width = left 

619 elif alignment == 'right': 

620 line_width = right 

621 else: 

622 line_width = 2 * min(left, right) 

623 

624 return line_width

626 def _get_dist_to_box(self, rotation, x0, y0, figure_box): 

627 """ 

628 Return the distance from the given points to the boundaries of a 

629 rotated box, in pixels. 

630 """ 

631 if rotation > 270: 

632 quad = rotation - 270 

633 h1 = y0 / math.cos(math.radians(quad)) 

634 h2 = (figure_box.x1 - x0) / math.cos(math.radians(90 - quad)) 

635 elif rotation > 180: 

636 quad = rotation - 180 

637 h1 = x0 / math.cos(math.radians(quad)) 

638 h2 = y0 / math.cos(math.radians(90 - quad)) 

639 elif rotation > 90: 

640 quad = rotation - 90 

641 h1 = (figure_box.y1 - y0) / math.cos(math.radians(quad)) 

642 h2 = x0 / math.cos(math.radians(90 - quad)) 

643 else: 

644 h1 = (figure_box.x1 - x0) / math.cos(math.radians(rotation)) 

645 h2 = (figure_box.y1 - y0) / math.cos(math.radians(90 - rotation)) 

646 

647 return min(h1, h2)

659 def _get_wrapped_text(self): 

660 """ 

661 Return a copy of the text with new lines added, so that 

662 the text is wrapped relative to the parent figure. 

663 """ 

664 # Not fit to handle breaking up latex syntax correctly, so 

665 # ignore latex for now. 

666 if self.get_usetex(): 

667 return self.get_text() 

668 

669 # Build the line incrementally, for a more accurate measure of length 

670 line_width = self._get_wrap_line_width() 

671 wrapped_str = "" 

672 line = "" 

673 

674 for word in self.get_text().split(' '): 

675 # New lines in the user's test need to force a split, so that it's 

676 # not using the longest current line width in the line being built 

677 sub_words = word.split('\n') 

678 for i in range(len(sub_words)): 

679 current_width = self._get_rendered_text_width( 

680 line + ' ' + sub_words[i]) 

681 

682 # Split long lines, and each newline found in the current word 

683 if current_width > line_width or i > 0: 

684 wrapped_str += line + '\n' 

685 line = ""

1420 def update_coords(self, renderer): 

1421 """ 

1422 Computes the actual *x*, *y* coordinates for text based on the 

1423 input *x*, *y* and the *dashlength*. Since the rotation is 

1424 with respect to the actual canvas's coordinates we need to map 

1425 back and forth. 

1426 """ 

1427 dashx, dashy = self.get_unitless_position() 

1428 dashlength = self.get_dashlength() 

1429 # Shortcircuit this process if we don't have a dash 

1430 if dashlength == 0.0: 

1431 self._x, self._y = dashx, dashy 

1432 return 

1433 

1434 dashrotation = self.get_dashrotation() 

1435 dashdirection = self.get_dashdirection() 

1436 dashpad = self.get_dashpad() 

1437 dashpush = self.get_dashpush() 

1438 

1439 angle = get_rotation(dashrotation) 

1440 theta = np.pi * (angle / 180.0 + dashdirection - 1) 

1441 cos_theta, sin_theta = np.cos(theta), np.sin(theta) 

1442 

1443 transform = self.get_transform() 

1444 

1445 # Compute the dash end points 

1446 # The 'c' prefix is for canvas coordinates 

1447 cxy = transform.transform_point((dashx, dashy)) 

1448 cd = np.array([cos_theta, sin_theta]) 

1449 c1 = cxy + dashpush * cd 

1450 c2 = cxy + (dashpush + dashlength) * cd 

1451 

1452 inverse = transform.inverted() 

1453 (x1, y1) = inverse.transform_point(tuple(c1)) 

1454 (x2, y2) = inverse.transform_point(tuple(c2)) 

1455 self.dashline.set_data((x1, x2), (y1, y2)) 

1456 

1457 # We now need to extend this vector out to 

1458 # the center of the text area. 

1459 # The basic problem here is that we're "rotating" 

1460 # two separate objects but want it to appear as 

1461 # if they're rotated together. 

1462 # This is made non-trivial because of the 

1463 # interaction between text rotation and alignment - 

1464 # text alignment is based on the bbox after rotation. 

1465 # We reset/force both alignments to 'center' 

1466 # so we can do something relatively reasonable. 

1467 # There's probably a better way to do this by 

1468 # embedding all this in the object's transformations, 

1469 # but I don't grok the transformation stuff 

1470 # well enough yet. 

1471 we = Text.get_window_extent(self, renderer=renderer) 

1472 w, h = we.width, we.height 

1473 # Watch for zeros 

1474 if sin_theta == 0.0: 

1475 dx = w 

1476 dy = 0.0 

1477 elif cos_theta == 0.0: 

1478 dx = 0.0 

1479 dy = h 

1480 else: 

1481 tan_theta = sin_theta / cos_theta 

1482 dx = w 

1483 dy = w * tan_theta 

1484 if dy > h or dy < -h: 

1485 dy = h 

1486 dx = h / tan_theta 

1487 cwd = np.array([dx, dy]) / 2 

1488 cwd *= 1 + dashpad / np.sqrt(np.dot(cwd, cwd)) 

1489 cw = c2 + (dashdirection * 2 - 1) * cwd 

1490 

1491 newx, newy = inverse.transform_point(tuple(cw)) 

1492 self._x, self._y = newx, newy 

1493 

1494 # Now set the window extent 

1495 # I'm not at all sure this is the right way to do this. 

1496 we = Text.get_window_extent(self, renderer=renderer) 

1497 self._twd_window_extent = we.frozen() 

1498 self._twd_window_extent.update_from_data_xy(np.array([c1]), False) 

1499 

1500 # Finally, make text align center 

1501 Text.set_horizontalalignment(self, 'center') 

1502 Text.set_verticalalignment(self, 'center')

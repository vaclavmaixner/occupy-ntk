273def get_cls_kwargs(cls, _set=None): 

274 r"""Return the full set of inherited kwargs for the given `cls`. 

275 

276 Probes a class's __init__ method, collecting all named arguments. If the 

277 __init__ defines a \**kwargs catch-all, then the constructor is presumed 

278 to pass along unrecognized keywords to its base classes, and the 

279 collection process is repeated recursively on each of the bases. 

280 

281 Uses a subset of inspect.getfullargspec() to cut down on method overhead, 

282 as this is used within the Core typing system to create copies of type 

283 objects which is a performance-sensitive operation. 

284 

285 No anonymous tuple arguments please ! 

286 

287 """ 

288 toplevel = _set is None 

289 if toplevel: 

290 _set = set() 

291 

292 ctr = cls.__dict__.get("__init__", False) 

293 

294 has_init = ( 

295 ctr 

296 and isinstance(ctr, types.FunctionType) 

297 and isinstance(ctr.__code__, types.CodeType) 

298 ) 

299 

300 if has_init: 

301 names, has_kw = _inspect_func_args(ctr) 

302 _set.update(names) 

303 

304 if not has_kw and not toplevel: 

305 return None 

306 

307 if not has_init or has_kw: 

308 for c in cls.__bases__: 

309 if get_cls_kwargs(c, _set) is None: 

310 break 

311 

312 _set.discard("self") 

313 return _set

395 self_arg

399 args, re-written in calling rather than receiving syntax. Arguments are

620def class_hierarchy(cls): 

621 """Return an unordered sequence of all classes related to cls. 

622 

623 Traverses diamond hierarchies. 

624 

625 Fibs slightly: subclasses of builtin types are not returned. Thus 

626 class_hierarchy(class A(object)) returns (A, object), not A plus every 

627 class systemwide that derives from object. 

628 

629 Old-style classes are discarded and hierarchies rooted on them 

630 will not be descended. 

631 

632 """ 

633 if compat.py2k: 

634 if isinstance(cls, types.ClassType): 

635 return list() 

636 

637 hier = {cls} 

638 process = list(cls.__mro__) 

639 while process: 

640 c = process.pop() 

641 if compat.py2k: 

642 if isinstance(c, types.ClassType): 

643 continue 

644 bases = ( 

645 _ 

646 for _ in c.__bases__ 

647 if _ not in hier and not isinstance(_, types.ClassType) 

648 ) 

649 else: 

650 bases = (_ for _ in c.__bases__ if _ not in hier) 

651 

652 for b in bases: 

653 process.append(b) 

654 hier.add(b) 

655 

656 if compat.py3k: 

657 if c.__module__ == "builtins" or not hasattr(c, "__subclasses__"):

660 if c.__module__ == "__builtin__" or not hasattr( 

661 c, "__subclasses__" 

662 ): 

663 continue 

664 

665 for s in [_ for _ in c.__subclasses__() if _ not in hier]: 

666 process.append(s) 

667 hier.add(s)

758def as_interface(obj, cls=None, methods=None, required=None): 

759 """Ensure basic interface compliance for an instance or dict of callables. 

760 

761 Checks that ``obj`` implements public methods of ``cls`` or has members 

762 listed in ``methods``. If ``required`` is not supplied, implementing at 

763 least one interface method is sufficient. Methods present on ``obj`` that 

764 are not in the interface are ignored. 

765 

766 If ``obj`` is a dict and ``dict`` does not meet the interface 

767 requirements, the keys of the dictionary are inspected. Keys present in 

768 ``obj`` that are not in the interface will raise TypeErrors. 

769 

770 Raises TypeError if ``obj`` does not meet the interface criteria. 

771 

772 In all passing cases, an object with callable members is returned. In the 

773 simple case, ``obj`` is returned as-is; if dict processing kicks in then 

774 an anonymous class is returned. 

775 

776 obj 

777 A type, instance, or dictionary of callables. 

778 cls 

779 Optional, a type. All public methods of cls are considered the 

780 interface. An ``obj`` instance of cls will always pass, ignoring 

781 ``required``.. 

782 methods 

783 Optional, a sequence of method names to consider as the interface. 

784 required 

785 Optional, a sequence of mandatory implementations. If omitted, an 

786 ``obj`` that provides at least one interface method is considered 

787 sufficient. As a convenience, required may be a type, in which case 

788 all public methods of the type are required. 

789 

790 """ 

791 if not cls and not methods: 

792 raise TypeError("a class or collection of method names are required") 

793 

794 if isinstance(cls, type) and isinstance(obj, cls): 

795 return obj 

796 

797 interface = set(methods or [m for m in dir(cls) if not m.startswith("_")]) 

798 implemented = set(dir(obj)) 

799 

800 complies = operator.ge 

801 if isinstance(required, type): 

802 required = interface 

803 elif not required: 

804 required = set() 

805 complies = operator.gt 

806 else: 

807 required = set(required) 

808 

809 if complies(implemented.intersection(interface), required): 

810 return obj 

811 

812 # No dict duck typing here. 

813 if not isinstance(obj, dict): 

814 qualifier = complies is operator.gt and "any of" or "all of" 

815 raise TypeError( 

816 "%r does not implement %s: %s" 

817 % (obj, qualifier, ", ".join(interface)) 

818 )

967class dependencies(object): 

968 """Apply imported dependencies as arguments to a function. 

969 

970 E.g.:: 

971 

972 @util.dependencies( 

973 "sqlalchemy.sql.widget", 

974 "sqlalchemy.engine.default" 

975 ); 

976 def some_func(self, widget, default, arg1, arg2, **kw): 

977 # ... 

978 

979 Rationale is so that the impact of a dependency cycle can be 

980 associated directly with the few functions that cause the cycle, 

981 and not pollute the module-level namespace. 

982 

983 """

986 self.import_deps = [] 

987 for dep in deps: 

988 tokens = dep.split(".") 

989 self.import_deps.append( 

990 dependencies._importlater(".".join(tokens[0:-1]), tokens[-1]) 

991 )

996 

997 spec_zero = list(spec[0]) 

998 hasself = spec_zero[0] in ("self", "cls") 

999 

1000 for i in range(len(import_deps)): 

1001 spec[0][i + (1 if hasself else 0)] = "import_deps[%r]" % i 

1002 

1003 inner_spec = format_argspec_plus(spec, grouped=False) 

1004 

1005 for impname in import_deps: 

1006 del spec_zero[1 if hasself else 0] 

1007 spec[0][:] = spec_zero 

1008 

1009 outer_spec = format_argspec_plus(spec, grouped=False) 

1010 

1011 code = "lambda %(args)s: fn(%(apply_kw)s)" % { 

1012 "args": outer_spec["args"], 

1013 "apply_kw": inner_spec["apply_kw"], 

1014 }

1016 decorated = eval(code, locals()) 

1017 decorated.__defaults__ = getattr(fn, "im_func", fn).__defaults__

1580def quoted_token_parser(value): 

1581 """Parse a dotted identifier with accommodation for quoted names. 

1582 

1583 Includes support for SQL-style double quotes as a literal character. 

1584 

1585 E.g.:: 

1586 

1587 >>> quoted_token_parser("name") 

1588 ["name"] 

1589 >>> quoted_token_parser("schema.name") 

1590 ["schema", "name"] 

1591 >>> quoted_token_parser('"Schema"."Name"') 

1592 ['Schema', 'Name'] 

1593 >>> quoted_token_parser('"Schema"."Name""Foo"') 

1594 ['Schema', 'Name""Foo'] 

1595 

1596 """ 

1597 

1598 if '"' not in value: 

1599 return value.split(".") 

1600 

1601 # 0 = outside of quotes 

1602 # 1 = inside of quotes 

1603 state = 0 

1604 result = [[]] 

1605 idx = 0 

1606 lv = len(value) 

1607 while idx < lv: 

1608 char = value[idx] 

1609 if char == '"': 

1610 if state == 1 and idx < lv - 1 and value[idx + 1] == '"': 

1611 result[-1].append('"') 

1612 idx += 1 

1613 else: 

1614 state ^= 1 

1615 elif char == "." and state == 0: 

1616 result.append([]) 

1617 else: 

1618 result[-1].append(char) 

1619 idx += 1 

1620 

1621 return ["".join(token) for token in result]

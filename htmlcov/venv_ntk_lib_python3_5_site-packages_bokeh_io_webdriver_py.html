54def kill_proc_tree(pid, including_parent=True): 

55 psutil = import_optional('psutil') 

56 if psutil is not None: 

57 parent = psutil.Process(pid) 

58 children = parent.children(recursive=True) 

59 for child in children: 

60 child.kill() 

61 psutil.wait_procs(children) 

62 if including_parent: 

63 parent.kill() 

64 parent.wait(5)

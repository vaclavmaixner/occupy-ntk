81 :param message: 

82 Error message to raise in case of a validation error. Can be 

83 interpolated using `%(min)d` and `%(max)d` if desired. Useful defaults

86 def __init__(self, min=-1, max=-1, message=None): 

87 assert min != -1 or max != -1, 'At least one of `min` or `max` must be specified.' 

88 assert max == -1 or min <= max, '`min` cannot be more than `max`.'

104 else: 

105 message = field.gettext('Field must be between %(min)d and %(max)d characters long.')

200 self.message = message

205 message = field.gettext('This field is required.') 

206 else: 

207 message = self.message

326 """ 

327 Validates an IP address. 

328 

329 :param ipv4: 

330 If True, accept IPv4 addresses as valid (default True)

336 def __init__(self, ipv4=True, ipv6=False, message=None): 

337 if not ipv4 and not ipv6: 

338 raise ValueError('IP Address Validator must have at least one of ipv4 or ipv6 enabled.')

345 valid = False 

346 if value: 

347 valid = (self.ipv4 and self.check_ipv4(value)) or (self.ipv6 and self.check_ipv6(value))

355 @classmethod 

356 def check_ipv4(cls, value): 

357 parts = value.split('.') 

358 if len(parts) == 4 and all(x.isdigit() for x in parts): 

359 numbers = list(int(x) for x in parts) 

360 return all(num >= 0 and num < 256 for num in numbers) 

361 return False 

362 

363 @classmethod 

364 def check_ipv6(cls, value): 

365 parts = value.split(':') 

366 if len(parts) > 8: 

367 return False 

368 

369 num_blank = 0 

370 for part in parts: 

371 if not part: 

372 num_blank += 1 

373 else: 

374 try: 

375 value = int(part, 16) 

376 except ValueError: 

377 return False 

378 else: 

379 if value < 0 or value >= 65536: 

380 return False 

381 

382 if num_blank < 2: 

383 return True 

384 elif num_blank == 2 and not parts[0] and not parts[1]: 

385 return True 

386 return False

542 try: 

543 hostname = hostname.encode('idna')

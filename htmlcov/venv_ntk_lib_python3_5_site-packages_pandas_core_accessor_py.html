13class DirNamesMixin(object): 

14 _accessors = frozenset() 

15 _deprecations = frozenset( 

16 ['asobject', 'base', 'data', 'flags', 'itemsize', 'strides']) 

17 

18 def _dir_deletions(self): 

19 """ delete unwanted __dir__ for this object """ 

20 return self._accessors | self._deprecations 

21 

22 def _dir_additions(self): 

23 """ add additional __dir__ for this object """ 

24 rv = set() 

25 for accessor in self._accessors: 

26 try: 

27 getattr(self, accessor) 

28 rv.add(accessor) 

29 except AttributeError: 

30 pass 

31 return rv 

32 

33 def __dir__(self): 

34 """ 

35 Provide method name lookup and completion 

36 Only provide 'public' methods 

37 """ 

38 rv = set(dir(type(self))) 

39 rv = (rv - self._dir_deletions()) | self._dir_additions() 

40 return sorted(rv)

43class PandasDelegate(object): 

44 """ 

45 an abstract base class for delegating methods/properties 

46 """ 

47 

48 def _delegate_property_get(self, name, *args, **kwargs): 

49 raise TypeError("You cannot access the " 

50 "property {name}".format(name=name)) 

51 

52 def _delegate_property_set(self, name, value, *args, **kwargs): 

53 raise TypeError("The property {name} cannot be set".format(name=name)) 

54 

55 def _delegate_method(self, name, *args, **kwargs): 

56 raise TypeError("You cannot call method {name}".format(name=name))

58 @classmethod 

59 def _add_delegate_accessors(cls, delegate, accessors, typ, 

60 overwrite=False): 

61 """ 

62 Add accessors to cls from the delegate class. 

63 

64 Parameters 

65 ---------- 

66 cls : the class to add the methods/properties to 

67 delegate : the class to get methods/properties & doc-strings 

68 acccessors : string list of accessors to add 

69 typ : 'property' or 'method' 

70 overwrite : boolean, default False 

71 overwrite the method/property in the target class if it exists 

72 """ 

73 

74 def _create_delegator_property(name): 

75 

76 def _getter(self): 

77 return self._delegate_property_get(name) 

78 

79 def _setter(self, new_values): 

80 return self._delegate_property_set(name, new_values) 

81 

82 _getter.__name__ = name 

83 _setter.__name__ = name 

84 

85 return property(fget=_getter, fset=_setter, 

86 doc=getattr(delegate, name).__doc__) 

87 

88 def _create_delegator_method(name): 

89 

90 def f(self, *args, **kwargs): 

91 return self._delegate_method(name, *args, **kwargs) 

92 

93 f.__name__ = name 

94 f.__doc__ = getattr(delegate, name).__doc__ 

95 

96 return f 

97 

98 for name in accessors: 

99 

100 if typ == 'property': 

101 f = _create_delegator_property(name) 

102 else: 

103 f = _create_delegator_method(name) 

104 

105 # don't overwrite existing methods/properties 

106 if overwrite or not hasattr(cls, name): 

107 setattr(cls, name, f)

145# Ported with modifications from xarray 

146# https://github.com/pydata/xarray/blob/master/xarray/core/extensions.py 

147# 1. We don't need to catch and re-raise AttributeErrors as RuntimeErrors 

148# 2. We use a UserWarning instead of a custom Warning

150class CachedAccessor(object): 

151 """ 

152 Custom property-like object (descriptor) for caching accessors. 

153 

154 Parameters 

155 ---------- 

156 name : str 

157 The namespace this will be accessed under, e.g. ``df.foo`` 

158 accessor : cls 

159 The class with the extension methods. The class' __init__ method 

160 should expect one of a ``Series``, ``DataFrame`` or ``Index`` as 

161 the single argument ``data`` 

162 """ 

163 def __init__(self, name, accessor): 

164 self._name = name 

165 self._accessor = accessor 

166 

167 def __get__(self, obj, cls): 

168 if obj is None: 

169 # we're accessing the attribute of the class, i.e., Dataset.geo 

170 return self._accessor 

171 accessor_obj = self._accessor(obj) 

172 # Replace the property with the accessor object. Inspired by: 

173 # http://www.pydanny.com/cached-property.html 

174 # We need to use object.__setattr__ because we overwrite __setattr__ on 

175 # NDFrame 

176 object.__setattr__(obj, self._name, accessor_obj) 

177 return accessor_obj

180def _register_accessor(name, cls): 

181 def decorator(accessor): 

182 if hasattr(cls, name): 

183 warnings.warn( 

184 'registration of accessor {!r} under name {!r} for type ' 

185 '{!r} is overriding a preexisting attribute with the same ' 

186 'name.'.format(accessor, name, cls), 

187 UserWarning, 

188 stacklevel=2) 

189 setattr(cls, name, CachedAccessor(name, accessor)) 

190 cls._accessors.add(name) 

191 return accessor 

192 return decorator

208Notes

210When accessed, your accessor will be initialized with the pandas object 

211the user is interacting with. So the signature must be

932 if self.float_format is None:

935 fmt_str = ('{{x: .{prec:d}g}}'

987class FloatArrayFormatter(GenericArrayFormatter): 

988 """ 

989 

990 """ 

991 

992 def __init__(self, *args, **kwargs): 

993 GenericArrayFormatter.__init__(self, *args, **kwargs) 

994 

995 # float_format is expected to be a string 

996 # formatter should be used to pass a function 

997 if self.float_format is not None and self.formatter is None:

1004 def _value_formatter(self, float_format=None, threshold=None): 

1005 """Returns a function to be applied on each value to format it 

1006 """ 

1007 

1008 # the float_format parameter supersedes self.float_format 

1009 if float_format is None: 

1010 float_format = self.float_format 

1011 

1012 # we are going to compose different functions, to first convert to 

1013 # a string, then replace the decimal symbol, and finally chop according 

1014 # to the threshold 

1015 

1016 # when there is no float_format, we use str instead of '%g' 

1017 # because str(0.0) = '0.0' while '%g' % 0.0 = '0' 

1018 if float_format: 

1019 def base_formatter(v): 

1020 return float_format(value=v) if notna(v) else self.na_rep 

1021 else: 

1022 def base_formatter(v): 

1023 return str(v) if notna(v) else self.na_rep 

1024 

1025 if self.decimal != '.': 

1026 def decimal_formatter(v): 

1027 return base_formatter(v).replace('.', self.decimal, 1) 

1028 else: 

1029 decimal_formatter = base_formatter 

1030 

1031 if threshold is None: 

1032 return decimal_formatter 

1033 

1034 def formatter(value): 

1035 if notna(value): 

1036 if abs(value) > threshold: 

1037 return decimal_formatter(value) 

1038 else: 

1039 return decimal_formatter(0.0) 

1040 else: 

1041 return self.na_rep 

1042 

1043 return formatter

1045 def get_result_as_array(self): 

1046 """ 

1047 Returns the float values converted into strings using 

1048 the parameters given at initialisation, as a numpy array 

1049 """ 

1050 

1051 if self.formatter is not None: 

1052 return np.array([self.formatter(x) for x in self.values]) 

1053 

1054 if self.fixed_width: 

1055 threshold = get_option("display.chop_threshold") 

1056 else: 

1057 threshold = None 

1058 

1059 # if we have a fixed_width, we'll need to try different float_format 

1060 def format_values_with(float_format): 

1061 formatter = self._value_formatter(float_format, threshold) 

1062 

1063 # default formatter leaves a space to the left when formatting 

1064 # floats, must be consistent for left-justifying NaNs (GH #25061) 

1065 if self.justify == 'left': 

1066 na_rep = ' ' + self.na_rep 

1067 else: 

1068 na_rep = self.na_rep 

1069 

1070 # separate the wheat from the chaff 

1071 values = self.values 

1072 mask = isna(values) 

1073 if hasattr(values, 'to_dense'): # sparse numpy ndarray 

1074 values = values.to_dense() 

1075 values = np.array(values, dtype='object') 

1076 values[mask] = na_rep 

1077 imask = (~mask).ravel() 

1078 values.flat[imask] = np.array([formatter(val) 

1079 for val in values.ravel()[imask]]) 

1080 

1081 if self.fixed_width: 

1082 return _trim_zeros(values, na_rep) 

1083 

1084 return values 

1085 

1086 # There is a special default string when we are fixed-width 

1087 # The default is otherwise to use str instead of a formatting string 

1088 if self.float_format is None: 

1089 if self.fixed_width: 

1090 float_format = partial('{value: .{digits:d}f}'.format, 

1091 digits=self.digits) 

1092 else: 

1093 float_format = self.float_format 

1094 else: 

1095 float_format = lambda value: self.float_format % value 

1096 

1097 formatted_values = format_values_with(float_format) 

1098 

1099 if not self.fixed_width: 

1100 return formatted_values 

1101 

1102 # we need do convert to engineering format if some values are too small 

1103 # and would appear as 0, or if some values are too big and take too 

1104 # much space 

1105 

1106 if len(formatted_values) > 0: 

1107 maxlen = max(len(x) for x in formatted_values) 

1108 too_long = maxlen > self.digits + 6 

1109 else: 

1110 too_long = False 

1111 

1112 with np.errstate(invalid='ignore'): 

1113 abs_vals = np.abs(self.values) 

1114 # this is pretty arbitrary for now 

1115 # large values: more that 8 characters including decimal symbol 

1116 # and first digit, hence > 1e6 

1117 has_large_values = (abs_vals > 1e6).any() 

1118 has_small_values = ((abs_vals < 10**(-self.digits)) & 

1119 (abs_vals > 0)).any() 

1120 

1121 if has_small_values or (too_long and has_large_values): 

1122 float_format = partial('{value: .{digits:d}e}'.format, 

1123 digits=self.digits) 

1124 formatted_values = format_values_with(float_format) 

1125 

1126 return formatted_values

1131 return [self.formatter(x) for x in self.values]

1193def format_percentiles(percentiles): 

1194 """ 

1195 Outputs rounded and formatted percentiles. 

1196 

1197 Parameters 

1198 ---------- 

1199 percentiles : list-like, containing floats from interval [0,1] 

1200 

1201 Returns 

1202 ------- 

1203 formatted : list of strings 

1204 

1205 Notes 

1206 ----- 

1207 Rounding precision is chosen so that: (1) if any two elements of 

1208 ``percentiles`` differ, they remain different after rounding 

1209 (2) no entry is *rounded* to 0% or 100%. 

1210 Any non-integer is always rounded to at least 1 decimal place. 

1211 

1212 Examples 

1213 -------- 

1214 Keeps all entries different after rounding: 

1215 

1216 >>> format_percentiles([0.01999, 0.02001, 0.5, 0.666666, 0.9999]) 

1217 ['1.999%', '2.001%', '50%', '66.667%', '99.99%'] 

1218 

1219 No element is rounded to 0% or 100% (unless already equal to it). 

1220 Duplicates are allowed: 

1221 

1222 >>> format_percentiles([0, 0.5, 0.02001, 0.5, 0.666666, 0.9999]) 

1223 ['0%', '50%', '2.0%', '50%', '66.67%', '99.99%'] 

1224 """ 

1225 

1226 percentiles = np.asarray(percentiles) 

1227 

1228 # It checks for np.NaN as well 

1229 with np.errstate(invalid='ignore'): 

1230 if not is_numeric_dtype(percentiles) or not np.all(percentiles >= 0) \ 

1231 or not np.all(percentiles <= 1): 

1232 raise ValueError("percentiles should all be in the interval [0,1]") 

1233 

1234 percentiles = 100 * percentiles 

1235 int_idx = (percentiles.astype(int) == percentiles) 

1236 

1237 if np.all(int_idx): 

1238 out = percentiles.astype(int).astype(str) 

1239 return [i + '%' for i in out] 

1240 

1241 unique_pcts = np.unique(percentiles) 

1242 to_begin = unique_pcts[0] if unique_pcts[0] > 0 else None 

1243 to_end = 100 - unique_pcts[-1] if unique_pcts[-1] < 100 else None 

1244 

1245 # Least precision that keeps percentiles unique after rounding 

1246 prec = -np.floor(np.log10(np.min( 

1247 np.ediff1d(unique_pcts, to_begin=to_begin, to_end=to_end) 

1248 ))).astype(int) 

1249 prec = max(1, prec) 

1250 out = np.empty_like(percentiles, dtype=object) 

1251 out[int_idx] = percentiles[int_idx].astype(int).astype(str) 

1252 out[~int_idx] = percentiles[~int_idx].round(prec).astype(str) 

1253 return [i + '%' for i in out]

1441class EngFormatter(object): 

1442 """ 

1443 Formats float values according to engineering format. 

1444 

1445 Based on matplotlib.ticker.EngFormatter 

1446 """ 

1447 

1448 # The SI engineering prefixes 

1449 ENG_PREFIXES = { 

1450 -24: "y", 

1451 -21: "z", 

1452 -18: "a", 

1453 -15: "f", 

1454 -12: "p", 

1455 -9: "n", 

1456 -6: "u", 

1457 -3: "m", 

1458 0: "", 

1459 3: "k", 

1460 6: "M", 

1461 9: "G", 

1462 12: "T", 

1463 15: "P", 

1464 18: "E", 

1465 21: "Z", 

1466 24: "Y" 

1467 } 

1468 

1469 def __init__(self, accuracy=None, use_eng_prefix=False): 

1470 self.accuracy = accuracy 

1471 self.use_eng_prefix = use_eng_prefix 

1472 

1473 def __call__(self, num): 

1474 """ Formats a number in engineering notation, appending a letter 

1475 representing the power of 1000 of the original number. Some examples: 

1476 

1477 >>> format_eng(0) # for self.accuracy = 0 

1478 ' 0' 

1479 

1480 >>> format_eng(1000000) # for self.accuracy = 1, 

1481 # self.use_eng_prefix = True 

1482 ' 1.0M' 

1483 

1484 >>> format_eng("-1e-6") # for self.accuracy = 2 

1485 # self.use_eng_prefix = False 

1486 '-1.00E-06' 

1487 

1488 @param num: the value to represent 

1489 @type num: either a numeric value or a string that can be converted to 

1490 a numeric value (as per decimal.Decimal constructor) 

1491 

1492 @return: engineering formatted string 

1493 """ 

1494 import decimal 

1495 import math 

1496 dnum = decimal.Decimal(str(num)) 

1497 

1498 if decimal.Decimal.is_nan(dnum): 

1499 return 'NaN' 

1500 

1501 if decimal.Decimal.is_infinite(dnum): 

1502 return 'inf' 

1503 

1504 sign = 1 

1505 

1506 if dnum < 0: # pragma: no cover 

1507 sign = -1 

1508 dnum = -dnum 

1509 

1510 if dnum != 0: 

1511 pow10 = decimal.Decimal(int(math.floor(dnum.log10() / 3) * 3)) 

1512 else: 

1513 pow10 = decimal.Decimal(0) 

1514 

1515 pow10 = pow10.min(max(self.ENG_PREFIXES.keys())) 

1516 pow10 = pow10.max(min(self.ENG_PREFIXES.keys())) 

1517 int_pow10 = int(pow10) 

1518 

1519 if self.use_eng_prefix: 

1520 prefix = self.ENG_PREFIXES[int_pow10] 

1521 else: 

1522 if int_pow10 < 0: 

1523 prefix = 'E-{pow10:02d}'.format(pow10=-int_pow10) 

1524 else: 

1525 prefix = 'E+{pow10:02d}'.format(pow10=int_pow10) 

1526 

1527 mant = sign * dnum / (10**pow10) 

1528 

1529 if self.accuracy is None: # pragma: no cover 

1530 format_str = u("{mant: g}{prefix}") 

1531 else: 

1532 format_str = (u("{{mant: .{acc:d}f}}{{prefix}}") 

1533 .format(acc=self.accuracy)) 

1534 

1535 formatted = format_str.format(mant=mant, prefix=prefix) 

1536 

1537 return formatted # .strip()

1576 Parameters

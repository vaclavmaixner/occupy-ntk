156 idle in the pool. The setting may be appropriate in the 

157 case of :class:`.NullPool` or special circumstances where 

158 the connection pool in use is not being used to maintain connection

403class _ConnectionRecord(object): 

404 

405 """Internal object which maintains an individual DBAPI connection 

406 referenced by a :class:`.Pool`. 

407 

408 The :class:`._ConnectionRecord` object always exists for any particular 

409 DBAPI connection whether or not that DBAPI connection has been 

410 "checked out". This is in contrast to the :class:`._ConnectionFairy`

424 pool events, including :meth:`.PoolEvents.connect` and

428 .. seealso::

744 do_commit() methods. 

745 

746 In practice, a :class:`.Connection` assigns a :class:`.Transaction` object 

747 to this variable when one is in scope so that the :class:`.Transaction` 

748 takes the job of committing or rolling back on return if 

749 :meth:`.Connection.close` is called while the :class:`.Transaction` 

750 still exists. 

751 

752 This is essentially an "event handler" of sorts but is simplified as an 

753 instance variable both for performance/simplicity as well as that there 

754 can only be one "reset agent" at a time. 

755 """ 

756 

757 @classmethod 

758 def _checkout(cls, pool, threadconns=None, fairy=None): 

759 if not fairy: 

760 fairy = _ConnectionRecord.checkout(pool) 

761 

762 fairy._pool = pool 

763 fairy._counter = 0 

764 

765 if threadconns is not None: 

766 threadconns.current = weakref.ref(fairy) 

767 

768 if fairy.connection is None: 

769 raise exc.InvalidRequestError("This connection is closed") 

770 fairy._counter += 1 

771 

772 if ( 

773 not pool.dispatch.checkout and not pool._pre_ping 

774 ) or fairy._counter != 1: 

775 return fairy 

776 

777 # Pool listeners can trigger a reconnection on checkout, as well 

778 # as the pre-pinger. 

779 # there are three attempts made here, but note that if the database 

780 # is not accessible from a connection standpoint, those won't proceed 

781 # here. 

782 attempts = 2 

783 while attempts > 0: 

784 try: 

785 if pool._pre_ping: 

786 if fairy._echo: 

787 pool.logger.debug( 

788 "Pool pre-ping on connection %s", fairy.connection 

789 ) 

790 

791 result = pool._dialect.do_ping(fairy.connection) 

792 if not result: 

793 if fairy._echo: 

794 pool.logger.debug( 

795 "Pool pre-ping on connection %s failed, " 

796 "will invalidate pool", 

797 fairy.connection, 

798 ) 

799 raise exc.InvalidatePoolError() 

800 

801 pool.dispatch.checkout( 

802 fairy.connection, fairy._connection_record, fairy 

803 ) 

804 return fairy 

805 except exc.DisconnectionError as e: 

806 if e.invalidate_pool: 

807 pool.logger.info( 

808 "Disconnection detected on checkout, " 

809 "invalidating all pooled connections prior to " 

810 "current timestamp (reason: %r)", 

811 e, 

812 ) 

813 fairy._connection_record.invalidate(e) 

814 pool._invalidate(fairy, e, _checkin=False) 

815 else: 

816 pool.logger.info( 

817 "Disconnection detected on checkout, " 

818 "invalidating individual connection %s (reason: %r)", 

819 fairy.connection, 

820 e, 

821 ) 

822 fairy._connection_record.invalidate(e) 

823 try: 

824 fairy.connection = ( 

825 fairy._connection_record.get_connection() 

826 ) 

827 except Exception as err: 

828 with util.safe_reraise(): 

829 fairy._connection_record._checkin_failed(err) 

830 

831 attempts -= 1 

832 

833 pool.logger.info("Reconnection attempts exhausted on checkout") 

834 fairy.invalidate() 

835 raise exc.InvalidRequestError("This connection is closed")

838 return _ConnectionFairy._checkout(self._pool, fairy=self)

843 self._connection_record, 

844 self._pool, 

845 None, 

846 self._echo, 

847 fairy=self, 

848 ) 

849 self.connection = None

854 def _reset(self, pool): 

855 if pool.dispatch.reset: 

856 pool.dispatch.reset(self, self._connection_record) 

857 if pool._reset_on_return is reset_rollback: 

858 if self._echo: 

859 pool.logger.debug( 

860 "Connection %s rollback-on-return%s", 

861 self.connection, 

862 ", via agent" if self._reset_agent else "", 

863 ) 

864 if self._reset_agent: 

865 self._reset_agent.rollback() 

866 else: 

867 pool._dialect.do_rollback(self) 

868 elif pool._reset_on_return is reset_commit: 

869 if self._echo: 

870 pool.logger.debug( 

871 "Connection %s commit-on-return%s", 

872 self.connection, 

873 ", via agent" if self._reset_agent else "", 

874 ) 

875 if self._reset_agent: 

876 self._reset_agent.commit() 

877 else: 

878 pool._dialect.do_commit(self)

927 def invalidate(self, e=None, soft=False): 

928 """Mark this connection as invalidated. 

929 

930 This method can be called directly, and is also called as a result 

931 of the :meth:`.Connection.invalidate` method. When invoked, 

932 the DBAPI connection is immediately closed and discarded from 

933 further use by the pool. The invalidation mechanism proceeds 

934 via the :meth:`._ConnectionRecord.invalidate` internal method. 

935 

936 :param e: an exception object indicating a reason for the invalidation. 

937 

938 :param soft: if True, the connection isn't closed; instead, this 

939 connection will be recycled on next checkout. 

940 

941 .. versionadded:: 1.0.3 

942 

943 .. seealso:: 

944 

945 :ref:`pool_connection_invalidation` 

946 

947 """ 

948 

949 if self.connection is None: 

950 util.warn("Can't invalidate an already-closed connection.") 

951 return 

952 if self._connection_record: 

953 self._connection_record.invalidate(e=e, soft=soft) 

954 if not soft: 

955 self.connection = None 

956 self._checkin()

971 """Separate this connection from its Pool. 

972 

973 This means that the connection will no longer be returned to the 

974 pool when closed, and will instead be literally closed. The 

975 containing ConnectionRecord is separated from the DB-API connection, 

976 and will create a new connection when next used. 

977 

978 Note that any overall connection limiting constraints imposed by a 

979 Pool implementation may be violated after a detach, as the detached 

980 connection is removed from the pool's knowledge and control. 

981 """ 

982

985 _refs.remove(rec) 

986 rec.fairy_ref = None 

987 rec.connection = None 

988 # TODO: should this be _return_conn? 

989 self._pool._do_return_conn(self._connection_record) 

990 self.info = self.info.copy() 

991 self._connection_record = None

993 if self._pool.dispatch.detach: 

994 self._pool.dispatch.detach(self.connection, rec)

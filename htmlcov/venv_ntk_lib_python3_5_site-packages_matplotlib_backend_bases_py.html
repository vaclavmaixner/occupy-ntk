204 def draw_path_collection(self, gc, master_transform, paths, all_transforms, 

205 offsets, offsetTrans, facecolors, edgecolors, 

206 linewidths, linestyles, antialiaseds, urls, 

207 offset_position): 

208 """ 

209 Draws a collection of paths selecting drawing properties from 

210 the lists *facecolors*, *edgecolors*, *linewidths*, 

211 *linestyles* and *antialiaseds*. *offsets* is a list of 

212 offsets to apply to each of the paths. The offsets in 

213 *offsets* are first transformed by *offsetTrans* before being 

214 applied. *offset_position* may be either "screen" or "data" 

215 depending on the space that the offsets are in. 

216 

217 This provides a fallback implementation of 

218 :meth:`draw_path_collection` that makes multiple calls to 

219 :meth:`draw_path`. Some backends may want to override this in 

220 order to render each set of path data only once, and then 

221 reference that path multiple times with the different offsets, 

222 colors, styles etc. The generator methods 

223 :meth:`_iter_collection_raw_paths` and 

224 :meth:`_iter_collection` are provided to help with (and 

225 standardize) the implementation across backends. It is highly 

226 recommended to use those generators, so that changes to the 

227 behavior of :meth:`draw_path_collection` can be made globally. 

228 """ 

229 path_ids = [] 

230 for path, transform in self._iter_collection_raw_paths( 

231 master_transform, paths, all_transforms): 

232 path_ids.append((path, transforms.Affine2D(transform))) 

233 

234 for xo, yo, path_id, gc0, rgbFace in self._iter_collection( 

235 gc, master_transform, all_transforms, path_ids, offsets, 

236 offsetTrans, facecolors, edgecolors, linewidths, linestyles, 

237 antialiaseds, urls, offset_position): 

238 path, transform = path_id 

239 transform = transforms.Affine2D( 

240 transform.get_matrix()).translate(xo, yo) 

241 self.draw_path(gc0, path, transform, rgbFace)

302 def _iter_collection_raw_paths(self, master_transform, paths, 

303 all_transforms): 

304 """ 

305 This is a helper method (along with :meth:`_iter_collection`) to make 

306 it easier to write a space-efficient :meth:`draw_path_collection` 

307 implementation in a backend. 

308 

309 This method yields all of the base path/transform 

310 combinations, given a master transform, a list of paths and 

311 list of transforms. 

312 

313 The arguments should be exactly what is passed in to 

314 :meth:`draw_path_collection`. 

315 

316 The backend should take each yielded path and transform and 

317 create an object that can be referenced (reused) later. 

318 """ 

319 Npaths = len(paths) 

320 Ntransforms = len(all_transforms) 

321 N = max(Npaths, Ntransforms) 

322 

323 if Npaths == 0: 

324 return 

325 

326 transform = transforms.IdentityTransform() 

327 for i in range(N): 

328 path = paths[i % Npaths] 

329 if Ntransforms: 

330 transform = Affine2D(all_transforms[i % Ntransforms]) 

331 yield path, transform + master_transform

333 def _iter_collection_uses_per_path(self, paths, all_transforms, 

334 offsets, facecolors, edgecolors): 

335 """ 

336 Compute how many times each raw path object returned by 

337 _iter_collection_raw_paths would be used when calling 

338 _iter_collection. This is intended for the backend to decide 

339 on the tradeoff between using the paths in-line and storing 

340 them once and reusing. Rounds up in case the number of uses 

341 is not the same for every path. 

342 """ 

343 Npaths = len(paths) 

344 if Npaths == 0 or len(facecolors) == len(edgecolors) == 0: 

345 return 0 

346 Npath_ids = max(Npaths, len(all_transforms)) 

347 N = max(Npath_ids, len(offsets)) 

348 return (N + Npath_ids - 1) // Npath_ids

350 def _iter_collection(self, gc, master_transform, all_transforms, 

351 path_ids, offsets, offsetTrans, facecolors, 

352 edgecolors, linewidths, linestyles, 

353 antialiaseds, urls, offset_position): 

354 """ 

355 This is a helper method (along with 

356 :meth:`_iter_collection_raw_paths`) to make it easier to write 

357 a space-efficient :meth:`draw_path_collection` implementation in a 

358 backend. 

359 

360 This method yields all of the path, offset and graphics 

361 context combinations to draw the path collection. The caller 

362 should already have looped over the results of 

363 :meth:`_iter_collection_raw_paths` to draw this collection. 

364 

365 The arguments should be the same as that passed into 

366 :meth:`draw_path_collection`, with the exception of 

367 *path_ids*, which is a list of arbitrary objects that the 

368 backend will use to reference one of the paths created in the 

369 :meth:`_iter_collection_raw_paths` stage. 

370 

371 Each yielded result is of the form:: 

372 

373 xo, yo, path_id, gc, rgbFace 

374 

375 where *xo*, *yo* is an offset; *path_id* is one of the elements of 

376 *path_ids*; *gc* is a graphics context and *rgbFace* is a color to 

377 use for filling the path. 

378 """ 

379 Ntransforms = len(all_transforms) 

380 Npaths = len(path_ids) 

381 Noffsets = len(offsets) 

382 N = max(Npaths, Noffsets) 

383 Nfacecolors = len(facecolors) 

384 Nedgecolors = len(edgecolors) 

385 Nlinewidths = len(linewidths) 

386 Nlinestyles = len(linestyles) 

387 Naa = len(antialiaseds) 

388 Nurls = len(urls) 

389 

390 if (Nfacecolors == 0 and Nedgecolors == 0) or Npaths == 0: 

391 return 

392 if Noffsets: 

393 toffsets = offsetTrans.transform(offsets) 

394 

395 gc0 = self.new_gc() 

396 gc0.copy_properties(gc) 

397 

398 if Nfacecolors == 0: 

399 rgbFace = None 

400 

401 if Nedgecolors == 0: 

402 gc0.set_linewidth(0.0) 

403 

404 xo, yo = 0, 0 

405 for i in range(N): 

406 path_id = path_ids[i % Npaths] 

407 if Noffsets: 

408 xo, yo = toffsets[i % Noffsets] 

409 if offset_position == 'data': 

410 if Ntransforms: 

411 transform = ( 

412 Affine2D(all_transforms[i % Ntransforms]) + 

413 master_transform) 

414 else: 

415 transform = master_transform 

416 xo, yo = transform.transform_point((xo, yo)) 

417 xp, yp = transform.transform_point((0, 0)) 

418 xo = -(xp - xo) 

419 yo = -(yp - yo) 

420 if not (np.isfinite(xo) and np.isfinite(yo)): 

421 continue 

422 if Nfacecolors: 

423 rgbFace = facecolors[i % Nfacecolors] 

424 if Nedgecolors: 

425 if Nlinewidths: 

426 gc0.set_linewidth(linewidths[i % Nlinewidths]) 

427 if Nlinestyles: 

428 gc0.set_dashes(*linestyles[i % Nlinestyles]) 

429 fg = edgecolors[i % Nedgecolors] 

430 if len(fg) == 4: 

431 if fg[3] == 0.0: 

432 gc0.set_linewidth(0) 

433 else: 

434 gc0.set_foreground(fg) 

435 else: 

436 gc0.set_foreground(fg) 

437 if rgbFace is not None and len(rgbFace) == 4: 

438 if rgbFace[3] == 0: 

439 rgbFace = None 

440 gc0.set_antialiased(antialiaseds[i % Naa]) 

441 if Nurls: 

442 gc0.set_url(urls[i % Nurls]) 

443 

444 yield xo, yo, path_id, gc0, rgbFace 

445 gc0.restore()

1950 def print_figure(self, filename, dpi=None, facecolor=None, edgecolor=None, 

1951 orientation='portrait', format=None, 

1952 *, bbox_inches=None, **kwargs): 

1953 """ 

1954 Render the figure to hardcopy. Set the figure patch face and edge 

1955 colors. This is useful because some of the GUIs have a gray figure 

1956 face color background and you'll probably want to override this on 

1957 hardcopy. 

1958 

1959 Parameters 

1960 ---------- 

1961 filename 

1962 can also be a file object on image backends 

1963 

1964 orientation : {'landscape', 'portrait'}, optional 

1965 only currently applies to PostScript printing. 

1966 

1967 dpi : scalar, optional 

1968 the dots per inch to save the figure in; if None, use savefig.dpi 

1969 

1970 facecolor : color spec or None, optional 

1971 the facecolor of the figure; if None, defaults to savefig.facecolor 

1972 

1973 edgecolor : color spec or None, optional 

1974 the edgecolor of the figure; if None, defaults to savefig.edgecolor 

1975 

1976 format : str, optional 

1977 when set, forcibly set the file format to save to 

1978 

1979 bbox_inches : str or `~matplotlib.transforms.Bbox`, optional 

1980 Bbox in inches. Only the given portion of the figure is 

1981 saved. If 'tight', try to figure out the tight bbox of 

1982 the figure. If None, use savefig.bbox 

1983 

1984 pad_inches : scalar, optional 

1985 Amount of padding around the figure when bbox_inches is 

1986 'tight'. If None, use savefig.pad_inches 

1987 

1988 bbox_extra_artists : list of `~matplotlib.artist.Artist`, optional 

1989 A list of extra artists that will be considered when the 

1990 tight bbox is calculated. 

1991 

1992 """ 

1993 if format is None: 

1994 # get format from filename, or from backend's default filetype 

1995 if isinstance(filename, getattr(os, "PathLike", ())): 

1996 filename = os.fspath(filename) 

1997 if isinstance(filename, str): 

1998 format = os.path.splitext(filename)[1][1:] 

1999 if format is None or format == '': 

2000 format = self.get_default_filetype() 

2001 if isinstance(filename, str): 

2002 filename = filename.rstrip('.') + '.' + format 

2003 format = format.lower() 

2004 

2005 # get canvas object and print method for format 

2006 canvas = self._get_output_canvas(format) 

2007 print_method = getattr(canvas, 'print_%s' % format) 

2008 

2009 if dpi is None: 

2010 dpi = rcParams['savefig.dpi'] 

2011 if dpi == 'figure': 

2012 dpi = getattr(self.figure, '_original_dpi', self.figure.dpi) 

2013 

2014 # Remove the figure manager, if any, to avoid resizing the GUI widget. 

2015 # Some code (e.g. Figure.show) differentiates between having *no* 

2016 # manager and a *None* manager, which should be fixed at some point, 

2017 # but this should be fine. 

2018 with cbook._setattr_cm(self, _is_saving=True, manager=None), \ 

2019 cbook._setattr_cm(self.figure, dpi=dpi): 

2020 

2021 if facecolor is None: 

2022 facecolor = rcParams['savefig.facecolor'] 

2023 if edgecolor is None: 

2024 edgecolor = rcParams['savefig.edgecolor'] 

2025 

2026 origfacecolor = self.figure.get_facecolor() 

2027 origedgecolor = self.figure.get_edgecolor() 

2028 

2029 self.figure.dpi = dpi 

2030 self.figure.set_facecolor(facecolor) 

2031 self.figure.set_edgecolor(edgecolor) 

2032 

2033 if bbox_inches is None: 

2034 bbox_inches = rcParams['savefig.bbox'] 

2035 

2036 if bbox_inches: 

2037 # call adjust_bbox to save only the given area 

2038 if bbox_inches == "tight": 

2039 # When bbox_inches == "tight", it saves the figure twice. 

2040 # The first save command (to a BytesIO) is just to estimate 

2041 # the bounding box of the figure. 

2042 result = print_method( 

2043 io.BytesIO(), 

2044 dpi=dpi, 

2045 facecolor=facecolor, 

2046 edgecolor=edgecolor, 

2047 orientation=orientation, 

2048 dryrun=True, 

2049 **kwargs) 

2050 renderer = self.figure._cachedRenderer 

2051 bbox_artists = kwargs.pop("bbox_extra_artists", None) 

2052 bbox_inches = self.figure.get_tightbbox(renderer, 

2053 bbox_extra_artists=bbox_artists) 

2054 pad = kwargs.pop("pad_inches", None) 

2055 if pad is None: 

2056 pad = rcParams['savefig.pad_inches'] 

2057 

2058 bbox_inches = bbox_inches.padded(pad) 

2059 

2060 restore_bbox = tight_bbox.adjust_bbox(self.figure, bbox_inches, 

2061 canvas.fixed_dpi) 

2062 

2063 _bbox_inches_restore = (bbox_inches, restore_bbox) 

2064 else: 

2065 _bbox_inches_restore = None 

2066 

2067 try: 

2068 result = print_method( 

2069 filename, 

2070 dpi=dpi, 

2071 facecolor=facecolor, 

2072 edgecolor=edgecolor, 

2073 orientation=orientation, 

2074 bbox_inches_restore=_bbox_inches_restore, 

2075 **kwargs) 

2076 finally: 

2077 if bbox_inches and restore_bbox: 

2078 restore_bbox() 

2079 

2080 self.figure.set_facecolor(origfacecolor)

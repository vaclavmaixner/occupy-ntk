706 def _sub_datetime_arraylike(self, other): 

707 """subtract DatetimeArray/Index or ndarray[datetime64]"""

709 raise ValueError("cannot add indices of unequal length") 

710 

711 if isinstance(other, np.ndarray): 

712 assert is_datetime64_dtype(other)

715 if not self._has_same_tz(other):

718 "timezones or no timezones" 

719 .format(cls=type(self).__name__))

721 self_i8 = self.asi8 

722 other_i8 = other.asi8 

723 arr_mask = self._isnan | other._isnan 

724 new_values = checked_add_with_arr(self_i8, -other_i8, 

725 arr_mask=arr_mask) 

726 if self._hasnans or other._hasnans: 

727 new_values[arr_mask] = iNaT 

728 return new_values.view('timedelta64[ns]')

796 def tz_convert(self, tz): 

797 """ 

798 Convert tz-aware Datetime Array/Index from one time zone to another. 

799 

800 Parameters 

801 ---------- 

802 tz : string, pytz.timezone, dateutil.tz.tzfile or None 

803 Time zone for time. Corresponding timestamps would be converted 

804 to this time zone of the Datetime Array/Index. A `tz` of None will 

805 convert to UTC and remove the timezone information. 

806 

807 Returns 

808 ------- 

809 normalized : same type as self 

810 

811 Raises 

812 ------ 

813 TypeError 

814 If Datetime Array/Index is tz-naive. 

815 

816 See Also 

817 -------- 

818 DatetimeIndex.tz : A timezone that has a variable offset from UTC. 

819 DatetimeIndex.tz_localize : Localize tz-naive DatetimeIndex to a 

820 given time zone, or remove timezone from a tz-aware DatetimeIndex. 

821 

822 Examples 

823 -------- 

824 With the `tz` parameter, we can change the DatetimeIndex 

825 to other time zones: 

826 

827 >>> dti = pd.date_range(start='2014-08-01 09:00', 

828 ... freq='H', periods=3, tz='Europe/Berlin') 

829 

830 >>> dti 

831 DatetimeIndex(['2014-08-01 09:00:00+02:00', 

832 '2014-08-01 10:00:00+02:00', 

833 '2014-08-01 11:00:00+02:00'], 

834 dtype='datetime64[ns, Europe/Berlin]', freq='H') 

835 

836 >>> dti.tz_convert('US/Central') 

837 DatetimeIndex(['2014-08-01 02:00:00-05:00', 

838 '2014-08-01 03:00:00-05:00', 

839 '2014-08-01 04:00:00-05:00'], 

840 dtype='datetime64[ns, US/Central]', freq='H') 

841 

842 With the ``tz=None``, we can remove the timezone (after converting 

843 to UTC if necessary): 

844 

845 >>> dti = pd.date_range(start='2014-08-01 09:00',freq='H', 

846 ... periods=3, tz='Europe/Berlin') 

847 

848 >>> dti 

849 DatetimeIndex(['2014-08-01 09:00:00+02:00', 

850 '2014-08-01 10:00:00+02:00', 

851 '2014-08-01 11:00:00+02:00'], 

852 dtype='datetime64[ns, Europe/Berlin]', freq='H') 

853 

854 >>> dti.tz_convert(None) 

855 DatetimeIndex(['2014-08-01 07:00:00', 

856 '2014-08-01 08:00:00', 

857 '2014-08-01 09:00:00'], 

858 dtype='datetime64[ns]', freq='H') 

859 """ 

860 tz = timezones.maybe_get_tz(tz) 

861 

862 if self.tz is None: 

863 # tz naive, use tz_localize 

864 raise TypeError('Cannot convert tz-naive timestamps, use ' 

865 'tz_localize to localize') 

866 

867 # No conversion since timestamps are all UTC to begin with 

868 dtype = tz_to_dtype(tz) 

869 return self._simple_new(self.asi8, dtype=dtype, freq=self.freq)

871 def tz_localize(self, tz, ambiguous='raise', nonexistent='raise', 

872 errors=None): 

873 """ 

874 Localize tz-naive Datetime Array/Index to tz-aware 

875 Datetime Array/Index. 

876 

877 This method takes a time zone (tz) naive Datetime Array/Index object 

878 and makes this time zone aware. It does not move the time to another 

879 time zone. 

880 Time zone localization helps to switch from time zone aware to time 

881 zone unaware objects. 

882 

883 Parameters 

884 ---------- 

885 tz : string, pytz.timezone, dateutil.tz.tzfile or None 

886 Time zone to convert timestamps to. Passing ``None`` will 

887 remove the time zone information preserving local time. 

888 ambiguous : 'infer', 'NaT', bool array, default 'raise' 

889 When clocks moved backward due to DST, ambiguous times may arise. 

890 For example in Central European Time (UTC+01), when going from 

891 03:00 DST to 02:00 non-DST, 02:30:00 local time occurs both at 

892 00:30:00 UTC and at 01:30:00 UTC. In such a situation, the 

893 `ambiguous` parameter dictates how ambiguous times should be 

894 handled. 

895 

896 - 'infer' will attempt to infer fall dst-transition hours based on 

897 order 

898 - bool-ndarray where True signifies a DST time, False signifies a 

899 non-DST time (note that this flag is only applicable for 

900 ambiguous times) 

901 - 'NaT' will return NaT where there are ambiguous times 

902 - 'raise' will raise an AmbiguousTimeError if there are ambiguous 

903 times 

904 

905 nonexistent : 'shift_forward', 'shift_backward, 'NaT', timedelta, 

906 default 'raise' 

907 A nonexistent time does not exist in a particular timezone 

908 where clocks moved forward due to DST. 

909 

910 - 'shift_forward' will shift the nonexistent time forward to the 

911 closest existing time 

912 - 'shift_backward' will shift the nonexistent time backward to the 

913 closest existing time 

914 - 'NaT' will return NaT where there are nonexistent times 

915 - timedelta objects will shift nonexistent times by the timedelta 

916 - 'raise' will raise an NonExistentTimeError if there are 

917 nonexistent times 

918 

919 .. versionadded:: 0.24.0 

920 

921 errors : {'raise', 'coerce'}, default None 

922 

923 - 'raise' will raise a NonExistentTimeError if a timestamp is not 

924 valid in the specified time zone (e.g. due to a transition from 

925 or to DST time). Use ``nonexistent='raise'`` instead. 

926 - 'coerce' will return NaT if the timestamp can not be converted 

927 to the specified time zone. Use ``nonexistent='NaT'`` instead. 

928 

929 .. deprecated:: 0.24.0 

930 

931 Returns 

932 ------- 

933 result : same type as self 

934 Array/Index converted to the specified time zone. 

935 

936 Raises 

937 ------ 

938 TypeError 

939 If the Datetime Array/Index is tz-aware and tz is not None. 

940 

941 See Also 

942 -------- 

943 DatetimeIndex.tz_convert : Convert tz-aware DatetimeIndex from 

944 one time zone to another. 

945 

946 Examples 

947 -------- 

948 >>> tz_naive = pd.date_range('2018-03-01 09:00', periods=3) 

949 >>> tz_naive 

950 DatetimeIndex(['2018-03-01 09:00:00', '2018-03-02 09:00:00', 

951 '2018-03-03 09:00:00'], 

952 dtype='datetime64[ns]', freq='D') 

953 

954 Localize DatetimeIndex in US/Eastern time zone: 

955 

956 >>> tz_aware = tz_naive.tz_localize(tz='US/Eastern') 

957 >>> tz_aware 

958 DatetimeIndex(['2018-03-01 09:00:00-05:00', 

959 '2018-03-02 09:00:00-05:00', 

960 '2018-03-03 09:00:00-05:00'], 

961 dtype='datetime64[ns, US/Eastern]', freq='D') 

962 

963 With the ``tz=None``, we can remove the time zone information 

964 while keeping the local time (not converted to UTC): 

965 

966 >>> tz_aware.tz_localize(None) 

967 DatetimeIndex(['2018-03-01 09:00:00', '2018-03-02 09:00:00', 

968 '2018-03-03 09:00:00'], 

969 dtype='datetime64[ns]', freq='D') 

970 

971 Be careful with DST changes. When there is sequential data, pandas can 

972 infer the DST time: 

973 >>> s = pd.to_datetime(pd.Series([ 

974 ... '2018-10-28 01:30:00', 

975 ... '2018-10-28 02:00:00', 

976 ... '2018-10-28 02:30:00', 

977 ... '2018-10-28 02:00:00', 

978 ... '2018-10-28 02:30:00', 

979 ... '2018-10-28 03:00:00', 

980 ... '2018-10-28 03:30:00'])) 

981 >>> s.dt.tz_localize('CET', ambiguous='infer') 

982 2018-10-28 01:30:00+02:00 0 

983 2018-10-28 02:00:00+02:00 1 

984 2018-10-28 02:30:00+02:00 2 

985 2018-10-28 02:00:00+01:00 3 

986 2018-10-28 02:30:00+01:00 4 

987 2018-10-28 03:00:00+01:00 5 

988 2018-10-28 03:30:00+01:00 6 

989 dtype: int64 

990 

991 In some cases, inferring the DST is impossible. In such cases, you can 

992 pass an ndarray to the ambiguous parameter to set the DST explicitly 

993 

994 >>> s = pd.to_datetime(pd.Series([ 

995 ... '2018-10-28 01:20:00', 

996 ... '2018-10-28 02:36:00', 

997 ... '2018-10-28 03:46:00'])) 

998 >>> s.dt.tz_localize('CET', ambiguous=np.array([True, True, False])) 

999 0 2018-10-28 01:20:00+02:00 

1000 1 2018-10-28 02:36:00+02:00 

1001 2 2018-10-28 03:46:00+01:00 

1002 dtype: datetime64[ns, CET] 

1003 

1004 If the DST transition causes nonexistent times, you can shift these 

1005 dates forward or backwards with a timedelta object or `'shift_forward'` 

1006 or `'shift_backwards'`. 

1007 >>> s = pd.to_datetime(pd.Series([ 

1008 ... '2015-03-29 02:30:00', 

1009 ... '2015-03-29 03:30:00'])) 

1010 >>> s.dt.tz_localize('Europe/Warsaw', nonexistent='shift_forward') 

1011 0 2015-03-29 03:00:00+02:00 

1012 1 2015-03-29 03:30:00+02:00 

1013 dtype: datetime64[ns, 'Europe/Warsaw'] 

1014 >>> s.dt.tz_localize('Europe/Warsaw', nonexistent='shift_backward') 

1015 0 2015-03-29 01:59:59.999999999+01:00 

1016 1 2015-03-29 03:30:00+02:00 

1017 dtype: datetime64[ns, 'Europe/Warsaw'] 

1018 >>> s.dt.tz_localize('Europe/Warsaw', nonexistent=pd.Timedelta('1H')) 

1019 0 2015-03-29 03:30:00+02:00 

1020 1 2015-03-29 03:30:00+02:00 

1021 dtype: datetime64[ns, 'Europe/Warsaw'] 

1022 """ 

1023 if errors is not None: 

1024 warnings.warn("The errors argument is deprecated and will be " 

1025 "removed in a future release. Use " 

1026 "nonexistent='NaT' or nonexistent='raise' " 

1027 "instead.", FutureWarning) 

1028 if errors == 'coerce': 

1029 nonexistent = 'NaT' 

1030 elif errors == 'raise': 

1031 nonexistent = 'raise' 

1032 else: 

1033 raise ValueError("The errors argument must be either 'coerce' " 

1034 "or 'raise'.") 

1035 

1036 nonexistent_options = ('raise', 'NaT', 'shift_forward', 

1037 'shift_backward') 

1038 if nonexistent not in nonexistent_options and not isinstance( 

1039 nonexistent, timedelta): 

1040 raise ValueError("The nonexistent argument must be one of 'raise'," 

1041 " 'NaT', 'shift_forward', 'shift_backward' or" 

1042 " a timedelta object") 

1043 

1044 if self.tz is not None: 

1045 if tz is None: 

1046 new_dates = conversion.tz_convert(self.asi8, timezones.UTC, 

1047 self.tz) 

1048 else: 

1049 raise TypeError("Already tz-aware, use tz_convert to convert.") 

1050 else: 

1051 tz = timezones.maybe_get_tz(tz) 

1052 # Convert to UTC 

1053 

1054 new_dates = conversion.tz_localize_to_utc( 

1055 self.asi8, tz, ambiguous=ambiguous, nonexistent=nonexistent, 

1056 ) 

1057 new_dates = new_dates.view(_NS_DTYPE) 

1058 dtype = tz_to_dtype(tz) 

1059 return self._simple_new(new_dates, dtype=dtype, freq=self.freq)

1681def sequence_to_dt64ns(data, dtype=None, copy=False, 

1682 tz=None, 

1683 dayfirst=False, yearfirst=False, ambiguous='raise', 

1684 int_as_wall_time=False): 

1685 """ 

1686 Parameters 

1687 ---------- 

1688 data : list-like 

1689 dtype : dtype, str, or None, default None 

1690 copy : bool, default False 

1691 tz : tzinfo, str, or None, default None 

1692 dayfirst : bool, default False 

1693 yearfirst : bool, default False 

1694 ambiguous : str, bool, or arraylike, default 'raise' 

1695 See pandas._libs.tslibs.conversion.tz_localize_to_utc 

1696 int_as_wall_time : bool, default False 

1697 Whether to treat ints as wall time in specified timezone, or as 

1698 nanosecond-precision UNIX epoch (wall time in UTC). 

1699 This is used in DatetimeIndex.__init__ to deprecate the wall-time 

1700 behaviour. 

1701 

1702 ..versionadded:: 0.24.0 

1703 

1704 Returns 

1705 ------- 

1706 result : numpy.ndarray 

1707 The sequence converted to a numpy array with dtype ``datetime64[ns]``. 

1708 tz : tzinfo or None 

1709 Either the user-provided tzinfo or one inferred from the data. 

1710 inferred_freq : Tick or None 

1711 The inferred frequency of the sequence. 

1712 

1713 Raises 

1714 ------ 

1715 TypeError : PeriodDType data is passed 

1716 """ 

1717 

1718 inferred_freq = None 

1719 

1720 dtype = _validate_dt64_dtype(dtype) 

1721 

1722 if not hasattr(data, "dtype"): 

1723 # e.g. list, tuple 

1724 if np.ndim(data) == 0: 

1725 # i.e. generator 

1726 data = list(data) 

1727 data = np.asarray(data) 

1728 copy = False 

1729 elif isinstance(data, ABCSeries): 

1730 data = data._values 

1731 if isinstance(data, ABCPandasArray): 

1732 data = data.to_numpy() 

1733 

1734 if hasattr(data, "freq"): 

1735 # i.e. DatetimeArray/Index 

1736 inferred_freq = data.freq 

1737 

1738 # if dtype has an embedded tz, capture it 

1739 tz = validate_tz_from_dtype(dtype, tz) 

1740 

1741 if isinstance(data, ABCIndexClass): 

1742 data = data._data 

1743 

1744 # By this point we are assured to have either a numpy array or Index 

1745 data, copy = maybe_convert_dtype(data, copy) 

1746 

1747 if is_object_dtype(data) or is_string_dtype(data): 

1748 # TODO: We do not have tests specific to string-dtypes, 

1749 # also complex or categorical or other extension 

1750 copy = False 

1751 if lib.infer_dtype(data, skipna=False) == 'integer': 

1752 data = data.astype(np.int64) 

1753 else: 

1754 # data comes back here as either i8 to denote UTC timestamps 

1755 # or M8[ns] to denote wall times 

1756 data, inferred_tz = objects_to_datetime64ns( 

1757 data, dayfirst=dayfirst, yearfirst=yearfirst) 

1758 tz = maybe_infer_tz(tz, inferred_tz) 

1759 # When a sequence of timestamp objects is passed, we always 

1760 # want to treat the (now i8-valued) data as UTC timestamps, 

1761 # not wall times. 

1762 int_as_wall_time = False 

1763 

1764 # `data` may have originally been a Categorical[datetime64[ns, tz]], 

1765 # so we need to handle these types. 

1766 if is_datetime64tz_dtype(data): 

1767 # DatetimeArray -> ndarray 

1768 tz = maybe_infer_tz(tz, data.tz) 

1769 result = data._data 

1770 

1771 elif is_datetime64_dtype(data): 

1772 # tz-naive DatetimeArray or ndarray[datetime64] 

1773 data = getattr(data, "_data", data) 

1774 if data.dtype != _NS_DTYPE: 

1775 data = conversion.ensure_datetime64ns(data) 

1776 

1777 if tz is not None: 

1778 # Convert tz-naive to UTC 

1779 tz = timezones.maybe_get_tz(tz) 

1780 data = conversion.tz_localize_to_utc(data.view('i8'), tz, 

1781 ambiguous=ambiguous) 

1782 data = data.view(_NS_DTYPE) 

1783 

1784 assert data.dtype == _NS_DTYPE, data.dtype 

1785 result = data 

1786 

1787 else: 

1788 # must be integer dtype otherwise 

1789 # assume this data are epoch timestamps 

1790 if tz: 

1791 tz = timezones.maybe_get_tz(tz) 

1792 

1793 if data.dtype != _INT64_DTYPE: 

1794 data = data.astype(np.int64, copy=False) 

1795 if int_as_wall_time and tz is not None and not timezones.is_utc(tz): 

1796 warnings.warn(_i8_message, FutureWarning, stacklevel=4) 

1797 data = conversion.tz_localize_to_utc(data.view('i8'), tz, 

1798 ambiguous=ambiguous) 

1799 data = data.view(_NS_DTYPE) 

1800 result = data.view(_NS_DTYPE) 

1801 

1802 if copy: 

1803 # TODO: should this be deepcopy? 

1804 result = result.copy() 

1805 

1806 assert isinstance(result, np.ndarray), type(result) 

1807 assert result.dtype == 'M8[ns]', result.dtype 

1808 

1809 # We have to call this again after possibly inferring a tz above 

1810 validate_tz_from_dtype(dtype, tz) 

1811 

1812 return result, tz, inferred_freq

1815def objects_to_datetime64ns(data, dayfirst, yearfirst, 

1816 utc=False, errors="raise", 

1817 require_iso8601=False, allow_object=False): 

1818 """ 

1819 Convert data to array of timestamps. 

1820 

1821 Parameters 

1822 ---------- 

1823 data : np.ndarray[object] 

1824 dayfirst : bool 

1825 yearfirst : bool 

1826 utc : bool, default False 

1827 Whether to convert timezone-aware timestamps to UTC 

1828 errors : {'raise', 'ignore', 'coerce'} 

1829 allow_object : bool 

1830 Whether to return an object-dtype ndarray instead of raising if the 

1831 data contains more than one timezone. 

1832 

1833 Returns 

1834 ------- 

1835 result : ndarray 

1836 np.int64 dtype if returned values represent UTC timestamps 

1837 np.datetime64[ns] if returned values represent wall times 

1838 object if mixed timezones 

1839 inferred_tz : tzinfo or None 

1840 

1841 Raises 

1842 ------ 

1843 ValueError : if data cannot be converted to datetimes 

1844 """ 

1845 assert errors in ["raise", "ignore", "coerce"] 

1846 

1847 # if str-dtype, convert 

1848 data = np.array(data, copy=False, dtype=np.object_) 

1849 

1850 try: 

1851 result, tz_parsed = tslib.array_to_datetime( 

1852 data, 

1853 errors=errors, 

1854 utc=utc, 

1855 dayfirst=dayfirst, 

1856 yearfirst=yearfirst, 

1857 require_iso8601=require_iso8601 

1858 ) 

1859 except ValueError as e: 

1860 try: 

1861 values, tz_parsed = conversion.datetime_to_datetime64(data) 

1862 # If tzaware, these values represent unix timestamps, so we 

1863 # return them as i8 to distinguish from wall times 

1864 return values.view('i8'), tz_parsed 

1865 except (ValueError, TypeError): 

1866 raise e 

1867 

1868 if tz_parsed is not None: 

1869 # We can take a shortcut since the datetime64 numpy array 

1870 # is in UTC 

1871 # Return i8 values to denote unix timestamps 

1872 return result.view('i8'), tz_parsed 

1873 elif is_datetime64_dtype(result): 

1874 # returning M8[ns] denotes wall-times; since tz is None 

1875 # the distinction is a thin one 

1876 return result, tz_parsed 

1877 elif is_object_dtype(result): 

1878 # GH#23675 when called via `pd.to_datetime`, returning an object-dtype 

1879 # array is allowed. When called via `pd.DatetimeIndex`, we can 

1880 # only accept datetime64 dtype, so raise TypeError if object-dtype 

1881 # is returned, as that indicates the values can be recognized as 

1882 # datetimes but they have conflicting timezones/awareness 

1883 if allow_object: 

1884 return result, tz_parsed 

1885 raise TypeError(result) 

1886 else: # pragma: no cover 

1887 # GH#23675 this TypeError should never be hit, whereas the TypeError 

1888 # in the object-dtype branch above is reachable. 

1889 raise TypeError(result)

1901 

1902 Returns 

1903 -------

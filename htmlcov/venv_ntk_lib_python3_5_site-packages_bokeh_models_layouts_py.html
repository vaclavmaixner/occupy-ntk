342@abstract 

343class Box(LayoutDOM): 

344 ''' Abstract base class for Row and Column. Do not use directly. 

345 

346 ''' 

347 

348 def __init__(self, *args, **kwargs): 

349 

350 if len(args) > 0 and "children" in kwargs: 

351 raise ValueError("'children' keyword cannot be used with positional arguments") 

352 elif len(args) > 0: 

353 kwargs["children"] = list(args) 

354 

355 super(Box, self).__init__(**kwargs) 

356 

357 @warning(EMPTY_LAYOUT) 

358 def _check_empty_layout(self): 

359 from itertools import chain 

360 if not list(chain(self.children)): 

361 return str(self) 

362 

363 @warning(BOTH_CHILD_AND_ROOT) 

364 def _check_child_is_also_root(self): 

365 problems = [] 

366 for c in self.children: 

367 if c.document is not None and c in c.document.roots: 

368 problems.append(str(c)) 

369 if problems: 

370 return ", ".join(problems) 

371 else: 

372 return None

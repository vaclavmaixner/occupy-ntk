59def export_png(obj, filename=None, height=None, width=None, webdriver=None, timeout=5): 

60 ''' Export the ``LayoutDOM`` object or document as a PNG. 

61 

62 If the filename is not given, it is derived from the script name (e.g. 

63 ``/foo/myplot.py`` will create ``/foo/myplot.png``) 

64 

65 Args: 

66 obj (LayoutDOM or Document) : a Layout (Row/Column), Plot or Widget 

67 object or Document to export. 

68 

69 filename (str, optional) : filename to save document under (default: None) 

70 If None, infer from the filename. 

71 

72 height (int) : the desired height of the exported layout obj only if 

73 it's a Plot instance. Otherwise the height kwarg is ignored. 

74 

75 width (int) : the desired width of the exported layout obj only if 

76 it's a Plot instance. Otherwise the width kwarg is ignored. 

77 

78 webdriver (selenium.webdriver) : a selenium webdriver instance to use 

79 to export the image. 

80 

81 timeout (int) : the maximum amount of time (in seconds) to wait for 

82 Bokeh to initialize (default: 5) (Added in 1.1.1). 

83 

84 Returns: 

85 filename (str) : the filename where the static file is saved. 

86 

87 If you would like to access an Image object directly, rather than save a 

88 file to disk, use the lower-level :func:`~bokeh.io.export.get_screenshot_as_png` 

89 function. 

90 

91 .. warning:: 

92 Responsive sizing_modes may generate layouts with unexpected size and 

93 aspect ratios. It is recommended to use the default ``fixed`` sizing mode. 

94 

95 ''' 

96 

97 image = get_screenshot_as_png(obj, height=height, width=width, driver=webdriver, timeout=timeout) 

98 

99 if filename is None: 

100 filename = default_filename("png") 

101 

102 if image.width == 0 or image.height == 0: 

103 raise ValueError("unable to save an empty image") 

104 

105 image.save(filename) 

106 

107 return abspath(filename)

109def export_svgs(obj, filename=None, height=None, width=None, webdriver=None, timeout=5): 

110 ''' Export the SVG-enabled plots within a layout. Each plot will result 

111 in a distinct SVG file. 

112 

113 If the filename is not given, it is derived from the script name 

114 (e.g. ``/foo/myplot.py`` will create ``/foo/myplot.svg``) 

115 

116 Args: 

117 obj (LayoutDOM object) : a Layout (Row/Column), Plot or Widget object to display 

118 

119 filename (str, optional) : filename to save document under (default: None) 

120 If None, infer from the filename. 

121 

122 height (int) : the desired height of the exported layout obj only if 

123 it's a Plot instance. Otherwise the height kwarg is ignored. 

124 

125 width (int) : the desired width of the exported layout obj only if 

126 it's a Plot instance. Otherwise the width kwarg is ignored. 

127 

128 webdriver (selenium.webdriver) : a selenium webdriver instance to use 

129 to export the image. 

130 

131 timeout (int) : the maximum amount of time (in seconds) to wait for 

132 Bokeh to initialize (default: 5) (Added in 1.1.1). 

133 

134 Returns: 

135 filenames (list(str)) : the list of filenames where the SVGs files are 

136 saved. 

137 

138 .. warning:: 

139 Responsive sizing_modes may generate layouts with unexpected size and 

140 aspect ratios. It is recommended to use the default ``fixed`` sizing mode. 

141 

142 ''' 

143 svgs = get_svgs(obj, height=height, width=width, driver=webdriver, timeout=timeout) 

144 

145 if len(svgs) == 0: 

146 log.warning("No SVG Plots were found.") 

147 return 

148 

149 if filename is None: 

150 filename = default_filename("svg") 

151 

152 filenames = [] 

153 

154 for i, svg in enumerate(svgs): 

155 if i == 0: 

156 filename = filename 

157 else: 

158 idx = filename.find(".svg") 

159 filename = filename[:idx] + "_{}".format(i) + filename[idx:] 

160 

161 with io.open(filename, mode="w", encoding="utf-8") as f: 

162 f.write(svg) 

163 

164 filenames.append(filename) 

165 

166 return filenames

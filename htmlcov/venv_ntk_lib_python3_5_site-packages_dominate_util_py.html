55def escape(data, quote=True): # stoled from std lib cgi 

56 ''' 

57 Escapes special characters into their html entities 

58 Replace special characters "&", "<" and ">" to HTML-safe sequences. 

59 If the optional flag quote is true, the quotation mark character (") 

60 is also translated. 

61 

62 This is used to escape content that appears in the body of an HTML cocument 

63 ''' 

64 data = data.replace("&", "&amp;") # Must be done first! 

65 data = data.replace("<", "&lt;") 

66 data = data.replace(">", "&gt;") 

67 if quote: 

68 data = data.replace('"', "&quot;") 

69 return data

85def unescape(data): 

86 ''' 

87 unescapes html entities. the opposite of escape. 

88 ''' 

89 cc = re.compile(r'&(?:(?:#(\d+))|([^;]+));') 

90 

91 result = [] 

92 m = cc.search(data) 

93 while m: 

94 result.append(data[0:m.start()]) 

95 d = m.group(1) 

96 if d: 

97 d = int(d) 

98 result.append(unichr(d)) 

99 else: 

100 d = _unescape.get(m.group(2), ord('?')) 

101 result.append(unichr(d)) 

102 

103 data = data[m.end():] 

104 m = cc.search(data) 

105 

106 result.append(data) 

107 return ''.join(result)

123class lazy(dom_tag): 

124 ''' 

125 delays function execution until rendered 

126 ''' 

127 def __new__(_cls, *args, **kwargs): 

128 ''' 

129 Need to reset this special method or else 

130 dom_tag will think it's being used as a dectorator. 

131 

132 This means lazy() can't be used as a dectorator, but 

133 thinking about when you might want that just confuses me. 

134 ''' 

135 return object.__new__(_cls) 

136 

137 def __init__(self, func, *args, **kwargs): 

138 super(lazy, self).__init__() 

139 self.func = func 

140 self.args = args 

141 self.kwargs = kwargs

144 def _render(self, sb, *a, **kw): 

145 r = self.func(*self.args, **self.kwargs)

55class Signer(object): 

56 """This class can sign and unsign bytes, validating the signature 

57 provided. 

58 

59 Salt can be used to namespace the hash, so that a signed string is 

60 only valid for a given namespace. Leaving this at the default value 

61 or re-using a salt value across different parts of your application 

62 where the same signed value in one part can mean something different 

63 in another part is a security risk. 

64 

65 See :ref:`the-salt` for an example of what the salt is doing and how 

66 you can utilize it. 

67 

68 .. versionadded:: 0.14 

69 ``key_derivation`` and ``digest_method`` were added as arguments 

70 to the class constructor. 

71 

72 .. versionadded:: 0.18 

73 ``algorithm`` was added as an argument to the class constructor. 

74 """

119 def derive_key(self): 

120 """This method is called to derive the key. The default key 

121 derivation choices can be overridden here. Key derivation is not 

122 intended to be used as a security method to make a complex key 

123 out of a short password. Instead you should use large random 

124 secret keys. 

125 """ 

126 salt = want_bytes(self.salt) 

127 if self.key_derivation == "concat": 

128 return self.digest_method(salt + self.secret_key).digest() 

129 elif self.key_derivation == "django-concat": 

130 return self.digest_method(salt + b"signer" + self.secret_key).digest() 

131 elif self.key_derivation == "hmac": 

132 mac = hmac.new(self.secret_key, digestmod=self.digest_method) 

133 mac.update(salt) 

134 return mac.digest() 

135 elif self.key_derivation == "none": 

136 return self.secret_key 

137 else: 

138 raise TypeError("Unknown key derivation method")

160 def unsign(self, signed_value): 

161 """Unsigns the given string.""" 

162 signed_value = want_bytes(signed_value) 

163 sep = want_bytes(self.sep) 

164 if sep not in signed_value: 

165 raise BadSignature("No %r found in value" % self.sep) 

166 value, sig = signed_value.rsplit(sep, 1) 

167 if self.verify_signature(value, sig): 

168 return value 

169 raise BadSignature("Signature %r does not match" % sig, payload=value)

72def segment_hits(cx, cy, x, y, radius): 

73 """ 

74 Determine if any line segments are within radius of a 

75 point. Returns the list of line segments that are within that 

76 radius. 

77 """ 

78 # Process single points specially 

79 if len(x) < 2: 

80 res, = np.nonzero((cx - x) ** 2 + (cy - y) ** 2 <= radius ** 2) 

81 return res 

82 

83 # We need to lop the last element off a lot. 

84 xr, yr = x[:-1], y[:-1] 

85 

86 # Only look at line segments whose nearest point to C on the line 

87 # lies within the segment. 

88 dx, dy = x[1:] - xr, y[1:] - yr 

89 Lnorm_sq = dx ** 2 + dy ** 2 # Possibly want to eliminate Lnorm==0 

90 u = ((cx - xr) * dx + (cy - yr) * dy) / Lnorm_sq 

91 candidates = (u >= 0) & (u <= 1) 

92 

93 # Note that there is a little area near one side of each point 

94 # which will be near neither segment, and another which will 

95 # be near both, depending on the angle of the lines. The 

96 # following radius test eliminates these ambiguities. 

97 point_hits = (cx - x) ** 2 + (cy - y) ** 2 <= radius ** 2 

98 candidates = candidates & ~(point_hits[:-1] | point_hits[1:]) 

99 

100 # For those candidates which remain, determine how far they lie away 

101 # from the line. 

102 px, py = xr + u * dx, yr + u * dy 

103 line_hits = (cx - px) ** 2 + (cy - py) ** 2 <= radius ** 2 

104 line_hits = line_hits & candidates 

105 points, = point_hits.ravel().nonzero() 

106 lines, = line_hits.ravel().nonzero() 

107 return np.concatenate((points, lines))

110def _mark_every_path(markevery, tpath, affine, ax_transform): 

111 """ 

112 Helper function that sorts out how to deal the input 

113 `markevery` and returns the points where markers should be drawn. 

114 

115 Takes in the `markevery` value and the line path and returns the 

116 sub-sampled path. 

117 """ 

118 # pull out the two bits of data we want from the path 

119 codes, verts = tpath.codes, tpath.vertices 

120 

121 def _slice_or_none(in_v, slc): 

122 ''' 

123 Helper function to cope with `codes` being an 

124 ndarray or `None` 

125 ''' 

126 if in_v is None: 

127 return None 

128 return in_v[slc] 

129 

130 # if just an int, assume starting at 0 and make a tuple 

131 if isinstance(markevery, Integral): 

132 markevery = (0, markevery) 

133 # if just a float, assume starting at 0.0 and make a tuple 

134 elif isinstance(markevery, Real): 

135 markevery = (0.0, markevery) 

136 

137 if isinstance(markevery, tuple): 

138 if len(markevery) != 2: 

139 raise ValueError('`markevery` is a tuple but its len is not 2; ' 

140 'markevery={}'.format(markevery)) 

141 start, step = markevery 

142 # if step is an int, old behavior 

143 if isinstance(step, Integral): 

144 # tuple of 2 int is for backwards compatibility, 

145 if not isinstance(start, Integral): 

146 raise ValueError( 

147 '`markevery` is a tuple with len 2 and second element is ' 

148 'an int, but the first element is not an int; markevery={}' 

149 .format(markevery)) 

150 # just return, we are done here 

151 

152 return Path(verts[slice(start, None, step)], 

153 _slice_or_none(codes, slice(start, None, step))) 

154 

155 elif isinstance(step, Real): 

156 if not isinstance(start, Real): 

157 raise ValueError( 

158 '`markevery` is a tuple with len 2 and second element is ' 

159 'a float, but the first element is not a float or an int; ' 

160 'markevery={}'.format(markevery)) 

161 # calc cumulative distance along path (in display coords): 

162 disp_coords = affine.transform(tpath.vertices) 

163 delta = np.empty((len(disp_coords), 2)) 

164 delta[0, :] = 0 

165 delta[1:, :] = disp_coords[1:, :] - disp_coords[:-1, :] 

166 delta = np.sum(delta**2, axis=1) 

167 delta = np.sqrt(delta) 

168 delta = np.cumsum(delta) 

169 # calc distance between markers along path based on the axes 

170 # bounding box diagonal being a distance of unity: 

171 scale = ax_transform.transform(np.array([[0, 0], [1, 1]])) 

172 scale = np.diff(scale, axis=0) 

173 scale = np.sum(scale**2) 

174 scale = np.sqrt(scale) 

175 marker_delta = np.arange(start * scale, delta[-1], step * scale) 

176 # find closest actual data point that is closest to 

177 # the theoretical distance along the path: 

178 inds = np.abs(delta[np.newaxis, :] - marker_delta[:, np.newaxis]) 

179 inds = inds.argmin(axis=1) 

180 inds = np.unique(inds) 

181 # return, we are done here 

182 return Path(verts[inds], 

183 _slice_or_none(codes, inds)) 

184 else: 

185 raise ValueError( 

186 '`markevery` is a tuple with len 2, but its second element is ' 

187 'not an int or a float; markevery=%s' % (markevery,)) 

188 

189 elif isinstance(markevery, slice): 

190 # mazol tov, it's already a slice, just return 

191 return Path(verts[markevery], _slice_or_none(codes, markevery)) 

192 

193 elif iterable(markevery): 

194 #fancy indexing 

195 try: 

196 return Path(verts[markevery], _slice_or_none(codes, markevery)) 

197 

198 except (ValueError, IndexError): 

199 raise ValueError('`markevery` is iterable but ' 

200 'not a valid form of numpy fancy indexing; ' 

201 'markevery=%s' % (markevery,)) 

202 else: 

203 raise ValueError('Value of `markevery` is not ' 

204 'recognized; ' 

205 'markevery=%s' % (markevery,))

430 def contains(self, mouseevent): 

431 """ 

432 Test whether the mouse event occurred on the line. The pick 

433 radius determines the precision of the location test (usually 

434 within five points of the value). Use 

435 :meth:`~matplotlib.lines.Line2D.get_pickradius` or 

436 :meth:`~matplotlib.lines.Line2D.set_pickradius` to view or 

437 modify it. 

438 

439 Returns *True* if any values are within the radius along with 

440 ``{'ind': pointlist}``, where *pointlist* is the set of points 

441 within the radius. 

442 

443 TODO: sort returned indices by distance 

444 """ 

445 if callable(self._contains): 

446 return self._contains(self, mouseevent) 

447 

448 if not isinstance(self.pickradius, Number): 

449 raise ValueError("pick radius should be a distance") 

450 

451 # Make sure we have data to plot 

452 if self._invalidy or self._invalidx: 

453 self.recache() 

454 if len(self._xy) == 0: 

455 return False, {} 

456 

457 # Convert points to pixels 

458 transformed_path = self._get_transformed_path() 

459 path, affine = transformed_path.get_transformed_path_and_affine() 

460 path = affine.transform_path(path) 

461 xy = path.vertices 

462 xt = xy[:, 0] 

463 yt = xy[:, 1] 

464 

465 # Convert pick radius from points to pixels 

466 if self.figure is None: 

467 warnings.warn('no figure set when check if mouse is on line') 

468 pixels = self.pickradius 

469 else: 

470 pixels = self.figure.dpi / 72. * self.pickradius 

471 

472 # the math involved in checking for containment (here and inside of 

473 # segment_hits) assumes that it is OK to overflow. In case the 

474 # application has set the error flags such that an exception is raised 

475 # on overflow, we temporarily set the appropriate error flags here and 

476 # set them back when we are finished. 

477 with np.errstate(all='ignore'): 

478 # Check for collision 

479 if self._linestyle in ['None', None]: 

480 # If no line, return the nearby point(s) 

481 d = (xt - mouseevent.x) ** 2 + (yt - mouseevent.y) ** 2 

482 ind, = np.nonzero(np.less_equal(d, pixels ** 2)) 

483 else: 

484 # If line, return the nearby segment(s) 

485 ind = segment_hits(mouseevent.x, mouseevent.y, xt, yt, pixels) 

486 if self._drawstyle.startswith("steps"): 

487 ind //= 2 

488 

489 ind += self.ind_offset 

490 

491 # Return the point(s) within radius 

492 return len(ind) > 0, dict(ind=ind)

527 """Set the markevery property to subsample the plot when using markers. 

528 

529 e.g., if `every=5`, every 5-th marker will be plotted. 

530 

531 Parameters 

532 ---------- 

533 every: None or int or (int, int) or slice or List[int] or float or \ 

534(float, float) 

535 Which markers to plot. 

536 

537 - every=None, every point will be plotted. 

538 - every=N, every N-th marker will be plotted starting with 

539 marker 0. 

540 - every=(start, N), every N-th marker, starting at point 

541 start, will be plotted. 

542 - every=slice(start, end, N), every N-th marker, starting at 

543 point start, up to but not including point end, will be plotted. 

544 - every=[i, j, m, n], only markers at points i, j, m, and n

547 approximately equal distances along the line; the distance 

548 along the line between markers is determined by multiplying the

550 by the value of every. 

551 - every=(0.5, 0.1) (i.e. a length-2 tuple of float), the

554 display-cordinate-diagonal-distance along the line. 

555

557 ----- 

558 Setting the markevery property will only show markers at actual data

561 spaced because the actual data points do not coincide with the 

562 theoretical spacing between markers. 

563 

564 When using a start offset to specify the first marker, the offset will 

565 be from the first data point which may be different from the first 

566 the visible data point if the plot is zoomed in. 

567 

568 If zooming in on a plot when using float arguments then the actual 

569 data points that have markers will change because the distance between 

570 markers is always determined from the display-coordinates 

571 axes-bounding-box-diagonal regardless of the actual axes data limits. 

572 

573 """ 

574 if self._markevery != every:

655 self.axes.get_xscale() == 'linear' and 

656 self._markevery is None and 

657 self.get_clip_on() is True):

660 if nanmask.any(): 

661 self._x_filled = self._x.copy() 

662 indices = np.arange(len(x)) 

663 self._x_filled[nanmask] = np.interp(indices[nanmask], 

664 indices[~nanmask], self._x[~nanmask]) 

665 else: 

666 self._x_filled = self._x 

667 

668 if self._path is not None: 

669 interpolation_steps = self._path._interpolation_steps 

670 else: 

671 interpolation_steps = 1 

672 xy = STEP_LOOKUP_MAP[self._drawstyle](*self._xy.T) 

673 self._path = Path(np.asarray(xy).T, 

674 _interpolation_steps=interpolation_steps) 

675 self._transformed_path = None 

676 self._invalidx = False

689 _interpolation_steps=self._path._interpolation_steps) 

690 else: 

691 _path = self._path

721 @allow_rasterization 

722 def draw(self, renderer): 

723 """draw the Line with `renderer` unless visibility is False""" 

724 if not self.get_visible(): 

725 return 

726 

727 if self._invalidy or self._invalidx: 

728 self.recache() 

729 self.ind_offset = 0 # Needed for contains() method. 

730 if self._subslice and self.axes: 

731 x0, x1 = self.axes.get_xbound() 

732 i0, = self._x_filled.searchsorted([x0], 'left') 

733 i1, = self._x_filled.searchsorted([x1], 'right') 

734 subslice = slice(max(i0 - 1, 0), i1 + 1) 

735 self.ind_offset = subslice.start 

736 self._transform_path(subslice) 

737 else: 

738 subslice = None 

739 

740 if self.get_path_effects(): 

741 from matplotlib.patheffects import PathEffectRenderer 

742 renderer = PathEffectRenderer(self.get_path_effects(), renderer) 

743 

744 renderer.open_group('line2d', self.get_gid()) 

745 if self._lineStyles[self._linestyle] != '_draw_nothing': 

746 tpath, affine = (self._get_transformed_path() 

747 .get_transformed_path_and_affine()) 

748 if len(tpath.vertices): 

749 gc = renderer.new_gc() 

750 self._set_gc_clip(gc) 

751 

752 lc_rgba = mcolors.to_rgba(self._color, self._alpha) 

753 gc.set_foreground(lc_rgba, isRGBA=True) 

754 

755 gc.set_antialiased(self._antialiased) 

756 gc.set_linewidth(self._linewidth) 

757 

758 if self.is_dashed(): 

759 cap = self._dashcapstyle 

760 join = self._dashjoinstyle 

761 else: 

762 cap = self._solidcapstyle 

763 join = self._solidjoinstyle 

764 gc.set_joinstyle(join) 

765 gc.set_capstyle(cap) 

766 gc.set_snap(self.get_snap()) 

767 if self.get_sketch_params() is not None: 

768 gc.set_sketch_params(*self.get_sketch_params()) 

769 

770 gc.set_dashes(self._dashOffset, self._dashSeq) 

771 renderer.draw_path(gc, tpath, affine.frozen()) 

772 gc.restore() 

773 

774 if self._marker and self._markersize > 0: 

775 gc = renderer.new_gc() 

776 self._set_gc_clip(gc) 

777 gc.set_linewidth(self._markeredgewidth) 

778 gc.set_antialiased(self._antialiased) 

779 

780 ec_rgba = mcolors.to_rgba( 

781 self.get_markeredgecolor(), self._alpha) 

782 fc_rgba = mcolors.to_rgba( 

783 self._get_markerfacecolor(), self._alpha) 

784 fcalt_rgba = mcolors.to_rgba( 

785 self._get_markerfacecolor(alt=True), self._alpha) 

786 # If the edgecolor is "auto", it is set according to the *line* 

787 # color but inherits the alpha value of the *face* color, if any. 

788 if (cbook._str_equal(self._markeredgecolor, "auto") 

789 and not cbook._str_lower_equal( 

790 self.get_markerfacecolor(), "none")): 

791 ec_rgba = ec_rgba[:3] + (fc_rgba[3],) 

792 gc.set_foreground(ec_rgba, isRGBA=True) 

793 if self.get_sketch_params() is not None: 

794 scale, length, randomness = self.get_sketch_params() 

795 gc.set_sketch_params(scale/2, length/2, 2*randomness) 

796 

797 marker = self._marker 

798 

799 # Markers *must* be drawn ignoring the drawstyle (but don't pay the 

800 # recaching if drawstyle is already "default"). 

801 if self.get_drawstyle() != "default": 

802 with cbook._setattr_cm( 

803 self, _drawstyle="default", _transformed_path=None): 

804 self.recache() 

805 self._transform_path(subslice) 

806 tpath, affine = (self._get_transformed_path() 

807 .get_transformed_path_and_affine()) 

808 else: 

809 tpath, affine = (self._get_transformed_path() 

810 .get_transformed_path_and_affine()) 

811 

812 if len(tpath.vertices): 

813 # subsample the markers if markevery is not None 

814 markevery = self.get_markevery() 

815 if markevery is not None: 

816 subsampled = _mark_every_path(markevery, tpath, 

817 affine, self.axes.transAxes) 

818 else: 

819 subsampled = tpath 

820 

821 snap = marker.get_snap_threshold() 

822 if isinstance(snap, Real): 

823 snap = renderer.points_to_pixels(self._markersize) >= snap 

824 gc.set_snap(snap) 

825 gc.set_joinstyle(marker.get_joinstyle()) 

826 gc.set_capstyle(marker.get_capstyle()) 

827 marker_path = marker.get_path() 

828 marker_trans = marker.get_transform() 

829 w = renderer.points_to_pixels(self._markersize) 

830 

831 if cbook._str_equal(marker.get_marker(), ","): 

832 gc.set_linewidth(0) 

833 else: 

834 # Don't scale for pixels, and don't stroke them 

835 marker_trans = marker_trans.scale(w) 

836 renderer.draw_markers(gc, marker_path, marker_trans, 

837 subsampled, affine.frozen(), 

838 fc_rgba) 

839 

840 alt_marker_path = marker.get_alt_path() 

841 if alt_marker_path: 

842 alt_marker_trans = marker.get_alt_transform() 

843 alt_marker_trans = alt_marker_trans.scale(w) 

844 renderer.draw_markers( 

845 gc, alt_marker_path, alt_marker_trans, subsampled, 

846 affine.frozen(), fcalt_rgba) 

847 

848 gc.restore() 

849 

850 renderer.close_group('line2d') 

851 self.stale = False

1020 '''Split drawstyle from linestyle string

1044 def set_linestyle(self, ls): 

1045 """ 

1046 Set the linestyle of the line (also accepts drawstyles, 

1047 e.g., ``'steps--'``) 

1048 

1049 

1050 =========================== ================= 

1051 linestyle description 

1052 =========================== ================= 

1053 ``'-'`` or ``'solid'`` solid line 

1054 ``'--'`` or ``'dashed'`` dashed line 

1055 ``'-.'`` or ``'dashdot'`` dash-dotted line 

1056 ``':'`` or ``'dotted'`` dotted line 

1057 ``'None'`` draw nothing 

1058 ``' '`` draw nothing 

1059 ``''`` draw nothing 

1060 =========================== ================= 

1061 

1062 'steps' is equivalent to 'steps-pre' and is maintained for 

1063 backward-compatibility. 

1064 

1065 Alternatively a dash tuple of the following form can be provided:: 

1066 

1067 (offset, onoffseq), 

1068 

1069 where ``onoffseq`` is an even length tuple of on and off ink in points. 

1070 

1071 .. seealso:: 

1072 

1073 :meth:`set_drawstyle` 

1074 To set the drawing style (stepping) of the plot. 

1075 

1076 Parameters 

1077 ---------- 

1078 ls : {'-', '--', '-.', ':', '', (offset, on-off-seq), ...} 

1079 The line style. 

1080 """ 

1081 if isinstance(ls, str): 

1082 ds, ls = self._split_drawstyle_linestyle(ls) 

1083 if ds is not None: 

1084 self.set_drawstyle(ds) 

1085 

1086 if ls in [' ', '', 'none']: 

1087 ls = 'None' 

1088 

1089 if ls not in self._lineStyles: 

1090 try: 

1091 ls = ls_mapper_r[ls] 

1092 except KeyError: 

1093 raise ValueError("Invalid linestyle {!r}; see docs of " 

1094 "Line2D.set_linestyle for valid values" 

1095 .format(ls)) 

1096 self._linestyle = ls 

1097 else: 

1098 self._linestyle = '--' 

1099 

1100 # get the unscaled dashes 

1101 self._us_dashOffset, self._us_dashSeq = _get_dash_pattern(ls) 

1102 # compute the linewidth scaled dashes 

1103 self._dashOffset, self._dashSeq = _scale_dashes( 

1104 self._us_dashOffset, self._us_dashSeq, self._linewidth)

1411 """When the line is picked, update the set of selected indices."""

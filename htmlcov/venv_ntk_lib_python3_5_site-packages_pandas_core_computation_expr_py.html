56def _replace_booleans(tok): 

57 """Replace ``&`` with ``and`` and ``|`` with ``or`` so that bitwise

65 Returns 

66 ------- 

67 t : tuple of int, str 

68 Either the input or token or the replacement values

70 toknum, tokval = tok

72 if tokval == '&':

74 elif tokval == '|':

80def _replace_locals(tok): 

81 """Replace local variables with a syntactically valid name.

116def _preparse(source, f=_compose(_replace_locals, _replace_booleans, 

117 _rewrite_assign)): 

118 """Compose a collection of tokenization functions 

119 

120 Parameters 

121 ---------- 

122 source : str 

123 A Python source code string 

124 f : callable 

125 This takes a tuple of (toknum, tokval) as its argument and returns a 

126 tuple with the same structure but possibly different elements. Defaults 

127 to the composition of ``_rewrite_assign``, ``_replace_booleans``, and 

128 ``_replace_locals``. 

129 

130 Returns 

131 ------- 

132 s : str 

133 Valid Python source code 

134 

135 Notes 

136 ----- 

137 The `f` parameter can be any callable that takes *and* returns input of the 

138 form ``(toknum, tokval)``, where ``toknum`` is one of the constants from 

139 the ``tokenize`` module and ``tokval`` is a string. 

140 """ 

141 assert callable(f), 'f must be callable' 

142 return tokenize.untokenize(lmap(f, tokenize_string(source)))

160def _filter_nodes(superclass, all_nodes=_all_nodes): 

161 """Filter out AST nodes that are subclasses of ``superclass``.""" 

162 node_names = (node.__name__ for node in all_nodes 

163 if issubclass(node, superclass)) 

164 return frozenset(node_names)

218def disallow(nodes): 

219 """Decorator to disallow certain nodes from parsing. Raises a 

220 NotImplementedError instead. 

221 

222 Returns 

223 ------- 

224 disallowed : callable 

225 """ 

226 def disallowed(cls): 

227 cls.unsupported_nodes = () 

228 for node in nodes: 

229 new_method = _node_not_implemented(node, cls) 

230 name = 'visit_{node}'.format(node=node) 

231 cls.unsupported_nodes += (name,)

233 return cls 

234 return disallowed

260def add_ops(op_classes): 

261 """Decorator to add default implementation of ops.""" 

262 def f(cls): 

263 for op_attr_name, op_class in compat.iteritems(op_classes): 

264 ops = getattr(cls, '{name}_ops'.format(name=op_attr_name)) 

265 ops_map = getattr(cls, '{name}_op_nodes_map'.format( 

266 name=op_attr_name)) 

267 for op in ops: 

268 op_node = ops_map[op] 

269 if op_node is not None: 

270 made_op = _op_maker(op_class, op) 

271 setattr(cls, 'visit_{node}'.format(node=op_node), made_op) 

272 return cls 

273 return f

342 def _rewrite_membership_op(self, node, left, right): 

343 # the kind of the operator (is actually an instance) 

344 op_instance = node.op 

345 op_type = type(op_instance) 

346 

347 # must be two terms and the comparison operator must be ==/!=/in/not in 

348 if is_term(left) and is_term(right) and op_type in self.rewrite_map: 

349 

350 left_list, right_list = map(_is_list, (left, right)) 

351 left_str, right_str = map(_is_str, (left, right)) 

352 

353 # if there are any strings or lists in the expression 

354 if left_list or right_list or left_str or right_str: 

355 op_instance = self.rewrite_map[op_type]() 

356 

357 # pop the string variable out of locals and replace it with a list 

358 # of one string, kind of a hack 

359 if right_str: 

360 name = self.env.add_tmp([right.value]) 

361 right = self.term_type(name, self.env) 

362 

363 if left_str: 

364 name = self.env.add_tmp([left.value]) 

365 left = self.term_type(name, self.env) 

366 

367 op = self.visit(op_instance) 

368 return op, op_instance, left, right

370 def _maybe_transform_eq_ne(self, node, left=None, right=None): 

371 if left is None: 

372 left = self.visit(node.left, side='left') 

373 if right is None: 

374 right = self.visit(node.right, side='right') 

375 op, op_class, left, right = self._rewrite_membership_op(node, left, 

376 right) 

377 return op, op_class, left, right

379 def _maybe_downcast_constants(self, left, right): 

380 f32 = np.dtype(np.float32) 

381 if left.is_scalar and not right.is_scalar and right.return_type == f32: 

382 # right is a float32 array, left is a scalar 

383 name = self.env.add_tmp(np.float32(left.value)) 

384 left = self.term_type(name, self.env) 

385 if right.is_scalar and not left.is_scalar and left.return_type == f32: 

386 # left is a float32 array, right is a scalar 

387 name = self.env.add_tmp(np.float32(right.value)) 

388 right = self.term_type(name, self.env) 

389 

390 return left, right

392 def _maybe_eval(self, binop, eval_in_python): 

393 # eval `in` and `not in` (for now) in "partial" python space 

394 # things that can be evaluated in "eval" space will be turned into 

395 # temporary variables. for example, 

396 # [1,2] in a + 2 * b 

397 # in that case a + 2 * b will be evaluated using numexpr, and the "in" 

398 # call will be evaluated using isin (in python space) 

399 return binop.evaluate(self.env, self.engine, self.parser, 

400 self.term_type, eval_in_python)

402 def _maybe_evaluate_binop(self, op, op_class, lhs, rhs, 

403 eval_in_python=('in', 'not in'), 

404 maybe_eval_in_python=('==', '!=', '<', '>', 

405 '<=', '>=')): 

406 res = op(lhs, rhs) 

407 

408 if res.has_invalid_return_type: 

409 raise TypeError("unsupported operand type(s) for {op}:" 

410 " '{lhs}' and '{rhs}'".format(op=res.op, 

411 lhs=lhs.type, 

412 rhs=rhs.type)) 

413 

414 if self.engine != 'pytables': 

415 if (res.op in _cmp_ops_syms and 

416 getattr(lhs, 'is_datetime', False) or 

417 getattr(rhs, 'is_datetime', False)): 

418 # all date ops must be done in python bc numexpr doesn't work 

419 # well with NaT 

420 return self._maybe_eval(res, self.binary_ops) 

421 

422 if res.op in eval_in_python: 

423 # "in"/"not in" ops are always evaluated in python 

424 return self._maybe_eval(res, eval_in_python) 

425 elif self.engine != 'pytables': 

426 if (getattr(lhs, 'return_type', None) == object or 

427 getattr(rhs, 'return_type', None) == object): 

428 # evaluate "==" and "!=" in python if either of our operands 

429 # has an object return type 

430 return self._maybe_eval(res, eval_in_python + 

431 maybe_eval_in_python) 

432 return res

501 def visit_Assign(self, node, **kwargs): 

502 """ 

503 support a single assignment node, like 

504 

505 c = a + b 

506 

507 set the assigner at the top level, must be a Name node which 

508 might or might not exist in the resolvers 

509 

510 """ 

511 

512 if len(node.targets) != 1: 

513 raise SyntaxError('can only assign a single expression') 

514 if not isinstance(node.targets[0], ast.Name): 

515 raise SyntaxError('left hand side of an assignment must be a ' 

516 'single name') 

517 if self.env.target is None: 

518 raise ValueError('cannot assign without a target object')

521 assigner = self.visit(node.targets[0], **kwargs) 

522 except UndefinedVariableError: 

523 assigner = node.targets[0].id

526 if self.assigner is None: 

527 raise SyntaxError('left hand side of an assignment must be a ' 

528 'single resolvable name')

658 def visit_Compare(self, node, **kwargs): 

659 ops = node.ops 

660 comps = node.comparators 

661 

662 # base case: we have something like a CMP b 

663 if len(comps) == 1: 

664 op = self.translate_In(ops[0]) 

665 binop = ast.BinOp(op=op, left=node.left, right=comps[0]) 

666 return self.visit(binop) 

667 

668 # recursive case: we have a chained comparison, a CMP b CMP c, etc. 

669 left = node.left 

670 values = [] 

671 for op, comp in zip(ops, comps): 

672 new_node = self.visit(ast.Compare(comparators=[comp], left=left, 

673 ops=[self.translate_In(op)])) 

674 left = comp 

675 values.append(new_node) 

676 return self.visit(ast.BoolOp(op=ast.And(), values=values)) 

677 

678 def _try_visit_binop(self, bop): 

679 if isinstance(bop, (Op, Term)): 

680 return bop 

681 return self.visit(bop) 

682 

683 def visit_BoolOp(self, node, **kwargs): 

684 def visitor(x, y): 

685 lhs = self._try_visit_binop(x) 

686 rhs = self._try_visit_binop(y) 

687 

688 op, op_class, lhs, rhs = self._maybe_transform_eq_ne( 

689 node, lhs, rhs) 

690 return self._maybe_evaluate_binop(op, node.op, lhs, rhs) 

691 

692 operands = node.values 

693 return reduce(visitor, operands)

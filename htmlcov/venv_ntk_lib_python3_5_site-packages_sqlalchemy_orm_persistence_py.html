361def _organize_states_for_save(base_mapper, states, uowtransaction): 

362 """Make an initial pass across a set of states for INSERT or 

363 UPDATE. 

364 

365 This includes splitting out into distinct lists for 

366 each, calling before_insert/before_update, obtaining 

367 key information for each state including its dictionary, 

368 mapper, the connection to use for the execution per state, 

369 and the identity flag. 

370 

371 """ 

372 

373 for state, dict_, mapper, connection in _connections_for_states( 

374 base_mapper, uowtransaction, states 

375 ): 

376 

377 has_identity = bool(state.key) 

378 

379 instance_key = state.key or mapper._identity_key_from_state(state) 

380 

381 row_switch = update_version_id = None 

382 

383 # call before_XXX extensions 

384 if not has_identity: 

385 mapper.dispatch.before_insert(mapper, connection, state) 

386 else: 

387 mapper.dispatch.before_update(mapper, connection, state) 

388 

389 if mapper._validate_polymorphic_identity: 

390 mapper._validate_polymorphic_identity(mapper, state, dict_) 

391 

392 # detect if we have a "pending" instance (i.e. has 

393 # no instance_key attached to it), and another instance 

394 # with the same identity key already exists as persistent. 

395 # convert to an UPDATE if so. 

396 if ( 

397 not has_identity 

398 and instance_key in uowtransaction.session.identity_map 

399 ): 

400 instance = uowtransaction.session.identity_map[instance_key] 

401 existing = attributes.instance_state(instance) 

402 

403 if not uowtransaction.was_already_deleted(existing): 

404 if not uowtransaction.is_deleted(existing): 

405 raise orm_exc.FlushError( 

406 "New instance %s with identity key %s conflicts " 

407 "with persistent instance %s" 

408 % (state_str(state), instance_key, state_str(existing)) 

409 ) 

410 

411 base_mapper._log_debug( 

412 "detected row switch for identity %s. " 

413 "will update %s, remove %s from " 

414 "transaction", 

415 instance_key, 

416 state_str(state), 

417 state_str(existing), 

418 ) 

419 

420 # remove the "delete" flag from the existing element 

421 uowtransaction.remove_state_actions(existing) 

422 row_switch = existing 

423 

424 if (has_identity or row_switch) and mapper.version_id_col is not None: 

425 update_version_id = mapper._get_committed_state_attr_by_column( 

426 row_switch if row_switch else state, 

427 row_switch.dict if row_switch else dict_, 

428 mapper.version_id_col, 

429 ) 

430 

431 yield ( 

432 state, 

433 dict_, 

434 mapper, 

435 connection, 

436 has_identity, 

437 row_switch, 

438 update_version_id, 

439 )

478def _collect_insert_commands(

481 bulk=False, 

482 return_defaults=False, 

483 render_nulls=False, 

484): 

485 """Identify sets of values to use in INSERT statements for a

487 

488 """

491 continue

494 value_params = {} 

495 

496 propkey_to_col = mapper._propkey_to_col[table] 

497 

498 eval_none = mapper._insert_cols_evaluating_none[table] 

499 

500 for propkey in set(propkey_to_col).intersection(state_dict): 

501 value = state_dict[propkey] 

502 col = propkey_to_col[propkey] 

503 if value is None and col not in eval_none and not render_nulls: 

504 continue 

505 elif ( 

506 not bulk 

507 and hasattr(value, "__clause_element__") 

508 or isinstance(value, sql.ClauseElement) 

509 ):

511 value.__clause_element__() 

512 if hasattr(value, "__clause_element__")

516 params[col.key] = value 

517

523 # and also produces confusion, given that "missing" and None 

524 # now have distinct meanings

526 mapper._insert_cols_as_none[table] 

527 .difference(params) 

528 .difference([c.key for c in value_params]) 

529 ): 

530 params[colkey] = None 

531 

532 if not bulk or return_defaults: 

533 # params are in terms of Column key objects, so 

534 # compare to pk_keys_by_table 

535 has_all_pks = mapper._pk_keys_by_table[table].issubset(params) 

536 

537 if mapper.base_mapper.eager_defaults: 

538 has_all_defaults = mapper._server_default_cols[table].issubset( 

539 params 

540 ) 

541 else: 

542 has_all_defaults = True 

543 else: 

544 has_all_defaults = has_all_pks = True 

545 

546 if ( 

547 mapper.version_id_generator is not False 

548 and mapper.version_id_col is not None 

549 and mapper.version_id_col in mapper._cols_by_table[table] 

550 ): 

551 params[mapper.version_id_col.key] = mapper.version_id_generator( 

552 None 

553 ) 

554

556 state, 

557 state_dict, 

558 params, 

559 mapper, 

560 connection, 

561 value_params, 

562 has_all_pks, 

563 has_all_defaults,

568 uowtransaction, table, states_to_update, bulk=False 

569): 

570 """Identify sets of values to use in UPDATE statements for a 

571 list of states. 

572 

573 This function works intricately with the history system 

574 to determine exactly what values should be updated 

575 as well as how the row should be matched within an UPDATE 

576 statement. Includes some tricky scenarios where the primary 

577 key of an object might have been changed. 

578 

579 """

582 state, 

583 state_dict, 

584 mapper, 

585 connection, 

586 update_version_id,

588

590 continue

592 pks = mapper._pks_by_table[table] 

593 

594 value_params = {} 

595 

596 propkey_to_col = mapper._propkey_to_col[table] 

597 

598 if bulk: 

599 # keys here are mapped attribute keys, so 

600 # look at mapper attribute keys for pk 

601 params = dict( 

602 (propkey_to_col[propkey].key, state_dict[propkey]) 

603 for propkey in set(propkey_to_col) 

604 .intersection(state_dict) 

605 .difference(mapper._pk_attr_keys_by_table[table]) 

606 ) 

607 has_all_defaults = True 

608 else:

610 for propkey in set(propkey_to_col).intersection( 

611 state.committed_state 

612 ): 

613 value = state_dict[propkey] 

614 col = propkey_to_col[propkey] 

615 

616 if hasattr(value, "__clause_element__") or isinstance( 

617 value, sql.ClauseElement 

618 ): 

619 value_params[col] = ( 

620 value.__clause_element__() 

621 if hasattr(value, "__clause_element__") 

622 else value 

623 ) 

624 # guard against values that generate non-__nonzero__ 

625 # objects for __eq__() 

626 elif ( 

627 state.manager[propkey].impl.is_equal( 

628 value, state.committed_state[propkey] 

629 ) 

630 is not True 

631 ): 

632 params[col.key] = value 

633 

634 if mapper.base_mapper.eager_defaults: 

635 has_all_defaults = ( 

636 mapper._server_onupdate_default_cols[table] 

637 ).issubset(params) 

638 else: 

639 has_all_defaults = True

656 break

658 # no net change, break 

659 continue 

660

663 params[col._label] = update_version_id 

664 

665 if ( 

666 bulk or col.key not in params 

667 ) and mapper.version_id_generator is not False: 

668 val = mapper.version_id_generator(update_version_id) 

669 params[col.key] = val 

670 elif mapper.version_id_generator is False and no_params: 

671 # no version id generator, no values set on the table, 

672 # and version id wasn't manually incremented. 

673 # set version id to itself so we get an UPDATE 

674 # statement 

675 params[col.key] = update_version_id 

676 

677 elif not (params or value_params): 

678 continue 

679 

680 has_all_pks = True 

681 expect_pk_cascaded = False 

682 if bulk: 

683 # keys here are mapped attribute keys, so 

684 # look at mapper attribute keys for pk 

685 pk_params = dict( 

686 (propkey_to_col[propkey]._label, state_dict.get(propkey)) 

687 for propkey in set(propkey_to_col).intersection( 

688 mapper._pk_attr_keys_by_table[table]

690 ) 

691 else:

693 for col in pks:

695

697 state, state_dict, attributes.PASSIVE_OFF 

698 ) 

699

701 if ( 

702 not history.deleted 

703 or ("pk_cascaded", state, col) 

704 in uowtransaction.attributes 

705 ): 

706 expect_pk_cascaded = True 

707 pk_params[col._label] = history.added[0] 

708 params.pop(col.key, None) 

709 else: 

710 # else, use the old value to locate the row 

711 pk_params[col._label] = history.deleted[0]

713 has_all_pks = False 

714 else: 

715 pk_params[col._label] = history.unchanged[0]

726 state_dict,

48def register_function(identifier, fn, package="_default"): 

49 """Associate a callable with a particular func. name. 

50 

51 This is normally called by _GenericMeta, but is also 

52 available by itself so that a non-Function construct 

53 can be associated with the :data:`.func` accessor (i.e. 

54 CAST, EXTRACT). 

55 

56 """ 

57 reg = _registry[package] 

58 case_sensitive_reg = _case_sensitive_registry[package] 

59 raw_identifier = identifier 

60 identifier = identifier.lower() 

61 

62 # Check if a function with the same lowercase identifier is registered. 

63 if identifier in reg and reg[identifier] is not _CASE_SENSITIVE: 

64 if raw_identifier in case_sensitive_reg[identifier]: 

65 util.warn( 

66 "The GenericFunction '{}' is already registered and " 

67 "is going to be overriden.".format(identifier) 

68 ) 

69 reg[identifier] = fn 

70 else: 

71 # If a function with the same lowercase identifier is registered, 

72 # then these 2 functions are considered as case-sensitive. 

73 # Note: This case should raise an error in a later release. 

74 util.warn_deprecated( 

75 "GenericFunction '{}' is already registered with " 

76 "different letter case, so the previously registered function " 

77 "'{}' is switched into case-sensitive mode. " 

78 "GenericFunction objects will be fully case-insensitive in a " 

79 "future release.".format( 

80 raw_identifier, 

81 list(case_sensitive_reg[identifier].keys())[0], 

82 ) 

83 ) 

84 reg[identifier] = _CASE_SENSITIVE 

85 

86 # Check if a function with different letter case identifier is registered. 

87 elif identifier in case_sensitive_reg: 

88 # Note: This case will be removed in a later release. 

89 if raw_identifier not in case_sensitive_reg[identifier]: 

90 util.warn_deprecated( 

91 "GenericFunction(s) '{}' are already registered with " 

92 "different letter cases and might interact with '{}'. " 

93 "GenericFunction objects will be fully case-insensitive in a " 

94 "future release.".format( 

95 sorted(case_sensitive_reg[identifier].keys()), 

96 raw_identifier, 

97 ) 

98 ) 

99 

100 else: 

101 util.warn( 

102 "The GenericFunction '{}' is already registered and " 

103 "is going to be overriden.".format(raw_identifier) 

104 ) 

105 

106 # Register by default 

107 else: 

108 reg[identifier] = fn 

109 

110 # Always register in case-sensitive registry 

111 case_sensitive_reg[identifier][raw_identifier] = fn

259 def as_comparison(self, left_index, right_index): 

260 """Interpret this expression as a boolean comparison between two values. 

261 

262 A hypothetical SQL function "is_equal()" which compares to values 

263 for equality would be written in the Core expression language as:: 

264 

265 expr = func.is_equal("a", "b") 

266 

267 If "is_equal()" above is comparing "a" and "b" for equality, the 

268 :meth:`.FunctionElement.as_comparison` method would be invoked as:: 

269 

270 expr = func.is_equal("a", "b").as_comparison(1, 2) 

271 

272 Where above, the integer value "1" refers to the first argument of the 

273 "is_equal()" function and the integer value "2" refers to the second. 

274 

275 This would create a :class:`.BinaryExpression` that is equivalent to:: 

276 

277 BinaryExpression("a", "b", operator=op.eq) 

278 

279 However, at the SQL level it would still render as 

280 "is_equal('a', 'b')". 

281 

282 The ORM, when it loads a related object or collection, needs to be able 

283 to manipulate the "left" and "right" sides of the ON clause of a JOIN 

284 expression. The purpose of this method is to provide a SQL function 

285 construct that can also supply this information to the ORM, when used 

286 with the :paramref:`.relationship.primaryjoin` parameter. The return 

287 value is a containment object called :class:`.FunctionAsBinary`. 

288 

289 An ORM example is as follows:: 

290 

291 class Venue(Base): 

292 __tablename__ = 'venue' 

293 id = Column(Integer, primary_key=True) 

294 name = Column(String) 

295 

296 descendants = relationship( 

297 "Venue", 

298 primaryjoin=func.instr( 

299 remote(foreign(name)), name + "/" 

300 ).as_comparison(1, 2) == 1, 

301 viewonly=True, 

302 order_by=name 

303 ) 

304 

305 Above, the "Venue" class can load descendant "Venue" objects by 

306 determining if the name of the parent Venue is contained within the 

307 start of the hypothetical descendant value's name, e.g. "parent1" would 

308 match up to "parent1/child1", but not to "parent2/child1". 

309 

310 Possible use cases include the "materialized path" example given above, 

311 as well as making use of special SQL functions such as geometric 

312 functions to create join conditions. 

313 

314 :param left_index: the integer 1-based index of the function argument 

315 that serves as the "left" side of the expression. 

316 :param right_index: the integer 1-based index of the function argument 

317 that serves as the "right" side of the expression. 

318 

319 .. versionadded:: 1.3 

320 

321 """ 

322 return FunctionAsBinary(self, left_index, right_index)

483class _FunctionGenerator(object): 

484 """Generate SQL function expressions. 

485 

486 :data:`.func` is a special object instance which generates SQL 

487 functions based on name-based attributes, e.g.:: 

488 

489 >>> print(func.count(1)) 

490 count(:param_1) 

491 

492 The returned object is an instance of :class:`.Function`, and is a 

493 column-oriented SQL element like any other, and is used in that way:: 

494 

495 >>> print(select([func.count(table.c.id)])) 

496 SELECT count(sometable.id) FROM sometable 

497 

498 Any name can be given to :data:`.func`. If the function name is unknown to 

499 SQLAlchemy, it will be rendered exactly as is. For common SQL functions 

500 which SQLAlchemy is aware of, the name may be interpreted as a *generic 

501 function* which will be compiled appropriately to the target database:: 

502 

503 >>> print(func.current_timestamp()) 

504 CURRENT_TIMESTAMP 

505 

506 To call functions which are present in dot-separated packages, 

507 specify them in the same manner:: 

508 

509 >>> print(func.stats.yield_curve(5, 10)) 

510 stats.yield_curve(:yield_curve_1, :yield_curve_2) 

511 

512 SQLAlchemy can be made aware of the return type of functions to enable 

513 type-specific lexical and result-based behavior. For example, to ensure 

514 that a string-based function returns a Unicode value and is similarly 

515 treated as a string in expressions, specify 

516 :class:`~sqlalchemy.types.Unicode` as the type: 

517 

518 >>> print(func.my_string(u'hi', type_=Unicode) + ' ' + 

519 ... func.my_string(u'there', type_=Unicode)) 

520 my_string(:my_string_1) || :my_string_2 || my_string(:my_string_3) 

521 

522 The object returned by a :data:`.func` call is usually an instance of 

523 :class:`.Function`. 

524 This object meets the "column" interface, including comparison and labeling 

525 functions. The object can also be passed the :meth:`~.Connectable.execute` 

526 method of a :class:`.Connection` or :class:`.Engine`, where it will be 

527 wrapped inside of a SELECT statement first:: 

528 

529 print(connection.execute(func.current_timestamp()).scalar()) 

530 

531 In a few exception cases, the :data:`.func` accessor 

532 will redirect a name to a built-in expression such as :func:`.cast` 

533 or :func:`.extract`, as these names have well-known meaning 

534 but are not exactly the same as "functions" from a SQLAlchemy 

535 perspective. 

536 

537 Functions which are interpreted as "generic" functions know how to 

538 calculate their return type automatically. For a listing of known generic 

539 functions, see :ref:`generic_functions`. 

540 

541 .. note:: 

542 

543 The :data:`.func` construct has only limited support for calling 

544 standalone "stored procedures", especially those with special 

545 parameterization concerns. 

546 

547 See the section :ref:`stored_procedures` for details on how to use 

548 the DBAPI-level ``callproc()`` method for fully traditional stored 

549 procedures. 

550 

551 .. seealso:: 

552 

553 :ref:`coretutorial_functions` - in the Core Tutorial 

554 

555 :class:`.Function` 

556 

557 """ 

558

560 self.__names = [] 

561 self.opts = opts 

562 

563 def __getattr__(self, name): 

564 # passthru __ attributes; fixes pydoc 

565 if name.startswith("__"): 

566 try: 

567 return self.__dict__[name] 

568 except KeyError: 

569 raise AttributeError(name) 

570 

571 elif name.endswith("_"): 

572 name = name[0:-1] 

573 f = _FunctionGenerator(**self.opts) 

574 f.__names = list(self.__names) + [name] 

575 return f

578 o = self.opts.copy() 

579 o.update(kwargs) 

580

582 

583 if tokens == 2:

593 case_sensitive_reg = _case_sensitive_registry[package] 

594 func = case_sensitive_reg.get(fname.lower()).get(fname) 

595 

596 if func is not None:

598 

599 return Function( 

600 self.__names[-1], packagenames=self.__names[0:-1], *c, **o 

601 )

694class GenericFunction(util.with_metaclass(_GenericMeta, Function)): 

695 """Define a 'generic' function. 

696 

697 A generic function is a pre-established :class:`.Function` 

698 class that is instantiated automatically when called 

699 by name from the :data:`.func` attribute. Note that 

700 calling any name from :data:`.func` has the effect that 

701 a new :class:`.Function` instance is created automatically, 

702 given that name. The primary use case for defining 

703 a :class:`.GenericFunction` class is so that a function 

704 of a particular name may be given a fixed return type. 

705 It can also include custom argument parsing schemes as well 

706 as additional methods. 

707 

708 Subclasses of :class:`.GenericFunction` are automatically 

709 registered under the name of the class. For 

710 example, a user-defined function ``as_utc()`` would 

711 be available immediately:: 

712 

713 from sqlalchemy.sql.functions import GenericFunction 

714 from sqlalchemy.types import DateTime 

715 

716 class as_utc(GenericFunction): 

717 type = DateTime 

718 

719 print select([func.as_utc()]) 

720 

721 User-defined generic functions can be organized into 

722 packages by specifying the "package" attribute when defining 

723 :class:`.GenericFunction`. Third party libraries 

724 containing many functions may want to use this in order 

725 to avoid name conflicts with other systems. For example, 

726 if our ``as_utc()`` function were part of a package

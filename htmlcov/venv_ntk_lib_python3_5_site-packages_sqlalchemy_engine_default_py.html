360 def _check_unicode_returns(self, connection, additional_tests=None): 

361 if util.py2k and not self.supports_unicode_statements: 

362 cast_to = util.binary_type 

363 else: 

364 cast_to = util.text_type 

365 

366 if self.positional: 

367 parameters = self.execute_sequence_format() 

368 else: 

369 parameters = {} 

370 

371 def check_unicode(test): 

372 statement = cast_to( 

373 expression.select([test]).compile(dialect=self) 

374 ) 

375 try: 

376 cursor = connection.connection.cursor() 

377 connection._cursor_execute(cursor, statement, parameters) 

378 row = cursor.fetchone() 

379 cursor.close() 

380 except exc.DBAPIError as de: 

381 # note that _cursor_execute() will have closed the cursor 

382 # if an exception is thrown. 

383 util.warn( 

384 "Exception attempting to " 

385 "detect unicode returns: %r" % de 

386 ) 

387 return False 

388 else: 

389 return isinstance(row[0], util.text_type) 

390 

391 tests = [ 

392 # detect plain VARCHAR 

393 expression.cast( 

394 expression.literal_column("'test plain returns'"), 

395 sqltypes.VARCHAR(60), 

396 ), 

397 # detect if there's an NVARCHAR type with different behavior 

398 # available 

399 expression.cast( 

400 expression.literal_column("'test unicode returns'"), 

401 sqltypes.Unicode(60), 

402 ), 

403 ] 

404 

405 if additional_tests: 

406 tests += additional_tests 

407 

408 results = {check_unicode(test) for test in tests} 

409 

410 if results.issuperset([True, False]): 

411 return "conditional" 

412 else: 

413 return results == {True}

595 def normalize_name(self, name): 

596 if name is None: 

597 return None 

598 if util.py2k: 

599 if isinstance(name, str): 

600 name = name.decode(self.encoding) 

601 

602 name_lower = name.lower() 

603 name_upper = name.upper() 

604 

605 if name_upper == name_lower: 

606 # name has no upper/lower conversion, e.g. non-european characters. 

607 # return unchanged 

608 return name 

609 elif name_upper == name and not ( 

610 self.identifier_preparer._requires_quotes 

611 )(name_lower): 

612 # name is all uppercase and doesn't require quoting; normalize 

613 # to all lower case 

614 return name_lower 

615 elif name_lower == name: 

616 # name is all lower case, which if denormalized means we need to 

617 # force quoting on it 

618 return quoted_name(name, quote=True) 

619 else: 

620 # name is mixed case, means it will be quoted in SQL when used 

621 # later, no normalizes 

622 return name

624 def denormalize_name(self, name): 

625 if name is None: 

626 return None 

627 

628 name_lower = name.lower() 

629 name_upper = name.upper() 

630 

631 if name_upper == name_lower: 

632 # name has no upper/lower conversion, e.g. non-european characters. 

633 # return unchanged 

634 return name 

635 elif name_lower == name and not ( 

636 self.identifier_preparer._requires_quotes 

637 )(name_lower): 

638 name = name_upper 

639 if util.py2k: 

640 if not self.supports_unicode_binds: 

641 name = name.encode(self.encoding) 

642 else: 

643 name = unicode(name) # noqa 

644 return name

834 def _expand_in_parameters(self, compiled, processors): 

835 """handle special 'expanding' parameters, IN tuples that are rendered 

836 on a per-parameter basis for an otherwise fixed SQL statement string. 

837 

838 """ 

839 if self.executemany: 

840 raise exc.InvalidRequestError( 

841 "'expanding' parameters can't be used with " "executemany()" 

842 ) 

843 

844 if self.compiled.positional and self.compiled._numeric_binds: 

845 # I'm not familiar with any DBAPI that uses 'numeric' 

846 raise NotImplementedError( 

847 "'expanding' bind parameters not supported with " 

848 "'numeric' paramstyle at this time."

852

854 if compiled.positional: 

855 positiontup = [] 

856 else: 

857 positiontup = None 

858

860 to_update_sets = {}

862 for name in ( 

863 self.compiled.positiontup 

864 if compiled.positional 

865 else self.compiled.binds 

866 ):

868 if parameter.expanding: 

869 

870 if name in replacement_expressions: 

871 to_update = to_update_sets[name] 

872 else: 

873 # we are removing the parameter from compiled_params 

874 # because it is a list value, which is not expected by 

875 # TypeEngine objects that would otherwise be asked to 

876 # process it. the single name is being replaced with 

877 # individual numbered parameters for each value in the 

878 # param. 

879 values = compiled_params.pop(name) 

880 

881 if not values: 

882 to_update = to_update_sets[name] = [] 

883 replacement_expressions[ 

884 name 

885 ] = self.compiled.visit_empty_set_expr( 

886 parameter._expanding_in_types 

887 if parameter._expanding_in_types 

888 else [parameter.type] 

889 ) 

890 

891 elif isinstance(values[0], (tuple, list)): 

892 to_update = to_update_sets[name] = [ 

893 ("%s_%s_%s" % (name, i, j), value) 

894 for i, tuple_element in enumerate(values, 1) 

895 for j, value in enumerate(tuple_element, 1) 

896 ] 

897 replacement_expressions[name] = ( 

898 "VALUES " if self.dialect.tuple_in_values else "" 

899 ) + ", ".join( 

900 "(%s)" 

901 % ", ".join( 

902 self.compiled.bindtemplate 

903 % { 

904 "name": to_update[ 

905 i * len(tuple_element) + j 

906 ][0] 

907 } 

908 for j, value in enumerate(tuple_element) 

909 ) 

910 for i, tuple_element in enumerate(values) 

911 ) 

912 else:

914 ("%s_%s" % (name, i), value) 

915 for i, value in enumerate(values, 1) 

916 ] 

917 replacement_expressions[name] = ", ".join( 

918 self.compiled.bindtemplate % {"name": key} 

919 for key, value in to_update

923 processors.update( 

924 (key, processors[name]) 

925 for key, value in to_update 

926 if name in processors 

927 ) 

928 if compiled.positional: 

929 positiontup.extend(name for name, value in to_update) 

930 self._expanded_parameters[name] = [ 

931 expand_key for expand_key, value in to_update 

932 ] 

933 elif compiled.positional:

937 return replacement_expressions[m.group(1)]

940 r"\[EXPANDING_(\S+)\]", process_expanding, self.statement

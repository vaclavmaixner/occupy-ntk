462class Image(XYGlyph): 

463 ''' Render images given as scalar data together with a color mapper. 

464 

465 In addition to the defined model properties, ``Image`` also can accept 

466 a keyword argument ``palette`` in place of an explicit ``color_mapper``. 

467 The value should be a list of colors, or the name of one of the built-in 

468 palettes in ``bokeh.palettes``. This palette will be used to automatically 

469 construct a ``ColorMapper`` model for the ``color_mapper`` property. 

470 

471 If both ``palette`` and ``color_mapper`` are passed, a ``ValueError`` 

472 exception will be raised. If neither is passed, then the ``Greys9`` 

473 palette will be used as a default. 

474 

475 ''' 

476 

477 def __init__(self, **kwargs): 

478 if 'palette' in kwargs and 'color_mapper' in kwargs: 

479 raise ValueError("only one of 'palette' and 'color_mapper' may be specified") 

480 elif 'color_mapper' not in kwargs: 

481 # Use a palette (given or default) 

482 palette = kwargs.pop('palette', 'Greys9') 

483 mapper = LinearColorMapper(palette) 

484 kwargs['color_mapper'] = mapper 

485 

486 super(Image, self).__init__(**kwargs) 

487 

488 # a canonical order for positional args that can be used for any 

489 # functions derived from this class 

490 _args = ('image', 'x', 'y', 'dw', 'dh', 'global_alpha', 'dilate') 

491 

492 # a hook to specify any additional kwargs handled by an initializer 

493 _extra_kws = { 

494 'palette': ( 

495 'str or list[color value]', 

496 'a palette to construct a value for the color mapper property from' 

497 ) 

498 } 

499 

500 image = NumberSpec(help=""" 

501 The arrays of scalar data for the images to be colormapped. 

502 """) 

503 

504 x = NumberSpec(help=""" 

505 The x-coordinates to locate the image anchors.

518 """)

78class MetaHasProps(type): 

79 ''' Specialize the construction of |HasProps| classes. 

80 

81 This class is a `metaclass`_ for |HasProps| that is responsible for 

82 creating and adding the |PropertyDescriptor| instances that delegate 

83 validation and serialization to |Property| attributes. 

84 

85 .. _metaclass: https://docs.python.org/3/reference/datamodel.html#metaclasses 

86 

87 ''' 

88 def __new__(meta_cls, class_name, bases, class_dict): 

89 ''' 

90 

91 ''' 

92 names_with_refs = set() 

93 container_names = set() 

94 

95 # Now handle all the Override 

96 overridden_defaults = {} 

97 for name, prop in class_dict.items(): 

98 if not isinstance(prop, Override): 

99 continue 

100 if prop.default_overridden: 

101 overridden_defaults[name] = prop.default 

102 

103 for name, default in overridden_defaults.items(): 

104 del class_dict[name] 

105 

106 generators = dict() 

107 for name, generator in class_dict.items(): 

108 if isinstance(generator, PropertyDescriptorFactory): 

109 generators[name] = generator 

110 elif isinstance(generator, type) and issubclass(generator, PropertyDescriptorFactory): 

111 # Support the user adding a property without using parens, 

112 # i.e. using just the Property subclass instead of an 

113 # instance of the subclass 

114 generators[name] = generator.autocreate() 

115 

116 dataspecs = {} 

117 new_class_attrs = {} 

118 

119 for name, generator in generators.items(): 

120 prop_descriptors = generator.make_descriptors(name) 

121 replaced_self = False 

122 for prop_descriptor in prop_descriptors: 

123 if prop_descriptor.name in generators: 

124 if generators[prop_descriptor.name] is generator: 

125 # a generator can replace itself, this is the 

126 # standard case like `foo = Int()` 

127 replaced_self = True 

128 prop_descriptor.add_prop_descriptor_to_class(class_name, new_class_attrs, names_with_refs, container_names, dataspecs) 

129 else: 

130 # if a generator tries to overwrite another 

131 # generator that's been explicitly provided, 

132 # use the prop that was manually provided 

133 # and ignore this one. 

134 pass 

135 else: 

136 prop_descriptor.add_prop_descriptor_to_class(class_name, new_class_attrs, names_with_refs, container_names, dataspecs) 

137 # if we won't overwrite ourselves anyway, delete the generator 

138 if not replaced_self: 

139 del class_dict[name] 

140 

141 class_dict.update(new_class_attrs) 

142 

143 class_dict["__properties__"] = set(new_class_attrs) 

144 class_dict["__properties_with_refs__"] = names_with_refs 

145 class_dict["__container_props__"] = container_names 

146 if len(overridden_defaults) > 0: 

147 class_dict["__overridden_defaults__"] = overridden_defaults 

148 if dataspecs: 

149 class_dict["__dataspecs__"] = dataspecs 

150 

151 if "__example__" in class_dict: 

152 path = class_dict["__example__"] 

153 

154 # running python with -OO will discard docstrings -> __doc__ is None 

155 if "__doc__" in class_dict and class_dict["__doc__"] is not None: 

156 class_dict["__doc__"] += _EXAMPLE_TEMPLATE % dict(path=path) 

157 

158 return super(MetaHasProps, meta_cls).__new__(meta_cls, class_name, bases, class_dict)

160 def __init__(cls, class_name, bases, nmspc): 

161 if class_name == 'HasProps': 

162 return 

163 # Check for improperly overriding a Property attribute. 

164 # Overriding makes no sense except through the Override 

165 # class which can be used to tweak the default. 

166 # Historically code also tried changing the Property's 

167 # type or changing from Property to non-Property: these 

168 # overrides are bad conceptually because the type of a 

169 # read-write property is invariant. 

170 cls_attrs = cls.__dict__.keys() # we do NOT want inherited attrs here 

171 for attr in cls_attrs: 

172 for base in bases: 

173 if issubclass(base, HasProps) and attr in base.properties(): 

174 warn(('Property "%s" in class %s was overridden by a class attribute ' + \ 

175 '"%s" in class %s; it never makes sense to do this. ' + \ 

176 'Either %s.%s or %s.%s should be removed, or %s.%s should not ' + \ 

177 'be a Property, or use Override(), depending on the intended effect.') % 

178 (attr, base.__name__, attr, class_name, 

179 base.__name__, attr, 

180 class_name, attr, 

181 base.__name__, attr), 

182 RuntimeWarning, stacklevel=2) 

183 

184 if "__overridden_defaults__" in cls.__dict__: 

185 our_props = cls.properties() 

186 for key in cls.__dict__["__overridden_defaults__"].keys(): 

187 if key not in our_props: 

188 warn(('Override() of %s in class %s does not override anything.') % (key, class_name), 

189 RuntimeWarning, stacklevel=2)

191def accumulate_from_superclasses(cls, propname): 

192 ''' Traverse the class hierarchy and accumulate the special sets of names 

193 ``MetaHasProps`` stores on classes: 

194 

195 Args: 

196 name (str) : name of the special attribute to collect. 

197 

198 Typically meaningful values are: ``__container_props__``, 

199 ``__properties__``, ``__properties_with_refs__`` 

200 

201 ''' 

202 cachename = "__cached_all" + propname 

203 # we MUST use cls.__dict__ NOT hasattr(). hasattr() would also look at base 

204 # classes, and the cache must be separate for each class 

205 if cachename not in cls.__dict__: 

206 s = set() 

207 for c in inspect.getmro(cls): 

208 if issubclass(c, HasProps) and hasattr(c, propname): 

209 base = getattr(c, propname) 

210 s.update(base) 

211 setattr(cls, cachename, s) 

212 return cls.__dict__[cachename]

214def accumulate_dict_from_superclasses(cls, propname): 

215 ''' Traverse the class hierarchy and accumulate the special dicts 

216 ``MetaHasProps`` stores on classes: 

217 

218 Args: 

219 name (str) : name of the special attribute to collect. 

220 

221 Typically meaningful values are: ``__dataspecs__``, 

222 ``__overridden_defaults__`` 

223 

224 ''' 

225 cachename = "__cached_all" + propname 

226 # we MUST use cls.__dict__ NOT hasattr(). hasattr() would also look at base 

227 # classes, and the cache must be separate for each class 

228 if cachename not in cls.__dict__: 

229 d = dict() 

230 for c in inspect.getmro(cls): 

231 if issubclass(c, HasProps) and hasattr(c, propname): 

232 base = getattr(c, propname) 

233 for k,v in base.items(): 

234 if k not in d: 

235 d[k] = v 

236 setattr(cls, cachename, d) 

237 return cls.__dict__[cachename]

255 def __setattr__(self, name, value): 

256 ''' Intercept attribute setting on HasProps in order to special case 

257 a few situations: 

258 

259 * short circuit all property machinery for ``_private`` attributes 

260 * suggest similar attribute names on attribute errors 

261 

262 Args: 

263 name (str) : the name of the attribute to set on this object 

264 value (obj) : the value to set 

265 

266 Returns: 

267 None 

268 

269 ''' 

270 # self.properties() below can be expensive so avoid it 

271 # if we're just setting a private underscore field 

272 if name.startswith("_"): 

273 super(HasProps, self).__setattr__(name, value) 

274 return 

275 

276 props = sorted(self.properties()) 

277 descriptor = getattr(self.__class__, name, None) 

278 

279 if name in props or (descriptor is not None and descriptor.fset is not None): 

280 super(HasProps, self).__setattr__(name, value) 

281 else: 

282 matches, text = difflib.get_close_matches(name.lower(), props), "similar" 

283 

284 if not matches: 

285 matches, text = props, "possible" 

286 

287 raise AttributeError("unexpected attribute '%s' to %s, %s attributes are %s" % 

288 (name, self.__class__.__name__, text, nice_join(matches)))

295 def equals(self, other): 

296 ''' Structural equality of models. 

297 

298 Args: 

299 other (HasProps) : the other instance to compare to 

300 

301 Returns: 

302 True, if properties are structurally equal, otherwise False 

303 

304 ''' 

305 

306 # NOTE: don't try to use this to implement __eq__. Because then 

307 # you will be tempted to implement __hash__, which would interfere 

308 # with mutability of models. However, not implementing __hash__ 

309 # will make bokeh unusable in Python 3, where proper implementation 

310 # of __hash__ is required when implementing __eq__. 

311 if not isinstance(other, self.__class__): 

312 return False 

313 else: 

314 return self.properties_with_values() == other.properties_with_values()

317 ''' Set a property value on this object from JSON. 

318 

319 Args: 

320 name: (str) : name of the attribute to set 

321 

322 json: (JSON-value) : value to set to the attribute to 

323 

324 models (dict or None, optional) :

326 

327 This is needed in cases where the attributes to update also 

328 have values that have references.

331 This is used to prevent "boomerang" updates to Bokeh apps.

528 def query_properties_with_values(self, query, include_defaults=True): 

529 ''' Query the properties values of |HasProps| instances with a 

530 predicate. 

531 

532 Args: 

533 query (callable) : 

534 A callable that accepts property descriptors and returns True 

535 or False 

536 

537 include_defaults (bool, optional) : 

538 Whether to include properties that have not been explicitly 

539 set by a user (default: True) 

540 

541 Returns: 

542 dict : mapping of property names and values for matching properties 

543 

544 ''' 

545 themed_keys = set() 

546 result = dict() 

547 if include_defaults: 

548 keys = self.properties() 

549 else: 

550 # TODO (bev) For now, include unstable default values. Things rely on Instances 

551 # always getting serialized, even defaults, and adding unstable defaults here 

552 # accomplishes that. Unmodified defaults for property value containers will be 

553 # weeded out below. 

554 keys = set(self._property_values.keys()) | set(self._unstable_default_values.keys()) 

555 if self.themed_values(): 

556 themed_keys = set(self.themed_values().keys()) 

557 keys |= themed_keys 

558 

559 for key in keys: 

560 descriptor = self.lookup(key) 

561 if not query(descriptor): 

562 continue 

563 

564 value = descriptor.serializable_value(self) 

565 if not include_defaults and key not in themed_keys: 

566 if isinstance(value, PropertyValueContainer) and key in self._unstable_default_values: 

567 continue 

568 result[key] = value 

569 

570 return result

584 def apply_theme(self, property_values): 

585 ''' Apply a set of theme values which will be used rather than 

586 defaults, but will not override application-set values. 

587 

588 The passed-in dictionary may be kept around as-is and shared with 

589 other instances to save memory (so neither the caller nor the 

590 |HasProps| instance should modify it). 

591 

592 Args: 

593 property_values (dict) : theme values to use in place of defaults 

594 

595 Returns: 

596 None 

597 

598 ''' 

599 old_dict = self.themed_values() 

600 

601 # if the same theme is set again, it should reuse the same dict 

602 if old_dict is property_values: 

603 return 

604 

605 removed = set() 

606 # we're doing a little song-and-dance to avoid storing __themed_values__ or 

607 # an empty dict, if there's no theme that applies to this HasProps instance. 

608 if old_dict is not None: 

609 removed.update(set(old_dict.keys())) 

610 added = set(property_values.keys()) 

611 old_values = dict() 

612 for k in added.union(removed): 

613 old_values[k] = getattr(self, k) 

614 

615 if len(property_values) > 0: 

616 setattr(self, '__themed_values__', property_values) 

617 elif hasattr(self, '__themed_values__'): 

618 delattr(self, '__themed_values__') 

619 

620 # Property container values might be cached even if unmodified. Invalidate 

621 # any cached values that are not modified at this point. 

622 for k, v in old_values.items(): 

623 if k in self._unstable_themed_values: 

624 del self._unstable_themed_values[k] 

625 

626 # Emit any change notifications that result 

627 for k, v in old_values.items(): 

628 descriptor = self.lookup(k) 

629 descriptor.trigger_if_changed(self, v)

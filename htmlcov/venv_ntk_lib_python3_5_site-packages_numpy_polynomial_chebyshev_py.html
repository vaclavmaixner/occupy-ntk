68The implementations of multiplication, division, integration, and

70 

71.. math ::

116def _cseries_to_zseries(c): 

117 """Covert Chebyshev series to z-series. 

118 

119 Covert a Chebyshev series to the equivalent z-series. The result is 

120 never an empty array. The dtype of the return is the same as that of 

121 the input. No checks are run on the arguments as this routine is for 

122 internal use. 

123 

124 Parameters 

125 ---------- 

126 c : 1-D ndarray 

127 Chebyshev coefficients, ordered from low to high 

128 

129 Returns 

130 ------- 

131 zs : 1-D ndarray 

132 Odd length symmetric z-series, ordered from low to high. 

133 

134 """ 

135 n = c.size 

136 zs = np.zeros(2*n-1, dtype=c.dtype) 

137 zs[n-1:] = c/2 

138 return zs + zs[::-1]

193def _zseries_div(z1, z2): 

194 """Divide the first z-series by the second. 

195 

196 Divide `z1` by `z2` and return the quotient and remainder as z-series. 

197 Warning: this implementation only applies when both z1 and z2 have the 

198 same symmetry, which is sufficient for present purposes. 

199 

200 Parameters 

201 ---------- 

202 z1, z2 : 1-D ndarray 

203 The arrays must be 1-D and have the same symmetry, but this is not 

204 checked. 

205 

206 Returns 

207 ------- 

208 

209 (quotient, remainder) : 1-D ndarrays 

210 Quotient and remainder as z-series. 

211 

212 Notes 

213 ----- 

214 This is not the same as polynomial division on account of the desired form 

215 of the remainder. If symmetric/anti-symmetric z-series are denoted by S/A 

216 then the following rules apply: 

217 

218 S/S -> S,S 

219 A/A -> S,A 

220 

221 The restriction to types of the same symmetry could be fixed but seems like 

222 unneeded generality. There is no natural form for the remainder in the case 

223 where there is no symmetry. 

224 

225 """ 

226 z1 = z1.copy() 

227 z2 = z2.copy() 

228 lc1 = len(z1) 

229 lc2 = len(z2) 

230 if lc2 == 1: 

231 z1 /= z2 

232 return z1, z1[:1]*0 

233 elif lc1 < lc2: 

234 return z1[:1]*0, z1 

235 else: 

236 dlen = lc1 - lc2 

237 scl = z2[0] 

238 z2 /= scl 

239 quo = np.empty(dlen + 1, dtype=z1.dtype) 

240 i = 0 

241 j = dlen 

242 while i < j: 

243 r = z1[i] 

244 quo[i] = z1[i] 

245 quo[dlen - i] = r 

246 tmp = r*z2 

247 z1[i:i+lc2] -= tmp 

248 z1[j:j+lc2] -= tmp 

249 i += 1 

250 j -= 1 

251 r = z1[i] 

252 quo[i] = r 

253 tmp = r*z2 

254 z1[i:i+lc2] -= tmp 

255 quo /= scl 

256 rem = z1[i+1:i-1+lc2].copy() 

257 return quo, rem

260def _zseries_der(zs): 

261 """Differentiate a z-series. 

262 

263 The derivative is with respect to x, not z. This is achieved using the 

264 chain rule and the value of dx/dz given in the module notes. 

265 

266 Parameters 

267 ---------- 

268 zs : z-series 

269 The z-series to differentiate. 

270 

271 Returns 

272 ------- 

273 derivative : z-series 

274 The derivative 

275 

276 Notes 

277 ----- 

278 The zseries for x (ns) has been multiplied by two in order to avoid 

279 using floats that are incompatible with Decimal and likely other 

280 specialized scalar types. This scaling has been compensated by 

281 multiplying the value of zs by two also so that the two cancels in the 

282 division. 

283 

284 """ 

285 n = len(zs)//2 

286 ns = np.array([-1, 0, 1], dtype=zs.dtype) 

287 zs *= np.arange(-n, n+1)*2 

288 d, r = _zseries_div(zs, ns) 

289 return d

292def _zseries_int(zs): 

293 """Integrate a z-series. 

294 

295 The integral is with respect to x, not z. This is achieved by a change 

296 of variable using dx/dz given in the module notes. 

297 

298 Parameters 

299 ---------- 

300 zs : z-series 

301 The z-series to integrate 

302 

303 Returns 

304 ------- 

305 integral : z-series 

306 The indefinite integral 

307 

308 Notes 

309 ----- 

310 The zseries for x (ns) has been multiplied by two in order to avoid 

311 using floats that are incompatible with Decimal and likely other 

312 specialized scalar types. This scaling has been compensated by 

313 dividing the resulting zs by two. 

314 

315 """ 

316 n = 1 + len(zs)//2 

317 ns = np.array([-1, 0, 1], dtype=zs.dtype) 

318 zs = _zseries_mul(zs, ns) 

319 div = np.arange(-n, n+1)*2 

320 zs[:n] /= div[:n] 

321 zs[n+1:] /= div[n+1:] 

322 zs[n] = 0 

323 return zs

330def poly2cheb(pol): 

331 """ 

332 Convert a polynomial to a Chebyshev series. 

333 

334 Convert an array representing the coefficients of a polynomial (relative 

335 to the "standard" basis) ordered from lowest degree to highest, to an 

336 array of the coefficients of the equivalent Chebyshev series, ordered 

337 from lowest to highest degree. 

338 

339 Parameters 

340 ---------- 

341 pol : array_like 

342 1-D array containing the polynomial coefficients 

343 

344 Returns 

345 ------- 

346 c : ndarray 

347 1-D array containing the coefficients of the equivalent Chebyshev 

348 series. 

349 

350 See Also 

351 -------- 

352 cheb2poly 

353 

354 Notes 

355 ----- 

356 The easy way to do conversions between polynomial basis sets 

357 is to use the convert method of a class instance. 

358 

359 Examples 

360 -------- 

361 >>> from numpy import polynomial as P 

362 >>> p = P.Polynomial(range(4)) 

363 >>> p 

364 Polynomial([0., 1., 2., 3.], domain=[-1, 1], window=[-1, 1])

366 >>> c 

367 Chebyshev([1. , 3.25, 1. , 0.75], domain=[-1., 1.], window=[-1., 1.]) 

368 >>> P.chebyshev.poly2cheb(range(4))

373 deg = len(pol) - 1

380def cheb2poly(c): 

381 """ 

382 Convert a Chebyshev series to a polynomial. 

383 

384 Convert an array representing the coefficients of a Chebyshev series, 

385 ordered from lowest degree to highest, to an array of the coefficients 

386 of the equivalent polynomial (relative to the "standard" basis) ordered 

387 from lowest to highest degree. 

388 

389 Parameters 

390 ---------- 

391 c : array_like 

392 1-D array containing the Chebyshev series coefficients, ordered 

393 from lowest order term to highest. 

394 

395 Returns 

396 ------- 

397 pol : ndarray 

398 1-D array containing the coefficients of the equivalent polynomial 

399 (relative to the "standard" basis) ordered from lowest order term 

400 to highest. 

401 

402 See Also 

403 -------- 

404 poly2cheb 

405 

406 Notes 

407 ----- 

408 The easy way to do conversions between polynomial basis sets 

409 is to use the convert method of a class instance. 

410 

411 Examples 

412 -------- 

413 >>> from numpy import polynomial as P 

414 >>> c = P.Chebyshev(range(4)) 

415 >>> c 

416 Chebyshev([0., 1., 2., 3.], domain=[-1, 1], window=[-1, 1]) 

417 >>> p = c.convert(kind=P.Polynomial) 

418 >>> p 

419 Polynomial([-2., -8., 4., 12.], domain=[-1., 1.], window=[-1., 1.]) 

420 >>> P.chebyshev.cheb2poly(range(4)) 

421 array([-2., -8., 4., 12.]) 

422 

423 """ 

424 from .polynomial import polyadd, polysub, polymulx 

425 

426 [c] = pu.as_series([c]) 

427 n = len(c) 

428 if n < 3: 

429 return c 

430 else: 

431 c0 = c[-2] 

432 c1 = c[-1] 

433 # i is the current degree of c1 

434 for i in range(n - 1, 1, -1): 

435 tmp = c0 

436 c0 = polysub(c[i - 2], c1) 

437 c1 = polyadd(tmp, polymulx(c1)*2) 

438 return polyadd(c0, polymulx(c1))

495def chebfromroots(roots): 

496 """ 

497 Generate a Chebyshev series with given roots. 

498 

499 The function returns the coefficients of the polynomial 

500 

501 .. math:: p(x) = (x - r_0) * (x - r_1) * ... * (x - r_n), 

502 

503 in Chebyshev form, where the `r_n` are the roots specified in `roots`. 

504 If a zero has multiplicity n, then it must appear in `roots` n times. 

505 For instance, if 2 is a root of multiplicity three and 3 is a root of 

506 multiplicity 2, then `roots` looks something like [2, 2, 2, 3, 3]. The 

507 roots can appear in any order. 

508 

509 If the returned coefficients are `c`, then 

510 

511 .. math:: p(x) = c_0 + c_1 * T_1(x) + ... + c_n * T_n(x) 

512 

513 The coefficient of the last term is not generally 1 for monic 

514 polynomials in Chebyshev form. 

515 

516 Parameters 

517 ---------- 

518 roots : array_like 

519 Sequence containing the roots. 

520 

521 Returns 

522 ------- 

523 out : ndarray 

524 1-D array of coefficients. If all roots are real then `out` is a 

525 real array, if some of the roots are complex, then `out` is complex 

526 even if all the coefficients in the result are real (see Examples 

527 below). 

528 

529 See Also 

530 -------- 

531 polyfromroots, legfromroots, lagfromroots, hermfromroots, hermefromroots 

532 

533 Examples 

534 --------

633 """Multiply a Chebyshev series by x. 

634 

635 Multiply the polynomial `c` by x, where x is the independent 

636 variable. 

637 

638 

639 Parameters 

640 ---------- 

641 c : array_like

678def chebmul(c1, c2): 

679 """ 

680 Multiply one Chebyshev series by another. 

681 

682 Returns the product of two Chebyshev series `c1` * `c2`. The arguments 

683 are sequences of coefficients, from lowest order "term" to highest, 

684 e.g., [1,2,3] represents the series ``T_0 + 2*T_1 + 3*T_2``. 

685 

686 Parameters 

687 ---------- 

688 c1, c2 : array_like 

689 1-D arrays of Chebyshev series coefficients ordered from low to 

690 high. 

691 

692 Returns 

693 ------- 

694 out : ndarray 

695 Of Chebyshev series coefficients representing their product. 

696 

697 See Also 

698 -------- 

699 chebadd, chebsub, chebmulx, chebdiv, chebpow 

700 

701 Notes 

702 ----- 

703 In general, the (polynomial) product of two C-series results in terms 

704 that are not in the Chebyshev polynomial basis set. Thus, to express 

705 the product as a C-series, it is typically necessary to "reproject" 

706 the product onto said basis set, which typically produces 

707 "unintuitive live" (but correct) results; see Examples section below. 

708 

709 Examples 

710 -------- 

711 >>> from numpy.polynomial import chebyshev as C 

712 >>> c1 = (1,2,3) 

713 >>> c2 = (3,2,1) 

714 >>> C.chebmul(c1,c2) # multiplication requires "reprojection" 

715 array([ 6.5, 12. , 12. , 4. , 1.5]) 

716 

717 """ 

718 # c1, c2 are trimmed copies 

719 [c1, c2] = pu.as_series([c1, c2]) 

720 z1 = _cseries_to_zseries(c1) 

721 z2 = _cseries_to_zseries(c2) 

722 prd = _zseries_mul(z1, z2) 

723 ret = _zseries_to_cseries(prd) 

724 return pu.trimseq(ret)

727def chebdiv(c1, c2): 

728 """ 

729 Divide one Chebyshev series by another. 

730 

731 Returns the quotient-with-remainder of two Chebyshev series 

732 `c1` / `c2`. The arguments are sequences of coefficients from lowest 

733 order "term" to highest, e.g., [1,2,3] represents the series 

734 ``T_0 + 2*T_1 + 3*T_2``. 

735 

736 Parameters 

737 ---------- 

738 c1, c2 : array_like 

739 1-D arrays of Chebyshev series coefficients ordered from low to 

740 high. 

741 

742 Returns 

743 ------- 

744 [quo, rem] : ndarrays 

745 Of Chebyshev series coefficients representing the quotient and 

746 remainder. 

747 

748 See Also 

749 -------- 

750 chebadd, chebsub, chemulx, chebmul, chebpow 

751 

752 Notes 

753 ----- 

754 In general, the (polynomial) division of one C-series by another 

755 results in quotient and remainder terms that are not in the Chebyshev 

756 polynomial basis set. Thus, to express these results as C-series, it 

757 is typically necessary to "reproject" the results onto said basis 

758 set, which typically produces "unintuitive" (but correct) results; 

759 see Examples section below. 

760 

761 Examples 

762 -------- 

763 >>> from numpy.polynomial import chebyshev as C 

764 >>> c1 = (1,2,3) 

765 >>> c2 = (3,2,1) 

766 >>> C.chebdiv(c1,c2) # quotient "intuitive," remainder not 

767 (array([3.]), array([-8., -4.])) 

768 >>> c2 = (0,1,2,3) 

769 >>> C.chebdiv(c2,c1) # neither "intuitive" 

770 (array([0., 2.]), array([-2., -4.])) 

771 

772 """ 

773 # c1, c2 are trimmed copies 

774 [c1, c2] = pu.as_series([c1, c2]) 

775 if c2[-1] == 0: 

776 raise ZeroDivisionError() 

777 

778 # note: this is more efficient than `pu._div(chebmul, c1, c2)` 

779 lc1 = len(c1) 

780 lc2 = len(c2) 

781 if lc1 < lc2: 

782 return c1[:1]*0, c1 

783 elif lc2 == 1: 

784 return c1/c2[-1], c1[:1]*0 

785 else: 

786 z1 = _cseries_to_zseries(c1) 

787 z2 = _cseries_to_zseries(c2) 

788 quo, rem = _zseries_div(z1, z2) 

789 quo = pu.trimseq(_zseries_to_cseries(quo)) 

790 rem = pu.trimseq(_zseries_to_cseries(rem)) 

791 return quo, rem

794def chebpow(c, pow, maxpower=16): 

795 """Raise a Chebyshev series to a power. 

796 

797 Returns the Chebyshev series `c` raised to the power `pow`. The 

798 argument `c` is a sequence of coefficients ordered from low to high. 

799 i.e., [1,2,3] is the series ``T_0 + 2*T_1 + 3*T_2.`` 

800 

801 Parameters 

802 ---------- 

803 c : array_like 

804 1-D array of Chebyshev series coefficients ordered from low to 

805 high. 

806 pow : integer 

807 Power to which the series will be raised 

808 maxpower : integer, optional 

809 Maximum power allowed. This is mainly to limit growth of the series 

810 to unmanageable size. Default is 16 

811 

812 Returns 

813 ------- 

814 coef : ndarray 

815 Chebyshev series of power. 

816 

817 See Also 

818 -------- 

819 chebadd, chebsub, chebmulx, chebmul, chebdiv 

820 

821 Examples 

822 -------- 

823 >>> from numpy.polynomial import chebyshev as C 

824 >>> C.chebpow([1, 2, 3, 4], 2) 

825 array([15.5, 22. , 16. , ..., 12.5, 12. , 8. ]) 

826 

827 """ 

828 # note: this is more efficient than `pu._pow(chebmul, c1, c2)`, as it 

829 # avoids converting between z and c series repeatedly 

830 

831 # c is a trimmed copy 

832 [c] = pu.as_series([c]) 

833 power = int(pow) 

834 if power != pow or power < 0: 

835 raise ValueError("Power must be a non-negative integer.") 

836 elif maxpower is not None and power > maxpower: 

837 raise ValueError("Power is too large") 

838 elif power == 0: 

839 return np.array([1], dtype=c.dtype) 

840 elif power == 1: 

841 return c 

842 else: 

843 # This can be made more efficient by using powers of two 

844 # in the usual way. 

845 zs = _cseries_to_zseries(c) 

846 prd = zs 

847 for i in range(2, power + 1): 

848 prd = np.convolve(prd, zs) 

849 return _zseries_to_cseries(prd)

852def chebder(c, m=1, scl=1, axis=0): 

853 """ 

854 Differentiate a Chebyshev series. 

855 

856 Returns the Chebyshev series coefficients `c` differentiated `m` times 

857 along `axis`. At each iteration the result is multiplied by `scl` (the 

858 scaling factor is for use in a linear change of variable). The argument 

859 `c` is an array of coefficients from low to high degree along each 

860 axis, e.g., [1,2,3] represents the series ``1*T_0 + 2*T_1 + 3*T_2`` 

861 while [[1,2],[1,2]] represents ``1*T_0(x)*T_0(y) + 1*T_1(x)*T_0(y) + 

862 2*T_0(x)*T_1(y) + 2*T_1(x)*T_1(y)`` if axis=0 is ``x`` and axis=1 is 

863 ``y``. 

864 

865 Parameters 

866 ---------- 

867 c : array_like 

868 Array of Chebyshev series coefficients. If c is multidimensional 

869 the different axis correspond to different variables with the 

870 degree in each axis given by the corresponding index. 

871 m : int, optional 

872 Number of derivatives taken, must be non-negative. (Default: 1) 

873 scl : scalar, optional 

874 Each differentiation is multiplied by `scl`. The end result is 

875 multiplication by ``scl**m``. This is for use in a linear change of 

876 variable. (Default: 1) 

877 axis : int, optional 

878 Axis over which the derivative is taken. (Default: 0). 

879 

880 .. versionadded:: 1.7.0 

881 

882 Returns 

883 ------- 

884 der : ndarray 

885 Chebyshev series of the derivative. 

886 

887 See Also 

888 -------- 

889 chebint 

890 

891 Notes 

892 ----- 

893 In general, the result of differentiating a C-series needs to be 

894 "reprojected" onto the C-series basis set. Thus, typically, the 

895 result of this function is "unintuitive," albeit correct; see Examples 

896 section below. 

897 

898 Examples 

899 -------- 

900 >>> from numpy.polynomial import chebyshev as C 

901 >>> c = (1,2,3,4) 

902 >>> C.chebder(c) 

903 array([14., 12., 24.]) 

904 >>> C.chebder(c,3) 

905 array([96.]) 

906 >>> C.chebder(c,scl=-1) 

907 array([-14., -12., -24.]) 

908 >>> C.chebder(c,2,-1) 

909 array([12., 96.]) 

910 

911 """ 

912 c = np.array(c, ndmin=1, copy=True) 

913 if c.dtype.char in '?bBhHiIlLqQpP': 

914 c = c.astype(np.double) 

915 cnt = pu._deprecate_as_int(m, "the order of derivation") 

916 iaxis = pu._deprecate_as_int(axis, "the axis") 

917 if cnt < 0: 

918 raise ValueError("The order of derivation must be non-negative") 

919 iaxis = normalize_axis_index(iaxis, c.ndim) 

920 

921 if cnt == 0: 

922 return c 

923 

924 c = np.moveaxis(c, iaxis, 0) 

925 n = len(c) 

926 if cnt >= n: 

927 c = c[:1]*0 

928 else: 

929 for i in range(cnt): 

930 n = n - 1 

931 c *= scl 

932 der = np.empty((n,) + c.shape[1:], dtype=c.dtype) 

933 for j in range(n, 2, -1): 

934 der[j - 1] = (2*j)*c[j] 

935 c[j - 2] += (j*c[j])/(j - 2) 

936 if n > 1: 

937 der[1] = 4*c[2] 

938 der[0] = c[1] 

939 c = der 

940 c = np.moveaxis(c, 0, iaxis) 

941 return c

944def chebint(c, m=1, k=[], lbnd=0, scl=1, axis=0): 

945 """ 

946 Integrate a Chebyshev series. 

947 

948 Returns the Chebyshev series coefficients `c` integrated `m` times from 

949 `lbnd` along `axis`. At each iteration the resulting series is 

950 **multiplied** by `scl` and an integration constant, `k`, is added. 

951 The scaling factor is for use in a linear change of variable. ("Buyer 

952 beware": note that, depending on what one is doing, one may want `scl` 

953 to be the reciprocal of what one might expect; for more information, 

954 see the Notes section below.) The argument `c` is an array of 

955 coefficients from low to high degree along each axis, e.g., [1,2,3] 

956 represents the series ``T_0 + 2*T_1 + 3*T_2`` while [[1,2],[1,2]] 

957 represents ``1*T_0(x)*T_0(y) + 1*T_1(x)*T_0(y) + 2*T_0(x)*T_1(y) + 

958 2*T_1(x)*T_1(y)`` if axis=0 is ``x`` and axis=1 is ``y``. 

959 

960 Parameters 

961 ---------- 

962 c : array_like 

963 Array of Chebyshev series coefficients. If c is multidimensional 

964 the different axis correspond to different variables with the 

965 degree in each axis given by the corresponding index. 

966 m : int, optional 

967 Order of integration, must be positive. (Default: 1) 

968 k : {[], list, scalar}, optional 

969 Integration constant(s). The value of the first integral at zero 

970 is the first value in the list, the value of the second integral 

971 at zero is the second value, etc. If ``k == []`` (the default), 

972 all constants are set to zero. If ``m == 1``, a single scalar can 

973 be given instead of a list. 

974 lbnd : scalar, optional 

975 The lower bound of the integral. (Default: 0) 

976 scl : scalar, optional 

977 Following each integration the result is *multiplied* by `scl` 

978 before the integration constant is added. (Default: 1) 

979 axis : int, optional 

980 Axis over which the integral is taken. (Default: 0). 

981 

982 .. versionadded:: 1.7.0 

983 

984 Returns 

985 ------- 

986 S : ndarray 

987 C-series coefficients of the integral. 

988 

989 Raises 

990 ------ 

991 ValueError 

992 If ``m < 1``, ``len(k) > m``, ``np.ndim(lbnd) != 0``, or 

993 ``np.ndim(scl) != 0``. 

994 

995 See Also 

996 -------- 

997 chebder 

998 

999 Notes 

1000 ----- 

1001 Note that the result of each integration is *multiplied* by `scl`. 

1002 Why is this important to note? Say one is making a linear change of 

1003 variable :math:`u = ax + b` in an integral relative to `x`. Then 

1004 :math:`dx = du/a`, so one will need to set `scl` equal to 

1005 :math:`1/a`- perhaps not what one would have first thought. 

1006 

1007 Also note that, in general, the result of integrating a C-series needs 

1008 to be "reprojected" onto the C-series basis set. Thus, typically, 

1009 the result of this function is "unintuitive," albeit correct; see 

1010 Examples section below. 

1011 

1012 Examples 

1013 -------- 

1014 >>> from numpy.polynomial import chebyshev as C 

1015 >>> c = (1,2,3) 

1016 >>> C.chebint(c) 

1017 array([ 0.5, -0.5, 0.5, 0.5]) 

1018 >>> C.chebint(c,3) 

1019 array([ 0.03125 , -0.1875 , 0.04166667, -0.05208333, 0.01041667, # may vary 

1020 0.00625 ]) 

1021 >>> C.chebint(c, k=3) 

1022 array([ 3.5, -0.5, 0.5, 0.5]) 

1023 >>> C.chebint(c,lbnd=-2) 

1024 array([ 8.5, -0.5, 0.5, 0.5]) 

1025 >>> C.chebint(c,scl=-2) 

1026 array([-1., 1., -1., -1.]) 

1027 

1028 """ 

1029 c = np.array(c, ndmin=1, copy=True) 

1030 if c.dtype.char in '?bBhHiIlLqQpP': 

1031 c = c.astype(np.double) 

1032 if not np.iterable(k): 

1033 k = [k] 

1034 cnt = pu._deprecate_as_int(m, "the order of integration") 

1035 iaxis = pu._deprecate_as_int(axis, "the axis") 

1036 if cnt < 0: 

1037 raise ValueError("The order of integration must be non-negative") 

1038 if len(k) > cnt: 

1039 raise ValueError("Too many integration constants") 

1040 if np.ndim(lbnd) != 0: 

1041 raise ValueError("lbnd must be a scalar.") 

1042 if np.ndim(scl) != 0: 

1043 raise ValueError("scl must be a scalar.") 

1044 iaxis = normalize_axis_index(iaxis, c.ndim) 

1045 

1046 if cnt == 0: 

1047 return c 

1048 

1049 c = np.moveaxis(c, iaxis, 0) 

1050 k = list(k) + [0]*(cnt - len(k)) 

1051 for i in range(cnt): 

1052 n = len(c) 

1053 c *= scl 

1054 if n == 1 and np.all(c[0] == 0): 

1055 c[0] += k[i] 

1056 else: 

1057 tmp = np.empty((n + 1,) + c.shape[1:], dtype=c.dtype) 

1058 tmp[0] = c[0]*0 

1059 tmp[1] = c[0] 

1060 if n > 1: 

1061 tmp[2] = c[1]/4 

1062 for j in range(2, n): 

1063 t = c[j]/(2*j + 1) # FIXME: t never used 

1064 tmp[j + 1] = c[j]/(2*(j + 1)) 

1065 tmp[j - 1] -= c[j]/(2*(j - 1)) 

1066 tmp[0] += k[i] - chebval(lbnd, tmp) 

1067 c = tmp 

1068 c = np.moveaxis(c, 0, iaxis) 

1069 return c

1072def chebval(x, c, tensor=True): 

1073 """ 

1074 Evaluate a Chebyshev series at points x. 

1075 

1076 If `c` is of length `n + 1`, this function returns the value: 

1077 

1078 .. math:: p(x) = c_0 * T_0(x) + c_1 * T_1(x) + ... + c_n * T_n(x) 

1079 

1080 The parameter `x` is converted to an array only if it is a tuple or a 

1081 list, otherwise it is treated as a scalar. In either case, either `x` 

1082 or its elements must support multiplication and addition both with 

1083 themselves and with the elements of `c`. 

1084 

1085 If `c` is a 1-D array, then `p(x)` will have the same shape as `x`. If 

1086 `c` is multidimensional, then the shape of the result depends on the 

1087 value of `tensor`. If `tensor` is true the shape will be c.shape[1:] + 

1088 x.shape. If `tensor` is false the shape will be c.shape[1:]. Note that 

1089 scalars have shape (,). 

1090 

1091 Trailing zeros in the coefficients will be used in the evaluation, so 

1092 they should be avoided if efficiency is a concern. 

1093 

1094 Parameters 

1095 ---------- 

1096 x : array_like, compatible object 

1097 If `x` is a list or tuple, it is converted to an ndarray, otherwise 

1098 it is left unchanged and treated as a scalar. In either case, `x` 

1099 or its elements must support addition and multiplication with 

1100 with themselves and with the elements of `c`. 

1101 c : array_like 

1102 Array of coefficients ordered so that the coefficients for terms of 

1103 degree n are contained in c[n]. If `c` is multidimensional the 

1104 remaining indices enumerate multiple polynomials. In the two 

1105 dimensional case the coefficients may be thought of as stored in 

1106 the columns of `c`. 

1107 tensor : boolean, optional 

1108 If True, the shape of the coefficient array is extended with ones 

1109 on the right, one for each dimension of `x`. Scalars have dimension 0 

1110 for this action. The result is that every column of coefficients in 

1111 `c` is evaluated for every element of `x`. If False, `x` is broadcast 

1112 over the columns of `c` for the evaluation. This keyword is useful 

1113 when `c` is multidimensional. The default value is True. 

1114 

1115 .. versionadded:: 1.7.0 

1116 

1117 Returns 

1118 ------- 

1119 values : ndarray, algebra_like 

1120 The shape of the return value is described above. 

1121 

1122 See Also 

1123 -------- 

1124 chebval2d, chebgrid2d, chebval3d, chebgrid3d 

1125 

1126 Notes 

1127 ----- 

1128 The evaluation uses Clenshaw recursion, aka synthetic division. 

1129 

1130 Examples 

1131 -------- 

1132 

1133 """ 

1134 c = np.array(c, ndmin=1, copy=True) 

1135 if c.dtype.char in '?bBhHiIlLqQpP': 

1136 c = c.astype(np.double) 

1137 if isinstance(x, (tuple, list)): 

1138 x = np.asarray(x) 

1139 if isinstance(x, np.ndarray) and tensor: 

1140 c = c.reshape(c.shape + (1,)*x.ndim) 

1141 

1142 if len(c) == 1: 

1143 c0 = c[0] 

1144 c1 = 0 

1145 elif len(c) == 2: 

1146 c0 = c[0] 

1147 c1 = c[1] 

1148 else: 

1149 x2 = 2*x 

1150 c0 = c[-2] 

1151 c1 = c[-1] 

1152 for i in range(3, len(c) + 1): 

1153 tmp = c0 

1154 c0 = c[-i] - c1 

1155 c1 = tmp + c1*x2 

1156 return c0 + c1*x

1368def chebvander(x, deg): 

1369 """Pseudo-Vandermonde matrix of given degree. 

1370 

1371 Returns the pseudo-Vandermonde matrix of degree `deg` and sample points 

1372 `x`. The pseudo-Vandermonde matrix is defined by 

1373 

1374 .. math:: V[..., i] = T_i(x), 

1375 

1376 where `0 <= i <= deg`. The leading indices of `V` index the elements of 

1377 `x` and the last index is the degree of the Chebyshev polynomial. 

1378 

1379 If `c` is a 1-D array of coefficients of length `n + 1` and `V` is the 

1380 matrix ``V = chebvander(x, n)``, then ``np.dot(V, c)`` and 

1381 ``chebval(x, c)`` are the same up to roundoff. This equivalence is 

1382 useful both for least squares fitting and for the evaluation of a large 

1383 number of Chebyshev series of the same degree and sample points. 

1384 

1385 Parameters 

1386 ---------- 

1387 x : array_like 

1388 Array of points. The dtype is converted to float64 or complex128 

1389 depending on whether any of the elements are complex. If `x` is 

1390 scalar it is converted to a 1-D array. 

1391 deg : int 

1392 Degree of the resulting matrix. 

1393 

1394 Returns 

1395 ------- 

1396 vander : ndarray 

1397 The pseudo Vandermonde matrix. The shape of the returned matrix is 

1398 ``x.shape + (deg + 1,)``, where The last index is the degree of the 

1399 corresponding Chebyshev polynomial. The dtype will be the same as 

1400 the converted `x`. 

1401 

1402 """ 

1403 ideg = pu._deprecate_as_int(deg, "deg") 

1404 if ideg < 0: 

1405 raise ValueError("deg must be non-negative") 

1406 

1407 x = np.array(x, copy=False, ndmin=1) + 0.0 

1408 dims = (ideg + 1,) + x.shape 

1409 dtyp = x.dtype 

1410 v = np.empty(dims, dtype=dtyp) 

1411 # Use forward recursion to generate the entries. 

1412 v[0] = x*0 + 1 

1413 if ideg > 0: 

1414 x2 = 2*x 

1415 v[1] = x 

1416 for i in range(2, ideg + 1): 

1417 v[i] = v[i-1]*x2 - v[i-2] 

1418 return np.moveaxis(v, 0, -1)

1528def chebfit(x, y, deg, rcond=None, full=False, w=None): 

1529 """ 

1530 Least squares fit of Chebyshev series to data. 

1531 

1532 Return the coefficients of a Chebyshev series of degree `deg` that is the 

1533 least squares fit to the data values `y` given at points `x`. If `y` is 

1534 1-D the returned coefficients will also be 1-D. If `y` is 2-D multiple 

1535 fits are done, one for each column of `y`, and the resulting 

1536 coefficients are stored in the corresponding columns of a 2-D return. 

1537 The fitted polynomial(s) are in the form 

1538 

1539 .. math:: p(x) = c_0 + c_1 * T_1(x) + ... + c_n * T_n(x), 

1540 

1541 where `n` is `deg`. 

1542 

1543 Parameters 

1544 ---------- 

1545 x : array_like, shape (M,) 

1546 x-coordinates of the M sample points ``(x[i], y[i])``. 

1547 y : array_like, shape (M,) or (M, K) 

1548 y-coordinates of the sample points. Several data sets of sample 

1549 points sharing the same x-coordinates can be fitted at once by 

1550 passing in a 2D-array that contains one dataset per column. 

1551 deg : int or 1-D array_like 

1552 Degree(s) of the fitting polynomials. If `deg` is a single integer, 

1553 all terms up to and including the `deg`'th term are included in the 

1554 fit. For NumPy versions >= 1.11.0 a list of integers specifying the 

1555 degrees of the terms to include may be used instead. 

1556 rcond : float, optional 

1557 Relative condition number of the fit. Singular values smaller than 

1558 this relative to the largest singular value will be ignored. The 

1559 default value is len(x)*eps, where eps is the relative precision of 

1560 the float type, about 2e-16 in most cases. 

1561 full : bool, optional 

1562 Switch determining nature of return value. When it is False (the 

1563 default) just the coefficients are returned, when True diagnostic 

1564 information from the singular value decomposition is also returned. 

1565 w : array_like, shape (`M`,), optional 

1566 Weights. If not None, the contribution of each point 

1567 ``(x[i],y[i])`` to the fit is weighted by `w[i]`. Ideally the 

1568 weights are chosen so that the errors of the products ``w[i]*y[i]`` 

1569 all have the same variance. The default value is None. 

1570 

1571 .. versionadded:: 1.5.0 

1572 

1573 Returns 

1574 ------- 

1575 coef : ndarray, shape (M,) or (M, K) 

1576 Chebyshev coefficients ordered from low to high. If `y` was 2-D, 

1577 the coefficients for the data in column k of `y` are in column 

1578 `k`. 

1579 

1580 [residuals, rank, singular_values, rcond] : list 

1581 These values are only returned if `full` = True 

1582 

1583 resid -- sum of squared residuals of the least squares fit 

1584 rank -- the numerical rank of the scaled Vandermonde matrix 

1585 sv -- singular values of the scaled Vandermonde matrix 

1586 rcond -- value of `rcond`. 

1587 

1588 For more details, see `linalg.lstsq`. 

1589 

1590 Warns 

1591 ----- 

1592 RankWarning 

1593 The rank of the coefficient matrix in the least-squares fit is 

1594 deficient. The warning is only raised if `full` = False. The 

1595 warnings can be turned off by 

1596 

1597 >>> import warnings 

1598 >>> warnings.simplefilter('ignore', np.RankWarning) 

1599 

1600 See Also 

1601 -------- 

1602 polyfit, legfit, lagfit, hermfit, hermefit 

1603 chebval : Evaluates a Chebyshev series. 

1604 chebvander : Vandermonde matrix of Chebyshev series. 

1605 chebweight : Chebyshev weight function. 

1606 linalg.lstsq : Computes a least-squares fit from the matrix. 

1607 scipy.interpolate.UnivariateSpline : Computes spline fits. 

1608 

1609 Notes 

1610 ----- 

1611 The solution is the coefficients of the Chebyshev series `p` that 

1612 minimizes the sum of the weighted squared errors 

1613 

1614 .. math:: E = \\sum_j w_j^2 * |y_j - p(x_j)|^2, 

1615 

1616 where :math:`w_j` are the weights. This problem is solved by setting up 

1617 as the (typically) overdetermined matrix equation 

1618 

1619 .. math:: V(x) * c = w * y, 

1620 

1621 where `V` is the weighted pseudo Vandermonde matrix of `x`, `c` are the 

1622 coefficients to be solved for, `w` are the weights, and `y` are the 

1623 observed values. This equation is then solved using the singular value 

1624 decomposition of `V`. 

1625 

1626 If some of the singular values of `V` are so small that they are 

1627 neglected, then a `RankWarning` will be issued. This means that the 

1628 coefficient values may be poorly determined. Using a lower order fit 

1629 will usually get rid of the warning. The `rcond` parameter can also be 

1630 set to a value smaller than its default, but the resulting fit may be 

1631 spurious and have large contributions from roundoff error. 

1632 

1633 Fits using Chebyshev series are usually better conditioned than fits 

1634 using power series, but much can depend on the distribution of the 

1635 sample points and the smoothness of the data. If the quality of the fit 

1636 is inadequate splines may be a good alternative. 

1637 

1638 References 

1639 ---------- 

1640 .. [1] Wikipedia, "Curve fitting", 

1641 https://en.wikipedia.org/wiki/Curve_fitting 

1642 

1643 Examples 

1644 -------- 

1645 

1646 """ 

1647 return pu._fit(chebvander, x, y, deg, rcond, full, w)

1650def chebcompanion(c): 

1651 """Return the scaled companion matrix of c. 

1652 

1653 The basis polynomials are scaled so that the companion matrix is 

1654 symmetric when `c` is a Chebyshev basis polynomial. This provides 

1655 better eigenvalue estimates than the unscaled case and for basis 

1656 polynomials the eigenvalues are guaranteed to be real if 

1657 `numpy.linalg.eigvalsh` is used to obtain them. 

1658 

1659 Parameters 

1660 ---------- 

1661 c : array_like 

1662 1-D array of Chebyshev series coefficients ordered from low to high 

1663 degree. 

1664 

1665 Returns 

1666 ------- 

1667 mat : ndarray 

1668 Scaled companion matrix of dimensions (deg, deg). 

1669 

1670 Notes 

1671 ----- 

1672 

1673 .. versionadded:: 1.7.0 

1674 

1675 """ 

1676 # c is a trimmed copy 

1677 [c] = pu.as_series([c]) 

1678 if len(c) < 2: 

1679 raise ValueError('Series must have maximum degree of at least 1.') 

1680 if len(c) == 2: 

1681 return np.array([[-c[0]/c[1]]]) 

1682 

1683 n = len(c) - 1 

1684 mat = np.zeros((n, n), dtype=c.dtype) 

1685 scl = np.array([1.] + [np.sqrt(.5)]*(n-1)) 

1686 top = mat.reshape(-1)[1::n+1] 

1687 bot = mat.reshape(-1)[n::n+1] 

1688 top[0] = np.sqrt(.5) 

1689 top[1:] = 1/2 

1690 bot[...] = top 

1691 mat[:, -1] -= (c[:-1]/c[-1])*(scl/scl[-1])*.5 

1692 return mat

1695def chebroots(c): 

1696 """ 

1697 Compute the roots of a Chebyshev series. 

1698 

1699 Return the roots (a.k.a. "zeros") of the polynomial 

1700 

1701 .. math:: p(x) = \\sum_i c[i] * T_i(x). 

1702 

1703 Parameters 

1704 ---------- 

1705 c : 1-D array_like 

1706 1-D array of coefficients. 

1707 

1708 Returns 

1709 ------- 

1710 out : ndarray 

1711 Array of the roots of the series. If all the roots are real, 

1712 then `out` is also real, otherwise it is complex. 

1713 

1714 See Also 

1715 -------- 

1716 polyroots, legroots, lagroots, hermroots, hermeroots 

1717 

1718 Notes 

1719 ----- 

1720 The root estimates are obtained as the eigenvalues of the companion 

1721 matrix, Roots far from the origin of the complex plane may have large 

1722 errors due to the numerical instability of the series for such 

1723 values. Roots with multiplicity greater than 1 will also show larger 

1724 errors as the value of the series near such points is relatively 

1725 insensitive to errors in the roots. Isolated roots near the origin can 

1726 be improved by a few iterations of Newton's method. 

1727 

1728 The Chebyshev series basis polynomials aren't powers of `x` so the 

1729 results of this function may seem unintuitive. 

1730 

1731 Examples 

1732 -------- 

1733 >>> import numpy.polynomial.chebyshev as cheb 

1734 >>> cheb.chebroots((-1, 1,-1, 1)) # T3 - T2 + T1 - T0 has real roots 

1735 array([ -5.00000000e-01, 2.60860684e-17, 1.00000000e+00]) # may vary 

1736 

1737 """ 

1738 # c is a trimmed copy 

1739 [c] = pu.as_series([c]) 

1740 if len(c) < 2: 

1741 return np.array([], dtype=c.dtype) 

1742 if len(c) == 2: 

1743 return np.array([-c[0]/c[1]]) 

1744 

1745 # rotated companion matrix reduces error 

1746 m = chebcompanion(c)[::-1,::-1] 

1747 r = la.eigvals(m) 

1748 r.sort() 

1749 return r

1752def chebinterpolate(func, deg, args=()): 

1753 """Interpolate a function at the Chebyshev points of the first kind. 

1754 

1755 Returns the Chebyshev series that interpolates `func` at the Chebyshev 

1756 points of the first kind in the interval [-1, 1]. The interpolating 

1757 series tends to a minmax approximation to `func` with increasing `deg` 

1758 if the function is continuous in the interval. 

1759 

1760 .. versionadded:: 1.14.0 

1761 

1762 Parameters 

1763 ---------- 

1764 func : function 

1765 The function to be approximated. It must be a function of a single 

1766 variable of the form ``f(x, a, b, c...)``, where ``a, b, c...`` are 

1767 extra arguments passed in the `args` parameter. 

1768 deg : int 

1769 Degree of the interpolating polynomial 

1770 args : tuple, optional 

1771 Extra arguments to be used in the function call. Default is no extra 

1772 arguments. 

1773 

1774 Returns 

1775 ------- 

1776 coef : ndarray, shape (deg + 1,) 

1777 Chebyshev coefficients of the interpolating series ordered from low to 

1778 high. 

1779 

1780 Examples 

1781 -------- 

1782 >>> import numpy.polynomial.chebyshev as C 

1783 >>> C.chebfromfunction(lambda x: np.tanh(x) + 0.5, 8) 

1784 array([ 5.00000000e-01, 8.11675684e-01, -9.86864911e-17, 

1785 -5.42457905e-02, -2.71387850e-16, 4.51658839e-03, 

1786 2.46716228e-17, -3.79694221e-04, -3.26899002e-16]) 

1787 

1788 Notes 

1789 ----- 

1790 

1791 The Chebyshev polynomials used in the interpolation are orthogonal when 

1792 sampled at the Chebyshev points of the first kind. If it is desired to 

1793 constrain some of the coefficients they can simply be set to the desired 

1794 value after the interpolation, no new interpolation or fit is needed. This 

1795 is especially useful if it is known apriori that some of coefficients are 

1796 zero. For instance, if the function is even then the coefficients of the 

1797 terms of odd degree in the result can be set to zero. 

1798 

1799 """ 

1800 deg = np.asarray(deg) 

1801 

1802 # check arguments. 

1803 if deg.ndim > 0 or deg.dtype.kind not in 'iu' or deg.size == 0: 

1804 raise TypeError("deg must be an int") 

1805 if deg < 0: 

1806 raise ValueError("expected deg >= 0") 

1807 

1808 order = deg + 1 

1809 xcheb = chebpts1(order) 

1810 yfunc = func(xcheb, *args) 

1811 m = chebvander(xcheb, deg) 

1812 c = np.dot(m.T, yfunc) 

1813 c[0] /= order 

1814 c[1:] /= 0.5*order 

1815 

1816 return c

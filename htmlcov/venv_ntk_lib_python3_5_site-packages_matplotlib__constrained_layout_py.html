1""" 

2This module provides the routine to adjust subplot layouts so that there are 

3no overlapping axes or axes decorations. All axes decorations are dealt with 

4(labels, ticks, titles, ticklabels) and some dependent artists are also dealt 

5with (colorbar, suptitle, legend). 

6 

7Layout is done via :meth:`~matplotlib.gridspec`, with one constraint per 

8gridspec, so it is possible to have overlapping axes if the gridspecs 

9overlap (i.e. using :meth:`~matplotlib.gridspec.GridSpecFromSubplotSpec`). 

10Axes placed using ``figure.subplots()`` or ``figure.add_subplots()`` will 

11participate in the layout. Axes manually placed via ``figure.add_axes()`` 

12will not. 

13 

14See Tutorial: :doc:`/tutorials/intermediate/constrainedlayout_guide` 

15 

16"""

83def do_constrained_layout(fig, renderer, h_pad, w_pad, 

84 hspace=None, wspace=None): 

85 

86 """ 

87 Do the constrained_layout. Called at draw time in 

88 ``figure.constrained_layout()`` 

89 

90 Parameters 

91 ---------- 

92 

93 

94 fig: Figure 

95 is the ``figure`` instance to do the layout in. 

96 

97 renderer: Renderer 

98 the renderer to use. 

99 

100 h_pad, w_pad : float 

101 are in figure-normalized units, and are a padding around the axes 

102 elements. 

103 

104 hspace, wspace : float 

105 are in fractions of the subplot sizes. 

106 

107 """ 

108 

109 ''' Steps: 

110 

111 1. get a list of unique gridspecs in this figure. Each gridspec will be 

112 constrained separately. 

113 2. Check for gaps in the gridspecs. i.e. if not every axes slot in the 

114 gridspec has been filled. If empty, add a ghost axis that is made so 

115 that it cannot be seen (though visible=True). This is needed to make 

116 a blank spot in the layout. 

117 3. Compare the tight_bbox of each axes to its `position`, and assume that 

118 the difference is the space needed by the elements around the edge of 

119 the axes (decorations) like the title, ticklabels, x-labels, etc. This 

120 can include legends who overspill the axes boundaries. 

121 4. Constrain gridspec elements to line up: 

122 a) if colnum0 neq colnumC, the two subplotspecs are stacked next to 

123 each other, with the appropriate order. 

124 b) if colnum0 == columnC line up the left or right side of the 

125 _poslayoutbox (depending if it is the min or max num that is equal). 

126 c) do the same for rows... 

127 5. The above doesn't constrain relative sizes of the _poslayoutboxes at 

128 all, and indeed zero-size is a solution that the solver often finds more 

129 convenient than expanding the sizes. Right now the solution is to compare 

130 subplotspec sizes (i.e. drowsC and drows0) and constrain the larger 

131 _poslayoutbox to be larger than the ratio of the sizes. i.e. if drows0 > 

132 drowsC, then ax._poslayoutbox > axc._poslayoutbox * drowsC / drows0. This 

133 works fine *if* the decorations are similar between the axes. If the 

134 larger subplotspec has much larger axes decorations, then the constraint 

135 above is incorrect. 

136 

137 We need the greater than in the above, in general, rather than an equals 

138 sign. Consider the case of the left column having 2 rows, and the right 

139 column having 1 row. We want the top and bottom of the _poslayoutboxes to 

140 line up. So that means if there are decorations on the left column axes 

141 they will be smaller than half as large as the right hand axis. 

142 

143 This can break down if the decoration size for the right hand axis (the 

144 margins) is very large. There must be a math way to check for this case. 

145 

146 ''' 

147 

148 invTransFig = fig.transFigure.inverted().transform_bbox 

149 

150 # list of unique gridspecs that contain child axes: 

151 gss = set() 

152 for ax in fig.axes: 

153 if hasattr(ax, 'get_subplotspec'): 

154 gs = ax.get_subplotspec().get_gridspec() 

155 if gs._layoutbox is not None: 

156 gss.add(gs) 

157 if len(gss) == 0: 

158 warnings.warn('There are no gridspecs with layoutboxes. ' 

159 'Possibly did not call parent GridSpec with the figure= ' 

160 'keyword') 

161 

162 if fig._layoutbox.constrained_layout_called < 1: 

163 for gs in gss: 

164 # fill in any empty gridspec slots w/ ghost axes... 

165 _make_ghost_gridspec_slots(fig, gs) 

166 

167 for nnn in range(2): 

168 # do the algrithm twice. This has to be done because decorators 

169 # change size after the first re-position (i.e. x/yticklabels get 

170 # larger/smaller). This second reposition tends to be much milder, 

171 # so doing twice makes things work OK. 

172 for ax in fig.axes: 

173 _log.debug(ax._layoutbox) 

174 if ax._layoutbox is not None: 

175 # make margins for each layout box based on the size of 

176 # the decorators. 

177 _make_layout_margins(ax, renderer, h_pad, w_pad) 

178 

179 # do layout for suptitle. 

180 if fig._suptitle is not None and fig._suptitle._layoutbox is not None: 

181 sup = fig._suptitle 

182 bbox = invTransFig(sup.get_window_extent(renderer=renderer)) 

183 height = bbox.y1 - bbox.y0 

184 sup._layoutbox.edit_height(height+h_pad) 

185 

186 # OK, the above lines up ax._poslayoutbox with ax._layoutbox 

187 # now we need to 

188 # 1) arrange the subplotspecs. We do it at this level because 

189 # the subplotspecs are meant to contain other dependent axes 

190 # like colorbars or legends. 

191 # 2) line up the right and left side of the ax._poslayoutbox 

192 # that have the same subplotspec maxes. 

193 

194 if fig._layoutbox.constrained_layout_called < 1: 

195 # arrange the subplotspecs... This is all done relative to each 

196 # other. Some subplotspecs conatain axes, and others contain 

197 # gridspecs the ones that contain gridspecs are a set proportion 

198 # of their parent gridspec. The ones that contain axes are 

199 # not so constrained. 

200 figlb = fig._layoutbox 

201 for child in figlb.children: 

202 if child._is_gridspec_layoutbox(): 

203 # This routine makes all the subplot spec containers 

204 # have the correct arrangement. It just stacks the 

205 # subplot layoutboxes in the correct order... 

206 _arrange_subplotspecs(child, hspace=hspace, wspace=wspace) 

207 

208 for gs in gss: 

209 _align_spines(fig, gs) 

210 

211 fig._layoutbox.constrained_layout_called += 1 

212 fig._layoutbox.update_variables() 

213 

214 # check if any axes collapsed to zero. If not, don't change positions: 

215 if _axes_all_finite_sized(fig): 

216 # Now set the position of the axes... 

217 for ax in fig.axes: 

218 if ax._layoutbox is not None: 

219 newpos = ax._poslayoutbox.get_rect() 

220 # Now set the new position. 

221 # ax.set_position will zero out the layout for 

222 # this axis, allowing users to hard-code the position, 

223 # so this does the same w/o zeroing layout. 

224 ax._set_position(newpos, which='original') 

225 else: 

226 warnings.warn('constrained_layout not applied. At least ' 

227 'one axes collapsed to zero width or height.')

230def _make_ghost_gridspec_slots(fig, gs): 

231 """ 

232 Check for unoccupied gridspec slots and make ghost axes for these 

233 slots... Do for each gs separately. This is a pretty big kludge 

234 but shoudn't have too much ill effect. The worst is that 

235 someone querrying the figure will wonder why there are more 

236 axes than they thought. 

237 """ 

238 nrows, ncols = gs.get_geometry() 

239 hassubplotspec = np.zeros(nrows * ncols, dtype=bool) 

240 axs = [] 

241 for ax in fig.axes: 

242 if (hasattr(ax, 'get_subplotspec') 

243 and ax._layoutbox is not None 

244 and ax.get_subplotspec().get_gridspec() == gs): 

245 axs += [ax] 

246 for ax in axs: 

247 ss0 = ax.get_subplotspec() 

248 if ss0.num2 is None: 

249 ss0.num2 = ss0.num1 

250 hassubplotspec[ss0.num1:(ss0.num2 + 1)] = True 

251 for nn, hss in enumerate(hassubplotspec): 

252 if not hss: 

253 # this gridspec slot doesn't have an axis so we 

254 # make a "ghost". 

255 ax = fig.add_subplot(gs[nn]) 

256 ax.set_frame_on(False) 

257 ax.set_xticks([]) 

258 ax.set_yticks([]) 

259 ax.set_facecolor((1, 0, 0, 0))

262def _make_layout_margins(ax, renderer, h_pad, w_pad): 

263 """ 

264 For each axes, make a margin between the *pos* layoutbox and the 

265 *axes* layoutbox be a minimum size that can accommodate the 

266 decorations on the axis. 

267 """ 

268 fig = ax.figure 

269 invTransFig = fig.transFigure.inverted().transform_bbox 

270 

271 pos = ax.get_position(original=True) 

272 tightbbox = ax.get_tightbbox(renderer=renderer) 

273 bbox = invTransFig(tightbbox) 

274 # use stored h_pad if it exists 

275 h_padt = ax._poslayoutbox.h_pad 

276 if h_padt is None: 

277 h_padt = h_pad 

278 w_padt = ax._poslayoutbox.w_pad 

279 if w_padt is None: 

280 w_padt = w_pad 

281 ax._poslayoutbox.edit_left_margin_min(-bbox.x0 + 

282 pos.x0 + w_padt) 

283 ax._poslayoutbox.edit_right_margin_min(bbox.x1 - 

284 pos.x1 + w_padt) 

285 ax._poslayoutbox.edit_bottom_margin_min( 

286 -bbox.y0 + pos.y0 + h_padt) 

287 ax._poslayoutbox.edit_top_margin_min(bbox.y1-pos.y1+h_padt) 

288 _log.debug('left %f', (-bbox.x0 + pos.x0 + w_pad)) 

289 _log.debug('right %f', (bbox.x1 - pos.x1 + w_pad)) 

290 _log.debug('bottom %f', (-bbox.y0 + pos.y0 + h_padt)) 

291 # Sometimes its possible for the solver to collapse 

292 # rather than expand axes, so they all have zero height 

293 # or width. This stops that... It *should* have been 

294 # taken into account w/ pref_width... 

295 if fig._layoutbox.constrained_layout_called < 1: 

296 ax._poslayoutbox.constrain_height_min(20, strength='weak') 

297 ax._poslayoutbox.constrain_width_min(20, strength='weak') 

298 ax._layoutbox.constrain_height_min(20, strength='weak') 

299 ax._layoutbox.constrain_width_min(20, strength='weak') 

300 ax._poslayoutbox.constrain_top_margin(0, strength='weak') 

301 ax._poslayoutbox.constrain_bottom_margin(0, 

302 strength='weak') 

303 ax._poslayoutbox.constrain_right_margin(0, strength='weak') 

304 ax._poslayoutbox.constrain_left_margin(0, strength='weak')

307def _align_spines(fig, gs): 

308 """ 

309 - Align right/left and bottom/top spines of appropriate subplots. 

310 - Compare size of subplotspec including height and width ratios 

311 and make sure that the axes spines are at least as large 

312 as they should be. 

313 """ 

314 # for each gridspec... 

315 nrows, ncols = gs.get_geometry() 

316 width_ratios = gs.get_width_ratios() 

317 height_ratios = gs.get_height_ratios() 

318 if width_ratios is None: 

319 width_ratios = np.ones(ncols) 

320 if height_ratios is None: 

321 height_ratios = np.ones(nrows) 

322 

323 # get axes in this gridspec.... 

324 axs = [] 

325 for ax in fig.axes: 

326 if (hasattr(ax, 'get_subplotspec') 

327 and ax._layoutbox is not None): 

328 if ax.get_subplotspec().get_gridspec() == gs: 

329 axs += [ax] 

330 rownummin = np.zeros(len(axs), dtype=np.int8) 

331 rownummax = np.zeros(len(axs), dtype=np.int8) 

332 colnummin = np.zeros(len(axs), dtype=np.int8) 

333 colnummax = np.zeros(len(axs), dtype=np.int8) 

334 width = np.zeros(len(axs)) 

335 height = np.zeros(len(axs)) 

336 

337 for n, ax in enumerate(axs): 

338 ss0 = ax.get_subplotspec() 

339 if ss0.num2 is None: 

340 ss0.num2 = ss0.num1 

341 rownummin[n], colnummin[n] = divmod(ss0.num1, ncols) 

342 rownummax[n], colnummax[n] = divmod(ss0.num2, ncols) 

343 width[n] = np.sum( 

344 width_ratios[colnummin[n]:(colnummax[n] + 1)]) 

345 height[n] = np.sum( 

346 height_ratios[rownummin[n]:(rownummax[n] + 1)]) 

347 

348 for nn, ax in enumerate(axs[:-1]): 

349 ss0 = ax.get_subplotspec() 

350 

351 # now compare ax to all the axs: 

352 # 

353 # If the subplotspecs have the same colnumXmax, then line 

354 # up their right sides. If they have the same min, then 

355 # line up their left sides (and vertical equivalents). 

356 rownum0min, colnum0min = rownummin[nn], colnummin[nn] 

357 rownum0max, colnum0max = rownummax[nn], colnummax[nn] 

358 width0, height0 = width[nn], height[nn] 

359 alignleft = False 

360 alignright = False 

361 alignbot = False 

362 aligntop = False 

363 alignheight = False 

364 alignwidth = False 

365 for mm in range(nn+1, len(axs)): 

366 axc = axs[mm] 

367 rownumCmin, colnumCmin = rownummin[mm], colnummin[mm] 

368 rownumCmax, colnumCmax = rownummax[mm], colnummax[mm] 

369 widthC, heightC = width[mm], height[mm] 

370 # Horizontally align axes spines if they have the 

371 # same min or max: 

372 if not alignleft and colnum0min == colnumCmin: 

373 # we want the _poslayoutboxes to line up on left 

374 # side of the axes spines... 

375 layoutbox.align([ax._poslayoutbox, 

376 axc._poslayoutbox], 

377 'left') 

378 alignleft = True 

379 

380 if not alignright and colnum0max == colnumCmax: 

381 # line up right sides of _poslayoutbox 

382 layoutbox.align([ax._poslayoutbox, 

383 axc._poslayoutbox], 

384 'right') 

385 alignright = True 

386 # Vertically align axes spines if they have the 

387 # same min or max: 

388 if not aligntop and rownum0min == rownumCmin: 

389 # line up top of _poslayoutbox 

390 _log.debug('rownum0min == rownumCmin') 

391 layoutbox.align([ax._poslayoutbox, axc._poslayoutbox], 

392 'top') 

393 aligntop = True 

394 

395 if not alignbot and rownum0max == rownumCmax: 

396 # line up bottom of _poslayoutbox 

397 _log.debug('rownum0max == rownumCmax') 

398 layoutbox.align([ax._poslayoutbox, axc._poslayoutbox], 

399 'bottom') 

400 alignbot = True 

401 ########### 

402 # Now we make the widths and heights of position boxes 

403 # similar. (i.e the spine locations) 

404 # This allows vertically stacked subplots to have 

405 # different sizes if they occupy different amounts 

406 # of the gridspec: i.e. 

407 # gs = gridspec.GridSpec(3,1) 

408 # ax1 = gs[0,:] 

409 # ax2 = gs[1:,:] 

410 # then drows0 = 1, and drowsC = 2, and ax2 

411 # should be at least twice as large as ax1. 

412 # But it can be more than twice as large because 

413 # it needs less room for the labeling. 

414 # 

415 # For height, this only needs to be done if the 

416 # subplots share a column. For width if they 

417 # share a row. 

418 

419 drowsC = (rownumCmax - rownumCmin + 1) 

420 drows0 = (rownum0max - rownum0min + 1) 

421 dcolsC = (colnumCmax - colnumCmin + 1) 

422 dcols0 = (colnum0max - colnum0min + 1) 

423 

424 if not alignheight and drows0 == drowsC: 

425 ax._poslayoutbox.constrain_height( 

426 axc._poslayoutbox.height * height0 / heightC) 

427 alignheight = True 

428 elif _in_same_column(colnum0min, colnum0max, 

429 colnumCmin, colnumCmax): 

430 if height0 > heightC: 

431 ax._poslayoutbox.constrain_height_min( 

432 axc._poslayoutbox.height * height0 / heightC) 

433 # these constraints stop the smaller axes from 

434 # being allowed to go to zero height... 

435 axc._poslayoutbox.constrain_height_min( 

436 ax._poslayoutbox.height * heightC / 

437 (height0*1.8)) 

438 elif height0 < heightC: 

439 axc._poslayoutbox.constrain_height_min( 

440 ax._poslayoutbox.height * heightC / height0) 

441 ax._poslayoutbox.constrain_height_min( 

442 ax._poslayoutbox.height * height0 / 

443 (heightC*1.8)) 

444 # widths... 

445 if not alignwidth and dcols0 == dcolsC: 

446 ax._poslayoutbox.constrain_width( 

447 axc._poslayoutbox.width * width0 / widthC) 

448 alignwidth = True 

449 elif _in_same_row(rownum0min, rownum0max, 

450 rownumCmin, rownumCmax): 

451 if width0 > widthC: 

452 ax._poslayoutbox.constrain_width_min( 

453 axc._poslayoutbox.width * width0 / widthC) 

454 axc._poslayoutbox.constrain_width_min( 

455 ax._poslayoutbox.width * widthC / 

456 (width0*1.8)) 

457 elif width0 < widthC: 

458 axc._poslayoutbox.constrain_width_min( 

459 ax._poslayoutbox.width * widthC / width0) 

460 ax._poslayoutbox.constrain_width_min( 

461 axc._poslayoutbox.width * width0 / 

462 (widthC*1.8))

465def _arrange_subplotspecs(gs, hspace=0, wspace=0): 

466 """ 

467 arrange the subplotspec children of this gridspec, and then recursively 

468 do the same of any gridspec children of those gridspecs... 

469 """ 

470 sschildren = [] 

471 for child in gs.children: 

472 if child._is_subplotspec_layoutbox(): 

473 for child2 in child.children: 

474 # check for gridspec children... 

475 if child2._is_gridspec_layoutbox(): 

476 _arrange_subplotspecs(child2, hspace=hspace, wspace=wspace) 

477 sschildren += [child] 

478 # now arrange the subplots... 

479 for child0 in sschildren: 

480 ss0 = child0.artist 

481 nrows, ncols = ss0.get_gridspec().get_geometry() 

482 if ss0.num2 is None: 

483 ss0.num2 = ss0.num1 

484 rowNum0min, colNum0min = divmod(ss0.num1, ncols) 

485 rowNum0max, colNum0max = divmod(ss0.num2, ncols) 

486 sschildren = sschildren[1:] 

487 for childc in sschildren: 

488 ssc = childc.artist 

489 rowNumCmin, colNumCmin = divmod(ssc.num1, ncols) 

490 if ssc.num2 is None: 

491 ssc.num2 = ssc.num1 

492 rowNumCmax, colNumCmax = divmod(ssc.num2, ncols) 

493 # OK, this tells us the relative layout of ax 

494 # with axc 

495 thepad = wspace / ncols 

496 if colNum0max < colNumCmin: 

497 layoutbox.hstack([ss0._layoutbox, ssc._layoutbox], 

498 padding=thepad) 

499 if colNumCmax < colNum0min: 

500 layoutbox.hstack([ssc._layoutbox, ss0._layoutbox], 

501 padding=thepad) 

502 

503 #### 

504 # vertical alignment 

505 thepad = hspace / nrows 

506 if rowNum0max < rowNumCmin: 

507 layoutbox.vstack([ss0._layoutbox, 

508 ssc._layoutbox], 

509 padding=thepad) 

510 if rowNumCmax < rowNum0min: 

511 layoutbox.vstack([ssc._layoutbox, 

512 ss0._layoutbox], 

513 padding=thepad)

516def layoutcolorbarsingle(ax, cax, shrink, aspect, location, pad=0.05): 

517 """ 

518 Do the layout for a colorbar, to not oeverly pollute colorbar.py 

519 

520 `pad` is in fraction of the original axis size. 

521 """ 

522 axlb = ax._layoutbox 

523 axpos = ax._poslayoutbox 

524 axsslb = ax.get_subplotspec()._layoutbox 

525 lb = layoutbox.LayoutBox( 

526 parent=axsslb, 

527 name=axsslb.name + '.cbar', 

528 artist=cax) 

529 

530 if location in ('left', 'right'): 

531 lbpos = layoutbox.LayoutBox( 

532 parent=lb, 

533 name=lb.name + '.pos', 

534 tightwidth=False, 

535 pos=True, 

536 subplot=False, 

537 artist=cax) 

538 

539 if location == 'right': 

540 # arrange to right of parent axis 

541 layoutbox.hstack([axlb, lb], padding=pad * axlb.width, 

542 strength='strong') 

543 else: 

544 layoutbox.hstack([lb, axlb], padding=pad * axlb.width) 

545 # constrain the height and center... 

546 layoutbox.match_heights([axpos, lbpos], [1, shrink]) 

547 layoutbox.align([axpos, lbpos], 'v_center') 

548 # set the width of the pos box 

549 lbpos.constrain_width(shrink * axpos.height * (1/aspect), 

550 strength='strong') 

551 elif location in ('bottom', 'top'): 

552 lbpos = layoutbox.LayoutBox( 

553 parent=lb, 

554 name=lb.name + '.pos', 

555 tightheight=True, 

556 pos=True, 

557 subplot=False, 

558 artist=cax) 

559 

560 if location == 'bottom': 

561 layoutbox.vstack([axlb, lb], padding=pad * axlb.height) 

562 else: 

563 layoutbox.vstack([lb, axlb], padding=pad * axlb.height) 

564 # constrain the height and center... 

565 layoutbox.match_widths([axpos, lbpos], 

566 [1, shrink], strength='strong') 

567 layoutbox.align([axpos, lbpos], 'h_center') 

568 # set the height of the pos box 

569 lbpos.constrain_height(axpos.width * aspect * shrink, 

570 strength='medium') 

571 

572 return lb, lbpos

575def _getmaxminrowcolumn(axs): 

576 # helper to get the min/max rows and columns of a list of axes. 

577 maxrow = -100000 

578 minrow = 1000000 

579 maxax = None 

580 minax = None 

581 maxcol = -100000 

582 mincol = 1000000 

583 maxax_col = None 

584 minax_col = None 

585 

586 for ax in axs: 

587 subspec = ax.get_subplotspec() 

588 nrows, ncols, row_start, row_stop, col_start, col_stop = \ 

589 subspec.get_rows_columns() 

590 if row_stop > maxrow: 

591 maxrow = row_stop 

592 maxax = ax 

593 if row_start < minrow: 

594 minrow = row_start 

595 minax = ax 

596 if col_stop > maxcol: 

597 maxcol = col_stop 

598 maxax_col = ax 

599 if col_start < mincol: 

600 mincol = col_start 

601 minax_col = ax 

602 return (minrow, maxrow, minax, maxax, mincol, maxcol, minax_col, maxax_col)

605def layoutcolorbargridspec(parents, cax, shrink, aspect, location, pad=0.05): 

606 """ 

607 Do the layout for a colorbar, to not oeverly pollute colorbar.py 

608 

609 `pad` is in fraction of the original axis size. 

610 """ 

611 

612 gs = parents[0].get_subplotspec().get_gridspec() 

613 # parent layout box.... 

614 gslb = gs._layoutbox 

615 

616 lb = layoutbox.LayoutBox(parent=gslb.parent, 

617 name=gslb.parent.name + '.cbar', 

618 artist=cax) 

619 # figure out the row and column extent of the parents. 

620 (minrow, maxrow, minax_row, maxax_row, 

621 mincol, maxcol, minax_col, maxax_col) = _getmaxminrowcolumn(parents) 

622 

623 if location in ('left', 'right'): 

624 lbpos = layoutbox.LayoutBox( 

625 parent=lb, 

626 name=lb.name + '.pos', 

627 tightwidth=False, 

628 pos=True, 

629 subplot=False, 

630 artist=cax) 

631 for ax in parents: 

632 if location == 'right': 

633 order = [ax._layoutbox, lb] 

634 else: 

635 order = [lb, ax._layoutbox] 

636 layoutbox.hstack(order, padding=pad * gslb.width, 

637 strength='strong') 

638 # constrain the height and center... 

639 # This isn't quite right. We'd like the colorbar 

640 # pos to line up w/ the axes poss, not the size of the 

641 # gs. 

642 

643 # Horizontal Layout: need to check all the axes in this gridspec 

644 for ch in gslb.children: 

645 subspec = ch.artist 

646 nrows, ncols, row_start, row_stop, col_start, col_stop = \ 

647 subspec.get_rows_columns() 

648 if location == 'right': 

649 if col_stop <= maxcol: 

650 order = [subspec._layoutbox, lb] 

651 # arrange to right of the parents 

652 if col_start > maxcol: 

653 order = [lb, subspec._layoutbox] 

654 elif location == 'left': 

655 if col_start >= mincol: 

656 order = [lb, subspec._layoutbox] 

657 if col_stop < mincol: 

658 order = [subspec._layoutbox, lb] 

659 layoutbox.hstack(order, padding=pad * gslb.width, 

660 strength='strong') 

661 

662 # Vertical layout: 

663 maxposlb = minax_row._poslayoutbox 

664 minposlb = maxax_row._poslayoutbox 

665 # now we want the height of the colorbar pos to be 

666 # set by the top and bottom of the min/max axes... 

667 # bottom top 

668 # b t 

669 # h = (top-bottom)*shrink 

670 # b = bottom + (top-bottom - h) / 2. 

671 lbpos.constrain_height( 

672 (maxposlb.top - minposlb.bottom) * 

673 shrink, strength='strong') 

674 lbpos.constrain_bottom( 

675 (maxposlb.top - minposlb.bottom) * 

676 (1 - shrink)/2 + minposlb.bottom, 

677 strength='strong') 

678 

679 # set the width of the pos box 

680 lbpos.constrain_width(lbpos.height * (shrink / aspect), 

681 strength='strong') 

682 elif location in ('bottom', 'top'): 

683 lbpos = layoutbox.LayoutBox( 

684 parent=lb, 

685 name=lb.name + '.pos', 

686 tightheight=True, 

687 pos=True, 

688 subplot=False, 

689 artist=cax) 

690 

691 for ax in parents: 

692 if location == 'bottom': 

693 order = [ax._layoutbox, lb] 

694 else: 

695 order = [lb, ax._layoutbox] 

696 layoutbox.vstack(order, padding=pad * gslb.width, 

697 strength='strong') 

698 

699 # Vertical Layout: need to check all the axes in this gridspec 

700 for ch in gslb.children: 

701 subspec = ch.artist 

702 nrows, ncols, row_start, row_stop, col_start, col_stop = \ 

703 subspec.get_rows_columns() 

704 if location == 'bottom': 

705 if row_stop <= minrow: 

706 order = [subspec._layoutbox, lb] 

707 if row_start > maxrow: 

708 order = [lb, subspec._layoutbox] 

709 elif location == 'top': 

710 if row_stop < minrow: 

711 order = [subspec._layoutbox, lb] 

712 if row_start >= maxrow: 

713 order = [lb, subspec._layoutbox] 

714 layoutbox.vstack(order, padding=pad * gslb.width, 

715 strength='strong') 

716 

717 # Do horizontal layout... 

718 maxposlb = maxax_col._poslayoutbox 

719 minposlb = minax_col._poslayoutbox 

720 lbpos.constrain_width((maxposlb.right - minposlb.left) * 

721 shrink) 

722 lbpos.constrain_left( 

723 (maxposlb.right - minposlb.left) * 

724 (1-shrink)/2 + minposlb.left) 

725 # set the height of the pos box 

726 lbpos.constrain_height(lbpos.width * shrink * aspect, 

727 strength='medium') 

728 

729 return lb, lbpos

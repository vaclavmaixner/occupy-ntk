38def _ensure_data(values, dtype=None): 

39 """ 

40 routine to ensure that our data is of the correct 

41 input dtype for lower-level routines 

42 

43 This will coerce: 

44 - ints -> int64 

45 - uint -> uint64 

46 - bool -> uint64 (TODO this should be uint8) 

47 - datetimelike -> i8 

48 - datetime64tz -> i8 (in local tz) 

49 - categorical -> codes 

50 

51 Parameters 

52 ---------- 

53 values : array-like 

54 dtype : pandas_dtype, optional 

55 coerce to this dtype 

56 

57 Returns 

58 ------- 

59 (ndarray, pandas_dtype, algo dtype as a string) 

60 

61 """ 

62 

63 # we check some simple dtypes first 

64 try: 

65 if is_object_dtype(dtype): 

66 return ensure_object(np.asarray(values)), 'object', 'object' 

67 if is_bool_dtype(values) or is_bool_dtype(dtype): 

68 # we are actually coercing to uint64 

69 # until our algos support uint8 directly (see TODO) 

70 return np.asarray(values).astype('uint64'), 'bool', 'uint64' 

71 elif is_signed_integer_dtype(values) or is_signed_integer_dtype(dtype): 

72 return ensure_int64(values), 'int64', 'int64' 

73 elif (is_unsigned_integer_dtype(values) or 

74 is_unsigned_integer_dtype(dtype)): 

75 return ensure_uint64(values), 'uint64', 'uint64' 

76 elif is_float_dtype(values) or is_float_dtype(dtype): 

77 return ensure_float64(values), 'float64', 'float64' 

78 elif is_object_dtype(values) and dtype is None: 

79 return ensure_object(np.asarray(values)), 'object', 'object' 

80 elif is_complex_dtype(values) or is_complex_dtype(dtype): 

81 

82 # ignore the fact that we are casting to float 

83 # which discards complex parts 

84 with catch_warnings(): 

85 simplefilter("ignore", np.ComplexWarning) 

86 values = ensure_float64(values) 

87 return values, 'float64', 'float64' 

88 

89 except (TypeError, ValueError, OverflowError): 

90 # if we are trying to coerce to a dtype 

91 # and it is incompat this will fall thru to here 

92 return ensure_object(values), 'object', 'object' 

93 

94 # datetimelike 

95 if (needs_i8_conversion(values) or 

96 is_period_dtype(dtype) or 

97 is_datetime64_any_dtype(dtype) or 

98 is_timedelta64_dtype(dtype)): 

99 if is_period_dtype(values) or is_period_dtype(dtype): 

100 from pandas import PeriodIndex 

101 values = PeriodIndex(values) 

102 dtype = values.dtype 

103 elif is_timedelta64_dtype(values) or is_timedelta64_dtype(dtype): 

104 from pandas import TimedeltaIndex 

105 values = TimedeltaIndex(values) 

106 dtype = values.dtype 

107 else: 

108 # Datetime 

109 from pandas import DatetimeIndex 

110 values = DatetimeIndex(values) 

111 dtype = values.dtype 

112 

113 return values.asi8, dtype, 'int64' 

114 

115 elif (is_categorical_dtype(values) and 

116 (is_categorical_dtype(dtype) or dtype is None)): 

117 values = getattr(values, 'values', values) 

118 values = values.codes 

119 dtype = 'category' 

120 

121 # we are actually coercing to int64 

122 # until our algos support int* directly (not all do) 

123 values = ensure_int64(values) 

124 

125 return values, dtype, 'int64' 

126 

127 # we have failed, return object 

128 values = np.asarray(values, dtype=np.object) 

129 return ensure_object(values), 'object', 'object'

874def checked_add_with_arr(arr, b, arr_mask=None, b_mask=None): 

875 """ 

876 Perform array addition that checks for underflow and overflow. 

877 

878 Performs the addition of an int64 array and an int64 integer (or array) 

879 but checks that they do not result in overflow first. For elements that 

880 are indicated to be NaN, whether or not there is overflow for that element 

881 is automatically ignored. 

882 

883 Parameters 

884 ---------- 

885 arr : array addend. 

886 b : array or scalar addend. 

887 arr_mask : boolean array or None 

888 array indicating which elements to exclude from checking 

889 b_mask : boolean array or boolean or None 

890 array or scalar indicating which element(s) to exclude from checking 

891 

892 Returns 

893 ------- 

894 sum : An array for elements x + b for each element x in arr if b is 

895 a scalar or an array for elements x + y for each element pair 

896 (x, y) in (arr, b). 

897 

898 Raises 

899 ------ 

900 OverflowError if any x + y exceeds the maximum or minimum int64 value. 

901 """ 

902 # For performance reasons, we broadcast 'b' to the new array 'b2' 

903 # so that it has the same size as 'arr'. 

904 b2 = np.broadcast_to(b, arr.shape) 

905 if b_mask is not None: 

906 # We do the same broadcasting for b_mask as well. 

907 b2_mask = np.broadcast_to(b_mask, arr.shape) 

908 else: 

909 b2_mask = None 

910 

911 # For elements that are NaN, regardless of their value, we should 

912 # ignore whether they overflow or not when doing the checked add. 

913 if arr_mask is not None and b2_mask is not None: 

914 not_nan = np.logical_not(arr_mask | b2_mask) 

915 elif arr_mask is not None: 

916 not_nan = np.logical_not(arr_mask) 

917 elif b_mask is not None: 

918 not_nan = np.logical_not(b2_mask) 

919 else: 

920 not_nan = np.empty(arr.shape, dtype=bool) 

921 not_nan.fill(True) 

922 

923 # gh-14324: For each element in 'arr' and its corresponding element 

924 # in 'b2', we check the sign of the element in 'b2'. If it is positive, 

925 # we then check whether its sum with the element in 'arr' exceeds 

926 # np.iinfo(np.int64).max. If so, we have an overflow error. If it 

927 # it is negative, we then check whether its sum with the element in 

928 # 'arr' exceeds np.iinfo(np.int64).min. If so, we have an overflow 

929 # error as well. 

930 mask1 = b2 > 0 

931 mask2 = b2 < 0 

932 

933 if not mask1.any(): 

934 to_raise = ((np.iinfo(np.int64).min - b2 > arr) & not_nan).any() 

935 elif not mask2.any(): 

936 to_raise = ((np.iinfo(np.int64).max - b2 < arr) & not_nan).any() 

937 else: 

938 to_raise = (((np.iinfo(np.int64).max - 

939 b2[mask1] < arr[mask1]) & not_nan[mask1]).any() or 

940 ((np.iinfo(np.int64).min - 

941 b2[mask2] > arr[mask2]) & not_nan[mask2]).any()) 

942 

943 if to_raise: 

944 raise OverflowError("Overflow in int64 addition") 

945 return arr + b

985 fractional part of the index surrounded by `i` and `j`. 

986 -lower: `i`. 

987 - higher: `j`. 

988

990 ------- 

991 score : float 

992 Score at percentile. 

993

995 --------

1450def take(arr, indices, axis=0, allow_fill=False, fill_value=None):

1495 

1496 Notes 

1497 -----

1499 is accepted by NumPy for `arr`.

1502

1504 -------- 

1505 numpy.take 

1506

1509 >>> from pandas.api.extensions import take

1518

1520 array([10., 10., nan]) 

1521

70 def __init__(self, old_name=None, new_name=None, message=None): 

71 self.old_name = old_name 

72 self.new_name = new_name 

73 self.message = message

75 def __call__(self, func, *args, **kwargs): 

76 """ 

77 Decorator call. Refer to ``decorate``. 

78 

79 """ 

80 old_name = self.old_name 

81 new_name = self.new_name 

82 message = self.message 

83 

84 if old_name is None: 

85 try: 

86 old_name = func.__name__ 

87 except AttributeError: 

88 old_name = func.__name__ 

89 if new_name is None: 

90 depdoc = "`%s` is deprecated!" % old_name 

91 else: 

92 depdoc = "`%s` is deprecated, use `%s` instead!" % \ 

93 (old_name, new_name) 

94 

95 if message is not None: 

96 depdoc += "\n" + message 

97 

98 def newfunc(*args,**kwds): 

99 """`arrayrange` is deprecated, use `arange` instead!""" 

100 warnings.warn(depdoc, DeprecationWarning, stacklevel=2) 

101 return func(*args, **kwds) 

102 

103 newfunc = _set_function_name(newfunc, old_name) 

104 doc = func.__doc__ 

105 if doc is None: 

106 doc = depdoc 

107 else: 

108 lines = doc.expandtabs().split('\n') 

109 indent = _get_indent(lines[1:]) 

110 if lines[0].lstrip(): 

111 # Indent the original first line to let inspect.cleandoc() 

112 # dedent the docstring despite the deprecation notice. 

113 doc = indent * ' ' + doc 

114 else: 

115 # Remove the same leading blank lines as cleandoc() would. 

116 skip = len(lines[0]) + 1 

117 for line in lines[1:]: 

118 if len(line) > indent: 

119 break 

120 skip += len(line) + 1 

121 doc = doc[skip:] 

122 doc = '\n\n'.join([depdoc, doc]) 

123 newfunc.__doc__ = doc 

124 try: 

125 d = func.__dict__ 

126 except AttributeError: 

127 pass 

128 else: 

129 newfunc.__dict__.update(d) 

130 return newfunc

133def _get_indent(lines): 

134 """ 

135 Determines the leading whitespace that could be removed from all the lines. 

136 """ 

137 indent = sys.maxsize 

138 for line in lines: 

139 content = len(line.lstrip()) 

140 if content: 

141 indent = min(indent, len(line) - content) 

142 if indent == sys.maxsize: 

143 indent = 0 

144 return indent

206def byte_bounds(a): 

207 """ 

208 Returns pointers to the end-points of an array. 

209 

210 Parameters 

211 ---------- 

212 a : ndarray 

213 Input array. It must conform to the Python-side of the array 

214 interface. 

215 

216 Returns 

217 ------- 

218 (low, high) : tuple of 2 integers 

219 The first integer is the first byte of the array, the second 

220 integer is just past the last byte of the array. If `a` is not 

221 contiguous it will not use every byte between the (`low`, `high`) 

222 values. 

223 

224 Examples 

225 -------- 

226 >>> I = np.eye(2, dtype='f'); I.dtype 

227 dtype('float32') 

228 >>> low, high = np.byte_bounds(I) 

229 >>> high - low == I.size*I.itemsize 

230 True 

231 >>> I = np.eye(2); I.dtype 

232 dtype('float64') 

233 >>> low, high = np.byte_bounds(I) 

234 >>> high - low == I.size*I.itemsize 

235 True 

236 

237 """ 

238 ai = a.__array_interface__ 

239 a_data = ai['data'][0] 

240 astrides = ai['strides'] 

241 ashape = ai['shape'] 

242 bytes_a = asarray(a).dtype.itemsize 

243 

244 a_low = a_high = a_data 

245 if astrides is None: 

246 # contiguous case 

247 a_high += a.size * bytes_a 

248 else: 

249 for shape, stride in zip(ashape, astrides): 

250 if stride < 0: 

251 a_low += (shape-1)*stride 

252 else: 

253 a_high += (shape-1)*stride 

254 a_high += bytes_a 

255 return a_low, a_high

263def who(vardict=None):

329 maxshape = 0 

330 maxbyte = 0 

331 totalbytes = 0 

332 for k in range(len(sta)): 

333 val = sta[k] 

334 if maxname < len(val[0]): 

335 maxname = len(val[0]) 

336 if maxshape < len(val[1]): 

337 maxshape = len(val[1]) 

338 if maxbyte < len(val[2]): 

339 maxbyte = len(val[2]) 

340 if val[4]:

342 

343 if len(sta) > 0: 

344 sp1 = max(10, maxname) 

345 sp2 = max(10, maxshape) 

346 sp3 = max(10, maxbyte) 

347 prval = "Name %s Shape %s Bytes %s Type" % (sp1*' ', sp2*' ', sp3*' ') 

348 print(prval + "\n" + "="*(len(prval)+5) + "\n") 

349 

350 for k in range(len(sta)): 

351 val = sta[k] 

352 print("%s %s %s %s %s %s %s" % (val[0], ' '*(sp1-len(val[0])+4), 

353 val[1], ' '*(sp2-len(val[1])+5), 

354 val[2], ' '*(sp3-len(val[2])+5), 

355 val[3]))

365# combine name and arguments and split to multiple lines of width 

366# characters. End lines on a comma and begin argument list indented with 

367# the rest of the arguments. 

368def _split_line(name, arguments, width): 

369 firstwidth = len(name) 

370 k = firstwidth 

371 newstr = name 

372 sepstr = ", " 

373 arglist = arguments.split(sepstr) 

374 for argument in arglist: 

375 if k == firstwidth: 

376 addstr = "" 

377 else: 

378 addstr = sepstr 

379 k = k + len(argument) + len(addstr) 

380 if k > width: 

381 k = firstwidth + 1 + len(argument) 

382 newstr = newstr + ",\n" + " "*(firstwidth+2) + argument 

383 else: 

384 newstr = newstr + addstr + argument 

385 return newstr

390# Traverse all module directories underneath globals 

391# to see if something is defined 

392def _makenamedict(module='numpy'): 

393 module = __import__(module, globals(), locals(), []) 

394 thedict = {module.__name__:module.__dict__} 

395 dictlist = [module.__name__] 

396 totraverse = [module.__dict__] 

397 while True: 

398 if len(totraverse) == 0: 

399 break 

400 thisdict = totraverse.pop(0) 

401 for x in thisdict.keys(): 

402 if isinstance(thisdict[x], types.ModuleType): 

403 modname = thisdict[x].__name__ 

404 if modname not in dictlist: 

405 moddict = thisdict[x].__dict__ 

406 dictlist.append(modname) 

407 totraverse.append(moddict) 

408 thedict[modname] = moddict 

409 return thedict, dictlist

730 output=None): 

731 """ 

732 Do a keyword search on docstrings. 

733 

734 A list of objects that matched the search is displayed, 

735 sorted by relevance. All given keywords need to be found in the 

736 docstring for it to be returned as a result, but the order does 

737 not matter. 

738 

739 Parameters 

740 ---------- 

741 what : str 

742 String containing words to look for. 

743 module : str or list, optional 

744 Name of module(s) whose docstrings to go through. 

745 import_modules : bool, optional 

746 Whether to import sub-modules in packages. Default is True. 

747 regenerate : bool, optional 

748 Whether to re-generate the docstring cache. Default is False. 

749 output : file-like, optional 

750 File-like object to write the output to. If omitted, use a pager. 

751 

752 See Also 

753 -------- 

754 source, info 

755 

756 Notes 

757 ----- 

758 Relevance is determined only roughly, by checking if the keywords occur 

759 in the function name, at the start of a docstring, etc. 

760 

761 Examples 

762 -------- 

763 >>> np.lookfor('binary representation') # doctest: +SKIP 

764 Search results for 'binary representation' 

765 ------------------------------------------ 

766 numpy.binary_repr 

767 Return the binary representation of the input number as a string. 

768 numpy.core.setup_common.long_double_representation 

769 Given a binary dump as given by GNU od -b, look for long double 

770 numpy.base_repr 

771 Return a string representation of a number in the given base system. 

772 ... 

773 

774 """ 

775 import pydoc 

776 

777 # Cache 

778 cache = _lookfor_generate_cache(module, import_modules, regenerate) 

779 

780 # Search 

781 # XXX: maybe using a real stemming search engine would be better? 

782 found = [] 

783 whats = str(what).lower().split() 

784 if not whats: 

785 return 

786 

787 for name, (docstring, kind, index) in cache.items(): 

788 if kind in ('module', 'object'): 

789 # don't show modules or objects 

790 continue 

791 doc = docstring.lower() 

792 if all(w in doc for w in whats): 

793 found.append(name) 

794 

795 # Relevance sort 

796 # XXX: this is full Harrison-Stetson heuristics now, 

797 # XXX: it probably could be improved 

798 

799 kind_relevance = {'func': 1000, 'class': 1000, 

800 'module': -1000, 'object': -1000} 

801 

802 def relevance(name, docstr, kind, index): 

803 r = 0 

804 # do the keywords occur within the start of the docstring? 

805 first_doc = "\n".join(docstr.lower().strip().split("\n")[:3]) 

806 r += sum([200 for w in whats if w in first_doc]) 

807 # do the keywords occur in the function name? 

808 r += sum([30 for w in whats if w in name]) 

809 # is the full name long? 

810 r += -len(name) * 5 

811 # is the object of bad type? 

812 r += kind_relevance.get(kind, -1000) 

813 # is the object deep in namespace hierarchy? 

814 r += -name.count('.') * 10 

815 r += max(-index / 100, -100) 

816 return r 

817 

818 def relevance_value(a): 

819 return relevance(a, *cache[a]) 

820 found.sort(key=relevance_value) 

821 

822 # Pretty-print 

823 s = "Search results for '%s'" % (' '.join(whats)) 

824 help_text = [s, "-"*len(s)] 

825 for name in found[::-1]: 

826 doc, kind, ix = cache[name] 

827 

828 doclines = [line.strip() for line in doc.strip().split("\n") 

829 if line.strip()] 

830 

831 # find a suitable short description 

832 try: 

833 first_doc = doclines[0].strip() 

834 if _function_signature_re.search(first_doc): 

835 first_doc = doclines[1].strip() 

836 except IndexError: 

837 first_doc = "" 

838 help_text.append("%s\n %s" % (name, first_doc)) 

839 

840 if not found: 

841 help_text.append("Nothing found.") 

842 

843 # Output 

844 if output is not None: 

845 output.write("\n".join(help_text)) 

846 elif len(help_text) > 10: 

847 pager = pydoc.getpager() 

848 pager("\n".join(help_text)) 

849 else: 

850 print("\n".join(help_text))

852def _lookfor_generate_cache(module, import_modules, regenerate): 

853 """ 

854 Generate docstring cache for given module. 

855 

856 Parameters 

857 ---------- 

858 module : str, None, module 

859 Module for which to generate docstring cache 

860 import_modules : bool 

861 Whether to import sub-modules in packages. 

862 regenerate : bool 

863 Re-generate the docstring cache 

864 

865 Returns 

866 ------- 

867 cache : dict {obj_full_name: (docstring, kind, index), ...} 

868 Docstring cache for the module, either cached one (regenerate=False) 

869 or newly generated. 

870 

871 """ 

872 global _lookfor_caches 

873 # Local import to speed up numpy's import time. 

874 import inspect 

875 

876 if sys.version_info[0] >= 3: 

877 # In Python3 stderr, stdout are text files. 

878 from io import StringIO 

879 else: 

880 from StringIO import StringIO 

881 

882 if module is None: 

883 module = "numpy" 

884 

885 if isinstance(module, str): 

886 try: 

887 __import__(module) 

888 except ImportError: 

889 return {} 

890 module = sys.modules[module] 

891 elif isinstance(module, list) or isinstance(module, tuple): 

892 cache = {} 

893 for mod in module: 

894 cache.update(_lookfor_generate_cache(mod, import_modules, 

895 regenerate)) 

896 return cache 

897 

898 if id(module) in _lookfor_caches and not regenerate: 

899 return _lookfor_caches[id(module)] 

900 

901 # walk items and collect docstrings 

902 cache = {} 

903 _lookfor_caches[id(module)] = cache 

904 seen = {} 

905 index = 0 

906 stack = [(module.__name__, module)] 

907 while stack: 

908 name, item = stack.pop(0) 

909 if id(item) in seen: 

910 continue 

911 seen[id(item)] = True 

912 

913 index += 1 

914 kind = "object" 

915 

916 if inspect.ismodule(item): 

917 kind = "module" 

918 try: 

919 _all = item.__all__ 

920 except AttributeError: 

921 _all = None 

922 

923 # import sub-packages 

924 if import_modules and hasattr(item, '__path__'): 

925 for pth in item.__path__: 

926 for mod_path in os.listdir(pth): 

927 this_py = os.path.join(pth, mod_path) 

928 init_py = os.path.join(pth, mod_path, '__init__.py') 

929 if (os.path.isfile(this_py) and 

930 mod_path.endswith('.py')): 

931 to_import = mod_path[:-3] 

932 elif os.path.isfile(init_py): 

933 to_import = mod_path 

934 else: 

935 continue 

936 if to_import == '__init__': 

937 continue 

938 

939 try: 

940 old_stdout = sys.stdout 

941 old_stderr = sys.stderr 

942 try: 

943 sys.stdout = StringIO() 

944 sys.stderr = StringIO() 

945 __import__("%s.%s" % (name, to_import)) 

946 finally: 

947 sys.stdout = old_stdout 

948 sys.stderr = old_stderr 

949 # Catch SystemExit, too 

950 except BaseException: 

951 continue 

952 

953 for n, v in _getmembers(item): 

954 try: 

955 item_name = getattr(v, '__name__', "%s.%s" % (name, n)) 

956 mod_name = getattr(v, '__module__', None) 

957 except NameError: 

958 # ref. SWIG's global cvars 

959 # NameError: Unknown C global variable 

960 item_name = "%s.%s" % (name, n) 

961 mod_name = None 

962 if '.' not in item_name and mod_name: 

963 item_name = "%s.%s" % (mod_name, item_name) 

964 

965 if not item_name.startswith(name + '.'): 

966 # don't crawl "foreign" objects 

967 if isinstance(v, ufunc): 

968 # ... unless they are ufuncs 

969 pass 

970 else: 

971 continue 

972 elif not (inspect.ismodule(v) or _all is None or n in _all): 

973 continue 

974 stack.append(("%s.%s" % (name, n), v)) 

975 elif inspect.isclass(item): 

976 kind = "class" 

977 for n, v in _getmembers(item): 

978 stack.append(("%s.%s" % (name, n), v)) 

979 elif hasattr(item, "__call__"): 

980 kind = "func" 

981 

982 try: 

983 doc = inspect.getdoc(item) 

984 except NameError: 

985 # ref SWIG's NameError: Unknown C global variable 

986 doc = None 

987 if doc is not None: 

988 cache[name] = (doc, kind, index) 

989 

990 return cache 

991 

992def _getmembers(item): 

993 import inspect 

994 try: 

995 members = inspect.getmembers(item) 

996 except Exception: 

997 members = [(x, getattr(item, x)) for x in dir(item) 

998 if hasattr(item, x)] 

999 return members

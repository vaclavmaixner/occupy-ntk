20def _flop_count(idx_contraction, inner, num_terms, size_dictionary): 

21 """ 

22 Computes the number of FLOPS in the contraction. 

23 

24 Parameters 

25 ---------- 

26 idx_contraction : iterable 

27 The indices involved in the contraction 

28 inner : bool 

29 Does this contraction require an inner product? 

30 num_terms : int 

31 The number of terms in a contraction 

32 size_dictionary : dict 

33 The size of each of the indices in idx_contraction 

34 

35 Returns 

36 ------- 

37 flop_count : int 

38 The total number of FLOPS required for the contraction. 

39 

40 Examples 

41 -------- 

42 

43 >>> _flop_count('abc', False, 1, {'a': 2, 'b':3, 'c':5}) 

44 30 

45 

46 >>> _flop_count('abc', True, 2, {'a': 2, 'b':3, 'c':5}) 

47 60 

48 

49 """ 

50 

51 overall_size = _compute_size_by_dict(idx_contraction, size_dictionary) 

52 op_factor = max(1, num_terms - 1) 

53 if inner: 

54 op_factor += 1 

55 

56 return overall_size * op_factor

87def _find_contraction(positions, input_sets, output_set): 

88 """ 

89 Finds the contraction for a given set of input and output sets. 

90 

91 Parameters 

92 ---------- 

93 positions : iterable 

94 Integer positions of terms used in the contraction. 

95 input_sets : list 

96 List of sets that represent the lhs side of the einsum subscript 

97 output_set : set 

98 Set that represents the rhs side of the overall einsum subscript 

99 

100 Returns 

101 ------- 

102 new_result : set 

103 The indices of the resulting contraction 

104 remaining : list 

105 List of sets that have not been contracted, the new set is appended to 

106 the end of this list 

107 idx_removed : set 

108 Indices removed from the entire contraction 

109 idx_contraction : set 

110 The indices used in the current contraction 

111 

112 Examples 

113 -------- 

114 

115 # A simple dot product test case 

116 >>> pos = (0, 1) 

117 >>> isets = [set('ab'), set('bc')] 

118 >>> oset = set('ac') 

119 >>> _find_contraction(pos, isets, oset) 

120 ({'a', 'c'}, [{'a', 'c'}], {'b'}, {'a', 'b', 'c'}) 

121 

122 # A more complex case with additional terms in the contraction 

123 >>> pos = (0, 2) 

124 >>> isets = [set('abd'), set('ac'), set('bdc')] 

125 >>> oset = set('ac') 

126 >>> _find_contraction(pos, isets, oset) 

127 ({'a', 'c'}, [{'a', 'c'}, {'a', 'c'}], {'b', 'd'}, {'a', 'b', 'c', 'd'}) 

128 """ 

129 

130 idx_contract = set() 

131 idx_remain = output_set.copy() 

132 remaining = [] 

133 for ind, value in enumerate(input_sets): 

134 if ind in positions: 

135 idx_contract |= value 

136 else: 

137 remaining.append(value) 

138 idx_remain |= value 

139 

140 new_result = idx_remain & idx_contract 

141 idx_removed = (idx_contract - new_result) 

142 remaining.append(new_result) 

143 

144 return (new_result, remaining, idx_removed, idx_contract)

147def _optimal_path(input_sets, output_set, idx_dict, memory_limit): 

148 """ 

149 Computes all possible pair contractions, sieves the results based 

150 on ``memory_limit`` and returns the lowest cost path. This algorithm 

151 scales factorial with respect to the elements in the list ``input_sets``. 

152 

153 Parameters 

154 ---------- 

155 input_sets : list 

156 List of sets that represent the lhs side of the einsum subscript 

157 output_set : set 

158 Set that represents the rhs side of the overall einsum subscript 

159 idx_dict : dictionary 

160 Dictionary of index sizes 

161 memory_limit : int 

162 The maximum number of elements in a temporary array 

163 

164 Returns 

165 ------- 

166 path : list 

167 The optimal contraction order within the memory limit constraint. 

168 

169 Examples 

170 -------- 

171 >>> isets = [set('abd'), set('ac'), set('bdc')] 

172 >>> oset = set() 

173 >>> idx_sizes = {'a': 1, 'b':2, 'c':3, 'd':4} 

174 >>> _optimal_path(isets, oset, idx_sizes, 5000) 

175 [(0, 2), (0, 1)] 

176 """ 

177 

178 full_results = [(0, [], input_sets)] 

179 for iteration in range(len(input_sets) - 1): 

180 iter_results = [] 

181 

182 # Compute all unique pairs 

183 for curr in full_results: 

184 cost, positions, remaining = curr 

185 for con in itertools.combinations(range(len(input_sets) - iteration), 2): 

186 

187 # Find the contraction 

188 cont = _find_contraction(con, remaining, output_set) 

189 new_result, new_input_sets, idx_removed, idx_contract = cont 

190 

191 # Sieve the results based on memory_limit 

192 new_size = _compute_size_by_dict(new_result, idx_dict) 

193 if new_size > memory_limit: 

194 continue 

195 

196 # Build (total_cost, positions, indices_remaining) 

197 total_cost = cost + _flop_count(idx_contract, idx_removed, len(con), idx_dict) 

198 new_pos = positions + [con] 

199 iter_results.append((total_cost, new_pos, new_input_sets)) 

200 

201 # Update combinatorial list, if we did not find anything return best 

202 # path + remaining contractions 

203 if iter_results: 

204 full_results = iter_results 

205 else: 

206 path = min(full_results, key=lambda x: x[0])[1] 

207 path += [tuple(range(len(input_sets) - iteration))] 

208 return path 

209 

210 # If we have not found anything return single einsum contraction 

211 if len(full_results) == 0: 

212 return [tuple(range(len(input_sets)))] 

213 

214 path = min(full_results, key=lambda x: x[0])[1] 

215 return path

217def _parse_possible_contraction(positions, input_sets, output_set, idx_dict, memory_limit, path_cost, naive_cost): 

218 """Compute the cost (removed size + flops) and resultant indices for 

219 performing the contraction specified by ``positions``. 

220 

221 Parameters 

222 ---------- 

223 positions : tuple of int 

224 The locations of the proposed tensors to contract. 

225 input_sets : list of sets 

226 The indices found on each tensors. 

227 output_set : set 

228 The output indices of the expression. 

229 idx_dict : dict 

230 Mapping of each index to its size. 

231 memory_limit : int 

232 The total allowed size for an intermediary tensor. 

233 path_cost : int 

234 The contraction cost so far. 

235 naive_cost : int 

236 The cost of the unoptimized expression. 

237 

238 Returns 

239 ------- 

240 cost : (int, int) 

241 A tuple containing the size of any indices removed, and the flop cost. 

242 positions : tuple of int 

243 The locations of the proposed tensors to contract. 

244 new_input_sets : list of sets 

245 The resulting new list of indices if this proposed contraction is performed. 

246 

247 """ 

248 

249 # Find the contraction 

250 contract = _find_contraction(positions, input_sets, output_set) 

251 idx_result, new_input_sets, idx_removed, idx_contract = contract 

252 

253 # Sieve the results based on memory_limit 

254 new_size = _compute_size_by_dict(idx_result, idx_dict) 

255 if new_size > memory_limit: 

256 return None 

257 

258 # Build sort tuple 

259 old_sizes = (_compute_size_by_dict(input_sets[p], idx_dict) for p in positions) 

260 removed_size = sum(old_sizes) - new_size 

261 

262 # NB: removed_size used to be just the size of any removed indices i.e.: 

263 # helpers.compute_size_by_dict(idx_removed, idx_dict) 

264 cost = _flop_count(idx_contract, idx_removed, len(positions), idx_dict) 

265 sort = (-removed_size, cost) 

266 

267 # Sieve based on total cost as well 

268 if (path_cost + cost) > naive_cost: 

269 return None 

270 

271 # Add contraction to possible choices 

272 return [sort, positions, new_input_sets]

275def _update_other_results(results, best): 

276 """Update the positions and provisional input_sets of ``results`` based on 

277 performing the contraction result ``best``. Remove any involving the tensors 

278 contracted. 

279 

280 Parameters 

281 ---------- 

282 results : list 

283 List of contraction results produced by ``_parse_possible_contraction``. 

284 best : list 

285 The best contraction of ``results`` i.e. the one that will be performed. 

286 

287 Returns 

288 ------- 

289 mod_results : list 

290 The list of modified results, updated with outcome of ``best`` contraction. 

291 """ 

292 

293 best_con = best[1] 

294 bx, by = best_con 

295 mod_results = [] 

296 

297 for cost, (x, y), con_sets in results: 

298 

299 # Ignore results involving tensors just contracted 

300 if x in best_con or y in best_con: 

301 continue 

302 

303 # Update the input_sets 

304 del con_sets[by - int(by > x) - int(by > y)] 

305 del con_sets[bx - int(bx > x) - int(bx > y)] 

306 con_sets.insert(-1, best[2][-1]) 

307 

308 # Update the position indices 

309 mod_con = x - int(x > bx) - int(x > by), y - int(y > bx) - int(y > by) 

310 mod_results.append((cost, mod_con, con_sets)) 

311 

312 return mod_results

314def _greedy_path(input_sets, output_set, idx_dict, memory_limit): 

315 """ 

316 Finds the path by contracting the best pair until the input list is 

317 exhausted. The best pair is found by minimizing the tuple 

318 ``(-prod(indices_removed), cost)``. What this amounts to is prioritizing 

319 matrix multiplication or inner product operations, then Hadamard like 

320 operations, and finally outer operations. Outer products are limited by 

321 ``memory_limit``. This algorithm scales cubically with respect to the 

322 number of elements in the list ``input_sets``. 

323 

324 Parameters 

325 ---------- 

326 input_sets : list 

327 List of sets that represent the lhs side of the einsum subscript 

328 output_set : set 

329 Set that represents the rhs side of the overall einsum subscript 

330 idx_dict : dictionary 

331 Dictionary of index sizes 

332 memory_limit_limit : int 

333 The maximum number of elements in a temporary array 

334 

335 Returns 

336 ------- 

337 path : list 

338 The greedy contraction order within the memory limit constraint. 

339 

340 Examples 

341 -------- 

342 >>> isets = [set('abd'), set('ac'), set('bdc')] 

343 >>> oset = set() 

344 >>> idx_sizes = {'a': 1, 'b':2, 'c':3, 'd':4} 

345 >>> _greedy_path(isets, oset, idx_sizes, 5000) 

346 [(0, 2), (0, 1)] 

347 """ 

348 

349 # Handle trivial cases that leaked through 

350 if len(input_sets) == 1: 

351 return [(0,)] 

352 elif len(input_sets) == 2: 

353 return [(0, 1)] 

354 

355 # Build up a naive cost 

356 contract = _find_contraction(range(len(input_sets)), input_sets, output_set) 

357 idx_result, new_input_sets, idx_removed, idx_contract = contract 

358 naive_cost = _flop_count(idx_contract, idx_removed, len(input_sets), idx_dict) 

359 

360 # Initially iterate over all pairs 

361 comb_iter = itertools.combinations(range(len(input_sets)), 2) 

362 known_contractions = [] 

363 

364 path_cost = 0 

365 path = [] 

366 

367 for iteration in range(len(input_sets) - 1): 

368 

369 # Iterate over all pairs on first step, only previously found pairs on subsequent steps 

370 for positions in comb_iter: 

371 

372 # Always initially ignore outer products 

373 if input_sets[positions[0]].isdisjoint(input_sets[positions[1]]): 

374 continue 

375 

376 result = _parse_possible_contraction(positions, input_sets, output_set, idx_dict, memory_limit, path_cost, 

377 naive_cost) 

378 if result is not None: 

379 known_contractions.append(result) 

380 

381 # If we do not have a inner contraction, rescan pairs including outer products 

382 if len(known_contractions) == 0: 

383 

384 # Then check the outer products 

385 for positions in itertools.combinations(range(len(input_sets)), 2): 

386 result = _parse_possible_contraction(positions, input_sets, output_set, idx_dict, memory_limit, 

387 path_cost, naive_cost) 

388 if result is not None: 

389 known_contractions.append(result) 

390 

391 # If we still did not find any remaining contractions, default back to einsum like behavior 

392 if len(known_contractions) == 0: 

393 path.append(tuple(range(len(input_sets)))) 

394 break 

395 

396 # Sort based on first index 

397 best = min(known_contractions, key=lambda x: x[0]) 

398 

399 # Now propagate as many unused contractions as possible to next iteration 

400 known_contractions = _update_other_results(known_contractions, best) 

401 

402 # Next iteration only compute contractions with the new tensor 

403 # All other contractions have been accounted for 

404 input_sets = best[2] 

405 new_tensor_pos = len(input_sets) - 1 

406 comb_iter = ((i, new_tensor_pos) for i in range(new_tensor_pos)) 

407 

408 # Update path and total cost 

409 path.append(best[1]) 

410 path_cost += best[0][1] 

411 

412 return path

415def _can_dot(inputs, result, idx_removed): 

416 """ 

417 Checks if we can use BLAS (np.tensordot) call and its beneficial to do so. 

418 

419 Parameters 

420 ---------- 

421 inputs : list of str 

422 Specifies the subscripts for summation. 

423 result : str 

424 Resulting summation. 

425 idx_removed : set 

426 Indices that are removed in the summation 

427 

428 

429 Returns 

430 ------- 

431 type : bool 

432 Returns true if BLAS should and can be used, else False 

433 

434 Notes 

435 ----- 

436 If the operations is BLAS level 1 or 2 and is not already aligned 

437 we default back to einsum as the memory movement to copy is more 

438 costly than the operation itself. 

439 

440 

441 Examples 

442 -------- 

443 

444 # Standard GEMM operation 

445 >>> _can_dot(['ij', 'jk'], 'ik', set('j')) 

446 True 

447 

448 # Can use the standard BLAS, but requires odd data movement 

449 >>> _can_dot(['ijj', 'jk'], 'ik', set('j')) 

450 False 

451 

452 # DDOT where the memory is not aligned 

453 >>> _can_dot(['ijk', 'ikj'], '', set('ijk')) 

454 False 

455 

456 """ 

457 

458 # All `dot` calls remove indices 

459 if len(idx_removed) == 0: 

460 return False 

461 

462 # BLAS can only handle two operands 

463 if len(inputs) != 2: 

464 return False 

465 

466 input_left, input_right = inputs 

467 

468 for c in set(input_left + input_right): 

469 # can't deal with repeated indices on same input or more than 2 total 

470 nl, nr = input_left.count(c), input_right.count(c) 

471 if (nl > 1) or (nr > 1) or (nl + nr > 2): 

472 return False 

473 

474 # can't do implicit summation or dimension collapse e.g. 

475 # "ab,bc->c" (implicitly sum over 'a') 

476 # "ab,ca->ca" (take diagonal of 'a') 

477 if nl + nr - 1 == int(c in result): 

478 return False 

479 

480 # Build a few temporaries 

481 set_left = set(input_left) 

482 set_right = set(input_right) 

483 keep_left = set_left - idx_removed 

484 keep_right = set_right - idx_removed 

485 rs = len(idx_removed) 

486 

487 # At this point we are a DOT, GEMV, or GEMM operation 

488 

489 # Handle inner products 

490 

491 # DDOT with aligned data 

492 if input_left == input_right: 

493 return True 

494 

495 # DDOT without aligned data (better to use einsum) 

496 if set_left == set_right: 

497 return False 

498 

499 # Handle the 4 possible (aligned) GEMV or GEMM cases 

500 

501 # GEMM or GEMV no transpose 

502 if input_left[-rs:] == input_right[:rs]: 

503 return True 

504 

505 # GEMM or GEMV transpose both 

506 if input_left[:rs] == input_right[-rs:]: 

507 return True 

508 

509 # GEMM or GEMV transpose right 

510 if input_left[-rs:] == input_right[-rs:]: 

511 return True 

512 

513 # GEMM or GEMV transpose left 

514 if input_left[:rs] == input_right[:rs]: 

515 return True 

516 

517 # Einsum is faster than GEMV if we have to copy data 

518 if not keep_left or not keep_right: 

519 return False 

520 

521 # We are a matrix-matrix product, but we need to copy data 

522 return True

525def _parse_einsum_input(operands): 

526 """ 

527 A reproduction of einsum c side einsum parsing in python. 

528 

529 Returns 

530 ------- 

531 input_strings : str 

532 Parsed input strings 

533 output_string : str 

534 Parsed output string 

535 operands : list of array_like 

536 The operands to use in the numpy contraction 

537 

538 Examples 

539 -------- 

540 The operand list is simplified to reduce printing: 

541 

542 >>> np.random.seed(123) 

543 >>> a = np.random.rand(4, 4) 

544 >>> b = np.random.rand(4, 4, 4) 

545 >>> _parse_einsum_input(('...a,...a->...', a, b)) 

546 ('za,xza', 'xz', [a, b]) # may vary 

547 

548 >>> _parse_einsum_input((a, [Ellipsis, 0], b, [Ellipsis, 0])) 

549 ('za,xza', 'xz', [a, b]) # may vary 

550 """ 

551 

552 if len(operands) == 0: 

553 raise ValueError("No input operands") 

554 

555 if isinstance(operands[0], basestring): 

556 subscripts = operands[0].replace(" ", "") 

557 operands = [asanyarray(v) for v in operands[1:]] 

558 

559 # Ensure all characters are valid 

560 for s in subscripts: 

561 if s in '.,->': 

562 continue 

563 if s not in einsum_symbols: 

564 raise ValueError("Character %s is not a valid symbol." % s) 

565 

566 else: 

567 tmp_operands = list(operands) 

568 operand_list = [] 

569 subscript_list = [] 

570 for p in range(len(operands) // 2): 

571 operand_list.append(tmp_operands.pop(0)) 

572 subscript_list.append(tmp_operands.pop(0)) 

573 

574 output_list = tmp_operands[-1] if len(tmp_operands) else None 

575 operands = [asanyarray(v) for v in operand_list] 

576 subscripts = "" 

577 last = len(subscript_list) - 1 

578 for num, sub in enumerate(subscript_list): 

579 for s in sub: 

580 if s is Ellipsis: 

581 subscripts += "..." 

582 elif isinstance(s, int): 

583 subscripts += einsum_symbols[s] 

584 else: 

585 raise TypeError("For this input type lists must contain " 

586 "either int or Ellipsis") 

587 if num != last: 

588 subscripts += "," 

589 

590 if output_list is not None: 

591 subscripts += "->" 

592 for s in output_list: 

593 if s is Ellipsis: 

594 subscripts += "..." 

595 elif isinstance(s, int): 

596 subscripts += einsum_symbols[s] 

597 else: 

598 raise TypeError("For this input type lists must contain " 

599 "either int or Ellipsis") 

600 # Check for proper "->" 

601 if ("-" in subscripts) or (">" in subscripts): 

602 invalid = (subscripts.count("-") > 1) or (subscripts.count(">") > 1) 

603 if invalid or (subscripts.count("->") != 1): 

604 raise ValueError("Subscripts can only contain one '->'.") 

605 

606 # Parse ellipses 

607 if "." in subscripts: 

608 used = subscripts.replace(".", "").replace(",", "").replace("->", "") 

609 unused = list(einsum_symbols_set - set(used)) 

610 ellipse_inds = "".join(unused) 

611 longest = 0 

612 

613 if "->" in subscripts: 

614 input_tmp, output_sub = subscripts.split("->") 

615 split_subscripts = input_tmp.split(",") 

616 out_sub = True 

617 else: 

618 split_subscripts = subscripts.split(',') 

619 out_sub = False 

620 

621 for num, sub in enumerate(split_subscripts): 

622 if "." in sub: 

623 if (sub.count(".") != 3) or (sub.count("...") != 1): 

624 raise ValueError("Invalid Ellipses.") 

625 

626 # Take into account numerical values 

627 if operands[num].shape == (): 

628 ellipse_count = 0 

629 else: 

630 ellipse_count = max(operands[num].ndim, 1) 

631 ellipse_count -= (len(sub) - 3) 

632 

633 if ellipse_count > longest: 

634 longest = ellipse_count 

635 

636 if ellipse_count < 0: 

637 raise ValueError("Ellipses lengths do not match.") 

638 elif ellipse_count == 0: 

639 split_subscripts[num] = sub.replace('...', '') 

640 else: 

641 rep_inds = ellipse_inds[-ellipse_count:] 

642 split_subscripts[num] = sub.replace('...', rep_inds) 

643 

644 subscripts = ",".join(split_subscripts) 

645 if longest == 0: 

646 out_ellipse = "" 

647 else: 

648 out_ellipse = ellipse_inds[-longest:] 

649 

650 if out_sub: 

651 subscripts += "->" + output_sub.replace("...", out_ellipse) 

652 else: 

653 # Special care for outputless ellipses 

654 output_subscript = "" 

655 tmp_subscripts = subscripts.replace(",", "") 

656 for s in sorted(set(tmp_subscripts)): 

657 if s not in (einsum_symbols): 

658 raise ValueError("Character %s is not a valid symbol." % s) 

659 if tmp_subscripts.count(s) == 1: 

660 output_subscript += s 

661 normal_inds = ''.join(sorted(set(output_subscript) - 

662 set(out_ellipse))) 

663 

664 subscripts += "->" + out_ellipse + normal_inds 

665 

666 # Build output string if does not exist 

667 if "->" in subscripts: 

668 input_subscripts, output_subscript = subscripts.split("->") 

669 else: 

670 input_subscripts = subscripts 

671 # Build output subscripts 

672 tmp_subscripts = subscripts.replace(",", "") 

673 output_subscript = "" 

674 for s in sorted(set(tmp_subscripts)): 

675 if s not in einsum_symbols: 

676 raise ValueError("Character %s is not a valid symbol." % s) 

677 if tmp_subscripts.count(s) == 1: 

678 output_subscript += s 

679 

680 # Make sure output subscripts are in the input 

681 for char in output_subscript: 

682 if char not in input_subscripts: 

683 raise ValueError("Output character %s did not appear in the input" 

684 % char) 

685 

686 # Make sure number operands is equivalent to the number of terms 

687 if len(input_subscripts.split(',')) != len(operands): 

688 raise ValueError("Number of einsum subscripts must be equal to the " 

689 "number of operands.") 

690 

691 return (input_subscripts, output_subscript, operands)

694def _einsum_path_dispatcher(*operands, **kwargs):

704@array_function_dispatch(_einsum_path_dispatcher, module='numpy') 

705def einsum_path(*operands, **kwargs): 

706 """ 

707 einsum_path(subscripts, *operands, optimize='greedy') 

708 

709 Evaluates the lowest cost contraction order for an einsum expression by 

710 considering the creation of intermediate arrays. 

711 

712 Parameters 

713 ---------- 

714 subscripts : str 

715 Specifies the subscripts for summation. 

716 *operands : list of array_like 

717 These are the arrays for the operation. 

718 optimize : {bool, list, tuple, 'greedy', 'optimal'} 

719 Choose the type of path. If a tuple is provided, the second argument is 

720 assumed to be the maximum intermediate size created. If only a single 

721 argument is provided the largest input or output array size is used 

722 as a maximum intermediate size. 

723 

724 * if a list is given that starts with ``einsum_path``, uses this as the 

725 contraction path 

726 * if False no optimization is taken 

727 * if True defaults to the 'greedy' algorithm 

728 * 'optimal' An algorithm that combinatorially explores all possible 

729 ways of contracting the listed tensors and choosest the least costly 

730 path. Scales exponentially with the number of terms in the 

731 contraction. 

732 * 'greedy' An algorithm that chooses the best pair contraction 

733 at each step. Effectively, this algorithm searches the largest inner, 

734 Hadamard, and then outer products at each step. Scales cubically with 

735 the number of terms in the contraction. Equivalent to the 'optimal' 

736 path for most contractions. 

737 

738 Default is 'greedy'. 

739 

740 Returns 

741 ------- 

742 path : list of tuples 

743 A list representation of the einsum path. 

744 string_repr : str 

745 A printable representation of the einsum path. 

746 

747 Notes 

748 ----- 

749 The resulting path indicates which terms of the input contraction should be 

750 contracted first, the result of this contraction is then appended to the 

751 end of the contraction list. This list can then be iterated over until all 

752 intermediate contractions are complete. 

753 

754 See Also 

755 -------- 

756 einsum, linalg.multi_dot 

757 

758 Examples 

759 -------- 

760 

761 We can begin with a chain dot example. In this case, it is optimal to 

762 contract the ``b`` and ``c`` tensors first as represented by the first 

763 element of the path ``(1, 2)``. The resulting tensor is added to the end 

764 of the contraction and the remaining contraction ``(0, 1)`` is then 

765 completed. 

766 

767 >>> np.random.seed(123) 

768 >>> a = np.random.rand(2, 2) 

769 >>> b = np.random.rand(2, 5) 

770 >>> c = np.random.rand(5, 2) 

771 >>> path_info = np.einsum_path('ij,jk,kl->il', a, b, c, optimize='greedy') 

772 >>> print(path_info[0]) 

773 ['einsum_path', (1, 2), (0, 1)] 

774 >>> print(path_info[1]) 

775 Complete contraction: ij,jk,kl->il # may vary 

776 Naive scaling: 4 

777 Optimized scaling: 3 

778 Naive FLOP count: 1.600e+02 

779 Optimized FLOP count: 5.600e+01 

780 Theoretical speedup: 2.857 

781 Largest intermediate: 4.000e+00 elements 

782 ------------------------------------------------------------------------- 

783 scaling current remaining 

784 ------------------------------------------------------------------------- 

785 3 kl,jk->jl ij,jl->il 

786 3 jl,ij->il il->il 

787 

788 

789 A more complex index transformation example. 

790 

791 >>> I = np.random.rand(10, 10, 10, 10) 

792 >>> C = np.random.rand(10, 10) 

793 >>> path_info = np.einsum_path('ea,fb,abcd,gc,hd->efgh', C, C, I, C, C, 

794 ... optimize='greedy') 

795 

796 >>> print(path_info[0]) 

797 ['einsum_path', (0, 2), (0, 3), (0, 2), (0, 1)] 

798 >>> print(path_info[1])  

799 Complete contraction: ea,fb,abcd,gc,hd->efgh # may vary 

800 Naive scaling: 8 

801 Optimized scaling: 5 

802 Naive FLOP count: 8.000e+08 

803 Optimized FLOP count: 8.000e+05 

804 Theoretical speedup: 1000.000 

805 Largest intermediate: 1.000e+04 elements 

806 -------------------------------------------------------------------------- 

807 scaling current remaining 

808 -------------------------------------------------------------------------- 

809 5 abcd,ea->bcde fb,gc,hd,bcde->efgh 

810 5 bcde,fb->cdef gc,hd,cdef->efgh 

811 5 cdef,gc->defg hd,defg->efgh 

812 5 defg,hd->efgh efgh->efgh 

813 """ 

814 

815 # Make sure all keywords are valid 

816 valid_contract_kwargs = ['optimize', 'einsum_call'] 

817 unknown_kwargs = [k for (k, v) in kwargs.items() if k 

818 not in valid_contract_kwargs] 

819 if len(unknown_kwargs): 

820 raise TypeError("Did not understand the following kwargs:" 

821 " %s" % unknown_kwargs) 

822 

823 # Figure out what the path really is 

824 path_type = kwargs.pop('optimize', True) 

825 if path_type is True: 

826 path_type = 'greedy' 

827 if path_type is None: 

828 path_type = False 

829 

830 memory_limit = None 

831 

832 # No optimization or a named path algorithm 

833 if (path_type is False) or isinstance(path_type, basestring): 

834 pass 

835 

836 # Given an explicit path 

837 elif len(path_type) and (path_type[0] == 'einsum_path'): 

838 pass 

839 

840 # Path tuple with memory limit 

841 elif ((len(path_type) == 2) and isinstance(path_type[0], basestring) and 

842 isinstance(path_type[1], (int, float))): 

843 memory_limit = int(path_type[1]) 

844 path_type = path_type[0] 

845 

846 else: 

847 raise TypeError("Did not understand the path: %s" % str(path_type)) 

848 

849 # Hidden option, only einsum should call this 

850 einsum_call_arg = kwargs.pop("einsum_call", False) 

851 

852 # Python side parsing 

853 input_subscripts, output_subscript, operands = _parse_einsum_input(operands) 

854 

855 # Build a few useful list and sets 

856 input_list = input_subscripts.split(',') 

857 input_sets = [set(x) for x in input_list] 

858 output_set = set(output_subscript) 

859 indices = set(input_subscripts.replace(',', '')) 

860 

861 # Get length of each unique dimension and ensure all dimensions are correct 

862 dimension_dict = {} 

863 broadcast_indices = [[] for x in range(len(input_list))] 

864 for tnum, term in enumerate(input_list): 

865 sh = operands[tnum].shape 

866 if len(sh) != len(term): 

867 raise ValueError("Einstein sum subscript %s does not contain the " 

868 "correct number of indices for operand %d." 

869 % (input_subscripts[tnum], tnum)) 

870 for cnum, char in enumerate(term): 

871 dim = sh[cnum] 

872 

873 # Build out broadcast indices 

874 if dim == 1: 

875 broadcast_indices[tnum].append(char) 

876 

877 if char in dimension_dict.keys(): 

878 # For broadcasting cases we always want the largest dim size 

879 if dimension_dict[char] == 1: 

880 dimension_dict[char] = dim 

881 elif dim not in (1, dimension_dict[char]): 

882 raise ValueError("Size of label '%s' for operand %d (%d) " 

883 "does not match previous terms (%d)." 

884 % (char, tnum, dimension_dict[char], dim)) 

885 else: 

886 dimension_dict[char] = dim 

887 

888 # Convert broadcast inds to sets 

889 broadcast_indices = [set(x) for x in broadcast_indices] 

890 

891 # Compute size of each input array plus the output array 

892 size_list = [_compute_size_by_dict(term, dimension_dict) 

893 for term in input_list + [output_subscript]] 

894 max_size = max(size_list) 

895 

896 if memory_limit is None: 

897 memory_arg = max_size 

898 else: 

899 memory_arg = memory_limit 

900 

901 # Compute naive cost 

902 # This isn't quite right, need to look into exactly how einsum does this 

903 inner_product = (sum(len(x) for x in input_sets) - len(indices)) > 0 

904 naive_cost = _flop_count(indices, inner_product, len(input_list), dimension_dict) 

905 

906 # Compute the path 

907 if (path_type is False) or (len(input_list) in [1, 2]) or (indices == output_set): 

908 # Nothing to be optimized, leave it to einsum 

909 path = [tuple(range(len(input_list)))] 

910 elif path_type == "greedy": 

911 path = _greedy_path(input_sets, output_set, dimension_dict, memory_arg) 

912 elif path_type == "optimal": 

913 path = _optimal_path(input_sets, output_set, dimension_dict, memory_arg) 

914 elif path_type[0] == 'einsum_path': 

915 path = path_type[1:] 

916 else: 

917 raise KeyError("Path name %s not found", path_type) 

918 

919 cost_list, scale_list, size_list, contraction_list = [], [], [], [] 

920 

921 # Build contraction tuple (positions, gemm, einsum_str, remaining) 

922 for cnum, contract_inds in enumerate(path): 

923 # Make sure we remove inds from right to left 

924 contract_inds = tuple(sorted(list(contract_inds), reverse=True)) 

925 

926 contract = _find_contraction(contract_inds, input_sets, output_set) 

927 out_inds, input_sets, idx_removed, idx_contract = contract 

928 

929 cost = _flop_count(idx_contract, idx_removed, len(contract_inds), dimension_dict) 

930 cost_list.append(cost) 

931 scale_list.append(len(idx_contract)) 

932 size_list.append(_compute_size_by_dict(out_inds, dimension_dict)) 

933 

934 bcast = set() 

935 tmp_inputs = [] 

936 for x in contract_inds: 

937 tmp_inputs.append(input_list.pop(x)) 

938 bcast |= broadcast_indices.pop(x) 

939 

940 new_bcast_inds = bcast - idx_removed 

941 

942 # If we're broadcasting, nix blas 

943 if not len(idx_removed & bcast): 

944 do_blas = _can_dot(tmp_inputs, out_inds, idx_removed) 

945 else: 

946 do_blas = False 

947 

948 # Last contraction 

949 if (cnum - len(path)) == -1: 

950 idx_result = output_subscript 

951 else: 

952 sort_result = [(dimension_dict[ind], ind) for ind in out_inds] 

953 idx_result = "".join([x[1] for x in sorted(sort_result)]) 

954 

955 input_list.append(idx_result) 

956 broadcast_indices.append(new_bcast_inds) 

957 einsum_str = ",".join(tmp_inputs) + "->" + idx_result 

958 

959 contraction = (contract_inds, idx_removed, einsum_str, input_list[:], do_blas) 

960 contraction_list.append(contraction) 

961 

962 opt_cost = sum(cost_list) + 1 

963 

964 if einsum_call_arg: 

965 return (operands, contraction_list) 

966 

967 # Return the path along with a nice string representation 

968 overall_contraction = input_subscripts + "->" + output_subscript 

969 header = ("scaling", "current", "remaining") 

970 

971 speedup = naive_cost / opt_cost 

972 max_i = max(size_list) 

973 

974 path_print = " Complete contraction: %s\n" % overall_contraction 

975 path_print += " Naive scaling: %d\n" % len(indices) 

976 path_print += " Optimized scaling: %d\n" % max(scale_list) 

977 path_print += " Naive FLOP count: %.3e\n" % naive_cost 

978 path_print += " Optimized FLOP count: %.3e\n" % opt_cost 

979 path_print += " Theoretical speedup: %3.3f\n" % speedup 

980 path_print += " Largest intermediate: %.3e elements\n" % max_i 

981 path_print += "-" * 74 + "\n" 

982 path_print += "%6s %24s %40s\n" % header 

983 path_print += "-" * 74 

984 

985 for n, contraction in enumerate(contraction_list): 

986 inds, idx_rm, einsum_str, remaining, blas = contraction 

987 remaining_str = ",".join(remaining) + "->" + output_subscript 

988 path_run = (scale_list[n], einsum_str, remaining_str) 

989 path_print += "\n%4d %24s %40s" % path_run 

990 

991 path = ['einsum_path'] + path 

992 return (path, path_print)

1004@array_function_dispatch(_einsum_dispatcher, module='numpy') 

1005def einsum(*operands, **kwargs): 

1006 """ 

1007 einsum(subscripts, *operands, out=None, dtype=None, order='K', 

1008 casting='safe', optimize=False) 

1009 

1010 Evaluates the Einstein summation convention on the operands. 

1011 

1012 Using the Einstein summation convention, many common multi-dimensional, 

1013 linear algebraic array operations can be represented in a simple fashion. 

1014 In *implicit* mode `einsum` computes these values. 

1015 

1016 In *explicit* mode, `einsum` provides further flexibility to compute 

1017 other array operations that might not be considered classical Einstein 

1018 summation operations, by disabling, or forcing summation over specified 

1019 subscript labels. 

1020 

1021 See the notes and examples for clarification. 

1022 

1023 Parameters 

1024 ---------- 

1025 subscripts : str 

1026 Specifies the subscripts for summation as comma separated list of 

1027 subscript labels. An implicit (classical Einstein summation) 

1028 calculation is performed unless the explicit indicator '->' is 

1029 included as well as subscript labels of the precise output form. 

1030 operands : list of array_like 

1031 These are the arrays for the operation. 

1032 out : ndarray, optional 

1033 If provided, the calculation is done into this array. 

1034 dtype : {data-type, None}, optional 

1035 If provided, forces the calculation to use the data type specified. 

1036 Note that you may have to also give a more liberal `casting` 

1037 parameter to allow the conversions. Default is None. 

1038 order : {'C', 'F', 'A', 'K'}, optional 

1039 Controls the memory layout of the output. 'C' means it should 

1040 be C contiguous. 'F' means it should be Fortran contiguous, 

1041 'A' means it should be 'F' if the inputs are all 'F', 'C' otherwise. 

1042 'K' means it should be as close to the layout as the inputs as 

1043 is possible, including arbitrarily permuted axes. 

1044 Default is 'K'. 

1045 casting : {'no', 'equiv', 'safe', 'same_kind', 'unsafe'}, optional 

1046 Controls what kind of data casting may occur. Setting this to 

1047 'unsafe' is not recommended, as it can adversely affect accumulations. 

1048 

1049 * 'no' means the data types should not be cast at all. 

1050 * 'equiv' means only byte-order changes are allowed. 

1051 * 'safe' means only casts which can preserve values are allowed. 

1052 * 'same_kind' means only safe casts or casts within a kind, 

1053 like float64 to float32, are allowed. 

1054 * 'unsafe' means any data conversions may be done. 

1055 

1056 Default is 'safe'. 

1057 optimize : {False, True, 'greedy', 'optimal'}, optional 

1058 Controls if intermediate optimization should occur. No optimization 

1059 will occur if False and True will default to the 'greedy' algorithm. 

1060 Also accepts an explicit contraction list from the ``np.einsum_path`` 

1061 function. See ``np.einsum_path`` for more details. Defaults to False. 

1062 

1063 Returns 

1064 ------- 

1065 output : ndarray 

1066 The calculation based on the Einstein summation convention. 

1067 

1068 See Also 

1069 -------- 

1070 einsum_path, dot, inner, outer, tensordot, linalg.multi_dot 

1071 

1072 Notes 

1073 ----- 

1074 .. versionadded:: 1.6.0 

1075 

1076 The Einstein summation convention can be used to compute 

1077 many multi-dimensional, linear algebraic array operations. `einsum` 

1078 provides a succinct way of representing these. 

1079 

1080 A non-exhaustive list of these operations, 

1081 which can be computed by `einsum`, is shown below along with examples: 

1082 

1083 * Trace of an array, :py:func:`numpy.trace`. 

1084 * Return a diagonal, :py:func:`numpy.diag`. 

1085 * Array axis summations, :py:func:`numpy.sum`. 

1086 * Transpositions and permutations, :py:func:`numpy.transpose`. 

1087 * Matrix multiplication and dot product, :py:func:`numpy.matmul` :py:func:`numpy.dot`. 

1088 * Vector inner and outer products, :py:func:`numpy.inner` :py:func:`numpy.outer`. 

1089 * Broadcasting, element-wise and scalar multiplication, :py:func:`numpy.multiply`. 

1090 * Tensor contractions, :py:func:`numpy.tensordot`. 

1091 * Chained array operations, in efficient calculation order, :py:func:`numpy.einsum_path`. 

1092 

1093 The subscripts string is a comma-separated list of subscript labels, 

1094 where each label refers to a dimension of the corresponding operand. 

1095 Whenever a label is repeated it is summed, so ``np.einsum('i,i', a, b)`` 

1096 is equivalent to :py:func:`np.inner(a,b) <numpy.inner>`. If a label 

1097 appears only once, it is not summed, so ``np.einsum('i', a)`` produces a 

1098 view of ``a`` with no changes. A further example ``np.einsum('ij,jk', a, b)`` 

1099 describes traditional matrix multiplication and is equivalent to 

1100 :py:func:`np.matmul(a,b) <numpy.matmul>`. Repeated subscript labels in one 

1101 operand take the diagonal. For example, ``np.einsum('ii', a)`` is equivalent 

1102 to :py:func:`np.trace(a) <numpy.trace>`. 

1103 

1104 In *implicit mode*, the chosen subscripts are important 

1105 since the axes of the output are reordered alphabetically. This 

1106 means that ``np.einsum('ij', a)`` doesn't affect a 2D array, while 

1107 ``np.einsum('ji', a)`` takes its transpose. Additionally, 

1108 ``np.einsum('ij,jk', a, b)`` returns a matrix multiplication, while, 

1109 ``np.einsum('ij,jh', a, b)`` returns the transpose of the 

1110 multiplication since subscript 'h' precedes subscript 'i'. 

1111 

1112 In *explicit mode* the output can be directly controlled by 

1113 specifying output subscript labels. This requires the 

1114 identifier '->' as well as the list of output subscript labels. 

1115 This feature increases the flexibility of the function since 

1116 summing can be disabled or forced when required. The call 

1117 ``np.einsum('i->', a)`` is like :py:func:`np.sum(a, axis=-1) <numpy.sum>`, 

1118 and ``np.einsum('ii->i', a)`` is like :py:func:`np.diag(a) <numpy.diag>`. 

1119 The difference is that `einsum` does not allow broadcasting by default. 

1120 Additionally ``np.einsum('ij,jh->ih', a, b)`` directly specifies the 

1121 order of the output subscript labels and therefore returns matrix 

1122 multiplication, unlike the example above in implicit mode. 

1123 

1124 To enable and control broadcasting, use an ellipsis. Default 

1125 NumPy-style broadcasting is done by adding an ellipsis 

1126 to the left of each term, like ``np.einsum('...ii->...i', a)``. 

1127 To take the trace along the first and last axes, 

1128 you can do ``np.einsum('i...i', a)``, or to do a matrix-matrix 

1129 product with the left-most indices instead of rightmost, one can do 

1130 ``np.einsum('ij...,jk...->ik...', a, b)``. 

1131 

1132 When there is only one operand, no axes are summed, and no output 

1133 parameter is provided, a view into the operand is returned instead 

1134 of a new array. Thus, taking the diagonal as ``np.einsum('ii->i', a)`` 

1135 produces a view (changed in version 1.10.0). 

1136 

1137 `einsum` also provides an alternative way to provide the subscripts 

1138 and operands as ``einsum(op0, sublist0, op1, sublist1, ..., [sublistout])``. 

1139 If the output shape is not provided in this format `einsum` will be 

1140 calculated in implicit mode, otherwise it will be performed explicitly. 

1141 The examples below have corresponding `einsum` calls with the two 

1142 parameter methods. 

1143 

1144 .. versionadded:: 1.10.0 

1145 

1146 Views returned from einsum are now writeable whenever the input array 

1147 is writeable. For example, ``np.einsum('ijk...->kji...', a)`` will now 

1148 have the same effect as :py:func:`np.swapaxes(a, 0, 2) <numpy.swapaxes>` 

1149 and ``np.einsum('ii->i', a)`` will return a writeable view of the diagonal 

1150 of a 2D array. 

1151 

1152 .. versionadded:: 1.12.0 

1153 

1154 Added the ``optimize`` argument which will optimize the contraction order 

1155 of an einsum expression. For a contraction with three or more operands this 

1156 can greatly increase the computational efficiency at the cost of a larger 

1157 memory footprint during computation. 

1158 

1159 Typically a 'greedy' algorithm is applied which empirical tests have shown 

1160 returns the optimal path in the majority of cases. In some cases 'optimal' 

1161 will return the superlative path through a more expensive, exhaustive search. 

1162 For iterative calculations it may be advisable to calculate the optimal path 

1163 once and reuse that path by supplying it as an argument. An example is given 

1164 below. 

1165 

1166 See :py:func:`numpy.einsum_path` for more details. 

1167

1355 if optimize_arg is False:

1363 valid_contract_kwargs = ['optimize'] + valid_einsum_kwargs 

1364 unknown_kwargs = [k for (k, v) in kwargs.items() if 

1365 k not in valid_contract_kwargs] 

1366 

1367 if len(unknown_kwargs):

1374 if out_array is not None: 

1375 specified_out = True

1378 operands, contraction_list = einsum_path(*operands, optimize=optimize_arg, 

1379 einsum_call=True) 

1380 

1381 handle_out = False 

1382 

1383 # Start contraction loop 

1384 for num, contraction in enumerate(contraction_list): 

1385 inds, idx_rm, einsum_str, remaining, blas = contraction 

1386 tmp_operands = [operands.pop(x) for x in inds] 

1387 

1388 # Do we need to deal with the output? 

1389 handle_out = specified_out and ((num + 1) == len(contraction_list)) 

1390 

1391 # Call tensordot if still possible 

1392 if blas: 

1393 # Checks have already been handled 

1394 input_str, results_index = einsum_str.split('->') 

1395 input_left, input_right = input_str.split(',') 

1396 

1397 tensor_result = input_left + input_right 

1398 for s in idx_rm: 

1399 tensor_result = tensor_result.replace(s, "") 

1400 

1401 # Find indices to contract over 

1402 left_pos, right_pos = [], [] 

1403 for s in sorted(idx_rm): 

1404 left_pos.append(input_left.find(s)) 

1405 right_pos.append(input_right.find(s)) 

1406 

1407 # Contract! 

1408 new_view = tensordot(*tmp_operands, axes=(tuple(left_pos), tuple(right_pos))) 

1409 

1410 # Build a new view if needed 

1411 if (tensor_result != results_index) or handle_out: 

1412 if handle_out: 

1413 einsum_kwargs["out"] = out_array 

1414 new_view = c_einsum(tensor_result + '->' + results_index, new_view, **einsum_kwargs) 

1415 

1416 # Call einsum 

1417 else: 

1418 # If out was specified 

1419 if handle_out: 

1420 einsum_kwargs["out"] = out_array 

1421

1423 new_view = c_einsum(einsum_str, *tmp_operands, **einsum_kwargs) 

1424

1426 operands.append(new_view) 

1427 del tmp_operands, new_view 

1428 

1429 if specified_out: 

1430 return out_array 

1431 else: 

1432 return operands[0]

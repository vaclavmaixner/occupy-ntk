60class AxesStack(Stack): 

61 """ 

62 Specialization of the `.Stack` to handle all tracking of 

63 `~matplotlib.axes.Axes` in a `.Figure`. 

64 This stack stores ``key, (ind, axes)`` pairs, where: 

65 

66 * **key** should be a hash of the args and kwargs 

67 used in generating the Axes. 

68 * **ind** is a serial number for tracking the order 

69 in which axes were added. 

70 

71 The AxesStack is a callable, where ``ax_stack()`` returns 

72 the current axes. Alternatively the :meth:`current_key_axes` will 

73 return the current key and associated axes. 

74 

75 """

83 """

88 def get(self, key):

100 "a new instance will always be created and returned. Meanwhile, "

121 """ 

122 Add Axes *a*, with key *key*, to the stack, and return the stack. 

123 

124 If *key* is unhashable, replace it by a unique, arbitrary object.

1021 def fixitems(items): 

1022 # items may have arrays and lists in them, so convert them 

1023 # to tuples for the key 

1024 ret = [] 

1025 for k, v in items: 

1026 # some objects can define __getitem__ without being 

1027 # iterable and in those cases the conversion to tuples 

1028 # will fail. So instead of using the iterable(v) function 

1029 # we simply try and convert to a tuple, and proceed if not. 

1030 try: 

1031 v = tuple(v) 

1032 except Exception: 

1033 pass 

1034 ret.append((k, v)) 

1035 return tuple(ret)

1375 def subplots(self, nrows=1, ncols=1, sharex=False, sharey=False, 

1376 squeeze=True, subplot_kw=None, gridspec_kw=None): 

1377 """ 

1378 Add a set of subplots to this figure. 

1379 

1380 This utility wrapper makes it convenient to create common layouts of 

1381 subplots in a single call. 

1382 

1383 Parameters 

1384 ---------- 

1385 nrows, ncols : int, optional, default: 1 

1386 Number of rows/columns of the subplot grid. 

1387 

1388 sharex, sharey : bool or {'none', 'all', 'row', 'col'}, default: False 

1389 Controls sharing of properties among x (`sharex`) or y (`sharey`) 

1390 axes: 

1391 

1392 - True or 'all': x- or y-axis will be shared among all 

1393 subplots. 

1394 - False or 'none': each subplot x- or y-axis will be 

1395 independent. 

1396 - 'row': each subplot row will share an x- or y-axis. 

1397 - 'col': each subplot column will share an x- or y-axis. 

1398 

1399 When subplots have a shared x-axis along a column, only the x tick 

1400 labels of the bottom subplot are created. Similarly, when subplots 

1401 have a shared y-axis along a row, only the y tick labels of the 

1402 first column subplot are created. To later turn other subplots' 

1403 ticklabels on, use `~matplotlib.axes.Axes.tick_params`. 

1404 

1405 squeeze : bool, optional, default: True 

1406 - If True, extra dimensions are squeezed out from the returned 

1407 array of Axes: 

1408 

1409 - if only one subplot is constructed (nrows=ncols=1), the 

1410 resulting single Axes object is returned as a scalar. 

1411 - for Nx1 or 1xM subplots, the returned object is a 1D numpy 

1412 object array of Axes objects. 

1413 - for NxM, subplots with N>1 and M>1 are returned 

1414 as a 2D array. 

1415 

1416 - If False, no squeezing at all is done: the returned Axes object 

1417 is always a 2D array containing Axes instances, even if it ends 

1418 up being 1x1. 

1419 

1420 subplot_kw : dict, optional 

1421 Dict with keywords passed to the 

1422 :meth:`~matplotlib.figure.Figure.add_subplot` call used to create 

1423 each subplot. 

1424 

1425 gridspec_kw : dict, optional 

1426 Dict with keywords passed to the 

1427 `~matplotlib.gridspec.GridSpec` constructor used to create 

1428 the grid the subplots are placed on. 

1429 

1430 Returns 

1431 ------- 

1432 ax : `~.axes.Axes` object or array of Axes objects. 

1433 *ax* can be either a single `~matplotlib.axes.Axes` object or 

1434 an array of Axes objects if more than one subplot was created. The 

1435 dimensions of the resulting array can be controlled with the 

1436 squeeze keyword, see above. 

1437 

1438 Examples 

1439 -------- 

1440 ::

1443 x = np.linspace(0, 2*np.pi, 400) 

1444 y = np.sin(x**2) 

1445 

1446 # Create a figure 

1447 plt.figure(1, clear=True) 

1448 

1449 # Creates a subplot 

1450 ax = fig.subplots() 

1451 ax.plot(x, y) 

1452 ax.set_title('Simple plot') 

1453 

1454 # Creates two subplots and unpacks the output array immediately 

1455 ax1, ax2 = fig.subplots(1, 2, sharey=True) 

1456 ax1.plot(x, y) 

1457 ax1.set_title('Sharing Y axis') 

1458 ax2.scatter(x, y) 

1459 

1460 # Creates four polar axes, and accesses them through the 

1461 # returned array 

1462 axes = fig.subplots(2, 2, subplot_kw=dict(polar=True)) 

1463 axes[0, 0].plot(x, y) 

1464 axes[1, 1].scatter(x, y) 

1465 

1466 # Share a X axis with each column of subplots 

1467 fig.subplots(2, 2, sharex='col') 

1468 

1469 # Share a Y axis with each row of subplots 

1470 fig.subplots(2, 2, sharey='row') 

1471 

1472 # Share both X and Y axes with all subplots 

1473 fig.subplots(2, 2, sharex='all', sharey='all') 

1474

1476 fig.subplots(2, 2, sharex=True, sharey=True) 

1477 

1478 See Also 

1479 -------- 

1480 .pyplot.subplots 

1481 .Figure.add_subplot

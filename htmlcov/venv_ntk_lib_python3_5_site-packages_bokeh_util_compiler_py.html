214class FromFile(Implementation): 

215 ''' A custom model implementation read from a separate source file. 

216 

217 Args: 

218 path (str) : 

219 The path to the file containing the extension source code 

220 

221 ''' 

222 def __init__(self, path): 

223 with io.open(path, encoding="utf-8") as f: 

224 self.code = f.read() 

225 self.file = path 

226 

227 @property 

228 def lang(self): 

229 if self.file.endswith(".coffee"): 

230 return "coffeescript" 

231 if self.file.endswith(".ts"): 

232 return "typescript" 

233 if self.file.endswith(".js"): 

234 return "javascript" 

235 if self.file.endswith((".css", ".less")): 

236 return "less"

241class CustomModel(object): 

242 ''' Represent a custom (user-defined) Bokeh model. 

243 

244 ''' 

245 def __init__(self, cls): 

246 self.cls = cls 

247 

248 @property 

249 def name(self): 

250 return self.cls.__name__ 

251 

252 @property 

253 def full_name(self): 

254 name = self.cls.__module__ + "." + self.name 

255 return name.replace("__main__.", "") 

256 

257 @property 

258 def file(self): 

259 module = sys.modules[self.cls.__module__] 

260 

261 if hasattr(module, "__file__"): 

262 return abspath(module.__file__) 

263 else: 

264 return None 

265 

266 @property 

267 def path(self): 

268 path = getattr(self.cls, "__base_path__", None) 

269 

270 if path is not None: 

271 return path 

272 elif self.file is not None: 

273 return dirname(self.file) 

274 else: 

275 return os.getcwd() 

276 

277 @property 

278 def implementation(self): 

279 impl = self.cls.__implementation__ 

280 

281 if isinstance(impl, six.string_types): 

282 if "\n" not in impl and impl.endswith(exts): 

283 impl = FromFile(impl if isabs(impl) else join(self.path, impl)) 

284 else: 

285 impl = CoffeeScript(impl) 

286 

287 if isinstance(impl, Inline) and impl.file is None: 

288 file = "%s%s.ts" % (self.file + ":" if self.file else "", self.name) 

289 impl = impl.__class__(impl.code, file) 

290 

291 return impl 

292 

293 @property 

294 def dependencies(self): 

295 return getattr(self.cls, "__dependencies__", {}) 

296 

297 @property 

298 def module(self):

312def calc_cache_key(custom_models): 

313 ''' Generate a key to cache a custom extension implementation with. 

314 

315 There is no metadata other than the Model classes, so this is the only 

316 base to generate a cache key. 

317 

318 We build the model keys from the list of ``model.full_name``. This is 

319 not ideal but possibly a better solution can be found found later. 

320 

321 ''' 

322 model_names = {model.full_name for model in custom_models.values()} 

323 encoded_names = ",".join(sorted(model_names)).encode('utf-8') 

324 return hashlib.sha256(encoded_names).hexdigest()

412def _detect_nodejs(): 

413 if settings.nodejs_path() is not None: 

414 nodejs_paths = [settings.nodejs_path()] 

415 else: 

416 nodejs_paths = ["nodejs", "node"] 

417 

418 for nodejs_path in nodejs_paths: 

419 try: 

420 proc = Popen([nodejs_path, "--version"], stdout=PIPE, stderr=PIPE) 

421 (stdout, _) = proc.communicate() 

422 except OSError: 

423 continue 

424 

425 if proc.returncode != 0: 

426 continue 

427 

428 match = re.match(r"^v(\d+)\.(\d+)\.(\d+).*$", stdout.decode("utf-8")) 

429 

430 if match is not None: 

431 version = tuple(int(v) for v in match.groups()) 

432 

433 if version >= nodejs_min_version: 

434 return nodejs_path 

435 

436 # if we've reached here, no valid version was found 

437 version = ".".join(map(str, nodejs_min_version)) 

438 raise RuntimeError('node.js v%s or higher is needed to allow compilation of custom models ' % version + 

439 '("conda install nodejs" or follow https://nodejs.org/en/download/)')

461def _run(app, argv, input=None): 

462 proc = Popen([app] + argv, stdout=PIPE, stderr=PIPE, stdin=PIPE) 

463 (stdout, errout) = proc.communicate(input=None if input is None else json.dumps(input).encode()) 

464 

465 if proc.returncode != 0: 

466 raise RuntimeError(errout.decode('utf-8')) 

467 else: 

468 return _crlf_cr_2_lf(stdout.decode('utf-8'))

507def _compile_models(custom_models): 

508 """Returns the compiled implementation of supplied `models`. """ 

509 ordered_models = sorted(custom_models.values(), key=lambda model: model.full_name) 

510 custom_impls = {} 

511 

512 dependencies = [] 

513 for model in ordered_models: 

514 dependencies.extend(list(model.dependencies.items())) 

515 

516 if dependencies: 

517 dependencies = sorted(dependencies, key=lambda name_version: name_version[0]) 

518 _run_npmjs(["install", "--no-progress"] + [ name + "@" + version for (name, version) in dependencies ]) 

519 

520 for model in ordered_models: 

521 impl = model.implementation 

522 compiled = _CACHING_IMPLEMENTATION(model, impl) 

523 if compiled is None: 

524 compiled = nodejs_compile(impl.code, lang=impl.lang, file=impl.file) 

525 

526 if "error" in compiled: 

527 raise CompilationError(compiled.error) 

528 

529 custom_impls[model.full_name] = compiled 

530 

531 return custom_impls

533def _bundle_models(custom_models): 

534 """ Create a JavaScript bundle with selected `models`. """ 

535 exports = [] 

536 modules = [] 

537 

538 with io.open(join(bokehjs_dir, "js", "bokeh.json"), encoding="utf-8") as f: 

539 bokeh = json.loads(f.read()) 

540 

541 known_modules = set() 

542 for artifact in bokeh["artifacts"]: 

543 canonical = artifact["module"].get("canonical") 

544 if canonical is not None: 

545 known_modules.add(canonical) 

546 

547 custom_impls = _compile_models(custom_models) 

548 

549 extra_modules = {} 

550 

551 def resolve_modules(to_resolve, root): 

552 resolved = {} 

553 for module in to_resolve: 

554 if module.startswith(("./", "../")): 

555 def mkpath(module, ext=""): 

556 return abspath(join(root, *module.split("/")) + ext) 

557 

558 if module.endswith(exts): 

559 path = mkpath(module) 

560 if not exists(path): 

561 raise RuntimeError("no such module: %s" % module) 

562 else: 

563 for ext in exts: 

564 path = mkpath(module, ext) 

565 if exists(path): 

566 break 

567 else: 

568 raise RuntimeError("no such module: %s" % module) 

569 

570 impl = FromFile(path) 

571 compiled = nodejs_compile(impl.code, lang=impl.lang, file=impl.file) 

572 

573 if "error" in compiled: 

574 raise CompilationError(compiled.error) 

575 

576 if impl.lang == "less": 

577 code = _style_template % dict(css=json.dumps(compiled.code)) 

578 deps = [] 

579 else: 

580 code = compiled.code 

581 deps = compiled.deps 

582 

583 sig = hashlib.sha256(code.encode('utf-8')).hexdigest() 

584 resolved[module] = sig 

585 

586 deps_map = resolve_deps(deps, dirname(path)) 

587 

588 if sig not in extra_modules: 

589 extra_modules[sig] = True 

590 modules.append((sig, code, deps_map)) 

591 else: 

592 index = module + ("" if module.endswith("/") else "/") + "index" 

593 if index not in known_modules: 

594 raise RuntimeError("no such module: %s" % module) 

595 

596 return resolved 

597 

598 def resolve_deps(deps, root): 

599 custom_modules = set(model.module for model in custom_models.values()) 

600 missing = set(deps) - known_modules - custom_modules 

601 return resolve_modules(missing, root) 

602 

603 for model in custom_models.values(): 

604 compiled = custom_impls[model.full_name] 

605 deps_map = resolve_deps(compiled.deps, model.path) 

606 

607 exports.append((model.name, model.module)) 

608 modules.append((model.module, compiled.code, deps_map)) 

609 

610 # sort everything by module name 

611 exports = sorted(exports, key=lambda spec: spec[1]) 

612 modules = sorted(modules, key=lambda spec: spec[0]) 

613 

614 for i, (module, code, deps) in enumerate(modules): 

615 for name, ref in deps.items(): 

616 code = code.replace("""require("%s")""" % name, """require("%s")""" % ref) 

617 code = code.replace("""require('%s')""" % name, """require('%s')""" % ref) 

618 modules[i] = (module, code) 

619 

620 sep = ",\n" 

621 

622 exports = sep.join(_export_template % dict(name=name, module=module) for (name, module) in exports) 

623 modules = sep.join(_module_template % dict(module=module, source=code) for (module, code) in modules) 

624 

625 content = _plugin_template % dict(prelude=_plugin_prelude, exports=exports, modules=modules) 

626 return _plugin_umd % dict(content=content)

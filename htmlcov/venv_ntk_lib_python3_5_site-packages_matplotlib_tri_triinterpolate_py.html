16class TriInterpolator(object): 

17 """ 

18 Abstract base class for classes used to perform interpolation on 

19 triangular grids. 

20 

21 Derived classes implement the following methods: 

22 

23 - ``__call__(x, y)`` , 

24 where x, y are array_like point coordinates of the same shape, and 

25 that returns a masked array of the same shape containing the 

26 interpolated z-values. 

27 

28 - ``gradient(x, y)`` , 

29 where x, y are array_like point coordinates of the same 

30 shape, and that returns a list of 2 masked arrays of the same shape 

31 containing the 2 derivatives of the interpolator (derivatives of 

32 interpolated z values with respect to x and y). 

33 

34 """ 

35 def __init__(self, triangulation, z, trifinder=None): 

36 if not isinstance(triangulation, Triangulation): 

37 raise ValueError("Expected a Triangulation object") 

38 self._triangulation = triangulation 

39 

40 self._z = np.asarray(z) 

41 if self._z.shape != self._triangulation.x.shape: 

42 raise ValueError("z array must have same length as triangulation x" 

43 " and y arrays") 

44 

45 if trifinder is not None and not isinstance(trifinder, TriFinder): 

46 raise ValueError("Expected a TriFinder object") 

47 self._trifinder = trifinder or self._triangulation.get_trifinder() 

48 

49 # Default scaling factors : 1.0 (= no scaling) 

50 # Scaling may be used for interpolations for which the order of 

51 # magnitude of x, y has an impact on the interpolant definition. 

52 # Please refer to :meth:`_interpolate_multikeys` for details. 

53 self._unit_x = 1.0 

54 self._unit_y = 1.0 

55 

56 # Default triangle renumbering: None (= no renumbering) 

57 # Renumbering may be used to avoid unnecessary computations 

58 # if complex calculations are done inside the Interpolator. 

59 # Please refer to :meth:`_interpolate_multikeys` for details. 

60 self._tri_renum = None 

61 

62 # __call__ and gradient docstrings are shared by all subclasses 

63 # (except, if needed, relevant additions). 

64 # However these methods are only implemented in subclasses to avoid 

65 # confusion in the documentation. 

66 _docstring__call__ = """ 

67 Returns a masked array containing interpolated values at the specified 

68 x,y points. 

69 

70 Parameters 

71 ---------- 

72 x, y : array-like 

73 x and y coordinates of the same shape and any number of 

74 dimensions. 

75 

76 Returns 

77 ------- 

78 z : np.ma.array 

79 Masked array of the same shape as *x* and *y*; values corresponding 

80 to (*x*, *y*) points outside of the triangulation are masked out. 

81 

82 """

106 def _interpolate_multikeys(self, x, y, tri_index=None, 

107 return_keys=('z',)): 

108 """ 

109 Versatile (private) method defined for all TriInterpolators. 

110 

111 :meth:`_interpolate_multikeys` is a wrapper around method 

112 :meth:`_interpolate_single_key` (to be defined in the child 

113 subclasses). 

114 :meth:`_interpolate_single_key actually performs the interpolation, 

115 but only for 1-dimensional inputs and at valid locations (inside 

116 unmasked triangles of the triangulation). 

117 

118 The purpose of :meth:`_interpolate_multikeys` is to implement the 

119 following common tasks needed in all subclasses implementations: 

120 

121 - calculation of containing triangles 

122 - dealing with more than one interpolation request at the same 

123 location (e.g., if the 2 derivatives are requested, it is 

124 unnecessary to compute the containing triangles twice) 

125 - scaling according to self._unit_x, self._unit_y 

126 - dealing with points outside of the grid (with fill value np.nan) 

127 - dealing with multi-dimensionnal *x*, *y* arrays: flattening for 

128 :meth:`_interpolate_params` call and final reshaping. 

129 

130 (Note that np.vectorize could do most of those things very well for 

131 you, but it does it by function evaluations over successive tuples of 

132 the input arrays. Therefore, this tends to be more time consuming than 

133 using optimized numpy functions - e.g., np.dot - which can be used 

134 easily on the flattened inputs, in the child-subclass methods 

135 :meth:`_interpolate_single_key`.) 

136 

137 It is guaranteed that the calls to :meth:`_interpolate_single_key` 

138 will be done with flattened (1-d) array_like input parameters `x`, `y` 

139 and with flattened, valid `tri_index` arrays (no -1 index allowed). 

140 

141 Parameters 

142 ---------- 

143 x, y : array_like 

144 x and y coordinates indicating where interpolated values are 

145 requested. 

146 tri_index : integer array_like, optional 

147 Array of the containing triangle indices, same shape as 

148 *x* and *y*. Defaults to None. If None, these indices 

149 will be computed by a TriFinder instance. 

150 (Note: For point outside the grid, tri_index[ipt] shall be -1). 

151 return_keys : tuple of keys from {'z', 'dzdx', 'dzdy'} 

152 Defines the interpolation arrays to return, and in which order. 

153 

154 Returns 

155 ------- 

156 ret : list of arrays 

157 Each array-like contains the expected interpolated values in the 

158 order defined by *return_keys* parameter. 

159 """ 

160 # Flattening and rescaling inputs arrays x, y 

161 # (initial shape is stored for output) 

162 x = np.asarray(x, dtype=np.float64) 

163 y = np.asarray(y, dtype=np.float64) 

164 sh_ret = x.shape 

165 if x.shape != y.shape: 

166 raise ValueError("x and y shall have same shapes." 

167 " Given: {0} and {1}".format(x.shape, y.shape)) 

168 x = np.ravel(x) 

169 y = np.ravel(y) 

170 x_scaled = x/self._unit_x 

171 y_scaled = y/self._unit_y 

172 size_ret = np.size(x_scaled) 

173 

174 # Computes & ravels the element indexes, extract the valid ones. 

175 if tri_index is None: 

176 tri_index = self._trifinder(x, y) 

177 else: 

178 if tri_index.shape != sh_ret: 

179 raise ValueError( 

180 "tri_index array is provided and shall" 

181 " have same shape as x and y. Given: " 

182 "{0} and {1}".format(tri_index.shape, sh_ret)) 

183 tri_index = np.ravel(tri_index) 

184 

185 mask_in = (tri_index != -1) 

186 if self._tri_renum is None: 

187 valid_tri_index = tri_index[mask_in] 

188 else: 

189 valid_tri_index = self._tri_renum[tri_index[mask_in]] 

190 valid_x = x_scaled[mask_in] 

191 valid_y = y_scaled[mask_in] 

192 

193 ret = [] 

194 for return_key in return_keys: 

195 # Find the return index associated with the key. 

196 try: 

197 return_index = {'z': 0, 'dzdx': 1, 'dzdy': 2}[return_key] 

198 except KeyError: 

199 raise ValueError("return_keys items shall take values in" 

200 " {'z', 'dzdx', 'dzdy'}") 

201 

202 # Sets the scale factor for f & df components 

203 scale = [1., 1./self._unit_x, 1./self._unit_y][return_index] 

204 

205 # Computes the interpolation 

206 ret_loc = np.empty(size_ret, dtype=np.float64) 

207 ret_loc[~mask_in] = np.nan 

208 ret_loc[mask_in] = self._interpolate_single_key( 

209 return_key, valid_tri_index, valid_x, valid_y) * scale 

210 ret += [np.ma.masked_invalid(ret_loc.reshape(sh_ret), copy=False)] 

211 

212 return ret

238 """ 

239 A LinearTriInterpolator performs linear interpolation on a triangular grid. 

240 

241 Each triangle is represented by a plane so that an interpolated value at 

242 point (x,y) lies on the plane of the triangle containing (x,y). 

243 Interpolated values are therefore continuous across the triangulation, but 

244 their first derivatives are discontinuous at edges between triangles. 

245 

246 Parameters 

247 ---------- 

248 triangulation : :class:`~matplotlib.tri.Triangulation` object 

249 The triangulation to interpolate over. 

250 z : array_like of shape (npoints,) 

251 Array of values, defined at grid points, to interpolate between. 

252 trifinder : :class:`~matplotlib.tri.TriFinder` object, optional 

253 If this is not specified, the Triangulation's default TriFinder will 

254 be used by calling 

255 :func:`matplotlib.tri.Triangulation.get_trifinder`. 

256 

257 Methods 

258 ------- 

259 `__call__` (x, y) : Returns interpolated values at x,y points 

260 `gradient` (x, y) : Returns interpolated derivatives at x,y points 

261 

262 """

293class CubicTriInterpolator(TriInterpolator): 

294 r""" 

295 A CubicTriInterpolator performs cubic interpolation on triangular grids. 

296 

297 In one-dimension - on a segment - a cubic interpolating function is 

298 defined by the values of the function and its derivative at both ends. 

299 This is almost the same in 2-d inside a triangle, except that the values 

300 of the function and its 2 derivatives have to be defined at each triangle 

301 node. 

302 

303 The CubicTriInterpolator takes the value of the function at each node - 

304 provided by the user - and internally computes the value of the 

305 derivatives, resulting in a smooth interpolation. 

306 (As a special feature, the user can also impose the value of the 

307 derivatives at each node, but this is not supposed to be the common 

308 usage.) 

309 

310 Parameters 

311 ---------- 

312 triangulation : :class:`~matplotlib.tri.Triangulation` object 

313 The triangulation to interpolate over. 

314 z : array_like of shape (npoints,) 

315 Array of values, defined at grid points, to interpolate between. 

316 kind : {'min_E', 'geom', 'user'}, optional 

317 Choice of the smoothing algorithm, in order to compute 

318 the interpolant derivatives (defaults to 'min_E'): 

319 

320 - if 'min_E': (default) The derivatives at each node is computed 

321 to minimize a bending energy. 

322 - if 'geom': The derivatives at each node is computed as a 

323 weighted average of relevant triangle normals. To be used for 

324 speed optimization (large grids). 

325 - if 'user': The user provides the argument `dz`, no computation 

326 is hence needed. 

327 

328 trifinder : :class:`~matplotlib.tri.TriFinder` object, optional 

329 If not specified, the Triangulation's default TriFinder will 

330 be used by calling 

331 :func:`matplotlib.tri.Triangulation.get_trifinder`. 

332 dz : tuple of array_likes (dzdx, dzdy), optional 

333 Used only if *kind* ='user'. In this case *dz* must be provided as 

334 (dzdx, dzdy) where dzdx, dzdy are arrays of the same shape as *z* and 

335 are the interpolant first derivatives at the *triangulation* points. 

336 

337 Methods 

338 ------- 

339 `__call__` (x, y) : Returns interpolated values at x,y points 

340 `gradient` (x, y) : Returns interpolated derivatives at x,y points 

341 

342 Notes 

343 ----- 

344 This note is a bit technical and details the way a 

345 :class:`~matplotlib.tri.CubicTriInterpolator` computes a cubic 

346 interpolation. 

347 

348 The interpolation is based on a Clough-Tocher subdivision scheme of 

349 the *triangulation* mesh (to make it clearer, each triangle of the 

350 grid will be divided in 3 child-triangles, and on each child triangle 

351 the interpolated function is a cubic polynomial of the 2 coordinates). 

352 This technique originates from FEM (Finite Element Method) analysis; 

353 the element used is a reduced Hsieh-Clough-Tocher (HCT) 

354 element. Its shape functions are described in [1]_. 

355 The assembled function is guaranteed to be C1-smooth, i.e. it is 

356 continuous and its first derivatives are also continuous (this 

357 is easy to show inside the triangles but is also true when crossing the 

358 edges). 

359 

360 In the default case (*kind* ='min_E'), the interpolant minimizes a 

361 curvature energy on the functional space generated by the HCT element 

362 shape functions - with imposed values but arbitrary derivatives at each 

363 node. The minimized functional is the integral of the so-called total 

364 curvature (implementation based on an algorithm from [2]_ - PCG sparse 

365 solver): 

366 

367 .. math:: 

368 

369 E(z) = \frac{1}{2} \int_{\Omega} \left( 

370 \left( \frac{\partial^2{z}}{\partial{x}^2} \right)^2 + 

371 \left( \frac{\partial^2{z}}{\partial{y}^2} \right)^2 + 

372 2\left( \frac{\partial^2{z}}{\partial{y}\partial{x}} \right)^2 

373 \right) dx\,dy 

374 

375 If the case *kind* ='geom' is chosen by the user, a simple geometric 

376 approximation is used (weighted average of the triangle normal 

377 vectors), which could improve speed on very large grids. 

378 

379 References 

380 ---------- 

381 .. [1] Michel Bernadou, Kamal Hassan, "Basis functions for general 

382 Hsieh-Clough-Tocher triangles, complete or reduced.", 

383 International Journal for Numerical Methods in Engineering, 

384 17(5):784 - 789. 2.01. 

385 .. [2] C.T. Kelley, "Iterative Methods for Optimization". 

386 

387 """

494 def _get_alpha_vec(x, y, tris_pts): 

495 """ 

496 Fast (vectorized) function to compute barycentric coordinates alpha. 

497 

498 Parameters 

499 ---------- 

500 x, y : array-like of dim 1 (shape (nx,)) 

501 Coordinates of the points whose points barycentric 

502 coordinates are requested 

503 tris_pts : array like of dim 3 (shape: (nx,3,2)) 

504 Coordinates of the containing triangles apexes. 

505 

506 Returns 

507 ------- 

508 alpha : array of dim 2 (shape (nx,3)) 

509 Barycentric coordinates of the points inside the containing 

510 triangles. 

511 """ 

512 ndim = tris_pts.ndim-2 

513 

514 a = tris_pts[:, 1, :] - tris_pts[:, 0, :] 

515 b = tris_pts[:, 2, :] - tris_pts[:, 0, :] 

516 abT = np.stack([a, b], axis=-1) 

517 ab = _transpose_vectorized(abT) 

518 OM = np.stack([x, y], axis=1) - tris_pts[:, 0, :] 

519 

520 metric = _prod_vectorized(ab, abT) 

521 # Here we try to deal with the colinear cases. 

522 # metric_inv is in this case set to the Moore-Penrose pseudo-inverse 

523 # meaning that we will still return a set of valid barycentric 

524 # coordinates. 

525 metric_inv = _pseudo_inv22sym_vectorized(metric) 

526 Covar = _prod_vectorized(ab, _transpose_vectorized( 

527 np.expand_dims(OM, ndim))) 

528 ksi = _prod_vectorized(metric_inv, Covar) 

529 alpha = _to_matrix_vectorized([ 

530 [1-ksi[:, 0, 0]-ksi[:, 1, 0]], [ksi[:, 0, 0]], [ksi[:, 1, 0]]]) 

531 return alpha

534 def _get_jacobian(tris_pts): 

535 """ 

536 Fast (vectorized) function to compute triangle jacobian matrix. 

537 

538 Parameters 

539 ---------- 

540 tris_pts : array like of dim 3 (shape: (nx,3,2)) 

541 Coordinates of the containing triangles apexes. 

542 

543 Returns 

544 ------- 

545 J : array of dim 3 (shape (nx,2,2)) 

546 Barycentric coordinates of the points inside the containing 

547 triangles. 

548 J[itri,:,:] is the jacobian matrix at apex 0 of the triangle 

549 itri, so that the following (matrix) relationship holds: 

550 [dz/dksi] = [J] x [dz/dx] 

551 with x: global coordinates 

552 ksi: element parametric coordinates in triangle first apex 

553 local basis. 

554 """ 

555 a = np.array(tris_pts[:, 1, :] - tris_pts[:, 0, :]) 

556 b = np.array(tris_pts[:, 2, :] - tris_pts[:, 0, :]) 

557 J = _to_matrix_vectorized([[a[:, 0], a[:, 1]], 

558 [b[:, 0], b[:, 1]]]) 

559 return J

562 def _compute_tri_eccentricities(tris_pts): 

563 """ 

564 Computes triangle eccentricities 

565 

566 Parameters 

567 ---------- 

568 tris_pts : array like of dim 3 (shape: (nx,3,2)) 

569 Coordinates of the triangles apexes. 

570 

571 Returns 

572 ------- 

573 ecc : array like of dim 2 (shape: (nx,3)) 

574 The so-called eccentricity parameters [1] needed for 

575 HCT triangular element. 

576 """ 

577 a = np.expand_dims(tris_pts[:, 2, :] - tris_pts[:, 1, :], axis=2) 

578 b = np.expand_dims(tris_pts[:, 0, :] - tris_pts[:, 2, :], axis=2) 

579 c = np.expand_dims(tris_pts[:, 1, :] - tris_pts[:, 0, :], axis=2) 

580 # Do not use np.squeeze, this is dangerous if only one triangle 

581 # in the triangulation... 

582 dot_a = _prod_vectorized(_transpose_vectorized(a), a)[:, 0, 0] 

583 dot_b = _prod_vectorized(_transpose_vectorized(b), b)[:, 0, 0] 

584 dot_c = _prod_vectorized(_transpose_vectorized(c), c)[:, 0, 0] 

585 # Note that this line will raise a warning for dot_a, dot_b or dot_c

690 def get_function_values(self, alpha, ecc, dofs): 

691 """ 

692 Parameters 

693 ---------- 

694 alpha : is a (N x 3 x 1) array (array of column-matrices) of 

695 barycentric coordinates, 

696 ecc : is a (N x 3 x 1) array (array of column-matrices) of triangle 

697 eccentricities, 

698 dofs : is a (N x 1 x 9) arrays (arrays of row-matrices) of computed 

699 degrees of freedom. 

700 

701 Returns 

702 ------- 

703 Returns the N-array of interpolated function values. 

704 """ 

705 subtri = np.argmin(alpha, axis=1)[:, 0] 

706 ksi = _roll_vectorized(alpha, -subtri, axis=0) 

707 E = _roll_vectorized(ecc, -subtri, axis=0) 

708 x = ksi[:, 0, 0] 

709 y = ksi[:, 1, 0] 

710 z = ksi[:, 2, 0] 

711 x_sq = x*x 

712 y_sq = y*y 

713 z_sq = z*z 

714 V = _to_matrix_vectorized([ 

715 [x_sq*x], [y_sq*y], [z_sq*z], [x_sq*z], [x_sq*y], [y_sq*x], 

716 [y_sq*z], [z_sq*y], [z_sq*x], [x*y*z]]) 

717 prod = _prod_vectorized(self.M, V) 

718 prod += _scalar_vectorized(E[:, 0, 0], 

719 _prod_vectorized(self.M0, V)) 

720 prod += _scalar_vectorized(E[:, 1, 0], 

721 _prod_vectorized(self.M1, V)) 

722 prod += _scalar_vectorized(E[:, 2, 0], 

723 _prod_vectorized(self.M2, V)) 

724 s = _roll_vectorized(prod, 3*subtri, axis=0) 

725 return _prod_vectorized(dofs, s)[:, 0, 0]

727 def get_function_derivatives(self, alpha, J, ecc, dofs): 

728 """ 

729 Parameters 

730 ---------- 

731 *alpha* is a (N x 3 x 1) array (array of column-matrices of 

732 barycentric coordinates) 

733 *J* is a (N x 2 x 2) array of jacobian matrices (jacobian matrix at 

734 triangle first apex) 

735 *ecc* is a (N x 3 x 1) array (array of column-matrices of triangle 

736 eccentricities) 

737 *dofs* is a (N x 1 x 9) arrays (arrays of row-matrices) of computed 

738 degrees of freedom. 

739 

740 Returns 

741 ------- 

742 Returns the values of interpolated function derivatives [dz/dx, dz/dy] 

743 in global coordinates at locations alpha, as a column-matrices of 

744 shape (N x 2 x 1). 

745 """ 

746 subtri = np.argmin(alpha, axis=1)[:, 0] 

747 ksi = _roll_vectorized(alpha, -subtri, axis=0) 

748 E = _roll_vectorized(ecc, -subtri, axis=0) 

749 x = ksi[:, 0, 0] 

750 y = ksi[:, 1, 0] 

751 z = ksi[:, 2, 0] 

752 x_sq = x*x 

753 y_sq = y*y 

754 z_sq = z*z 

755 dV = _to_matrix_vectorized([ 

756 [ -3.*x_sq, -3.*x_sq], 

757 [ 3.*y_sq, 0.], 

758 [ 0., 3.*z_sq], 

759 [ -2.*x*z, -2.*x*z+x_sq], 

760 [-2.*x*y+x_sq, -2.*x*y], 

761 [ 2.*x*y-y_sq, -y_sq], 

762 [ 2.*y*z, y_sq], 

763 [ z_sq, 2.*y*z], 

764 [ -z_sq, 2.*x*z-z_sq], 

765 [ x*z-y*z, x*y-y*z]]) 

766 # Puts back dV in first apex basis 

767 dV = _prod_vectorized(dV, _extract_submatrices( 

768 self.rotate_dV, subtri, block_size=2, axis=0)) 

769 

770 prod = _prod_vectorized(self.M, dV) 

771 prod += _scalar_vectorized(E[:, 0, 0], 

772 _prod_vectorized(self.M0, dV)) 

773 prod += _scalar_vectorized(E[:, 1, 0], 

774 _prod_vectorized(self.M1, dV)) 

775 prod += _scalar_vectorized(E[:, 2, 0], 

776 _prod_vectorized(self.M2, dV)) 

777 dsdksi = _roll_vectorized(prod, 3*subtri, axis=0) 

778 dfdksi = _prod_vectorized(dofs, dsdksi) 

779 # In global coordinates: 

780 # Here we try to deal with the simplest colinear cases, returning a 

781 # null matrix. 

782 J_inv = _safe_inv22_vectorized(J) 

783 dfdx = _prod_vectorized(J_inv, _transpose_vectorized(dfdksi)) 

784 return dfdx

855 def get_bending_matrices(self, J, ecc): 

856 """ 

857 Parameters 

858 ---------- 

859 *J* is a (N x 2 x 2) array of jacobian matrices (jacobian matrix at 

860 triangle first apex) 

861 *ecc* is a (N x 3 x 1) array (array of column-matrices) of triangle 

862 eccentricities 

863 

864 Returns 

865 ------- 

866 Returns the element K matrices for bending energy expressed in 

867 GLOBAL nodal coordinates. 

868 K_ij = integral [ (d2zi/dx2 + d2zi/dy2) * (d2zj/dx2 + d2zj/dy2) dA] 

869 tri_J is needed to rotate dofs from local basis to global basis 

870 """ 

871 n = np.size(ecc, 0) 

872 

873 # 1) matrix to rotate dofs in global coordinates 

874 J1 = _prod_vectorized(self.J0_to_J1, J) 

875 J2 = _prod_vectorized(self.J0_to_J2, J) 

876 DOF_rot = np.zeros([n, 9, 9], dtype=np.float64) 

877 DOF_rot[:, 0, 0] = 1 

878 DOF_rot[:, 3, 3] = 1 

879 DOF_rot[:, 6, 6] = 1 

880 DOF_rot[:, 1:3, 1:3] = J 

881 DOF_rot[:, 4:6, 4:6] = J1 

882 DOF_rot[:, 7:9, 7:9] = J2 

883 

884 # 2) matrix to rotate Hessian in global coordinates. 

885 H_rot, area = self.get_Hrot_from_J(J, return_area=True) 

886 

887 # 3) Computes stiffness matrix 

888 # Gauss quadrature. 

889 K = np.zeros([n, 9, 9], dtype=np.float64) 

890 weights = self.gauss_w 

891 pts = self.gauss_pts 

892 for igauss in range(self.n_gauss): 

893 alpha = np.tile(pts[igauss, :], n).reshape(n, 3) 

894 alpha = np.expand_dims(alpha, 2) 

895 weight = weights[igauss] 

896 d2Skdksi2 = self.get_d2Sidksij2(alpha, ecc) 

897 d2Skdx2 = _prod_vectorized(d2Skdksi2, H_rot) 

898 K += weight * _prod_vectorized(_prod_vectorized(d2Skdx2, self.E), 

899 _transpose_vectorized(d2Skdx2)) 

900 

901 # 4) With nodal (not elem) dofs 

902 K = _prod_vectorized(_prod_vectorized(_transpose_vectorized(DOF_rot), 

903 K), DOF_rot) 

904 

905 # 5) Need the area to compute total element energy 

906 return _scalar_vectorized(area, K)

908 def get_Hrot_from_J(self, J, return_area=False): 

909 """ 

910 Parameters 

911 ---------- 

912 *J* is a (N x 2 x 2) array of jacobian matrices (jacobian matrix at 

913 triangle first apex) 

914 

915 Returns 

916 ------- 

917 Returns H_rot used to rotate Hessian from local basis of first apex, 

918 to global coordinates. 

919 if *return_area* is True, returns also the triangle area (0.5*det(J)) 

920 """ 

921 # Here we try to deal with the simplest colinear cases; a null 

922 # energy and area is imposed. 

923 J_inv = _safe_inv22_vectorized(J) 

924 Ji00 = J_inv[:, 0, 0] 

925 Ji11 = J_inv[:, 1, 1] 

926 Ji10 = J_inv[:, 1, 0] 

927 Ji01 = J_inv[:, 0, 1] 

928 H_rot = _to_matrix_vectorized([ 

929 [Ji00*Ji00, Ji10*Ji10, Ji00*Ji10], 

930 [Ji01*Ji01, Ji11*Ji11, Ji01*Ji11], 

931 [2*Ji00*Ji01, 2*Ji11*Ji10, Ji00*Ji11+Ji10*Ji01]]) 

932 if not return_area: 

933 return H_rot 

934 else: 

935 area = 0.5 * (J[:, 0, 0]*J[:, 1, 1] - J[:, 0, 1]*J[:, 1, 0]) 

936 return H_rot, area

938 def get_Kff_and_Ff(self, J, ecc, triangles, Uc): 

939 """ 

940 Builds K and F for the following elliptic formulation: 

941 minimization of curvature energy with value of function at node 

942 imposed and derivatives 'free'. 

943 Builds the global Kff matrix in cco format. 

944 Builds the full Ff vec Ff = - Kfc x Uc 

945 

946 Parameters 

947 ---------- 

948 *J* is a (N x 2 x 2) array of jacobian matrices (jacobian matrix at 

949 triangle first apex) 

950 *ecc* is a (N x 3 x 1) array (array of column-matrices) of triangle 

951 eccentricities 

952 *triangles* is a (N x 3) array of nodes indexes. 

953 *Uc* is (N x 3) array of imposed displacements at nodes 

954 

955 Returns 

956 ------- 

957 (Kff_rows, Kff_cols, Kff_vals) Kff matrix in coo format - Duplicate 

958 (row, col) entries must be summed. 

959 Ff: force vector - dim npts * 3 

960 """ 

961 ntri = np.size(ecc, 0) 

962 vec_range = np.arange(ntri, dtype=np.int32) 

963 c_indices = -np.ones(ntri, dtype=np.int32) # for unused dofs, -1 

964 f_dof = [1, 2, 4, 5, 7, 8] 

965 c_dof = [0, 3, 6] 

966 

967 # vals, rows and cols indices in global dof numbering 

968 f_dof_indices = _to_matrix_vectorized([[ 

969 c_indices, triangles[:, 0]*2, triangles[:, 0]*2+1, 

970 c_indices, triangles[:, 1]*2, triangles[:, 1]*2+1, 

971 c_indices, triangles[:, 2]*2, triangles[:, 2]*2+1]]) 

972 

973 expand_indices = np.ones([ntri, 9, 1], dtype=np.int32) 

974 f_row_indices = _prod_vectorized(_transpose_vectorized(f_dof_indices), 

975 _transpose_vectorized(expand_indices)) 

976 f_col_indices = _prod_vectorized(expand_indices, f_dof_indices) 

977 K_elem = self.get_bending_matrices(J, ecc) 

978 

979 # Extracting sub-matrices 

980 # Explanation & notations: 

981 # * Subscript f denotes 'free' degrees of freedom (i.e. dz/dx, dz/dx) 

982 # * Subscript c denotes 'condensated' (imposed) degrees of freedom 

983 # (i.e. z at all nodes) 

984 # * F = [Ff, Fc] is the force vector 

985 # * U = [Uf, Uc] is the imposed dof vector 

986 # [ Kff Kfc ] 

987 # * K = [ ] is the laplacian stiffness matrix 

988 # [ Kcf Kff ] 

989 # * As F = K x U one gets straightforwardly: Ff = - Kfc x Uc 

990 

991 # Computing Kff stiffness matrix in sparse coo format 

992 Kff_vals = np.ravel(K_elem[np.ix_(vec_range, f_dof, f_dof)]) 

993 Kff_rows = np.ravel(f_row_indices[np.ix_(vec_range, f_dof, f_dof)]) 

994 Kff_cols = np.ravel(f_col_indices[np.ix_(vec_range, f_dof, f_dof)]) 

995 

996 # Computing Ff force vector in sparse coo format 

997 Kfc_elem = K_elem[np.ix_(vec_range, f_dof, c_dof)] 

998 Uc_elem = np.expand_dims(Uc, axis=2) 

999 Ff_elem = - _prod_vectorized(Kfc_elem, Uc_elem)[:, :, 0] 

1000 Ff_indices = f_dof_indices[np.ix_(vec_range, [0], f_dof)][:, 0, :] 

1001 

1002 # Extracting Ff force vector in dense format 

1003 # We have to sum duplicate indices - using bincount 

1004 Ff = np.bincount(np.ravel(Ff_indices), weights=np.ravel(Ff_elem)) 

1005 return Kff_rows, Kff_cols, Kff_vals, Ff

1089class _DOF_estimator_geom(_DOF_estimator): 

1090 """ Fast 'geometric' approximation, recommended for large arrays. """ 

1091 def compute_dz(self): 

1092 """ 

1093 self.df is computed as weighted average of _triangles sharing a common 

1094 node. On each triangle itri f is first assumed linear (= ~f), which 

1095 allows to compute d~f[itri] 

1096 Then the following approximation of df nodal values is then proposed: 

1097 f[ipt] = SUM ( w[itri] x d~f[itri] , for itri sharing apex ipt) 

1098 The weighted coeff. w[itri] are proportional to the angle of the 

1099 triangle itri at apex ipt 

1100 """ 

1101 el_geom_w = self.compute_geom_weights() 

1102 el_geom_grad = self.compute_geom_grads() 

1103 

1104 # Sum of weights coeffs 

1105 w_node_sum = np.bincount(np.ravel(self._triangles), 

1106 weights=np.ravel(el_geom_w)) 

1107 

1108 # Sum of weighted df = (dfx, dfy) 

1109 dfx_el_w = np.empty_like(el_geom_w) 

1110 dfy_el_w = np.empty_like(el_geom_w) 

1111 for iapex in range(3): 

1112 dfx_el_w[:, iapex] = el_geom_w[:, iapex]*el_geom_grad[:, 0] 

1113 dfy_el_w[:, iapex] = el_geom_w[:, iapex]*el_geom_grad[:, 1] 

1114 dfx_node_sum = np.bincount(np.ravel(self._triangles), 

1115 weights=np.ravel(dfx_el_w)) 

1116 dfy_node_sum = np.bincount(np.ravel(self._triangles), 

1117 weights=np.ravel(dfy_el_w)) 

1118 

1119 # Estimation of df 

1120 dfx_estim = dfx_node_sum/w_node_sum 

1121 dfy_estim = dfy_node_sum/w_node_sum 

1122 return np.vstack([dfx_estim, dfy_estim]).T 

1123 

1124 def compute_geom_weights(self): 

1125 """ 

1126 Builds the (nelems x 3) weights coeffs of _triangles angles, 

1127 renormalized so that np.sum(weights, axis=1) == np.ones(nelems) 

1128 """ 

1129 weights = np.zeros([np.size(self._triangles, 0), 3]) 

1130 tris_pts = self._tris_pts 

1131 for ipt in range(3): 

1132 p0 = tris_pts[:, (ipt) % 3, :] 

1133 p1 = tris_pts[:, (ipt+1) % 3, :] 

1134 p2 = tris_pts[:, (ipt-1) % 3, :] 

1135 alpha1 = np.arctan2(p1[:, 1]-p0[:, 1], p1[:, 0]-p0[:, 0]) 

1136 alpha2 = np.arctan2(p2[:, 1]-p0[:, 1], p2[:, 0]-p0[:, 0]) 

1137 # In the below formula we could take modulo 2. but 

1138 # modulo 1. is safer regarding round-off errors (flat triangles). 

1139 angle = np.abs(np.mod((alpha2-alpha1) / np.pi, 1.)) 

1140 # Weight proportional to angle up np.pi/2; null weight for 

1141 # degenerated cases 0 and np.pi (note that `angle` is normalized 

1142 # by np.pi). 

1143 weights[:, ipt] = 0.5 - np.abs(angle-0.5) 

1144 return weights 

1145 

1146 def compute_geom_grads(self): 

1147 """ 

1148 Compute the (global) gradient component of f assumed linear (~f). 

1149 returns array df of shape (nelems,2) 

1150 df[ielem].dM[ielem] = dz[ielem] i.e. df = dz x dM = dM.T^-1 x dz 

1151 """ 

1152 tris_pts = self._tris_pts 

1153 tris_f = self.z[self._triangles] 

1154 

1155 dM1 = tris_pts[:, 1, :] - tris_pts[:, 0, :] 

1156 dM2 = tris_pts[:, 2, :] - tris_pts[:, 0, :] 

1157 dM = np.dstack([dM1, dM2]) 

1158 # Here we try to deal with the simplest colinear cases: a null 

1159 # gradient is assumed in this case. 

1160 dM_inv = _safe_inv22_vectorized(dM) 

1161 

1162 dZ1 = tris_f[:, 1] - tris_f[:, 0] 

1163 dZ2 = tris_f[:, 2] - tris_f[:, 0] 

1164 dZ = np.vstack([dZ1, dZ2]).T 

1165 df = np.empty_like(dZ) 

1166 

1167 # With np.einsum : could be ej,eji -> ej 

1168 df[:, 0] = dZ[:, 0]*dM_inv[:, 0, 0] + dZ[:, 1]*dM_inv[:, 1, 0] 

1169 df[:, 1] = dZ[:, 0]*dM_inv[:, 0, 1] + dZ[:, 1]*dM_inv[:, 1, 1] 

1170 return df

1173class _DOF_estimator_min_E(_DOF_estimator_geom): 

1174 """ 

1175 The 'smoothest' approximation, df is computed through global minimization 

1176 of the bending energy: 

1177 E(f) = integral[(d2z/dx2 + d2z/dy2 + 2 d2z/dxdy)**2 dA] 

1178 """ 

1179 def __init__(self, Interpolator): 

1180 self._eccs = Interpolator._eccs 

1181 _DOF_estimator_geom.__init__(self, Interpolator) 

1182 

1183 def compute_dz(self): 

1184 """ 

1185 Elliptic solver for bending energy minimization. 

1186 Uses a dedicated 'toy' sparse Jacobi PCG solver. 

1187 """ 

1188 # Initial guess for iterative PCG solver. 

1189 dz_init = _DOF_estimator_geom.compute_dz(self) 

1190 Uf0 = np.ravel(dz_init) 

1191 

1192 reference_element = _ReducedHCT_Element() 

1193 J = CubicTriInterpolator._get_jacobian(self._tris_pts) 

1194 eccs = self._eccs 

1195 triangles = self._triangles 

1196 Uc = self.z[self._triangles] 

1197 

1198 # Building stiffness matrix and force vector in coo format 

1199 Kff_rows, Kff_cols, Kff_vals, Ff = reference_element.get_Kff_and_Ff( 

1200 J, eccs, triangles, Uc) 

1201 

1202 # Building sparse matrix and solving minimization problem 

1203 # We could use scipy.sparse direct solver; however to avoid this 

1204 # external dependency an implementation of a simple PCG solver with 

1205 # a simple diagonal Jacobi preconditioner is implemented. 

1206 tol = 1.e-10 

1207 n_dof = Ff.shape[0] 

1208 Kff_coo = _Sparse_Matrix_coo(Kff_vals, Kff_rows, Kff_cols, 

1209 shape=(n_dof, n_dof)) 

1210 Kff_coo.compress_csc() 

1211 Uf, err = _cg(A=Kff_coo, b=Ff, x0=Uf0, tol=tol) 

1212 # If the PCG did not converge, we return the best guess between Uf0 

1213 # and Uf. 

1214 err0 = np.linalg.norm(Kff_coo.dot(Uf0) - Ff) 

1215 if err0 < err: 

1216 # Maybe a good occasion to raise a warning here ? 

1217 warnings.warn("In TriCubicInterpolator initialization, PCG sparse" 

1218 " solver did not converge after 1000 iterations. " 

1219 "`geom` approximation is used instead of `min_E`") 

1220 Uf = Uf0 

1221 

1222 # Building dz from Uf 

1223 dz = np.empty([self._pts.shape[0], 2], dtype=np.float64) 

1224 dz[:, 0] = Uf[::2] 

1225 dz[:, 1] = Uf[1::2] 

1226 return dz

1303def _cg(A, b, x0=None, tol=1.e-10, maxiter=1000): 

1304 """ 

1305 Use Preconditioned Conjugate Gradient iteration to solve A x = b 

1306 A simple Jacobi (diagonal) preconditionner is used. 

1307 

1308 Parameters 

1309 ---------- 

1310 A: _Sparse_Matrix_coo 

1311 *A* must have been compressed before by compress_csc or 

1312 compress_csr method. 

1313 

1314 b: array 

1315 Right hand side of the linear system. 

1316 

1317 Returns 

1318 ------- 

1319 x: array. 

1320 The converged solution. 

1321 err: float 

1322 The absolute error np.linalg.norm(A.dot(x) - b) 

1323 

1324 Other parameters 

1325 ---------------- 

1326 x0: array. 

1327 Starting guess for the solution. 

1328 tol: float. 

1329 Tolerance to achieve. The algorithm terminates when the relative 

1330 residual is below tol. 

1331 maxiter: integer. 

1332 Maximum number of iterations. Iteration will stop 

1333 after maxiter steps even if the specified tolerance has not 

1334 been achieved. 

1335 """ 

1336 n = b.size 

1337 assert A.n == n 

1338 assert A.m == n 

1339 b_norm = np.linalg.norm(b) 

1340 

1341 # Jacobi pre-conditioner 

1342 kvec = A.diag 

1343 # For diag elem < 1e-6 we keep 1e-6. 

1344 kvec = np.where(kvec > 1.e-6, kvec, 1.e-6) 

1345 

1346 # Initial guess 

1347 if x0 is None: 

1348 x = np.zeros(n) 

1349 else: 

1350 x = x0 

1351 

1352 r = b - A.dot(x) 

1353 w = r/kvec 

1354 

1355 p = np.zeros(n) 

1356 beta = 0.0 

1357 rho = np.dot(r, w) 

1358 k = 0 

1359 

1360 # Following C. T. Kelley 

1361 while (np.sqrt(abs(rho)) > tol*b_norm) and (k < maxiter): 

1362 p = w + beta*p 

1363 z = A.dot(p) 

1364 alpha = rho/np.dot(p, z) 

1365 r = r - alpha*z 

1366 w = r/kvec 

1367 rhoold = rho 

1368 rho = np.dot(r, w) 

1369 x = x + alpha*p 

1370 beta = rho/rhoold 

1371 #err = np.linalg.norm(A.dot(x) - b) # absolute accuracy - not used 

1372 k += 1 

1373 err = np.linalg.norm(A.dot(x) - b) 

1374 return x, err

1377# The following private functions: 

1378# :func:`_safe_inv22_vectorized` 

1379# :func:`_pseudo_inv22sym_vectorized` 

1380# :func:`_prod_vectorized` 

1381# :func:`_scalar_vectorized` 

1382# :func:`_transpose_vectorized` 

1383# :func:`_roll_vectorized` 

1384# :func:`_to_matrix_vectorized` 

1385# :func:`_extract_submatrices` 

1386# provide fast numpy implementation of some standard operations on arrays of 

1387# matrices - stored as (:, n_rows, n_cols)-shaped np.arrays. 

1388 

1389# Development note: Dealing with pathologic 'flat' triangles in the 

1390# CubicTriInterpolator code and impact on (2,2)-matrix inversion functions 

1391# :func:`_safe_inv22_vectorized` and :func:`_pseudo_inv22sym_vectorized`. 

1392# 

1393# Goals: 

1394# 1) The CubicTriInterpolator should be able to handle flat or almost flat 

1395# triangles without raising an error, 

1396# 2) These degenerated triangles should have no impact on the automatic dof 

1397# calculation (associated with null weight for the _DOF_estimator_geom and 

1398# with null energy for the _DOF_estimator_min_E), 

1399# 3) Linear patch test should be passed exactly on degenerated meshes, 

1400# 4) Interpolation (with :meth:`_interpolate_single_key` or 

1401# :meth:`_interpolate_multi_key`) shall be correctly handled even *inside* 

1402# the pathologic triangles, to interact correctly with a TriRefiner class. 

1403# 

1404# Difficulties: 

1405# Flat triangles have rank-deficient *J* (so-called jacobian matrix) and 

1406# *metric* (the metric tensor = J x J.T). Computation of the local 

1407# tangent plane is also problematic. 

1408# 

1409# Implementation: 

1410# Most of the time, when computing the inverse of a rank-deficient matrix it 

1411# is safe to simply return the null matrix (which is the implementation in 

1412# :func:`_safe_inv22_vectorized`). This is because of point 2), itself 

1413# enforced by: 

1414# - null area hence null energy in :class:`_DOF_estimator_min_E` 

1415# - angles close or equal to 0 or np.pi hence null weight in 

1416# :class:`_DOF_estimator_geom`. 

1417# Note that the function angle -> weight is continuous and maximum for an 

1418# angle np.pi/2 (refer to :meth:`compute_geom_weights`) 

1419# The exception is the computation of barycentric coordinates, which is done 

1420# by inversion of the *metric* matrix. In this case, we need to compute a set 

1421# of valid coordinates (1 among numerous possibilities), to ensure point 4). 

1422# We benefit here from the symmetry of metric = J x J.T, which makes it easier 

1423# to compute a pseudo-inverse in :func:`_pseudo_inv22sym_vectorized` 

1424def _safe_inv22_vectorized(M): 

1425 """ 

1426 Inversion of arrays of (2,2) matrices, returns 0 for rank-deficient 

1427 matrices. 

1428 

1429 *M* : array of (2,2) matrices to inverse, shape (n,2,2) 

1430 """ 

1431 assert M.ndim == 3 

1432 assert M.shape[-2:] == (2, 2) 

1433 M_inv = np.empty_like(M) 

1434 prod1 = M[:, 0, 0]*M[:, 1, 1] 

1435 delta = prod1 - M[:, 0, 1]*M[:, 1, 0] 

1436 

1437 # We set delta_inv to 0. in case of a rank deficient matrix; a 

1438 # rank-deficient input matrix *M* will lead to a null matrix in output 

1439 rank2 = (np.abs(delta) > 1e-8*np.abs(prod1)) 

1440 if np.all(rank2): 

1441 # Normal 'optimized' flow. 

1442 delta_inv = 1./delta 

1443 else: 

1444 # 'Pathologic' flow. 

1445 delta_inv = np.zeros(M.shape[0]) 

1446 delta_inv[rank2] = 1./delta[rank2] 

1447 

1448 M_inv[:, 0, 0] = M[:, 1, 1]*delta_inv 

1449 M_inv[:, 0, 1] = -M[:, 0, 1]*delta_inv 

1450 M_inv[:, 1, 0] = -M[:, 1, 0]*delta_inv 

1451 M_inv[:, 1, 1] = M[:, 0, 0]*delta_inv 

1452 return M_inv

1455def _pseudo_inv22sym_vectorized(M): 

1456 """ 

1457 Inversion of arrays of (2,2) SYMMETRIC matrices; returns the 

1458 (Moore-Penrose) pseudo-inverse for rank-deficient matrices. 

1459 

1460 In case M is of rank 1, we have M = trace(M) x P where P is the orthogonal 

1461 projection on Im(M), and we return trace(M)^-1 x P == M / trace(M)**2 

1462 In case M is of rank 0, we return the null matrix. 

1463 

1464 *M* : array of (2,2) matrices to inverse, shape (n,2,2) 

1465 """ 

1466 assert M.ndim == 3 

1467 assert M.shape[-2:] == (2, 2) 

1468 M_inv = np.empty_like(M) 

1469 prod1 = M[:, 0, 0]*M[:, 1, 1] 

1470 delta = prod1 - M[:, 0, 1]*M[:, 1, 0] 

1471 rank2 = (np.abs(delta) > 1e-8*np.abs(prod1)) 

1472 

1473 if np.all(rank2): 

1474 # Normal 'optimized' flow. 

1475 M_inv[:, 0, 0] = M[:, 1, 1] / delta 

1476 M_inv[:, 0, 1] = -M[:, 0, 1] / delta 

1477 M_inv[:, 1, 0] = -M[:, 1, 0] / delta 

1478 M_inv[:, 1, 1] = M[:, 0, 0] / delta 

1479 else: 

1480 # 'Pathologic' flow. 

1481 # Here we have to deal with 2 sub-cases 

1482 # 1) First sub-case: matrices of rank 2: 

1483 delta = delta[rank2] 

1484 M_inv[rank2, 0, 0] = M[rank2, 1, 1] / delta 

1485 M_inv[rank2, 0, 1] = -M[rank2, 0, 1] / delta 

1486 M_inv[rank2, 1, 0] = -M[rank2, 1, 0] / delta 

1487 M_inv[rank2, 1, 1] = M[rank2, 0, 0] / delta 

1488 # 2) Second sub-case: rank-deficient matrices of rank 0 and 1: 

1489 rank01 = ~rank2 

1490 tr = M[rank01, 0, 0] + M[rank01, 1, 1] 

1491 tr_zeros = (np.abs(tr) < 1.e-8) 

1492 sq_tr_inv = (1.-tr_zeros) / (tr**2+tr_zeros) 

1493 #sq_tr_inv = 1. / tr**2 

1494 M_inv[rank01, 0, 0] = M[rank01, 0, 0] * sq_tr_inv 

1495 M_inv[rank01, 0, 1] = M[rank01, 0, 1] * sq_tr_inv 

1496 M_inv[rank01, 1, 0] = M[rank01, 1, 0] * sq_tr_inv 

1497 M_inv[rank01, 1, 1] = M[rank01, 1, 1] * sq_tr_inv 

1498 

1499 return M_inv

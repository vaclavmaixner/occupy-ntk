29class TreeBuilderRegistry(object): 

30 """A way of looking up TreeBuilder subclasses by their name or by desired 

31 features. 

32 """ 

33 

34 def __init__(self): 

35 self.builders_for_feature = defaultdict(list) 

36 self.builders = [] 

37 

38 def register(self, treebuilder_class): 

39 """Register a treebuilder based on its advertised features. 

40 

41 :param treebuilder_class: A subclass of Treebuilder. its .features 

42 attribute should list its features. 

43 """ 

44 for feature in treebuilder_class.features: 

45 self.builders_for_feature[feature].insert(0, treebuilder_class) 

46 self.builders.insert(0, treebuilder_class) 

47 

48 def lookup(self, *features): 

49 """Look up a TreeBuilder subclass with the desired features. 

50 

51 :param features: A list of features to look for. If none are 

52 provided, the most recently registered TreeBuilder subclass 

53 will be used. 

54 :return: A TreeBuilder subclass, or None if there's no 

55 registered subclass with all the requested features. 

56 """ 

57 if len(self.builders) == 0: 

58 # There are no builders at all. 

59 return None 

60 

61 if len(features) == 0: 

62 # They didn't ask for any features. Give them the most 

63 # recently registered builder. 

64 return self.builders[0] 

65 

66 # Go down the list of features in order, and eliminate any builders 

67 # that don't match every feature. 

68 features = list(features) 

69 features.reverse() 

70 candidates = None 

71 candidate_set = None 

72 while len(features) > 0: 

73 feature = features.pop() 

74 we_have_the_feature = self.builders_for_feature.get(feature, []) 

75 if len(we_have_the_feature) > 0: 

76 if candidates is None: 

77 candidates = we_have_the_feature 

78 candidate_set = set(candidates) 

79 else: 

80 # Eliminate any candidates that don't have this feature. 

81 candidate_set = candidate_set.intersection( 

82 set(we_have_the_feature)) 

83 

84 # The only valid candidates are the ones in candidate_set. 

85 # Go through the original list of candidates and pick the first one 

86 # that's in candidate_set. 

87 if candidate_set is None: 

88 return None 

89 for candidate in candidates: 

90 if candidate in candidate_set: 

91 return candidate 

92 return None

176 def can_be_empty_element(self, tag_name): 

177 """Might a tag with this name be an empty-element tag? 

178 

179 The final markup may or may not actually present this tag as 

180 self-closing. 

181 

182 For instance: an HTMLBuilder does not consider a <p> tag to be 

183 an empty-element tag (it's not in 

184 HTMLBuilder.empty_element_tags). This means an empty <p> tag 

185 will be presented as "<p></p>", not "<p/>" or "<p>". 

186 

187 The default implementation has no opinion about which tags are 

188 empty-element tags, so a tag will be presented as an 

189 empty-element tag if and only if it has no children. 

190 "<foo></foo>" will become "<foo/>", and "<foo>bar</foo>" will 

191 be left alone. 

192 

193 :param tag_name: The name of a markup tag. 

194 """ 

195 if self.empty_element_tags is None: 

196 return True 

197 return tag_name in self.empty_element_tags

265 def _replace_cdata_list_attribute_values(self, tag_name, attrs): 

266 """When an attribute value is associated with a tag that can 

267 have multiple values for that attribute, convert the string 

268 value to a list of strings. 

269 

270 Basically, replaces class="foo bar" with class=["foo", "bar"] 

271 

272 NOTE: This method modifies its input in place. 

273 

274 :param tag_name: The name of a tag. 

275 :param attrs: A dictionary containing the tag's attributes. 

276 Any appropriate attribute values will be modified in place. 

277 """ 

278 if not attrs: 

279 return attrs 

280 if self.cdata_list_attributes: 

281 universal = self.cdata_list_attributes.get('*', []) 

282 tag_specific = self.cdata_list_attributes.get( 

283 tag_name.lower(), None) 

284 for attr in list(attrs.keys()): 

285 if attr in universal or (tag_specific and attr in tag_specific): 

286 # We have a "class"-type attribute whose string 

287 # value is a whitespace-separated list of 

288 # values. Split it into a list. 

289 value = attrs[attr] 

290 if isinstance(value, str): 

291 values = nonwhitespace_re.findall(value) 

292 else: 

293 # html5lib sometimes calls setAttributes twice 

294 # for the same tag when rearranging the parse 

295 # tree. On the second call the attribute value 

296 # here is already a list. If this happens, 

297 # leave the value alone rather than trying to 

298 # split it again. 

299 values = value 

300 attrs[attr] = values 

301 return attrs

399 def set_up_substitutions(self, tag): 

400 """Replace the declared encoding in a <meta> tag with a placeholder, 

401 to be substituted when the tag is output to a string. 

402 

403 An HTML document may come in to Beautiful Soup as one 

404 encoding, but exit in a different encoding, and the <meta> tag 

405 needs to be changed to reflect this. 

406 

407 :param tag: A `Tag` 

408 :return: Whether or not a substitution was performed. 

409 """ 

410 # We are only interested in <meta> tags 

411 if tag.name != 'meta': 

412 return False 

413 

414 http_equiv = tag.get('http-equiv') 

415 content = tag.get('content') 

416 charset = tag.get('charset') 

417 

418 # We are interested in <meta> tags that say what encoding the 

419 # document was originally in. This means HTML 5-style <meta> 

420 # tags that provide the "charset" attribute. It also means 

421 # HTML 4-style <meta> tags that provide the "content" 

422 # attribute and have "http-equiv" set to "content-type". 

423 # 

424 # In both cases we will replace the value of the appropriate 

425 # attribute with a standin object that can take on any 

426 # encoding. 

427 meta_encoding = None 

428 if charset is not None: 

429 # HTML 5 style: 

430 # <meta charset="utf8"> 

431 meta_encoding = charset 

432 tag['charset'] = CharsetMetaAttributeValue(charset) 

433 

434 elif (content is not None and http_equiv is not None 

435 and http_equiv.lower() == 'content-type'): 

436 # HTML 4 style: 

437 # <meta http-equiv="content-type" content="text/html; charset=utf8"> 

438 tag['content'] = ContentMetaAttributeValue(content) 

439 

440 return (meta_encoding is not None)

442def register_treebuilders_from(module): 

443 """Copy TreeBuilders from the given module into this module.""" 

444 # I'm fairly sure this is not the best way to do this. 

445 this_module = sys.modules['bs4.builder'] 

446 for name in module.__all__: 

447 obj = getattr(module, name) 

448 

449 if issubclass(obj, TreeBuilder): 

450 setattr(this_module, name, obj) 

451 this_module.__all__.append(name) 

452 # Register the builder while we're at it. 

453 this_module.builder_registry.register(obj)

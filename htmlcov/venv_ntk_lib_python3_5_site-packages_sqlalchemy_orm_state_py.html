34@inspection._self_inspects 

35class InstanceState(interfaces.InspectionAttrInfo): 

36 """tracks state information at the instance level. 

37 

38 The :class:`.InstanceState` is a key object used by the 

39 SQLAlchemy ORM in order to track the state of an object; 

40 it is created the moment an object is instantiated, typically 

41 as a result of :term:`instrumentation` which SQLAlchemy applies 

42 to the ``__init__()`` method of the class. 

43 

44 :class:`.InstanceState` is also a semi-public object, 

45 available for runtime inspection as to the state of a 

46 mapped instance, including information such as its current 

47 status within a particular :class:`.Session` and details 

48 about data on individual attributes. The public API 

49 in order to acquire a :class:`.InstanceState` object 

50 is to use the :func:`.inspect` system:: 

51 

52 >>> from sqlalchemy import inspect 

53 >>> insp = inspect(some_mapped_object) 

54 

55 .. seealso:: 

56 

57 :ref:`core_inspection_toplevel` 

58

757 # to a session 

758 

759 inst = self.obj() 

760 if self.session_id: 

761 self._strong_obj = inst 

762 

763 if inst is None and attr: 

764 raise orm_exc.ObjectDereferencedError( 

765 "Can't emit change event for attribute '%s' - " 

766 "parent object of type %s has been garbage " 

767 "collected." 

768 % (self.manager[attr.key], base.state_class_str(self)) 

769 )

771 def _commit(self, dict_, keys): 

772 """Commit attributes. 

773 

774 This is used by a partial-attribute load operation to mark committed 

775 those attributes which were refreshed from the database. 

776 

777 Attributes marked as "expired" can potentially remain "expired" after 

778 this step if a value was not populated in state.dict. 

779 

780 """ 

781 for key in keys: 

782 self.committed_state.pop(key, None) 

783 

784 self.expired = False 

785 

786 self.expired_attributes.difference_update( 

787 set(keys).intersection(dict_) 

788 ) 

789 

790 # the per-keys commit removes object-level callables, 

791 # while that of commit_all does not. it's not clear 

792 # if this behavior has a clear rationale, however tests do 

793 # ensure this is what it does. 

794 if self.callables: 

795 for key in ( 

796 set(self.callables).intersection(keys).intersection(dict_) 

797 ): 

798 del self.callables[key]

800 def _commit_all(self, dict_, instance_dict=None): 

801 """commit all attributes unconditionally. 

802 

803 This is used after a flush() or a full load/refresh 

804 to remove all pending state from the instance. 

805 

806 - all attributes are marked as "committed" 

807 - the "strong dirty reference" is removed 

808 - the "modified" flag is set to False 

809 - any "expired" markers for scalar attributes loaded are removed. 

810 - lazy load callables for objects / collections *stay* 

811 

812 Attributes marked as "expired" can potentially remain 

813 "expired" after this step if a value was not populated in state.dict. 

814 

815 """ 

816 self._commit_all_states([(self, dict_)], instance_dict) 

817 

818 @classmethod 

819 def _commit_all_states(self, iter_, instance_dict=None): 

820 """Mass / highly inlined version of commit_all().""" 

821 

822 for state, dict_ in iter_: 

823 state_dict = state.__dict__ 

824 

825 state.committed_state.clear() 

826 

827 if "_pending_mutations" in state_dict: 

828 del state_dict["_pending_mutations"] 

829 

830 state.expired_attributes.difference_update(dict_) 

831 

832 if instance_dict and state.modified: 

833 instance_dict._modified.discard(state) 

834 

835 state.modified = state.expired = False 

836 state._strong_obj = None

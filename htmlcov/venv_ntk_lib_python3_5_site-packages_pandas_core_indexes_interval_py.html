50def _get_next_label(label): 

51 dtype = getattr(label, 'dtype', type(label)) 

52 if isinstance(label, (Timestamp, Timedelta)): 

53 dtype = 'datetime64' 

54 if is_datetime_or_timedelta_dtype(dtype) or is_datetime64tz_dtype(dtype): 

55 return label + np.timedelta64(1, 'ns') 

56 elif is_integer_dtype(dtype): 

57 return label + 1 

58 elif is_float_dtype(dtype): 

59 return np.nextafter(label, np.infty) 

60 else: 

61 raise TypeError('cannot determine next label for type {typ!r}' 

62 .format(typ=type(label)))

80def _get_interval_closed_bounds(interval): 

81 """ 

82 Given an Interval or IntervalIndex, return the corresponding interval with 

83 closed bounds.

85 left, right = interval.left, interval.right

87 left = _get_next_label(left)

597 def _maybe_convert_i8(self, key): 

598 """ 

599 Maybe convert a given key to it's equivalent i8 value(s). Used as a 

600 preprocessing step prior to IntervalTree queries (self._engine), which 

601 expects numeric data. 

602 

603 Parameters 

604 ---------- 

605 key : scalar or list-like 

606 The key that should maybe be converted to i8. 

607 

608 Returns 

609 ------- 

610 key: scalar or list-like 

611 The original key if no conversion occured, int if converted scalar, 

612 Int64Index if converted list-like. 

613 """ 

614 original = key 

615 if is_list_like(key): 

616 key = ensure_index(key) 

617 

618 if not self._needs_i8_conversion(key): 

619 return original 

620 

621 scalar = is_scalar(key) 

622 if is_interval_dtype(key) or isinstance(key, Interval): 

623 # convert left/right and reconstruct 

624 left = self._maybe_convert_i8(key.left) 

625 right = self._maybe_convert_i8(key.right) 

626 constructor = Interval if scalar else IntervalIndex.from_arrays 

627 return constructor(left, right, closed=self.closed) 

628 

629 if scalar: 

630 # Timestamp/Timedelta 

631 key_dtype, key_i8 = infer_dtype_from_scalar(key, pandas_dtype=True) 

632 else: 

633 # DatetimeIndex/TimedeltaIndex 

634 key_dtype, key_i8 = key.dtype, Index(key.asi8) 

635 if key.hasnans: 

636 # convert NaT from it's i8 value to np.nan so it's not viewed 

637 # as a valid value, maybe causing errors (e.g. is_overlapping) 

638 key_i8 = key_i8.where(~key._isnan) 

639 

640 # ensure consistency with IntervalIndex subtype 

641 subtype = self.dtype.subtype 

642 msg = ('Cannot index an IntervalIndex of subtype {subtype} with ' 

643 'values of dtype {other}') 

644 if not is_dtype_equal(subtype, key_dtype): 

645 raise ValueError(msg.format(subtype=subtype, other=key_dtype)) 

646 

647 return key_i8

723 

724 Parameters 

725 ---------- 

726 key : label 

727 method : {None}, optional

730 Returns 

731 -------

756 """

760 key = self._maybe_cast_indexed(key) 

761 

762 if self.is_non_overlapping_monotonic:

764 left = self._maybe_cast_slice_bound(key.left, 'left', None) 

765 right = self._maybe_cast_slice_bound(key.right, 'right', None)

769

772 if start is None or stop is None:

775 return start 

776 elif start < stop:

778 else: 

779 raise KeyError(original_key) 

780 

781 else: 

782 # use the interval tree

784 if isinstance(key, Interval): 

785 left, right = _get_interval_closed_bounds(key) 

786 return self._engine.get_loc_interval(left, right) 

787 else: 

788 return self._engine.get_loc(key)

841 def _get_reindexer(self, target): 

842 """ 

843 Return an indexer for a target IntervalIndex with self 

844 """ 

845 

846 # find the left and right indexers 

847 left = self._maybe_convert_i8(target.left) 

848 right = self._maybe_convert_i8(target.right) 

849 lindexer = self._engine.get_indexer(left.values) 

850 rindexer = self._engine.get_indexer(right.values) 

851 

852 # we want to return an indexer on the intervals 

853 # however, our keys could provide overlapping of multiple 

854 # intervals, so we iterate thru the indexers and construct 

855 # a set of indexers 

856 

857 indexer = [] 

858 n = len(self) 

859 

860 for i, (lhs, rhs) in enumerate(zip(lindexer, rindexer)): 

861 

862 target_value = target[i] 

863 

864 # matching on the lhs bound 

865 if (lhs != -1 and 

866 self.closed == 'right' and 

867 target_value.left == self[lhs].right): 

868 lhs += 1 

869 

870 # matching on the lhs bound 

871 if (rhs != -1 and 

872 self.closed == 'left' and 

873 target_value.right == self[rhs].left): 

874 rhs -= 1 

875 

876 # not found 

877 if lhs == -1 and rhs == -1: 

878 indexer.append(np.array([-1])) 

879 

880 elif rhs == -1: 

881 

882 indexer.append(np.arange(lhs, n)) 

883 

884 elif lhs == -1: 

885 

886 # care about left/right closed here 

887 value = self[i] 

888 

889 # target.closed same as self.closed 

890 if self.closed == target.closed: 

891 if target_value.left < value.left: 

892 indexer.append(np.array([-1])) 

893 continue 

894 

895 # target.closed == 'left' 

896 elif self.closed == 'right': 

897 if target_value.left <= value.left: 

898 indexer.append(np.array([-1])) 

899 continue 

900 

901 # target.closed == 'right' 

902 elif self.closed == 'left': 

903 if target_value.left <= value.left: 

904 indexer.append(np.array([-1])) 

905 continue 

906 

907 indexer.append(np.arange(0, rhs + 1)) 

908 

909 else: 

910 indexer.append(np.arange(lhs, rhs + 1)) 

911 

912 return np.concatenate(indexer)

1159def interval_range(start=None, end=None, periods=None, freq=None, 

1160 name=None, closed='right'): 

1161 """ 

1162 Return a fixed frequency IntervalIndex 

1163 

1164 Parameters 

1165 ---------- 

1166 start : numeric or datetime-like, default None 

1167 Left bound for generating intervals 

1168 end : numeric or datetime-like, default None 

1169 Right bound for generating intervals 

1170 periods : integer, default None 

1171 Number of periods to generate 

1172 freq : numeric, string, or DateOffset, default None 

1173 The length of each interval. Must be consistent with the type of start 

1174 and end, e.g. 2 for numeric, or '5H' for datetime-like. Default is 1 

1175 for numeric and 'D' for datetime-like. 

1176 name : string, default None 

1177 Name of the resulting IntervalIndex 

1178 closed : {'left', 'right', 'both', 'neither'}, default 'right' 

1179 Whether the intervals are closed on the left-side, right-side, both 

1180 or neither. 

1181 

1182 Returns 

1183 ------- 

1184 rng : IntervalIndex 

1185 

1186 See Also 

1187 -------- 

1188 IntervalIndex : An Index of intervals that are all closed on the same side. 

1189 

1190 Notes 

1191 ----- 

1192 Of the four parameters ``start``, ``end``, ``periods``, and ``freq``, 

1193 exactly three must be specified. If ``freq`` is omitted, the resulting 

1194 ``IntervalIndex`` will have ``periods`` linearly spaced elements between 

1195 ``start`` and ``end``, inclusively. 

1196 

1197 To learn more about datetime-like frequency strings, please see `this link 

1198 <http://pandas.pydata.org/pandas-docs/stable/timeseries.html#offset-aliases>`__. 

1199 

1200 Examples 

1201 -------- 

1202 Numeric ``start`` and ``end`` is supported. 

1203 

1204 >>> pd.interval_range(start=0, end=5) 

1205 IntervalIndex([(0, 1], (1, 2], (2, 3], (3, 4], (4, 5]], 

1206 closed='right', dtype='interval[int64]') 

1207 

1208 Additionally, datetime-like input is also supported. 

1209 

1210 >>> pd.interval_range(start=pd.Timestamp('2017-01-01'), 

1211 ... end=pd.Timestamp('2017-01-04')) 

1212 IntervalIndex([(2017-01-01, 2017-01-02], (2017-01-02, 2017-01-03], 

1213 (2017-01-03, 2017-01-04]], 

1214 closed='right', dtype='interval[datetime64[ns]]') 

1215 

1216 The ``freq`` parameter specifies the frequency between the left and right. 

1217 endpoints of the individual intervals within the ``IntervalIndex``. For 

1218 numeric ``start`` and ``end``, the frequency must also be numeric. 

1219 

1220 >>> pd.interval_range(start=0, periods=4, freq=1.5) 

1221 IntervalIndex([(0.0, 1.5], (1.5, 3.0], (3.0, 4.5], (4.5, 6.0]], 

1222 closed='right', dtype='interval[float64]') 

1223 

1224 Similarly, for datetime-like ``start`` and ``end``, the frequency must be 

1225 convertible to a DateOffset. 

1226 

1227 >>> pd.interval_range(start=pd.Timestamp('2017-01-01'), 

1228 ... periods=3, freq='MS') 

1229 IntervalIndex([(2017-01-01, 2017-02-01], (2017-02-01, 2017-03-01], 

1230 (2017-03-01, 2017-04-01]], 

1231 closed='right', dtype='interval[datetime64[ns]]') 

1232 

1233 Specify ``start``, ``end``, and ``periods``; the frequency is generated 

1234 automatically (linearly spaced). 

1235 

1236 >>> pd.interval_range(start=0, end=6, periods=4) 

1237 IntervalIndex([(0.0, 1.5], (1.5, 3.0], (3.0, 4.5], (4.5, 6.0]], 

1238 closed='right', 

1239 dtype='interval[float64]') 

1240 

1241 The ``closed`` parameter specifies which endpoints of the individual 

1242 intervals within the ``IntervalIndex`` are closed. 

1243 

1244 >>> pd.interval_range(end=5, periods=4, closed='both') 

1245 IntervalIndex([[1, 2], [2, 3], [3, 4], [4, 5]], 

1246 closed='both', dtype='interval[int64]') 

1247 """ 

1248 start = com.maybe_box_datetimelike(start) 

1249 end = com.maybe_box_datetimelike(end) 

1250 endpoint = start if start is not None else end 

1251 

1252 if freq is None and com._any_none(periods, start, end): 

1253 freq = 1 if is_number(endpoint) else 'D' 

1254 

1255 if com.count_not_none(start, end, periods, freq) != 3: 

1256 raise ValueError('Of the four parameters: start, end, periods, and ' 

1257 'freq, exactly three must be specified') 

1258 

1259 if not _is_valid_endpoint(start): 

1260 msg = 'start must be numeric or datetime-like, got {start}' 

1261 raise ValueError(msg.format(start=start)) 

1262 elif not _is_valid_endpoint(end): 

1263 msg = 'end must be numeric or datetime-like, got {end}' 

1264 raise ValueError(msg.format(end=end)) 

1265 

1266 if is_float(periods): 

1267 periods = int(periods) 

1268 elif not is_integer(periods) and periods is not None: 

1269 msg = 'periods must be a number, got {periods}' 

1270 raise TypeError(msg.format(periods=periods)) 

1271 

1272 if freq is not None and not is_number(freq): 

1273 try: 

1274 freq = to_offset(freq) 

1275 except ValueError: 

1276 raise ValueError('freq must be numeric or convertible to ' 

1277 'DateOffset, got {freq}'.format(freq=freq)) 

1278 

1279 # verify type compatibility 

1280 if not all([_is_type_compatible(start, end), 

1281 _is_type_compatible(start, freq), 

1282 _is_type_compatible(end, freq)]): 

1283 raise TypeError("start, end, freq need to be type compatible") 

1284 

1285 # +1 to convert interval count to breaks count (n breaks = n-1 intervals) 

1286 if periods is not None: 

1287 periods += 1 

1288 

1289 if is_number(endpoint): 

1290 # force consistency between start/end/freq (lower end if freq skips it) 

1291 if com._all_not_none(start, end, freq): 

1292 end -= (end - start) % freq 

1293 

1294 # compute the period/start/end if unspecified (at most one) 

1295 if periods is None: 

1296 periods = int((end - start) // freq) + 1 

1297 elif start is None: 

1298 start = end - (periods - 1) * freq 

1299 elif end is None: 

1300 end = start + (periods - 1) * freq 

1301 

1302 breaks = np.linspace(start, end, periods) 

1303 if all(is_integer(x) for x in com._not_none(start, end, freq)): 

1304 # np.linspace always produces float output 

1305 breaks = maybe_downcast_to_dtype(breaks, 'int64') 

1306 else: 

1307 # delegate to the appropriate range function 

1308 if isinstance(endpoint, Timestamp): 

1309 range_func = date_range 

1310 else: 

1311 range_func = timedelta_range 

1312 

1313 breaks = range_func(start=start, end=end, periods=periods, freq=freq) 

1314 

1315 return IntervalIndex.from_breaks(breaks, name=name, closed=closed)

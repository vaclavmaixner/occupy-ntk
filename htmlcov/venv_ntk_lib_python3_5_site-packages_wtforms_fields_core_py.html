178 def validate(self, form, extra_validators=tuple()): 

179 """ 

180 Validates the field and returns True or False. `self.errors` will 

181 contain any errors raised during validation. This is usually only 

182 called by `Form.validate`. 

183 

184 Subfields shouldn't override this, but rather override either 

185 `pre_validate`, `post_validate` or both, depending on needs. 

186 

187 :param form: The form the field belongs to. 

188 :param extra_validators: A sequence of extra validators to run. 

189 """ 

190 self.errors = list(self.process_errors) 

191 stop_validation = False 

192 

193 # Call pre_validate 

194 try: 

195 self.pre_validate(form) 

196 except StopValidation as e: 

197 if e.args and e.args[0]: 

198 self.errors.append(e.args[0]) 

199 stop_validation = True 

200 except ValueError as e: 

201 self.errors.append(e.args[0]) 

202 

203 # Run validators 

204 if not stop_validation: 

205 chain = itertools.chain(self.validators, extra_validators) 

206 stop_validation = self._run_validation_chain(form, chain) 

207 

208 # Call post_validate 

209 try: 

210 self.post_validate(form, stop_validation) 

211 except ValueError as e: 

212 self.errors.append(e.args[0]) 

213 

214 return len(self.errors) == 0

216 def _run_validation_chain(self, form, validators): 

217 """ 

218 Run a validation chain, stopping if any validator raises StopValidation. 

219 

220 :param form: The Form instance this field belongs to. 

221 :param validators: a sequence or iterable of validator callables. 

222 :return: True if validation was stopped, False otherwise. 

223 """ 

224 for validator in validators: 

225 try: 

226 validator(form, self) 

227 except StopValidation as e: 

228 if e.args and e.args[0]: 

229 self.errors.append(e.args[0]) 

230 return True 

231 except ValueError as e: 

232 self.errors.append(e.args[0]) 

233 

234 return False

256 def process(self, formdata, data=unset_value): 

257 """ 

258 Process incoming data, calling process_data, process_formdata as needed, 

259 and run filters. 

260 

261 If `data` is not provided, process_data will be called on the field's 

262 default. 

263 

264 Field subclasses usually won't override this, instead overriding the 

265 process_formdata and process_data methods. Only override this for 

266 special advanced processing, such as when a field encapsulates many 

267 inputs. 

268 """ 

269 self.process_errors = [] 

270 if data is unset_value: 

271 try: 

272 data = self.default() 

273 except TypeError: 

274 data = self.default 

275 

276 self.object_data = data 

277 

278 try: 

279 self.process_data(data) 

280 except ValueError as e: 

281 self.process_errors.append(e.args[0]) 

282 

283 if formdata is not None: 

284 if self.name in formdata: 

285 self.raw_data = formdata.getlist(self.name) 

286 else: 

287 self.raw_data = [] 

288

290 self.process_formdata(self.raw_data) 

291 except ValueError as e: 

292 self.process_errors.append(e.args[0]) 

293 

294 try: 

295 for filter in self.filters: 

296 self.data = filter(self.data) 

297 except ValueError as e:

594class DecimalField(LocaleAwareNumberField): 

595 """ 

596 A text field which displays and coerces data of the `decimal.Decimal` type. 

597 

598 :param places: 

599 How many decimal places to quantize the value to for display on form. 

600 If None, does not quantize value. 

601 :param rounding: 

602 How to round the value during quantize, for example 

603 `decimal.ROUND_UP`. If unset, uses the rounding value from the 

604 current thread's context. 

605 :param use_locale: 

606 If True, use locale-based number formatting. Locale-based number 

607 formatting requires the 'babel' package. 

608 :param number_format: 

609 Optional number format for locale. If omitted, use the default decimal 

610 format for the locale. 

611 """ 

612 widget = widgets.TextInput() 

613 

614 def __init__(self, label=None, validators=None, places=unset_value, rounding=None, **kwargs): 

615 super(DecimalField, self).__init__(label, validators, **kwargs) 

616 if self.use_locale and (places is not unset_value or rounding is not None): 

617 raise TypeError("When using locale-aware numbers, 'places' and 'rounding' are ignored.") 

618 

619 if places is unset_value: 

620 places = 2 

621 self.places = places 

622 self.rounding = rounding 

623 

624 def _value(self): 

625 if self.raw_data: 

626 return self.raw_data[0] 

627 elif self.data is not None: 

628 if self.use_locale: 

629 return text_type(self._format_decimal(self.data)) 

630 elif self.places is not None: 

631 if hasattr(self.data, 'quantize'): 

632 exp = decimal.Decimal('.1') ** self.places 

633 if self.rounding is None: 

634 quantized = self.data.quantize(exp) 

635 else: 

636 quantized = self.data.quantize(exp, rounding=self.rounding) 

637 return text_type(quantized) 

638 else: 

639 # If for some reason, data is a float or int, then format 

640 # as we would for floats using string formatting. 

641 format = '%%0.%df' % self.places 

642 return format % self.data 

643 else: 

644 return text_type(self.data) 

645 else: 

646 return '' 

647 

648 def process_formdata(self, valuelist): 

649 if valuelist: 

650 try: 

651 if self.use_locale: 

652 self.data = self._parse_decimal(valuelist[0]) 

653 else: 

654 self.data = decimal.Decimal(valuelist[0]) 

655 except (decimal.InvalidOperation, ValueError): 

656 self.data = None 

657 raise ValueError(self.gettext('Not a valid decimal value'))

853class FieldList(Field): 

854 """ 

855 Encapsulate an ordered list of multiple instances of the same field type, 

856 keeping data as a list. 

857 

858 >>> authors = FieldList(StringField('Name', [validators.required()])) 

859 

860 :param unbound_field: 

861 A partially-instantiated field definition, just like that would be 

862 defined on a form directly. 

863 :param min_entries: 

864 if provided, always have at least this many entries on the field, 

865 creating blank ones if the provided input does not specify a sufficient 

866 amount. 

867 :param max_entries: 

868 accept no more than this many entries as input, even if more exist in 

869 formdata. 

870 """ 

871 widget = widgets.ListWidget() 

872 

873 def __init__(self, unbound_field, label=None, validators=None, min_entries=0, 

874 max_entries=None, default=tuple(), **kwargs): 

875 super(FieldList, self).__init__(label, validators, default=default, **kwargs) 

876 if self.filters: 

877 raise TypeError('FieldList does not accept any filters. Instead, define them on the enclosed field.') 

878 assert isinstance(unbound_field, UnboundField), 'Field must be unbound, not a field class' 

879 self.unbound_field = unbound_field 

880 self.min_entries = min_entries 

881 self.max_entries = max_entries 

882 self.last_index = -1 

883 self._prefix = kwargs.get('_prefix', '') 

884 

885 def process(self, formdata, data=unset_value): 

886 self.entries = [] 

887 if data is unset_value or not data: 

888 try: 

889 data = self.default() 

890 except TypeError: 

891 data = self.default 

892 

893 self.object_data = data 

894 

895 if formdata: 

896 indices = sorted(set(self._extract_indices(self.name, formdata))) 

897 if self.max_entries: 

898 indices = indices[:self.max_entries] 

899 

900 idata = iter(data) 

901 for index in indices: 

902 try: 

903 obj_data = next(idata) 

904 except StopIteration: 

905 obj_data = unset_value 

906 self._add_entry(formdata, obj_data, index=index) 

907 else: 

908 for obj_data in data: 

909 self._add_entry(formdata, obj_data) 

910 

911 while len(self.entries) < self.min_entries: 

912 self._add_entry(formdata) 

913 

914 def _extract_indices(self, prefix, formdata): 

915 """ 

916 Yield indices of any keys with given prefix. 

917 

918 formdata must be an object which will produce keys when iterated. For 

919 example, if field 'foo' contains keys 'foo-0-bar', 'foo-1-baz', then 

920 the numbers 0 and 1 will be yielded, but not neccesarily in order. 

921 """ 

922 offset = len(prefix) + 1 

923 for k in formdata: 

924 if k.startswith(prefix): 

925 k = k[offset:].split('-', 1)[0] 

926 if k.isdigit(): 

927 yield int(k)

929 def validate(self, form, extra_validators=tuple()): 

930 """ 

931 Validate this FieldList. 

932 

933 Note that FieldList validation differs from normal field validation in 

934 that FieldList validates all its enclosed fields first before running any 

935 of its own validators. 

936 """ 

937 self.errors = [] 

938 

939 # Run validators on all entries within 

940 for subfield in self.entries: 

941 if not subfield.validate(form): 

942 self.errors.append(subfield.errors) 

943 

944 chain = itertools.chain(self.validators, extra_validators) 

945 self._run_validation_chain(form, chain) 

946 

947 return len(self.errors) == 0

949 def populate_obj(self, obj, name): 

950 values = getattr(obj, name, None) 

951 try: 

952 ivalues = iter(values) 

953 except TypeError: 

954 ivalues = iter([]) 

955 

956 candidates = itertools.chain(ivalues, itertools.repeat(None)) 

957 _fake = type(str('_fake'), (object, ), {}) 

958 output = [] 

959 for field, data in izip(self.entries, candidates): 

960 fake_obj = _fake() 

961 fake_obj.data = data 

962 field.populate_obj(fake_obj, 'data') 

963 output.append(fake_obj.data) 

964 

965 setattr(obj, name, output)

967 def _add_entry(self, formdata=None, data=unset_value, index=None): 

968 assert not self.max_entries or len(self.entries) < self.max_entries, \ 

969 'You cannot have more than max_entries entries in this FieldList' 

970 if index is None: 

971 index = self.last_index + 1 

972 self.last_index = index 

973 name = '%s-%d' % (self.short_name, index) 

974 id = '%s-%d' % (self.id, index) 

975 field = self.unbound_field.bind(form=None, name=name, prefix=self._prefix, id=id, _meta=self.meta, 

976 translations=self._translations) 

977 field.process(formdata, data) 

978 self.entries.append(field) 

979 return field

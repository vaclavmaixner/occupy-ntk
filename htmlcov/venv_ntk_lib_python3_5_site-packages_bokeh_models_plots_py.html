69 def select(self, *args, **kwargs): 

70 ''' Query this object and all of its references for objects that 

71 match the given selector. 

72 

73 There are a few different ways to call the ``select`` method. 

74 The most general is to supply a JSON-like query dictionary as the 

75 single argument or as keyword arguments: 

76 

77 Args: 

78 selector (JSON-like) : some sample text 

79 

80 Keyword Arguments: 

81 kwargs : query dict key/values as keyword arguments 

82 

83 Additionally, for compatibility with ``Model.select``, a selector 

84 dict may be passed as ``selector`` keyword argument, in which case 

85 the value of ``kwargs['selector']`` is used for the query. 

86 

87 For convenience, queries on just names can be made by supplying 

88 the ``name`` string as the single parameter: 

89 

90 Args: 

91 name (str) : the name to query on 

92 

93 Also queries on just type can be made simply by supplying the 

94 ``Model`` subclass as the single parameter: 

95 

96 Args: 

97 type (Model) : the type to query on 

98 

99 Returns: 

100 seq[Model] 

101 

102 Examples: 

103 

104 .. code-block:: python 

105 

106 # These three are equivalent 

107 p.select(selector={"type": HoverTool}) 

108 p.select({"type": HoverTool}) 

109 p.select(HoverTool) 

110 

111 # These two are also equivalent 

112 p.select({"name": "mycircle"}) 

113 p.select("mycircle") 

114 

115 # Keyword arguments can be supplied in place of selector dict 

116 p.select({"name": "foo", "type": HoverTool}) 

117 p.select(name="foo", type=HoverTool) 

118 

119 ''' 

120 

121 selector = _select_helper(args, kwargs) 

122 

123 # Want to pass selector that is a dictionary 

124 return _list_attr_splat(find(self.references(), selector, {'plot': self}))

349 incompatible.append("incompatibility on x-dimension: %s, %s" %(rng, self.x_scale))

369 @error(BAD_EXTRA_RANGE_NAME) 

370 def _check_bad_extra_range_name(self): 

371 msg = "" 

372 filt = lambda x: x is not self and isinstance(x, Plot) 

373 for ref in collect_filtered_models(filt, self): 

374 prop_names = ref.properties() 

375 bad = [] 

376 if 'x_range_name' in prop_names and 'y_range_name' in prop_names: 

377 if ref.x_range_name not in self.extra_x_ranges and ref.x_range_name != "default": 

378 bad.append(('x_range_name', ref.x_range_name)) 

379 if ref.y_range_name not in self.extra_y_ranges and ref.y_range_name != "default": 

380 bad.append(('y_range_name', ref.y_range_name)) 

381 if bad: 

382 if msg: msg += ", " 

383 msg += (", ".join("%s=%r" % (a, b) for (a,b) in bad) + " [%s]" % ref) 

384 if msg: 

385 return msg

736class _list_attr_splat(list): 

737 def __setattr__(self, attr, value): 

738 for x in self: 

739 setattr(x, attr, value) 

740 def __getattribute__(self, attr): 

741 if attr in dir(list): 

742 return list.__getattribute__(self, attr) 

743 if len(self) == 0: 

744 raise AttributeError("Trying to access %r attribute on an empty 'splattable' list" % attr) 

745 if len(self) == 1: 

746 return getattr(self[0], attr) 

747 try: 

748 return _list_attr_splat([getattr(x, attr) for x in self]) 

749 except Exception: 

750 raise AttributeError("Trying to access %r attribute on a 'splattable' list, but list items have no %r attribute" % (attr, attr))

753 if len(set(type(x) for x in self)) == 1:

770def _select_helper(args, kwargs): 

771 """ Allow flexible selector syntax. 

772 

773 Returns: 

774 dict 

775 

776 """ 

777 if len(args) > 1: 

778 raise TypeError("select accepts at most ONE positional argument.") 

779 

780 if len(args) > 0 and len(kwargs) > 0: 

781 raise TypeError("select accepts EITHER a positional argument, OR keyword arguments (not both).") 

782 

783 if len(args) == 0 and len(kwargs) == 0: 

784 raise TypeError("select requires EITHER a positional argument, OR keyword arguments.") 

785 

786 if args: 

787 arg = args[0] 

788 if isinstance(arg, dict): 

789 selector = arg 

790 elif isinstance(arg, string_types): 

791 selector = dict(name=arg) 

792 elif isinstance(arg, type) and issubclass(arg, Model): 

793 selector = {"type": arg} 

794 else: 

795 raise TypeError("selector must be a dictionary, string or plot object.") 

796 

797 elif 'selector' in kwargs: 

798 if len(kwargs) == 1: 

799 selector = kwargs['selector'] 

800 else: 

801 raise TypeError("when passing 'selector' keyword arg, not other keyword args may be present") 

802 

803 else: 

804 selector = kwargs 

805 

806 return selector

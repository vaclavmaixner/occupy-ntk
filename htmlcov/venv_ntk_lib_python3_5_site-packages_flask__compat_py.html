60def with_metaclass(meta, *bases): 

61 """Create a base class with a metaclass.""" 

62 # This requires a bit of explanation: the basic idea is to make a 

63 # dummy metaclass for one level of class instantiation that replaces 

64 # itself with the actual metaclass. 

65 class metaclass(type): 

66 def __new__(metacls, name, this_bases, d): 

67 return meta(name, bases, d)

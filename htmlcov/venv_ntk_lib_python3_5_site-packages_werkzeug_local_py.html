30def release_local(local): 

31 """Releases the contents of the local for the current context. 

32 This makes it possible to use locals without a manager. 

33 

34 Example:: 

35 

36 >>> loc = Local() 

37 >>> loc.foo = 42 

38 >>> release_local(loc) 

39 >>> hasattr(loc, 'foo') 

40 False 

41 

42 With this function one can release :class:`Local` objects as well 

43 as :class:`LocalStack` objects. However it is not possible to 

44 release data held by proxies that way, one always has to retain 

45 a reference to the underlying local object in order to be able 

46 to release it. 

47 

48 .. versionadded:: 0.6.1 

49 """ 

50 local.__release_local__() 

51

53class Local(object): 

54 __slots__ = ("__storage__", "__ident_func__") 

55 

56 def __init__(self): 

57 object.__setattr__(self, "__storage__", {}) 

58 object.__setattr__(self, "__ident_func__", get_ident) 

59 

60 def __iter__(self): 

61 return iter(self.__storage__.items()) 

62 

63 def __call__(self, proxy): 

64 """Create a proxy for a name.""" 

65 return LocalProxy(self, proxy) 

66 

67 def __release_local__(self): 

68 self.__storage__.pop(self.__ident_func__(), None) 

69 

70 def __getattr__(self, name): 

71 try: 

72 return self.__storage__[self.__ident_func__()][name] 

73 except KeyError: 

74 raise AttributeError(name) 

75 

76 def __setattr__(self, name, value): 

77 ident = self.__ident_func__() 

78 storage = self.__storage__ 

79 try: 

80 storage[ident][name] = value 

81 except KeyError: 

82 storage[ident] = {name: value} 

83 

84 def __delattr__(self, name): 

85 try: 

86 del self.__storage__[self.__ident_func__()][name] 

87 except KeyError: 

88 raise AttributeError(name)

91class LocalStack(object): 

92 """This class works similar to a :class:`Local` but keeps a stack 

93 of objects instead. This is best explained with an example:: 

94 

95 >>> ls = LocalStack() 

96 >>> ls.push(42) 

97 >>> ls.top 

98 42 

99 >>> ls.push(23) 

100 >>> ls.top 

101 23 

102 >>> ls.pop() 

103 23 

104 >>> ls.top 

105 42 

106 

107 They can be force released by using a :class:`LocalManager` or with 

108 the :func:`release_local` function but the correct way is to pop the 

109 item from the stack after using. When the stack is empty it will 

110 no longer be bound to the current context (and as such released). 

111 

112 By calling the stack without arguments it returns a proxy that resolves to 

113 the topmost item on the stack. 

114 

115 .. versionadded:: 0.6.1 

116 """ 

117 

118 def __init__(self): 

119 self._local = Local() 

120 

121 def __release_local__(self): 

122 self._local.__release_local__() 

123 

124 def _get__ident_func__(self): 

125 return self._local.__ident_func__ 

126 

127 def _set__ident_func__(self, value): 

128 object.__setattr__(self._local, "__ident_func__", value) 

129 

130 __ident_func__ = property(_get__ident_func__, _set__ident_func__) 

131 del _get__ident_func__, _set__ident_func__ 

132 

133 def __call__(self): 

134 def _lookup(): 

135 rv = self.top 

136 if rv is None: 

137 raise RuntimeError("object unbound") 

138 return rv 

139 

140 return LocalProxy(_lookup) 

141 

142 def push(self, obj): 

143 """Pushes a new item to the stack""" 

144 rv = getattr(self._local, "stack", None) 

145 if rv is None: 

146 self._local.stack = rv = [] 

147 rv.append(obj) 

148 return rv 

149 

150 def pop(self): 

151 """Removes the topmost item from the stack, will return the 

152 old value or `None` if the stack was already empty. 

153 """ 

154 stack = getattr(self._local, "stack", None) 

155 if stack is None: 

156 return None 

157 elif len(stack) == 1: 

158 release_local(self._local) 

159 return stack[-1] 

160 else: 

161 return stack.pop() 

162 

163 @property 

164 def top(self): 

165 """The topmost item on the stack. If the stack is empty, 

166 `None` is returned. 

167 """ 

168 try: 

169 return self._local.stack[-1] 

170 except (AttributeError, IndexError): 

171 return None

174class LocalManager(object): 

175 """Local objects cannot manage themselves. For that you need a local 

176 manager. You can pass a local manager multiple locals or add them later 

177 by appending them to `manager.locals`. Every time the manager cleans up, 

178 it will clean up all the data left in the locals for this context. 

179 

180 The `ident_func` parameter can be added to override the default ident 

181 function for the wrapped locals. 

182 

183 .. versionchanged:: 0.6.1 

184 Instead of a manager the :func:`release_local` function can be used 

185 as well. 

186 

187 .. versionchanged:: 0.7 

188 `ident_func` was added. 

189 """ 

190 

191 def __init__(self, locals=None, ident_func=None): 

192 if locals is None: 

193 self.locals = [] 

194 elif isinstance(locals, Local): 

195 self.locals = [locals] 

196 else: 

197 self.locals = list(locals) 

198 if ident_func is not None: 

199 self.ident_func = ident_func 

200 for local in self.locals: 

201 object.__setattr__(local, "__ident_func__", ident_func) 

202 else: 

203 self.ident_func = get_ident 

204 

205 def get_ident(self): 

206 """Return the context identifier the local objects use internally for 

207 this context. You cannot override this method to change the behavior 

208 but use it to link other context local objects (such as SQLAlchemy's 

209 scoped sessions) to the Werkzeug locals. 

210 

211 .. versionchanged:: 0.7 

212 You can pass a different ident function to the local manager that 

213 will then be propagated to all the locals passed to the 

214 constructor. 

215 """ 

216 return self.ident_func() 

217 

218 def cleanup(self): 

219 """Manually clean up the data in the locals for this context. Call 

220 this at the end of the request or use `make_middleware()`. 

221 """ 

222 for local in self.locals: 

223 release_local(local) 

224 

225 def make_middleware(self, app): 

226 """Wrap a WSGI application so that cleaning up happens after 

227 request end. 

228 """ 

229 

230 def application(environ, start_response): 

231 return ClosingIterator(app(environ, start_response), self.cleanup) 

232 

233 return application 

234 

235 def middleware(self, func): 

236 """Like `make_middleware` but for decorating functions. 

237 

238 Example usage:: 

239 

240 @manager.middleware 

241 def application(environ, start_response): 

242 ... 

243 

244 The difference to `make_middleware` is that the function passed 

245 will have all the arguments copied from the inner application 

246 (name, docstring, module). 

247 """ 

248 return update_wrapper(self.make_middleware(func), func)

251 return "<%s storages: %d>" % (self.__class__.__name__, len(self.locals))

255class LocalProxy(object): 

256 """Acts as a proxy for a werkzeug local. Forwards all operations to 

257 a proxied object. The only operations not supported for forwarding 

258 are right handed operands and any kind of assignment. 

259 

260 Example usage:: 

261 

262 from werkzeug.local import Local 

263 l = Local() 

264 

265 # these are proxies 

266 request = l('request') 

267 user = l('user') 

268 

269 

270 from werkzeug.local import LocalStack 

271 _response_local = LocalStack() 

272 

273 # this is a proxy 

274 response = _response_local() 

275 

276 Whenever something is bound to l.user / l.request the proxy objects 

277 will forward all operations. If no object is bound a :exc:`RuntimeError` 

278 will be raised. 

279 

280 To create proxies to :class:`Local` or :class:`LocalStack` objects, 

281 call the object as shown above. If you want to have a proxy to an 

282 object looked up by a function, you can (as of Werkzeug 0.6.1) pass 

283 a function to the :class:`LocalProxy` constructor:: 

284 

285 session = LocalProxy(lambda: get_current_request().session) 

286 

287 .. versionchanged:: 0.6.1 

288 The class can be instantiated with a callable as well now. 

289 """

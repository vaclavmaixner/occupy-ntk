62def _align_core(terms): 

63 term_index = [i for i, term in enumerate(terms) 

64 if hasattr(term.value, 'axes')] 

65 term_dims = [terms[i].value.ndim for i in term_index] 

66 ndims = pd.Series(dict(zip(term_index, term_dims))) 

67 

68 # initial axes are the axes of the largest-axis'd term 

69 biggest = terms[ndims.idxmax()].value 

70 typ = biggest._constructor 

71 axes = biggest.axes 

72 naxes = len(axes) 

73 gt_than_one_axis = naxes > 1 

74 

75 for value in (terms[i].value for i in term_index): 

76 is_series = isinstance(value, pd.Series) 

77 is_series_and_gt_one_axis = is_series and gt_than_one_axis 

78 

79 for axis, items in enumerate(value.axes): 

80 if is_series_and_gt_one_axis: 

81 ax, itm = naxes - 1, value.index 

82 else: 

83 ax, itm = axis, items 

84 

85 if not axes[ax].is_(itm): 

86 axes[ax] = axes[ax].join(itm, how='outer')

88 for i, ndim in compat.iteritems(ndims): 

89 for axis, items in zip(range(ndim), axes): 

90 ti = terms[i].value 

91 

92 if hasattr(ti, 'reindex'): 

93 transpose = isinstance(ti, pd.Series) and naxes > 1 

94 reindexer = axes[naxes - 1] if transpose else items 

95 

96 term_axis_size = len(ti.axes[axis]) 

97 reindexer_size = len(reindexer) 

98 

99 ordm = np.log10(max(1, abs(reindexer_size - term_axis_size)))

101 w = ('Alignment difference on axis {axis} is larger '

104 ).format(axis=axis, term=terms[i].name, ordm=ordm) 

105 warnings.warn(w, category=PerformanceWarning, stacklevel=6)

107 f = partial(ti.reindex, reindexer, axis=axis, copy=False)

109 terms[i].update(f())

116def _align(terms): 

117 """Align a set of terms""" 

118 try: 

119 # flatten the parse tree (a nested list, really) 

120 terms = list(com.flatten(terms)) 

121 except TypeError: 

122 # can't iterate so it must just be a constant or single variable 

123 if isinstance(terms.value, pd.core.generic.NDFrame): 

124 typ = type(terms.value) 

125 return typ, _zip_axes_from_type(typ, terms.value.axes) 

126 return np.result_type(terms.type), None 

127 

128 # if all resolved variables are numeric scalars 

129 if all(term.is_scalar for term in terms): 

130 return _result_type_many(*(term.value for term in terms)).type, None 

131 

132 # perform the main alignment 

133 typ, axes = _align_core(terms) 

134 return typ, axes

137def _reconstruct_object(typ, obj, axes, dtype): 

138 """Reconstruct an object given its type, raw value, and possibly empty 

139 (None) axes. 

140 

141 Parameters 

142 ---------- 

143 typ : object 

144 A type 

145 obj : object 

146 The value to use in the type constructor 

147 axes : dict 

148 The axes to use to construct the resulting pandas object 

149 

150 Returns 

151 ------- 

152 ret : typ 

153 An object of type ``typ`` with the value `obj` and possible axes 

154 `axes`. 

155 """ 

156 try: 

157 typ = typ.type 

158 except AttributeError: 

159 pass 

160 

161 res_t = np.result_type(obj.dtype, dtype) 

162 

163 if (not isinstance(typ, partial) and 

164 issubclass(typ, pd.core.generic.PandasObject)): 

165 return typ(obj, dtype=res_t, **axes) 

166 

167 # special case for pathological things like ~True/~False 

168 if hasattr(res_t, 'type') and typ == np.bool_ and res_t != np.bool_: 

169 ret_value = res_t.type(obj) 

170 else: 

171 ret_value = typ(obj).astype(res_t) 

172 # The condition is to distinguish 0-dim array (returned in case of 

173 # scalar) and 1 element array 

174 # e.g. np.array(0) and np.array([0]) 

175 if len(obj.shape) == 1 and len(obj) == 1: 

176 if not isinstance(ret_value, np.ndarray): 

177 ret_value = np.array([ret_value]).astype(res_t) 

178 

179 return ret_value

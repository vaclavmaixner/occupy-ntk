52# Note: in DirectoryHandler and in general we assume this is an 

53# immutable object, because we share it among sessions and we 

54# don't monitor it for changes. If you make this mutable by adding 

55# any kind of setter, you could have to refactor some other code. 

56class Theme(object): 

57 ''' Provide new default values for Bokeh models. 

58 

59 Bokeh Model properties all have some built-in default value. If a property 

60 has not been explicitly set (e.g. ``m.foo = 10``) then accessing the 

61 property with return the default value. It may be useful for users to be 

62 able to specify a different set of default values than the built-in 

63 default. The ``Theme`` class allows collections of custom default values 

64 to be easily applied to Bokeh documents. 

65 

66 The ``Theme`` class can be constructed either from a YAML file or from a 

67 JSON dict (but not both). Examples of both formats are shown below. 

68 

69 The plotting API's defaults override some theme properties. Namely: 

70 `fill_alpha`, `fill_color`, `line_alpha`, `line_color`, `text_alpha` and 

71 `text_color`. Those properties should therefore be set explicitly when 

72 using the plotting API. 

73 

74 Args: 

75 filename (str, optional) : path to a YAML theme file 

76 json (str, optional) : a JSON dictionary specifying theme values 

77 

78 Raises: 

79 ValueError 

80 If neither ``filename`` or ``json`` is supplied. 

81 

82 Examples: 

83 

84 Themes are specified by providing a top-level key ``attrs`` which 

85 has blocks for Model types to be themed. Each block has keys and 

86 values that specify the new property defaults for that type. 

87 

88 Here is an example theme in YAML format that sets various visual 

89 properties for all figures, grids, and titles: 

90 

91 .. code-block:: yaml 

92 

93 attrs: 

94 Figure: 

95 background_fill_color: '#2F2F2F' 

96 border_fill_color: '#2F2F2F' 

97 outline_line_color: '#444444' 

98 Grid: 

99 grid_line_dash: [6, 4] 

100 grid_line_alpha: .3 

101 Title: 

102 text_color: "white" 

103 

104 Here is the same theme, in JSON format: 

105 

106 .. code-block:: python 

107 

108 { 

109 'attrs' : { 

110 'Figure' : { 

111 'background_fill_color': '#2F2F2F', 

112 'border_fill_color': '#2F2F2F', 

113 'outline_line_color': '#444444', 

114 }, 

115 'Grid': { 

116 'grid_line_dash': [6, 4]', 

117 'grid_line_alpha': .3, 

118 }, 

119 'Title': { 

120 'text_color': 'white' 

121 } 

122 } 

123 

124 ''' 

125 def __init__(self, filename=None, json=None): 

126 if (filename is not None) and (json is not None): 

127 raise ValueError("Theme should be constructed from a file or from json not both") 

128 

129 if filename is not None: 

130 f = open(filename) 

131 try: 

132 json = yaml.safe_load(f) 

133 # empty docs result in None rather than {}, fix it. 

134 if json is None: 

135 json = {} 

136 finally: 

137 f.close() 

138 

139 if json is None: 

140 raise ValueError("Theme requires json or a filename to construct") 

141 

142 self._json = json 

143 

144 if 'attrs' not in self._json: 

145 self._json['attrs'] = {} 

146 

147 if not isinstance(self._json['attrs'], dict): 

148 raise ValueError("theme problem: attrs field should be a dictionary of class names, not %r" % (self._json['attrs'])) 

149 

150 for key, value in self._json['attrs'].items(): 

151 if not isinstance(value, dict): 

152 raise ValueError("theme problem: attrs.%s should be a dictionary of properties, not %r" % (key, value)) 

153 

154 self._line_defaults = self._json.get('line_defaults', _empty_dict) 

155 self._fill_defaults = self._json.get('fill_defaults', _empty_dict) 

156 self._text_defaults = self._json.get('text_defaults', _empty_dict) 

157 

158 # mapping from class name to the full set of properties 

159 # (including those merged in from base classes) for that 

160 # class. 

161 self._by_class_cache = {} 

162 

163 def _add_glyph_defaults(self, cls, props): 

164 from ..models.glyphs import Glyph 

165 if issubclass(cls, Glyph): 

166 if hasattr(cls, "line_alpha"): 

167 props.update(self._line_defaults) 

168 if hasattr(cls, "fill_alpha"): 

169 props.update(self._fill_defaults) 

170 if hasattr(cls, "text_alpha"): 

171 props.update(self._text_defaults) 

172 

173 def _for_class(self, cls): 

174 if cls.__name__ not in self._by_class_cache: 

175 attrs = self._json['attrs'] 

176 combined = {} 

177 # we go in reverse order so that subclass props override base class 

178 for base in cls.__mro__[-2::-1]: 

179 if not issubclass(base, HasProps): 

180 continue 

181 self._add_glyph_defaults(base, combined) 

182 combined.update(attrs.get(base.__name__, _empty_dict)) 

183 if len(combined) == 0: 

184 combined = _empty_dict 

185 self._by_class_cache[cls.__name__] = combined 

186 return self._by_class_cache[cls.__name__] 

187 

188 def apply_to_model(self, model): 

189 ''' Apply this theme to a model. 

190 

191 .. warning:: 

192 Typically, don't call this method directly. Instead, set the theme 

193 on the :class:`~bokeh.document.Document` the model is a part of. 

194 

195 ''' 

196 model.apply_theme(self._for_class(model.__class__)) 

197 

198 # a little paranoia because it would be Bad(tm) to mess 

199 # this up... would be nicer if python had a way to freeze 

200 # the dict. 

201 if len(_empty_dict) > 0: 

202 raise RuntimeError("Somebody put stuff in _empty_dict")

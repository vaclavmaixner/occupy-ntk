20def mask_missing(arr, values_to_mask): 

21 """ 

22 Return a masking array of same size/shape as arr 

23 with entries equaling any member of values_to_mask set to True 

24 """ 

25 dtype, values_to_mask = infer_dtype_from_array(values_to_mask) 

26 

27 try: 

28 values_to_mask = np.array(values_to_mask, dtype=dtype) 

29 

30 except Exception: 

31 values_to_mask = np.array(values_to_mask, dtype=object) 

32 

33 na_mask = isna(values_to_mask) 

34 nonna = values_to_mask[~na_mask] 

35 

36 mask = None 

37 for x in nonna: 

38 if mask is None: 

39 

40 # numpy elementwise comparison warning 

41 if is_numeric_v_string_like(arr, x): 

42 mask = False 

43 else: 

44 mask = arr == x 

45 

46 # if x is a string and arr is not, then we get False and we must 

47 # expand the mask to size arr.shape 

48 if is_scalar(mask): 

49 mask = np.zeros(arr.shape, dtype=bool) 

50 else: 

51 

52 # numpy elementwise comparison warning 

53 if is_numeric_v_string_like(arr, x): 

54 mask |= False 

55 else: 

56 mask |= arr == x 

57 

58 if na_mask.any(): 

59 if mask is None: 

60 mask = isna(arr) 

61 else: 

62 mask |= isna(arr) 

63 

64 # GH 21977 

65 if mask is None: 

66 mask = np.zeros(arr.shape, dtype=bool) 

67 

68 return mask

111def interpolate_1d(xvalues, yvalues, method='linear', limit=None, 

112 limit_direction='forward', limit_area=None, fill_value=None, 

113 bounds_error=False, order=None, **kwargs): 

114 """ 

115 Logic for the 1-d interpolation. The result should be 1-d, inputs 

116 xvalues and yvalues will each be 1-d arrays of the same length. 

117 

118 Bounds_error is currently hardcoded to False since non-scipy ones don't 

119 take it as an argumnet. 

120 """ 

121 # Treat the original, non-scipy methods first. 

122 

123 invalid = isna(yvalues) 

124 valid = ~invalid 

125 

126 if not valid.any(): 

127 # have to call np.asarray(xvalues) since xvalues could be an Index 

128 # which can't be mutated 

129 result = np.empty_like(np.asarray(xvalues), dtype=np.float64) 

130 result.fill(np.nan) 

131 return result 

132 

133 if valid.all(): 

134 return yvalues 

135 

136 if method == 'time': 

137 if not getattr(xvalues, 'is_all_dates', None): 

138 # if not issubclass(xvalues.dtype.type, np.datetime64): 

139 raise ValueError('time-weighted interpolation only works ' 

140 'on Series or DataFrames with a ' 

141 'DatetimeIndex') 

142 method = 'values' 

143 

144 valid_limit_directions = ['forward', 'backward', 'both'] 

145 limit_direction = limit_direction.lower() 

146 if limit_direction not in valid_limit_directions: 

147 msg = ('Invalid limit_direction: expecting one of {valid!r}, ' 

148 'got {invalid!r}.') 

149 raise ValueError(msg.format(valid=valid_limit_directions, 

150 invalid=limit_direction)) 

151 

152 if limit_area is not None: 

153 valid_limit_areas = ['inside', 'outside'] 

154 limit_area = limit_area.lower() 

155 if limit_area not in valid_limit_areas: 

156 raise ValueError('Invalid limit_area: expecting one of {}, got ' 

157 '{}.'.format(valid_limit_areas, limit_area))

159 # default limit is unlimited GH #16282 

160 if limit is None: 

161 # limit = len(xvalues) 

162 pass 

163 elif not is_integer(limit): 

164 raise ValueError('Limit must be an integer') 

165 elif limit < 1: 

166 raise ValueError('Limit must be greater than 0') 

167 

168 from pandas import Series 

169 ys = Series(yvalues) 

170 

171 # These are sets of index pointers to invalid values... i.e. {0, 1, etc... 

172 all_nans = set(np.flatnonzero(invalid)) 

173 start_nans = set(range(ys.first_valid_index())) 

174 end_nans = set(range(1 + ys.last_valid_index(), len(valid))) 

175 mid_nans = all_nans - start_nans - end_nans 

176

178 # but in this case, it is the final set of indices that need to be 

179 # preserved as NaN after the interpolation. 

180 

181 # For example if limit_direction='forward' then preserve_nans will 

182 # contain indices of NaNs at the beginning of the series, and NaNs that 

183 # are more than'limit' away from the prior non-NaN. 

184 

185 # set preserve_nans based on direction using _interp_limit 

186 if limit_direction == 'forward': 

187 preserve_nans = start_nans | set(_interp_limit(invalid, limit, 0)) 

188 elif limit_direction == 'backward': 

189 preserve_nans = end_nans | set(_interp_limit(invalid, 0, limit)) 

190 else: 

191 # both directions... just use _interp_limit 

192 preserve_nans = set(_interp_limit(invalid, limit, limit)) 

193 

194 # if limit_area is set, add either mid or outside indices 

195 # to preserve_nans GH #16284 

196 if limit_area == 'inside': 

197 # preserve NaNs on the outside 

198 preserve_nans |= start_nans | end_nans 

199 elif limit_area == 'outside': 

200 # preserve NaNs on the inside 

201 preserve_nans |= mid_nans 

202 

203 # sort preserve_nans and covert to list 

204 preserve_nans = sorted(preserve_nans) 

205 

206 xvalues = getattr(xvalues, 'values', xvalues) 

207 yvalues = getattr(yvalues, 'values', yvalues) 

208 result = yvalues.copy() 

209 

210 if method in ['linear', 'time', 'index', 'values']: 

211 if method in ('values', 'index'): 

212 inds = np.asarray(xvalues) 

213 # hack for DatetimeIndex, #1646 

214 if needs_i8_conversion(inds.dtype.type): 

215 inds = inds.view(np.int64) 

216 if inds.dtype == np.object_: 

217 inds = lib.maybe_convert_objects(inds) 

218 else: 

219 inds = xvalues 

220 result[invalid] = np.interp(inds[invalid], inds[valid], yvalues[valid]) 

221 result[preserve_nans] = np.nan 

222 return result

225 'barycentric', 'krogh', 'spline', 'polynomial', 

226 'from_derivatives', 'piecewise_polynomial', 'pchip', 'akima']

244def _interpolate_scipy_wrapper(x, y, new_x, method, fill_value=None, 

245 bounds_error=False, order=None, **kwargs): 

246 """ 

247 passed off to scipy.interpolate.interp1d. method is scipy's kind. 

248 Returns an array interpolated at new_x. Add any new methods to 

249 the list in _clean_interp_method 

250 """ 

251 try: 

252 from scipy import interpolate 

253 # TODO: Why is DatetimeIndex being imported here? 

254 from pandas import DatetimeIndex # noqa 

255 except ImportError: 

256 raise ImportError('{method} interpolation requires SciPy' 

257 .format(method=method)) 

258 

259 new_x = np.asarray(new_x) 

260 

261 # ignores some kwargs that could be passed along. 

262 alt_methods = { 

263 'barycentric': interpolate.barycentric_interpolate, 

264 'krogh': interpolate.krogh_interpolate, 

265 'from_derivatives': _from_derivatives, 

266 'piecewise_polynomial': _from_derivatives, 

267 } 

268 

269 if getattr(x, 'is_all_dates', False): 

270 # GH 5975, scipy.interp1d can't hande datetime64s 

271 x, new_x = x._values.astype('i8'), new_x.astype('i8') 

272 

273 if method == 'pchip': 

274 try: 

275 alt_methods['pchip'] = interpolate.pchip_interpolate 

276 except AttributeError: 

277 raise ImportError("Your version of Scipy does not support " 

278 "PCHIP interpolation.") 

279 elif method == 'akima': 

280 try: 

281 from scipy.interpolate import Akima1DInterpolator # noqa 

282 alt_methods['akima'] = _akima_interpolate 

283 except ImportError: 

284 raise ImportError("Your version of Scipy does not support " 

285 "Akima interpolation.") 

286 

287 interp1d_methods = ['nearest', 'zero', 'slinear', 'quadratic', 'cubic', 

288 'polynomial'] 

289 if method in interp1d_methods: 

290 if method == 'polynomial': 

291 method = order 

292 terp = interpolate.interp1d(x, y, kind=method, fill_value=fill_value, 

293 bounds_error=bounds_error) 

294 new_y = terp(new_x) 

295 elif method == 'spline': 

296 # GH #10633 

297 if not order: 

298 raise ValueError("order needs to be specified and greater than 0") 

299 terp = interpolate.UnivariateSpline(x, y, k=order, **kwargs) 

300 new_y = terp(new_x) 

301 else: 

302 # GH 7295: need to be able to write for some reason 

303 # in some circumstances: check all three 

304 if not x.flags.writeable: 

305 x = x.copy() 

306 if not y.flags.writeable: 

307 y = y.copy() 

308 if not new_x.flags.writeable: 

309 new_x = new_x.copy() 

310 method = alt_methods[method] 

311 new_y = method(x, y, new_x, **kwargs) 

312 return new_y

534def fill_zeros(result, x, y, name, fill): 

535 """ 

536 if this is a reversed op, then flip x,y 

537 

538 if we have an integer value (or array in y)

541

589def mask_zero_div_zero(x, y, result, copy=False): 

590 """ 

591 Set results of 0 / 0 or 0 // 0 to np.nan, regardless of the dtypes 

592 of the numerator or the denominator. 

593 

594 Parameters 

595 ---------- 

596 x : ndarray 

597 y : ndarray 

598 result : ndarray 

599 copy : bool (default False) 

600 Whether to always create a new array or try to fill in the existing 

601 array if possible. 

602 

603 Returns 

604 ------- 

605 filled_result : ndarray 

606 

607 Examples 

608 -------- 

609 >>> x = np.array([1, 0, -1], dtype=np.int64) 

610 >>> y = 0 # int 0; numpy behavior is different with float 

611 >>> result = x / y 

612 >>> result # raw numpy result does not fill division by zero 

613 array([0, 0, 0]) 

614 >>> mask_zero_div_zero(x, y, result) 

615 array([ inf, nan, -inf]) 

616 """ 

617 if is_scalar(y): 

618 y = np.array(y) 

619 

620 zmask = y == 0 

621 if zmask.any(): 

622 shape = result.shape 

623 

624 nan_mask = (zmask & (x == 0)).ravel() 

625 neginf_mask = (zmask & (x < 0)).ravel() 

626 posinf_mask = (zmask & (x > 0)).ravel() 

627 

628 if nan_mask.any() or neginf_mask.any() or posinf_mask.any(): 

629 # Fill negative/0 with -inf, positive/0 with +inf, 0/0 with NaN 

630 result = result.astype('float64', copy=copy).ravel() 

631 

632 np.putmask(result, nan_mask, np.nan) 

633 np.putmask(result, posinf_mask, np.inf) 

634 np.putmask(result, neginf_mask, -np.inf) 

635 

636 result = result.reshape(shape) 

637 

638 return result

642 """ 

643 Fill nulls caused by division by zero, casting to a diffferent dtype 

644 if necessary. 

645 

646 Parameters 

647 ---------- 

648 op : function (operator.add, operator.div, ...) 

649 left : object (Index for non-reversed ops) 

650 right : object (Index fof reversed ops) 

651 result : ndarray 

652

654 ------- 

655 result : ndarray 

656 """ 

657 opstr = '__{opname}__'.format(opname=op.__name__).replace('____', '__')

660 result = mask_zero_div_zero(left, right, result) 

661 elif op is operator.mod: 

662 result = fill_zeros(result, left, right, opstr, np.nan) 

663 elif op is divmod: 

664 res0 = mask_zero_div_zero(left, right, result[0]) 

665 res1 = fill_zeros(result[1], left, right, opstr, np.nan) 

666 result = (res0, res1) 

667 return result

670def _interp_limit(invalid, fw_limit, bw_limit): 

671 """ 

672 Get indexers of values that won't be filled 

673 because they exceed the limits. 

674 

675 Parameters 

676 ---------- 

677 invalid : boolean ndarray 

678 fw_limit : int or None 

679 forward limit to index 

680 bw_limit : int or None 

681 backward limit to index 

682 

683 Returns 

684 ------- 

685 set of indexers 

686 

687 Notes 

688 ----- 

689 This is equivalent to the more readable, but slower 

690 

691 .. code-block:: python 

692 

693 def _interp_limit(invalid, fw_limit, bw_limit): 

694 for x in np.where(invalid)[0]: 

695 if invalid[max(0, x - fw_limit):x + bw_limit + 1].all(): 

696 yield x 

697 """ 

698 # handle forward first; the backward direction is the same except 

699 # 1. operate on the reversed array 

700 # 2. subtract the returned indices from N - 1 

701 N = len(invalid) 

702 f_idx = set() 

703 b_idx = set() 

704 

705 def inner(invalid, limit): 

706 limit = min(limit, N) 

707 windowed = _rolling_window(invalid, limit + 1).all(1) 

708 idx = (set(np.where(windowed)[0] + limit) | 

709 set(np.where((~invalid[:limit + 1]).cumsum() == 0)[0])) 

710 return idx 

711 

712 if fw_limit is not None: 

713 

714 if fw_limit == 0: 

715 f_idx = set(np.where(invalid)[0]) 

716 else: 

717 f_idx = inner(invalid, fw_limit) 

718 

719 if bw_limit is not None: 

720 

721 if bw_limit == 0: 

722 # then we don't even need to care about backwards 

723 # just use forwards 

724 return f_idx 

725 else: 

726 b_idx = list(inner(invalid[::-1], bw_limit)) 

727 b_idx = set(N - 1 - np.asarray(b_idx)) 

728 if fw_limit == 0: 

729 return b_idx 

730 

731 return f_idx & b_idx

734def _rolling_window(a, window): 

735 """ 

736 [True, True, False, True, False], 2 -> 

737 

738 [ 

739 [True, True], 

740 [True, False], 

741 [False, True], 

742 [True, False], 

743 ] 

744 """ 

745 # https://stackoverflow.com/a/6811241 

746 shape = a.shape[:-1] + (a.shape[-1] - window + 1, window) 

747 strides = a.strides + (a.strides[-1],) 

748 return np.lib.stride_tricks.as_strided(a, shape=shape, strides=strides)

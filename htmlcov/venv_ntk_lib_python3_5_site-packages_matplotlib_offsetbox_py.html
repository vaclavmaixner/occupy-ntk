52def _get_packed_offsets(wd_list, total, sep, mode="fixed"): 

53 """ 

54 Geiven a list of (width, xdescent) of each boxes, calculate the 

55 total width and the x-offset positions of each items according to 

56 *mode*. xdescent is analogous to the usual descent, but along the 

57 x-direction. xdescent values are currently ignored. 

58 

59 *wd_list* : list of (width, xdescent) of boxes to be packed. 

60 *sep* : spacing between boxes 

61 *total* : Intended total length. None if not used. 

62 *mode* : packing mode. 'fixed', 'expand', or 'equal'. 

63 """ 

64 

65 w_list, d_list = zip(*wd_list) 

66 # d_list is currently not used. 

67 

68 if mode == "fixed": 

69 offsets_ = np.cumsum([0] + [w + sep for w in w_list]) 

70 offsets = offsets_[:-1] 

71 if total is None: 

72 total = offsets_[-1] - sep 

73 return total, offsets 

74 

75 elif mode == "expand": 

76 # This is a bit of a hack to avoid a TypeError when *total* 

77 # is None and used in conjugation with tight layout. 

78 if total is None: 

79 total = 1 

80 if len(w_list) > 1: 

81 sep = (total - sum(w_list)) / (len(w_list) - 1) 

82 else: 

83 sep = 0 

84 offsets_ = np.cumsum([0] + [w + sep for w in w_list]) 

85 offsets = offsets_[:-1] 

86 return total, offsets 

87 

88 elif mode == "equal": 

89 maxh = max(w_list) 

90 if total is None: 

91 total = (maxh + sep) * len(w_list) 

92 else: 

93 sep = total / len(w_list) - maxh 

94 offsets = (maxh + sep) * np.arange(len(w_list)) 

95 return total, offsets 

96 

97 else: 

98 raise ValueError("Unknown mode : %s" % (mode,))

101def _get_aligned_offsets(hd_list, height, align="baseline"): 

102 """ 

103 Given a list of (height, descent) of each boxes, align the boxes 

104 with *align* and calculate the y-offsets of each boxes. 

105 total width and the offset positions of each items according to 

106 *mode*. xdescent is analogous to the usual descent, but along the 

107 x-direction. xdescent values are currently ignored. 

108 

109 *hd_list* : list of (width, xdescent) of boxes to be aligned. 

110 *sep* : spacing between boxes 

111 *height* : Intended total length. None if not used. 

112 *align* : align mode. 'baseline', 'top', 'bottom', or 'center'. 

113 """ 

114 

115 if height is None: 

116 height = max(h for h, d in hd_list) 

117 

118 if align == "baseline": 

119 height_descent = max(h - d for h, d in hd_list) 

120 descent = max(d for h, d in hd_list) 

121 height = height_descent + descent 

122 offsets = [0. for h, d in hd_list] 

123 elif align in ["left", "top"]: 

124 descent = 0. 

125 offsets = [d for h, d in hd_list] 

126 elif align in ["right", "bottom"]: 

127 descent = 0. 

128 offsets = [height - h + d for h, d in hd_list] 

129 elif align == "center": 

130 descent = 0. 

131 offsets = [(height - h) * .5 + d for h, d in hd_list] 

132 else: 

133 raise ValueError("Unknown Align mode : %s" % (align,)) 

134 

135 return height, descent, offsets

380 def _parse_thead_tbody_tfoot(self, table_html): 

381 """ 

382 Given a table, return parsed header, body, and foot. 

383 

384 Parameters 

385 ---------- 

386 table_html : node-like 

387 

388 Returns 

389 ------- 

390 tuple of (header, body, footer), each a list of list-of-text rows. 

391 

392 Notes 

393 ----- 

394 Header and body are lists-of-lists. Top level list is a list of 

395 rows. Each row is a list of str text. 

396 

397 Logic: Use <thead>, <tbody>, <tfoot> elements to identify 

398 header, body, and footer, otherwise: 

399 - Put all rows into body 

400 - Move rows from top of body to header only if 

401 all elements inside row are <th> 

402 - Move rows from bottom of body to footer only if 

403 all elements inside row are <th> 

404 """ 

405 

406 header_rows = self._parse_thead_tr(table_html) 

407 body_rows = self._parse_tbody_tr(table_html) 

408 footer_rows = self._parse_tfoot_tr(table_html) 

409 

410 def row_is_all_th(row): 

411 return all(self._equals_tag(t, 'th') for t in 

412 self._parse_td(row)) 

413 

414 if not header_rows: 

415 # The table has no <thead>. Move the top all-<th> rows from 

416 # body_rows to header_rows. (This is a common case because many 

417 # tables in the wild have no <thead> or <tfoot> 

418 while body_rows and row_is_all_th(body_rows[0]): 

419 header_rows.append(body_rows.pop(0)) 

420 

421 header = self._expand_colspan_rowspan(header_rows) 

422 body = self._expand_colspan_rowspan(body_rows) 

423 footer = self._expand_colspan_rowspan(footer_rows) 

424 

425 return header, body, footer

427 def _expand_colspan_rowspan(self, rows): 

428 """ 

429 Given a list of <tr>s, return a list of text rows. 

430 

431 Parameters 

432 ---------- 

433 rows : list of node-like 

434 List of <tr>s 

435 

436 Returns 

437 ------- 

438 list of list 

439 Each returned row is a list of str text. 

440 

441 Notes 

442 ----- 

443 Any cell with ``rowspan`` or ``colspan`` will have its contents copied 

444 to subsequent cells. 

445 """ 

446 

447 all_texts = [] # list of rows, each a list of str 

448 remainder = [] # list of (index, text, nrows) 

449 

450 for tr in rows: 

451 texts = [] # the output for this row 

452 next_remainder = [] 

453 

454 index = 0 

455 tds = self._parse_td(tr) 

456 for td in tds: 

457 # Append texts from previous rows with rowspan>1 that come 

458 # before this <td> 

459 while remainder and remainder[0][0] <= index: 

460 prev_i, prev_text, prev_rowspan = remainder.pop(0) 

461 texts.append(prev_text) 

462 if prev_rowspan > 1: 

463 next_remainder.append((prev_i, prev_text, 

464 prev_rowspan - 1)) 

465 index += 1 

466 

467 # Append the text from this <td>, colspan times 

468 text = _remove_whitespace(self._text_getter(td)) 

469 rowspan = int(self._attr_getter(td, 'rowspan') or 1) 

470 colspan = int(self._attr_getter(td, 'colspan') or 1) 

471 

472 for _ in range(colspan): 

473 texts.append(text) 

474 if rowspan > 1: 

475 next_remainder.append((index, text, rowspan - 1)) 

476 index += 1 

477 

478 # Append texts from previous rows at the final position 

479 for prev_i, prev_text, prev_rowspan in remainder: 

480 texts.append(prev_text) 

481 if prev_rowspan > 1: 

482 next_remainder.append((prev_i, prev_text, 

483 prev_rowspan - 1)) 

484 

485 all_texts.append(texts) 

486 remainder = next_remainder 

487 

488 # Append rows that only appear because the previous row had non-1 

489 # rowspan 

490 while remainder: 

491 next_remainder = [] 

492 texts = [] 

493 for prev_i, prev_text, prev_rowspan in remainder: 

494 texts.append(prev_text) 

495 if prev_rowspan > 1: 

496 next_remainder.append((prev_i, prev_text, 

497 prev_rowspan - 1)) 

498 all_texts.append(texts) 

499 remainder = next_remainder 

500 

501 return all_texts

657 def _parse_td(self, row): 

658 # Look for direct children only: the "row" element here may be a

662 def _parse_tables(self, doc, match, kwargs): 

663 pattern = match.pattern 

664 

665 # 1. check all descendants for the given pattern and only search tables 

666 # 2. go up the tree until we find a table 

667 query = '//table//*[re:test(text(), {patt!r})]/ancestor::table' 

668 xpath_expr = u(query).format(patt=pattern) 

669 

670 # if any table attributes were given build an xpath expression to 

671 # search for them 

672 if kwargs: 

673 xpath_expr += _build_xpath_expr(kwargs) 

674 

675 tables = doc.xpath(xpath_expr, namespaces=_re_namespace) 

676 

677 tables = self._handle_hidden_tables(tables, "attrib") 

678 if self.displayed_only: 

679 for table in tables: 

680 # lxml utilizes XPATH 1.0 which does not have regex 

681 # support. As a result, we find all elements with a style 

682 # attribute and iterate them to check for display:none 

683 for elem in table.xpath('.//*[@style]'): 

684 if "display:none" in elem.attrib.get( 

685 "style", "").replace(" ", ""): 

686 elem.getparent().remove(elem) 

687 

688 if not tables: 

689 raise ValueError("No tables found matching regex {patt!r}" 

690 .format(patt=pattern)) 

691 return tables

860def _validate_flavor(flavor): 

861 if flavor is None: 

862 flavor = 'lxml', 'bs4' 

863 elif isinstance(flavor, string_types): 

864 flavor = flavor, 

865 elif isinstance(flavor, compat.Iterable): 

866 if not all(isinstance(flav, string_types) for flav in flavor): 

867 raise TypeError('Object of type {typ!r} is not an iterable of ' 

868 'strings' 

869 .format(typ=type(flavor).__name__)) 

870 else: 

871 fmt = '{flavor!r}' if isinstance(flavor, string_types) else '{flavor}' 

872 fmt += ' is not a valid flavor' 

873 raise ValueError(fmt.format(flavor=flavor)) 

874 

875 flavor = tuple(flavor) 

876 valid_flavors = set(_valid_parsers) 

877 flavor_set = set(flavor) 

878 

879 if not flavor_set & valid_flavors: 

880 raise ValueError('{invalid} is not a valid set of flavors, valid ' 

881 'flavors are {valid}' 

882 .format(invalid=_print_as_set(flavor_set), 

883 valid=_print_as_set(valid_flavors))) 

884 return flavor

887def _parse(flavor, io, match, attrs, encoding, displayed_only, **kwargs): 

888 flavor = _validate_flavor(flavor) 

889 compiled_match = re.compile(match) # you can pass a compiled regex here 

890 

891 # hack around python 3 deleting the exception variable 

892 retained = None 

893 for flav in flavor: 

894 parser = _parser_dispatch(flav) 

895 p = parser(io, compiled_match, attrs, encoding, displayed_only) 

896 

897 try: 

898 tables = p.parse_tables() 

899 except Exception as caught: 

900 # if `io` is an io-like object, check if it's seekable 

901 # and try to rewind it before trying the next parser 

902 if hasattr(io, 'seekable') and io.seekable(): 

903 io.seek(0) 

904 elif hasattr(io, 'seekable') and not io.seekable(): 

905 # if we couldn't rewind it, let the user know 

906 raise ValueError('The flavor {} failed to parse your input. ' 

907 'Since you passed a non-rewindable file ' 

908 'object, we can\'t rewind it to try ' 

909 'another parser. Try read_html() with a ' 

910 'different flavor.'.format(flav)) 

911 

912 retained = caught 

913 else: 

914 break 

915 else: 

916 raise_with_traceback(retained) 

917 

918 ret = [] 

919 for table in tables: 

920 try: 

921 ret.append(_data_to_frame(data=table, **kwargs)) 

922 except EmptyDataError: # empty table 

923 continue 

924 return ret

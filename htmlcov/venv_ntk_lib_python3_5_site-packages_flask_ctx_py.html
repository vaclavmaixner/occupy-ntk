98def after_this_request(f): 

99 """Executes a function after this request. This is useful to modify 

100 response objects. The function is passed the response object and has 

101 to return the same or a new one. 

102 

103 Example:: 

104 

105 @app.route('/') 

106 def index(): 

107 @after_this_request 

108 def add_header(response): 

109 response.headers['X-Foo'] = 'Parachute' 

110 return response 

111 return 'Hello World!' 

112 

113 This is more useful if a function other than the view function wants to 

114 modify a response. For instance think of a decorator that wants to add 

115 some headers without converting the return value into a response object. 

116 

117 .. versionadded:: 0.9 

118 """ 

119 _request_ctx_stack.top._after_request_functions.append(f) 

120 return f

123def copy_current_request_context(f): 

124 """A helper function that decorates a function to retain the current 

125 request context. This is useful when working with greenlets. The moment 

126 the function is decorated a copy of the request context is created and 

127 then pushed when the function is called. The current session is also 

128 included in the copied request context. 

129 

130 Example:: 

131 

132 import gevent 

133 from flask import copy_current_request_context 

134 

135 @app.route('/') 

136 def index(): 

137 @copy_current_request_context 

138 def do_some_work(): 

139 # do some work here, it can access flask.request or 

140 # flask.session like you would otherwise in the view function. 

141 ... 

142 gevent.spawn(do_some_work) 

143 return 'Regular response' 

144 

145 .. versionadded:: 0.10 

146 """ 

147 top = _request_ctx_stack.top 

148 if top is None: 

149 raise RuntimeError( 

150 "This decorator can only be used at local scopes " 

151 "when a request context is on the stack. For instance within " 

152 "view functions." 

153 ) 

154 reqctx = top.copy() 

155 

156 def wrapper(*args, **kwargs): 

157 with reqctx: 

158 return f(*args, **kwargs) 

159 

160 return update_wrapper(wrapper, f)

163def has_request_context(): 

164 """If you have code that wants to test if a request context is there or 

165 not this function can be used. For instance, you may want to take advantage 

166 of request information if the request object is available, but fail 

167 silently if it is unavailable. 

168 

169 :: 

170 

171 class User(db.Model): 

172 

173 def __init__(self, username, remote_addr=None): 

174 self.username = username 

175 if remote_addr is None and has_request_context(): 

176 remote_addr = request.remote_addr 

177 self.remote_addr = remote_addr 

178 

179 Alternatively you can also just test any of the context bound objects 

180 (such as :class:`request` or :class:`g`) for truthness:: 

181 

182 class User(db.Model): 

183 

184 def __init__(self, username, remote_addr=None): 

185 self.username = username 

186 if remote_addr is None and request: 

187 remote_addr = request.remote_addr 

188 self.remote_addr = remote_addr 

189 

190 .. versionadded:: 0.7 

191 """ 

192 return _request_ctx_stack.top is not None

205class AppContext(object): 

206 """The application context binds an application object implicitly 

207 to the current thread or greenlet, similar to how the 

208 :class:`RequestContext` binds request information. The application 

209 context is also implicitly created if a request context is created 

210 but the application is not on top of the individual application 

211 context. 

212 """ 

213 

214 def __init__(self, app): 

215 self.app = app 

216 self.url_adapter = app.create_url_adapter(None) 

217 self.g = app.app_ctx_globals_class() 

218 

219 # Like request context, app contexts can be pushed multiple times 

220 # but there a basic "refcount" is enough to track them. 

221 self._refcnt = 0 

222 

223 def push(self): 

224 """Binds the app context to the current context.""" 

225 self._refcnt += 1 

226 if hasattr(sys, "exc_clear"): 

227 sys.exc_clear() 

228 _app_ctx_stack.push(self) 

229 appcontext_pushed.send(self.app) 

230 

231 def pop(self, exc=_sentinel): 

232 """Pops the app context.""" 

233 try: 

234 self._refcnt -= 1 

235 if self._refcnt <= 0: 

236 if exc is _sentinel: 

237 exc = sys.exc_info()[1] 

238 self.app.do_teardown_appcontext(exc) 

239 finally: 

240 rv = _app_ctx_stack.pop() 

241 assert rv is self, "Popped wrong app context. (%r instead of %r)" % (rv, self) 

242 appcontext_popped.send(self.app) 

243 

244 def __enter__(self): 

245 self.push() 

246 return self 

247 

248 def __exit__(self, exc_type, exc_value, tb): 

249 self.pop(exc_value) 

250 

251 if BROKEN_PYPY_CTXMGR_EXIT and exc_type is not None: 

252 reraise(exc_type, exc_value, tb)

255class RequestContext(object): 

256 """The request context contains all request relevant information. It is 

257 created at the beginning of the request and pushed to the 

258 `_request_ctx_stack` and removed at the end of it. It will create the 

259 URL adapter and request object for the WSGI environment provided. 

260 

261 Do not attempt to use this class directly, instead use 

262 :meth:`~flask.Flask.test_request_context` and 

263 :meth:`~flask.Flask.request_context` to create this object. 

264 

265 When the request context is popped, it will evaluate all the 

266 functions registered on the application for teardown execution 

267 (:meth:`~flask.Flask.teardown_request`). 

268 

269 The request context is automatically popped at the end of the request 

270 for you. In debug mode the request context is kept around if 

271 exceptions happen so that interactive debuggers have a chance to 

272 introspect the data. With 0.4 this can also be forced for requests 

273 that did not fail and outside of ``DEBUG`` mode. By setting 

274 ``'flask._preserve_context'`` to ``True`` on the WSGI environment the 

275 context will not pop itself at the end of the request. This is used by 

276 the :meth:`~flask.Flask.test_client` for example to implement the 

277 deferred cleanup functionality. 

278 

279 You might find this helpful for unittests where you need the 

280 information from the context local around for a little longer. Make 

281 sure to properly :meth:`~werkzeug.LocalStack.pop` the stack yourself in 

282 that situation, otherwise your unittests will leak memory. 

283 """

286 self.app = app 

287 if request is None: 

288 request = app.request_class(environ) 

289 self.request = request 

290 self.url_adapter = None

293 except HTTPException as e:

355 def push(self): 

356 """Binds the request context to the current context.""" 

357 # If an exception occurs in debug mode or if context preservation is 

358 # activated under exception situations exactly one context stays 

359 # on the stack. The rationale is that you want to access that 

360 # information under debug situations. However if someone forgets to 

361 # pop that context again we want to make sure that on the next push 

362 # it's invalidated, otherwise we run at risk that something leaks 

363 # memory. This is usually only a problem in test suite since this 

364 # functionality is not active in production environments. 

365 top = _request_ctx_stack.top 

366 if top is not None and top.preserved: 

367 top.pop(top._preserved_exc) 

368 

369 # Before we push the request context we have to ensure that there 

370 # is an application context. 

371 app_ctx = _app_ctx_stack.top 

372 if app_ctx is None or app_ctx.app != self.app: 

373 app_ctx = self.app.app_context() 

374 app_ctx.push() 

375 self._implicit_app_ctx_stack.append(app_ctx) 

376 else: 

377 self._implicit_app_ctx_stack.append(None) 

378 

379 if hasattr(sys, "exc_clear"): 

380 sys.exc_clear() 

381 

382 _request_ctx_stack.push(self) 

383 

384 # Open the session at the moment that the request context is available. 

385 # This allows a custom open_session method to use the request context. 

386 # Only open a new session if this is the first time the request was 

387 # pushed, otherwise stream_with_context loses the session. 

388 if self.session is None: 

389 session_interface = self.app.session_interface 

390 self.session = session_interface.open_session(self.app, self.request) 

391 

392 if self.session is None: 

393 self.session = session_interface.make_null_session(self.app) 

394 

395 if self.url_adapter is not None: 

396 self.match_request()

398 def pop(self, exc=_sentinel): 

399 """Pops the request context and unbinds it by doing that. This will 

400 also trigger the execution of functions registered by the 

401 :meth:`~flask.Flask.teardown_request` decorator. 

402 

403 .. versionchanged:: 0.9 

404 Added the `exc` argument. 

405 """ 

406 app_ctx = self._implicit_app_ctx_stack.pop() 

407 

408 try: 

409 clear_request = False 

410 if not self._implicit_app_ctx_stack: 

411 self.preserved = False 

412 self._preserved_exc = None 

413 if exc is _sentinel: 

414 exc = sys.exc_info()[1] 

415 self.app.do_teardown_request(exc) 

416 

417 # If this interpreter supports clearing the exception information 

418 # we do that now. This will only go into effect on Python 2.x, 

419 # on 3.x it disappears automatically at the end of the exception 

420 # stack. 

421 if hasattr(sys, "exc_clear"): 

422 sys.exc_clear() 

423 

424 request_close = getattr(self.request, "close", None) 

425 if request_close is not None: 

426 request_close() 

427 clear_request = True 

428 finally: 

429 rv = _request_ctx_stack.pop() 

430 

431 # get rid of circular dependencies at the end of the request 

432 # so that we don't require the GC to be active. 

433 if clear_request: 

434 rv.request.environ["werkzeug.request"] = None 

435 

436 # Get rid of the app as well if necessary. 

437 if app_ctx is not None: 

438 app_ctx.pop(exc) 

439 

440 assert rv is self, "Popped wrong request context. (%r instead of %r)" % ( 

441 rv, 

442 self, 

443 )

445 def auto_pop(self, exc): 

446 if self.request.environ.get("flask._preserve_context") or ( 

447 exc is not None and self.app.preserve_context_on_exception 

448 ): 

449 self.preserved = True 

450 self._preserved_exc = exc 

451 else: 

452 self.pop(exc)

145 '''

1074class comment(html_tag): 

1075 ''' 

1076 Normal, one-line comment: 

1077 >>> print comment("Hello, comments!") 

1078 <!--Hello, comments!--> 

1079 

1080 For IE's "if" statement comments: 

1081 >>> print comment(p("Upgrade your browser."), condition='lt IE6') 

1082 <!--[if lt IE6]><p>Upgrade your browser.</p><![endif]--> 

1083 

1084 Downlevel conditional comments: 

1085 >>> print comment(p("You are using a ", em("downlevel"), " browser."), 

1086 condition='false', downlevel='revealed') 

1087 <![if false]><p>You are using a <em>downlevel</em> browser.</p><![endif]> 

1088 

1089 For more on conditional comments see: 

1090 http://msdn.microsoft.com/en-us/library/ms537512(VS.85).aspx 

1091 ''' 

1092 

1093 ATTRIBUTE_CONDITION = 'condition' 

1094 

1095 # Valid values are 'hidden', 'downlevel' or 'revealed' 

1096 ATTRIBUTE_DOWNLEVEL = 'downlevel' 

1097 

1098 def _render(self, sb, indent_level=1, indent_str=' ', pretty=True, xhtml=False): 

1099 has_condition = comment.ATTRIBUTE_CONDITION in self.attributes 

1100 is_revealed = comment.ATTRIBUTE_DOWNLEVEL in self.attributes and \ 

1101 self.attributes[comment.ATTRIBUTE_DOWNLEVEL] == 'revealed' 

1102 

1103 sb.append('<!') 

1104 if not is_revealed: 

1105 sb.append('--') 

1106 if has_condition: 

1107 sb.append('[if %s]>' % self.attributes[comment.ATTRIBUTE_CONDITION]) 

1108 

1109 pretty = self._render_children(sb, indent_level - 1, indent_str, pretty, xhtml) 

1110 

1111 # if len(self.children) > 1: 

1112 if any(isinstance(child, dom_tag) for child in self): 

1113 sb.append('\n') 

1114 sb.append(indent_str * (indent_level - 1)) 

1115 

1116 if has_condition: 

1117 sb.append('<![endif]') 

1118 if not is_revealed: 

1119 sb.append('--') 

1120 sb.append('>') 

1121 

1122 return sb

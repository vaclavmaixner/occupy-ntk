26def read_gbq(query, project_id=None, index_col=None, col_order=None, 

27 reauth=False, auth_local_webserver=False, dialect=None, 

28 location=None, configuration=None, credentials=None, 

29 private_key=None, verbose=None): 

30 """ 

31 Load data from Google BigQuery. 

32 

33 This function requires the `pandas-gbq package 

34 <https://pandas-gbq.readthedocs.io>`__. 

35 

36 See the `How to authenticate with Google BigQuery 

37 <https://pandas-gbq.readthedocs.io/en/latest/howto/authentication.html>`__ 

38 guide for authentication instructions. 

39 

40 Parameters 

41 ---------- 

42 query : str 

43 SQL-Like Query to return data values. 

44 project_id : str, optional 

45 Google BigQuery Account project ID. Optional when available from 

46 the environment. 

47 index_col : str, optional 

48 Name of result column to use for index in results DataFrame. 

49 col_order : list(str), optional 

50 List of BigQuery column names in the desired order for results 

51 DataFrame. 

52 reauth : boolean, default False 

53 Force Google BigQuery to re-authenticate the user. This is useful 

54 if multiple accounts are used. 

55 auth_local_webserver : boolean, default False 

56 Use the `local webserver flow`_ instead of the `console flow`_ 

57 when getting user credentials. 

58 

59 .. _local webserver flow: 

60 http://google-auth-oauthlib.readthedocs.io/en/latest/reference/google_auth_oauthlib.flow.html#google_auth_oauthlib.flow.InstalledAppFlow.run_local_server 

61 .. _console flow: 

62 http://google-auth-oauthlib.readthedocs.io/en/latest/reference/google_auth_oauthlib.flow.html#google_auth_oauthlib.flow.InstalledAppFlow.run_console 

63 

64 *New in version 0.2.0 of pandas-gbq*. 

65 dialect : str, default 'legacy' 

66 Note: The default value is changing to 'standard' in a future verion. 

67 

68 SQL syntax dialect to use. Value can be one of: 

69 

70 ``'legacy'`` 

71 Use BigQuery's legacy SQL dialect. For more information see 

72 `BigQuery Legacy SQL Reference 

73 <https://cloud.google.com/bigquery/docs/reference/legacy-sql>`__. 

74 ``'standard'`` 

75 Use BigQuery's standard SQL, which is 

76 compliant with the SQL 2011 standard. For more information 

77 see `BigQuery Standard SQL Reference 

78 <https://cloud.google.com/bigquery/docs/reference/standard-sql/>`__. 

79 

80 .. versionchanged:: 0.24.0 

81 location : str, optional 

82 Location where the query job should run. See the `BigQuery locations 

83 documentation 

84 <https://cloud.google.com/bigquery/docs/dataset-locations>`__ for a 

85 list of available locations. The location must match that of any 

86 datasets used in the query. 

87 

88 *New in version 0.5.0 of pandas-gbq*. 

89 configuration : dict, optional 

90 Query config parameters for job processing. 

91 For example: 

92 

93 configuration = {'query': {'useQueryCache': False}} 

94 

95 For more information see `BigQuery REST API Reference 

96 <https://cloud.google.com/bigquery/docs/reference/rest/v2/jobs#configuration.query>`__. 

97 credentials : google.auth.credentials.Credentials, optional 

98 Credentials for accessing Google APIs. Use this parameter to override 

99 default credentials, such as to use Compute Engine 

100 :class:`google.auth.compute_engine.Credentials` or Service Account 

101 :class:`google.oauth2.service_account.Credentials` directly. 

102 

103 *New in version 0.8.0 of pandas-gbq*. 

104 

105 .. versionadded:: 0.24.0 

106 private_key : str, deprecated 

107 Deprecated in pandas-gbq version 0.8.0. Use the ``credentials`` 

108 parameter and 

109 :func:`google.oauth2.service_account.Credentials.from_service_account_info` 

110 or 

111 :func:`google.oauth2.service_account.Credentials.from_service_account_file` 

112 instead. 

113 

114 Service account private key in JSON format. Can be file path 

115 or string contents. This is useful for remote server 

116 authentication (eg. Jupyter/IPython notebook on remote host). 

117 verbose : None, deprecated 

118 Deprecated in pandas-gbq version 0.4.0. Use the `logging module to 

119 adjust verbosity instead 

120 <https://pandas-gbq.readthedocs.io/en/latest/intro.html#logging>`__. 

121 

122 Returns 

123 ------- 

124 df: DataFrame 

125 DataFrame representing results of query. 

126 

127 See Also 

128 -------- 

129 pandas_gbq.read_gbq : This function in the pandas-gbq library. 

130 pandas.DataFrame.to_gbq : Write a DataFrame to Google BigQuery. 

131 """ 

132 pandas_gbq = _try_import() 

133 

134 if dialect is None: 

135 dialect = "legacy" 

136 warnings.warn( 

137 'The default value for dialect is changing to "standard" in a ' 

138 'future version of pandas-gbq. Pass in dialect="legacy" to ' 

139 "disable this warning.", 

140 FutureWarning, 

141 stacklevel=2, 

142 ) 

143 

144 return pandas_gbq.read_gbq( 

145 query, project_id=project_id, index_col=index_col, 

146 col_order=col_order, reauth=reauth, 

147 auth_local_webserver=auth_local_webserver, dialect=dialect, 

148 location=location, configuration=configuration, 

149 credentials=credentials, verbose=verbose, private_key=private_key)

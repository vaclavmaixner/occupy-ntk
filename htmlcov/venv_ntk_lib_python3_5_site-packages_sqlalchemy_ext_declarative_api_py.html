150class declared_attr(interfaces._MappedAttribute, property): 

151 """Mark a class-level method as representing the definition of 

152 a mapped property or special declarative member name. 

153 

154 @declared_attr turns the attribute into a scalar-like 

155 property that can be invoked from the uninstantiated class. 

156 Declarative treats attributes specifically marked with 

157 @declared_attr as returning a construct that is specific 

158 to mapping or declarative table configuration. The name 

159 of the attribute is that of what the non-dynamic version 

160 of the attribute would be. 

161 

162 @declared_attr is more often than not applicable to mixins, 

163 to define relationships that are to be applied to different 

164 implementors of the class:: 

165 

166 class ProvidesUser(object): 

167 "A mixin that adds a 'user' relationship to classes." 

168 

169 @declared_attr 

170 def user(self): 

171 return relationship("User") 

172 

173 It also can be applied to mapped classes, such as to provide 

174 a "polymorphic" scheme for inheritance:: 

175 

176 class Employee(Base): 

177 id = Column(Integer, primary_key=True) 

178 type = Column(String(50), nullable=False) 

179 

180 @declared_attr 

181 def __tablename__(cls): 

182 return cls.__name__.lower() 

183 

184 @declared_attr 

185 def __mapper_args__(cls): 

186 if cls.__name__ == 'Employee': 

187 return { 

188 "polymorphic_on":cls.type, 

189 "polymorphic_identity":"Employee" 

190 } 

191 else: 

192 return {"polymorphic_identity":cls.__name__} 

193 

194 """ 

195 

196 def __init__(self, fget, cascading=False): 

197 super(declared_attr, self).__init__(fget) 

198 self.__doc__ = fget.__doc__ 

199 self._cascading = cascading 

200 

201 def __get__(desc, self, cls): 

202 reg = cls.__dict__.get("_sa_declared_attr_reg", None) 

203 if reg is None: 

204 if ( 

205 not re.match(r"^__.+__$", desc.fget.__name__) 

206 and attributes.manager_of_class(cls) is None 

207 ): 

208 util.warn( 

209 "Unmanaged access of declarative attribute %s from " 

210 "non-mapped class %s" % (desc.fget.__name__, cls.__name__) 

211 ) 

212 return desc.fget(cls) 

213 elif desc in reg: 

214 return reg[desc] 

215 else: 

216 reg[desc] = obj = desc.fget(cls) 

217 return obj 

218 

219 @hybridmethod 

220 def _stateful(cls, **kw): 

221 return _stateful_declared_attr(**kw)

223 @hybridproperty 

224 def cascading(cls): 

225 """Mark a :class:`.declared_attr` as cascading. 

226 

227 This is a special-use modifier which indicates that a column 

228 or MapperProperty-based declared attribute should be configured 

229 distinctly per mapped subclass, within a mapped-inheritance scenario. 

230 

231 .. warning:: 

232 

233 The :attr:`.declared_attr.cascading` modifier has several 

234 limitations: 

235 

236 * The flag **only** applies to the use of :class:`.declared_attr` 

237 on declarative mixin classes and ``__abstract__`` classes; it 

238 currently has no effect when used on a mapped class directly. 

239 

240 * The flag **only** applies to normally-named attributes, e.g. 

241 not any special underscore attributes such as ``__tablename__``. 

242 On these attributes it has **no** effect. 

243 

244 * The flag currently **does not allow further overrides** down 

245 the class hierarchy; if a subclass tries to override the 

246 attribute, a warning is emitted and the overridden attribute 

247 is skipped. This is a limitation that it is hoped will be 

248 resolved at some point. 

249 

250 Below, both MyClass as well as MySubClass will have a distinct 

251 ``id`` Column object established:: 

252 

253 class HasIdMixin(object): 

254 @declared_attr.cascading 

255 def id(cls): 

256 if has_inherited_table(cls): 

257 return Column( 

258 ForeignKey('myclass.id'), primary_key=True) 

259 else: 

260 return Column(Integer, primary_key=True) 

261 

262 class MyClass(HasIdMixin, Base): 

263 __tablename__ = 'myclass' 

264 # ... 

265 

266 class MySubClass(MyClass): 

267 "" 

268 # ... 

269 

270 The behavior of the above configuration is that ``MySubClass`` 

271 will refer to both its own ``id`` column as well as that of 

272 ``MyClass`` underneath the attribute named ``some_id``. 

273 

274 .. seealso:: 

275 

276 :ref:`declarative_inheritance` 

277 

278 :ref:`mixin_inheritance_columns` 

279 

280 

281 """ 

282 return cls._stateful(cascading=True)

298def declarative_base( 

299 bind=None, 

300 metadata=None, 

301 mapper=None, 

302 cls=object, 

303 name="Base", 

304 constructor=_declarative_constructor, 

305 class_registry=None, 

306 metaclass=DeclarativeMeta, 

307): 

308 r"""Construct a base class for declarative class definitions. 

309 

310 The new base class will be given a metaclass that produces 

311 appropriate :class:`~sqlalchemy.schema.Table` objects and makes 

312 the appropriate :func:`~sqlalchemy.orm.mapper` calls based on the 

313 information provided declaratively in the class and any subclasses 

314 of the class. 

315 

316 :param bind: An optional 

317 :class:`~sqlalchemy.engine.Connectable`, will be assigned 

318 the ``bind`` attribute on the :class:`~sqlalchemy.schema.MetaData` 

319 instance. 

320 

321 :param metadata: 

322 An optional :class:`~sqlalchemy.schema.MetaData` instance. All 

323 :class:`~sqlalchemy.schema.Table` objects implicitly declared by 

324 subclasses of the base will share this MetaData. A MetaData instance 

325 will be created if none is provided. The 

326 :class:`~sqlalchemy.schema.MetaData` instance will be available via the 

327 `metadata` attribute of the generated declarative base class. 

328 

329 :param mapper: 

330 An optional callable, defaults to :func:`~sqlalchemy.orm.mapper`. Will 

331 be used to map subclasses to their Tables. 

332 

333 :param cls: 

334 Defaults to :class:`object`. A type to use as the base for the generated 

335 declarative base class. May be a class or tuple of classes. 

336 

337 :param name: 

338 Defaults to ``Base``. The display name for the generated 

339 class. Customizing this is not required, but can improve clarity in 

340 tracebacks and debugging. 

341 

342 :param constructor: 

343 Defaults to 

344 :func:`~sqlalchemy.ext.declarative.base._declarative_constructor`, an 

345 __init__ implementation that assigns \**kwargs for declared 

346 fields and relationships to an instance. If ``None`` is supplied, 

347 no __init__ will be provided and construction will fall back to 

348 cls.__init__ by way of the normal Python semantics. 

349 

350 :param class_registry: optional dictionary that will serve as the 

351 registry of class names-> mapped classes when string names 

352 are used to identify classes inside of :func:`.relationship` 

353 and others. Allows two or more declarative base classes 

354 to share the same registry of class names for simplified 

355 inter-base relationships. 

356 

357 :param metaclass: 

358 Defaults to :class:`.DeclarativeMeta`. A metaclass or __metaclass__ 

359 compatible callable to use as the meta type of the generated 

360 declarative base class. 

361 

362 .. versionchanged:: 1.1 if :paramref:`.declarative_base.cls` is a 

363 single class (rather than a tuple), the constructed base class will 

364 inherit its docstring. 

365 

366 .. seealso:: 

367 

368 :func:`.as_declarative` 

369 

370 """ 

371 lcl_metadata = metadata or MetaData() 

372 if bind: 

373 lcl_metadata.bind = bind 

374 

375 if class_registry is None: 

376 class_registry = weakref.WeakValueDictionary() 

377 

378 bases = not isinstance(cls, tuple) and (cls,) or cls 

379 class_dict = dict( 

380 _decl_class_registry=class_registry, metadata=lcl_metadata 

381 ) 

382 

383 if isinstance(cls, type): 

384 class_dict["__doc__"] = cls.__doc__ 

385 

386 if constructor: 

387 class_dict["__init__"] = constructor 

388 if mapper: 

389 class_dict["__mapper_cls__"] = mapper 

390 

391 return metaclass(name, bases, class_dict)

431class ConcreteBase(object): 

432 """A helper class for 'concrete' declarative mappings. 

433 

434 :class:`.ConcreteBase` will use the :func:`.polymorphic_union` 

435 function automatically, against all tables mapped as a subclass 

436 to this class. The function is called via the 

437 ``__declare_last__()`` function, which is essentially

443

445 

446 from sqlalchemy.ext.declarative import ConcreteBase

448 class Employee(ConcreteBase, Base): 

449 __tablename__ = 'employee' 

450 employee_id = Column(Integer, primary_key=True) 

451 name = Column(String(50)) 

452 __mapper_args__ = { 

453 'polymorphic_identity':'employee', 

454 'concrete':True} 

455 

456 class Manager(Employee): 

457 __tablename__ = 'manager' 

458 employee_id = Column(Integer, primary_key=True) 

459 name = Column(String(50)) 

460 manager_data = Column(String(40)) 

461 __mapper_args__ = { 

462 'polymorphic_identity':'manager', 

463 'concrete':True} 

464 

465 .. seealso:: 

466 

467 :class:`.AbstractConcreteBase` 

468 

469 :ref:`concrete_inheritance` 

470 

471 :ref:`inheritance_concrete_helpers` 

472 

473 

474 """ 

475 

476 @classmethod 

477 def _create_polymorphic_union(cls, mappers): 

478 return polymorphic_union( 

479 OrderedDict( 

480 (mp.polymorphic_identity, mp.local_table) for mp in mappers 

481 ), 

482 "type", 

483 "pjoin", 

484 ) 

485 

486 @classmethod 

487 def __declare_first__(cls): 

488 m = cls.__mapper__ 

489 if m.with_polymorphic: 

490 return

498class AbstractConcreteBase(ConcreteBase): 

499 """A helper class for 'concrete' declarative mappings. 

500 

501 :class:`.AbstractConcreteBase` will use the :func:`.polymorphic_union` 

502 function automatically, against all tables mapped as a subclass 

503 to this class. The function is called via the 

504 ``__declare_last__()`` function, which is essentially 

505 a hook for the :meth:`.after_configured` event. 

506 

507 :class:`.AbstractConcreteBase` does produce a mapped class 

508 for the base class, however it is not persisted to any table; it 

509 is instead mapped directly to the "polymorphic" selectable directly 

510 and is only used for selecting. Compare to :class:`.ConcreteBase`, 

511 which does create a persisted table for the base class. 

512 

513 .. note:: 

514 

515 The :class:`.AbstractConcreteBase` class does not intend to set up the 

516 mapping for the base class until all the subclasses have been defined, 

517 as it needs to create a mapping against a selectable that will include 

518 all subclass tables. In order to achieve this, it waits for the 

519 **mapper configuration event** to occur, at which point it scans 

520 through all the configured subclasses and sets up a mapping that will 

521 query against all subclasses at once. 

522 

523 While this event is normally invoked automatically, in the case of 

524 :class:`.AbstractConcreteBase`, it may be necessary to invoke it 

525 explicitly after **all** subclass mappings are defined, if the first 

526 operation is to be a query against this base class. To do so, invoke 

527 :func:`.configure_mappers` once all the desired classes have been 

528 configured:: 

529 

530 from sqlalchemy.orm import configure_mappers 

531 

532 configure_mappers() 

533 

534 .. seealso:: 

535 

536 :func:`.orm.configure_mappers` 

537 

538 

539 Example:: 

540 

541 from sqlalchemy.ext.declarative import AbstractConcreteBase 

542 

543 class Employee(AbstractConcreteBase, Base): 

544 pass 

545 

546 class Manager(Employee): 

547 __tablename__ = 'manager' 

548 employee_id = Column(Integer, primary_key=True) 

549 name = Column(String(50)) 

550 manager_data = Column(String(40)) 

551 

552 __mapper_args__ = { 

553 'polymorphic_identity':'manager', 

554 'concrete':True} 

555 

556 configure_mappers() 

557 

558 The abstract base class is handled by declarative in a special way; 

559 at class configuration time, it behaves like a declarative mixin 

560 or an ``__abstract__`` base class. Once classes are configured 

561 and mappings are produced, it then gets mapped itself, but 

562 after all of its descendants. This is a very unique system of mapping 

563 not found in any other SQLAlchemy system. 

564 

565 Using this approach, we can specify columns and properties 

566 that will take place on mapped subclasses, in the way that 

567 we normally do as in :ref:`declarative_mixins`:: 

568 

569 class Company(Base): 

570 __tablename__ = 'company' 

571 id = Column(Integer, primary_key=True) 

572 

573 class Employee(AbstractConcreteBase, Base): 

574 employee_id = Column(Integer, primary_key=True) 

575 

576 @declared_attr 

577 def company_id(cls): 

578 return Column(ForeignKey('company.id')) 

579 

580 @declared_attr 

581 def company(cls): 

582 return relationship("Company") 

583 

584 class Manager(Employee): 

585 __tablename__ = 'manager' 

586 

587 name = Column(String(50)) 

588 manager_data = Column(String(40)) 

589 

590 __mapper_args__ = { 

591 'polymorphic_identity':'manager', 

592 'concrete':True} 

593 

594 configure_mappers() 

595 

596 When we make use of our mappings however, both ``Manager`` and 

597 ``Employee`` will have an independently usable ``.company`` attribute:: 

598 

599 session.query(Employee).filter(Employee.company.has(id=5)) 

600 

601 .. versionchanged:: 1.0.0 - The mechanics of :class:`.AbstractConcreteBase` 

602 have been reworked to support relationships established directly 

603 on the abstract base, without any special configurational steps. 

604 

605 .. seealso:: 

606 

607 :class:`.ConcreteBase` 

608 

609 :ref:`concrete_inheritance` 

610 

611 :ref:`inheritance_concrete_helpers` 

612 

613 """ 

614 

615 __no_table__ = True 

616 

617 @classmethod 

618 def __declare_first__(cls): 

619 cls._sa_decl_prepare_nocascade() 

620 

621 @classmethod 

622 def _sa_decl_prepare_nocascade(cls): 

623 if getattr(cls, "__mapper__", None): 

624 return 

625 

626 to_map = _DeferredMapperConfig.config_for_cls(cls) 

627 

628 # can't rely on 'self_and_descendants' here 

629 # since technically an immediate subclass 

630 # might not be mapped, but a subclass 

631 # may be. 

632 mappers = [] 

633 stack = list(cls.__subclasses__()) 

634 while stack: 

635 klass = stack.pop() 

636 stack.extend(klass.__subclasses__()) 

637 mn = _mapper_or_none(klass) 

638 if mn is not None: 

639 mappers.append(mn) 

640 pjoin = cls._create_polymorphic_union(mappers) 

641 

642 # For columns that were declared on the class, these 

643 # are normally ignored with the "__no_table__" mapping, 

644 # unless they have a different attribute key vs. col name 

645 # and are in the properties argument. 

646 # In that case, ensure we update the properties entry 

647 # to the correct column from the pjoin target table. 

648 declared_cols = set(to_map.declared_columns) 

649 for k, v in list(to_map.properties.items()): 

650 if v in declared_cols: 

651 to_map.properties[k] = pjoin.c[v.key] 

652 

653 to_map.local_table = pjoin 

654 

655 m_args = to_map.mapper_args_fn or dict 

656 

657 def mapper_args(): 

658 args = m_args() 

659 args["polymorphic_on"] = pjoin.c.type 

660 return args 

661 

662 to_map.mapper_args_fn = mapper_args 

663 

664 m = to_map.map() 

665 

666 for scls in cls.__subclasses__(): 

667 sm = _mapper_or_none(scls) 

668 if sm and sm.concrete and cls in scls.__bases__: 

669 sm._set_concrete_base(m)

684class DeferredReflection(object): 

685 """A helper class for construction of mappings based on 

686 a deferred reflection step. 

687 

688 Normally, declarative can be used with reflection by 

689 setting a :class:`.Table` object using autoload=True 

690 as the ``__table__`` attribute on a declarative class. 

691 The caveat is that the :class:`.Table` must be fully 

692 reflected, or at the very least have a primary key column, 

693 at the point at which a normal declarative mapping is 

694 constructed, meaning the :class:`.Engine` must be available 

695 at class declaration time. 

696 

697 The :class:`.DeferredReflection` mixin moves the construction 

698 of mappers to be at a later point, after a specific 

699 method is called which first reflects all :class:`.Table` 

700 objects created so far. Classes can define it as such:: 

701 

702 from sqlalchemy.ext.declarative import declarative_base 

703 from sqlalchemy.ext.declarative import DeferredReflection 

704 Base = declarative_base() 

705 

706 class MyClass(DeferredReflection, Base): 

707 __tablename__ = 'mytable' 

708 

709 Above, ``MyClass`` is not yet mapped. After a series of 

710 classes have been defined in the above fashion, all tables 

711 can be reflected and mappings created using 

712 :meth:`.prepare`:: 

713 

714 engine = create_engine("someengine://...") 

715 DeferredReflection.prepare(engine)

717 The :class:`.DeferredReflection` mixin can be applied to individual 

718 classes, used as the base for the declarative base itself, 

719 or used in a custom abstract class. Using an abstract base 

720 allows that only a subset of classes to be prepared for a 

721 particular prepare step, which is necessary for applications 

722 that use more than one engine. For example, if an application 

723 has two engines, you might use two bases, and prepare each 

724 separately, e.g.:: 

725 

726 class ReflectedOne(DeferredReflection, Base): 

727 __abstract__ = True 

728 

729 class ReflectedTwo(DeferredReflection, Base): 

730 __abstract__ = True 

731 

732 class MyClass(ReflectedOne): 

733 __tablename__ = 'mytable' 

734 

735 class MyOtherClass(ReflectedOne): 

736 __tablename__ = 'myothertable' 

737 

738 class YetAnotherClass(ReflectedTwo): 

739 __tablename__ = 'yetanothertable' 

740 

741 # ... etc. 

742 

743 Above, the class hierarchies for ``ReflectedOne`` and 

744 ``ReflectedTwo`` can be configured separately:: 

745 

746 ReflectedOne.prepare(engine_one) 

747 ReflectedTwo.prepare(engine_two) 

748 

749 """

751 @classmethod 

752 def prepare(cls, engine): 

753 """Reflect all :class:`.Table` objects for all current 

754 :class:`.DeferredReflection` subclasses""" 

755 

756 to_map = _DeferredMapperConfig.classes_for_base(cls) 

757 for thingy in to_map: 

758 cls._sa_decl_prepare(thingy.local_table, engine) 

759 thingy.map() 

760 mapper = thingy.cls.__mapper__ 

761 metadata = mapper.class_.metadata 

762 for rel in mapper._props.values(): 

763 if ( 

764 isinstance(rel, properties.RelationshipProperty) 

765 and rel.secondary is not None 

766 ): 

767 if isinstance(rel.secondary, Table): 

768 cls._reflect_table(rel.secondary, engine) 

769 elif isinstance(rel.secondary, _class_resolver): 

770 rel.secondary._resolvers += ( 

771 cls._sa_deferred_table_resolver(engine, metadata), 

772 )

891 def _get_extension_lengths(self, frac, automin, automax, default=0.05): 

892 ''' 

893 Get the lengths of colorbar extensions. 

894 

895 A helper method for _uniform_y and _proportional_y. 

896 ''' 

897 # Set the default value. 

898 extendlength = np.array([default, default]) 

899 if isinstance(frac, str): 

900 if frac.lower() == 'auto': 

901 # Use the provided values when 'auto' is required. 

902 extendlength[:] = [automin, automax] 

903 else: 

904 # Any other string is invalid. 

905 raise ValueError('invalid value for extendfrac') 

906 elif frac is not None: 

907 try: 

908 # Try to set min and max extension fractions directly. 

909 extendlength[:] = frac 

910 # If frac is a sequence containing None then NaN may 

911 # be encountered. This is an error. 

912 if np.isnan(extendlength).any(): 

913 raise ValueError() 

914 except (TypeError, ValueError): 

915 # Raise an error on encountering an invalid value for frac. 

916 raise ValueError('invalid value for extendfrac') 

917 return extendlength

919 def _uniform_y(self, N): 

920 ''' 

921 Return colorbar data coordinates for *N* uniformly 

922 spaced boundaries, plus ends if required. 

923 ''' 

924 if self.extend == 'neither': 

925 y = np.linspace(0, 1, N) 

926 else: 

927 automin = automax = 1. / (N - 1.) 

928 extendlength = self._get_extension_lengths(self.extendfrac, 

929 automin, automax, 

930 default=0.05) 

931 if self.extend == 'both': 

932 y = np.zeros(N + 2, 'd') 

933 y[0] = 0. - extendlength[0] 

934 y[-1] = 1. + extendlength[1] 

935 elif self.extend == 'min': 

936 y = np.zeros(N + 1, 'd') 

937 y[0] = 0. - extendlength[0] 

938 else: 

939 y = np.zeros(N + 1, 'd') 

940 y[-1] = 1. + extendlength[1] 

941 y[self._inside] = np.linspace(0, 1, N) 

942 return y

944 def _proportional_y(self): 

945 ''' 

946 Return colorbar data coordinates for the boundaries of 

947 a proportional colorbar. 

948 ''' 

949 if isinstance(self.norm, colors.BoundaryNorm): 

950 y = (self._boundaries - self._boundaries[0]) 

951 y = y / (self._boundaries[-1] - self._boundaries[0]) 

952 else: 

953 y = self.norm(self._boundaries.copy()) 

954 y = np.ma.filled(y, np.nan) 

955 if self.extend == 'min': 

956 # Exclude leftmost interval of y. 

957 clen = y[-1] - y[1] 

958 automin = (y[2] - y[1]) / clen 

959 automax = (y[-1] - y[-2]) / clen 

960 elif self.extend == 'max': 

961 # Exclude rightmost interval in y. 

962 clen = y[-2] - y[0] 

963 automin = (y[1] - y[0]) / clen 

964 automax = (y[-2] - y[-3]) / clen 

965 elif self.extend == 'both': 

966 # Exclude leftmost and rightmost intervals in y. 

967 clen = y[-2] - y[1] 

968 automin = (y[2] - y[1]) / clen 

969 automax = (y[-2] - y[-3]) / clen 

970 if self.extend in ('both', 'min', 'max'): 

971 extendlength = self._get_extension_lengths(self.extendfrac, 

972 automin, automax, 

973 default=0.05) 

974 if self.extend in ('both', 'min'): 

975 y[0] = 0. - extendlength[0] 

976 if self.extend in ('both', 'max'): 

977 y[-1] = 1. + extendlength[1] 

978 yi = y[self._inside] 

979 norm = colors.Normalize(yi[0], yi[-1]) 

980 y[self._inside] = np.ma.filled(norm(yi), np.nan) 

981 return y

983 def _mesh(self): 

984 ''' 

985 Return X,Y, the coordinate arrays for the colorbar pcolormesh. 

986 These are suitable for a vertical colorbar; swapping and 

987 transposition for a horizontal colorbar are done outside 

988 this function. 

989 ''' 

990 # if boundaries and values are None, then we can go ahead and 

991 # scale this up for Auto tick location. Otherwise we 

992 # want to keep normalized between 0 and 1 and use manual tick 

993 # locations. 

994 

995 x = np.array([0.0, 1.0]) 

996 if self.spacing == 'uniform': 

997 y = self._uniform_y(self._central_N()) 

998 else: 

999 y = self._proportional_y() 

1000 if self._use_auto_colorbar_locator(): 

1001 y = self.norm.inverse(y) 

1002 x = self.norm.inverse(x) 

1003 self._y = y 

1004 X, Y = np.meshgrid(x, y) 

1005 if self._use_auto_colorbar_locator(): 

1006 xmid = self.norm.inverse(0.5) 

1007 else: 

1008 xmid = 0.5 

1009 if self._extend_lower() and not self.extendrect: 

1010 X[0, :] = xmid 

1011 if self._extend_upper() and not self.extendrect: 

1012 X[-1, :] = xmid 

1013 return X, Y

1015 def _locate(self, x): 

1016 ''' 

1017 Given a set of color data values, return their 

1018 corresponding colorbar data coordinates. 

1019 ''' 

1020 if isinstance(self.norm, (colors.NoNorm, colors.BoundaryNorm)): 

1021 b = self._boundaries 

1022 xn = x 

1023 else: 

1024 # Do calculations using normalized coordinates so 

1025 # as to make the interpolation more accurate. 

1026 b = self.norm(self._boundaries, clip=False).filled() 

1027 xn = self.norm(x, clip=False).filled() 

1028 

1029 # The rest is linear interpolation with extrapolation at ends. 

1030 ii = np.searchsorted(b, xn) 

1031 i0 = ii - 1 

1032 itop = (ii == len(b)) 

1033 ibot = (ii == 0) 

1034 i0[itop] -= 1 

1035 ii[itop] -= 1 

1036 i0[ibot] += 1 

1037 ii[ibot] += 1 

1038 

1039 db = np.take(b, ii) - np.take(b, i0) 

1040 y = self._y 

1041 dy = np.take(y, ii) - np.take(y, i0) 

1042 z = np.take(y, i0) + (xn - np.take(b, i0)) * dy / db 

1043 return z

1238def make_axes(parents, location=None, orientation=None, fraction=0.15, 

1239 shrink=1.0, aspect=20, **kw): 

1240 ''' 

1241 Resize and reposition parent axes, and return a child 

1242 axes suitable for a colorbar. 

1243 

1244 Keyword arguments may include the following (with defaults): 

1245 

1246 location : [None|'left'|'right'|'top'|'bottom'] 

1247 The position, relative to **parents**, where the colorbar axes 

1248 should be created. If None, the value will either come from the 

1249 given ``orientation``, else it will default to 'right'. 

1250 

1251 orientation : [None|'vertical'|'horizontal'] 

1252 The orientation of the colorbar. Typically, this keyword shouldn't 

1253 be used, as it can be derived from the ``location`` keyword. 

1254 

1255 %s 

1256 

1257 Returns (cax, kw), the child axes and the reduced kw dictionary to be 

1258 passed when creating the colorbar instance. 

1259 ''' 

1260 

1261 locations = ["left", "right", "top", "bottom"] 

1262 if orientation is not None and location is not None: 

1263 raise TypeError('position and orientation are mutually exclusive. ' 

1264 'Consider setting the position to any of {}' 

1265 .format(', '.join(locations))) 

1266 

1267 # provide a default location 

1268 if location is None and orientation is None: 

1269 location = 'right' 

1270 

1271 # allow the user to not specify the location by specifying the 

1272 # orientation instead 

1273 if location is None: 

1274 location = 'right' if orientation == 'vertical' else 'bottom' 

1275 

1276 if location not in locations: 

1277 raise ValueError('Invalid colorbar location. Must be one ' 

1278 'of %s' % ', '.join(locations)) 

1279 

1280 default_location_settings = {'left': {'anchor': (1.0, 0.5), 

1281 'panchor': (0.0, 0.5), 

1282 'pad': 0.10, 

1283 'orientation': 'vertical'}, 

1284 'right': {'anchor': (0.0, 0.5), 

1285 'panchor': (1.0, 0.5), 

1286 'pad': 0.05, 

1287 'orientation': 'vertical'}, 

1288 'top': {'anchor': (0.5, 0.0), 

1289 'panchor': (0.5, 1.0), 

1290 'pad': 0.05, 

1291 'orientation': 'horizontal'}, 

1292 'bottom': {'anchor': (0.5, 1.0), 

1293 'panchor': (0.5, 0.0), 

1294 'pad': 0.15, # backwards compat 

1295 'orientation': 'horizontal'}, 

1296 } 

1297 

1298 loc_settings = default_location_settings[location] 

1299 

1300 # put appropriate values into the kw dict for passing back to 

1301 # the Colorbar class 

1302 kw['orientation'] = loc_settings['orientation'] 

1303 kw['ticklocation'] = location 

1304 

1305 anchor = kw.pop('anchor', loc_settings['anchor']) 

1306 parent_anchor = kw.pop('panchor', loc_settings['panchor']) 

1307 

1308 parents_iterable = cbook.iterable(parents) 

1309 # turn parents into a list if it is not already. We do this w/ np 

1310 # because `plt.subplots` can return an ndarray and is natural to 

1311 # pass to `colorbar`. 

1312 parents = np.atleast_1d(parents).ravel() 

1313 

1314 # check if using constrained_layout: 

1315 try: 

1316 gs = parents[0].get_subplotspec().get_gridspec() 

1317 using_constrained_layout = (gs._layoutbox is not None) 

1318 except AttributeError: 

1319 using_constrained_layout = False 

1320 

1321 # defaults are not appropriate for constrained_layout: 

1322 pad0 = loc_settings['pad'] 

1323 if using_constrained_layout: 

1324 pad0 = 0.02 

1325 pad = kw.pop('pad', pad0) 

1326 

1327 fig = parents[0].get_figure() 

1328 if not all(fig is ax.get_figure() for ax in parents): 

1329 raise ValueError('Unable to create a colorbar axes as not all ' 

1330 'parents share the same figure.') 

1331 

1332 # take a bounding box around all of the given axes 

1333 parents_bbox = mtransforms.Bbox.union( 

1334 [ax.get_position(original=True).frozen() for ax in parents]) 

1335 

1336 pb = parents_bbox 

1337 if location in ('left', 'right'): 

1338 if location == 'left': 

1339 pbcb, _, pb1 = pb.splitx(fraction, fraction + pad) 

1340 else: 

1341 pb1, _, pbcb = pb.splitx(1 - fraction - pad, 1 - fraction) 

1342 pbcb = pbcb.shrunk(1.0, shrink).anchored(anchor, pbcb) 

1343 else: 

1344 if location == 'bottom': 

1345 pbcb, _, pb1 = pb.splity(fraction, fraction + pad) 

1346 else: 

1347 pb1, _, pbcb = pb.splity(1 - fraction - pad, 1 - fraction) 

1348 pbcb = pbcb.shrunk(shrink, 1.0).anchored(anchor, pbcb) 

1349 

1350 # define the aspect ratio in terms of y's per x rather than x's per y 

1351 aspect = 1.0 / aspect 

1352 

1353 # define a transform which takes us from old axes coordinates to 

1354 # new axes coordinates 

1355 shrinking_trans = mtransforms.BboxTransform(parents_bbox, pb1) 

1356 

1357 # transform each of the axes in parents using the new transform 

1358 for ax in parents: 

1359 new_posn = shrinking_trans.transform(ax.get_position(original=True)) 

1360 new_posn = mtransforms.Bbox(new_posn) 

1361 ax._set_position(new_posn) 

1362 if parent_anchor is not False: 

1363 ax.set_anchor(parent_anchor) 

1364 

1365 cax = fig.add_axes(pbcb) 

1366 

1367 # OK, now make a layoutbox for the cb axis. Later, we will use this 

1368 # to make the colorbar fit nicely. 

1369 if not using_constrained_layout: 

1370 # no layout boxes: 

1371 lb = None 

1372 lbpos = None 

1373 # and we need to set the aspect ratio by hand... 

1374 cax.set_aspect(aspect, anchor=anchor, adjustable='box') 

1375 else: 

1376 if not parents_iterable: 

1377 # this is a single axis... 

1378 ax = parents[0] 

1379 lb, lbpos = constrained_layout.layoutcolorbarsingle( 

1380 ax, cax, shrink, aspect, location, pad=pad) 

1381 else: # there is more than one parent, so lets use gridspec 

1382 # the colorbar will be a sibling of this gridspec, so the 

1383 # parent is the same parent as the gridspec. Either the figure, 

1384 # or a subplotspec. 

1385 

1386 lb, lbpos = constrained_layout.layoutcolorbargridspec( 

1387 parents, cax, shrink, aspect, location, pad) 

1388 

1389 cax._layoutbox = lb 

1390 cax._poslayoutbox = lbpos 

1391 

1392 return cax, kw

1396def make_axes_gridspec(parent, *, fraction=0.15, shrink=1.0, aspect=20, **kw): 

1397 ''' 

1398 Resize and reposition a parent axes, and return a child axes 

1399 suitable for a colorbar. This function is similar to 

1400 make_axes. Prmary differences are 

1401 

1402 * *make_axes_gridspec* only handles the *orientation* keyword 

1403 and cannot handle the "location" keyword. 

1404 

1405 * *make_axes_gridspec* should only be used with a subplot parent. 

1406 

1407 * *make_axes* creates an instance of Axes. *make_axes_gridspec* 

1408 creates an instance of Subplot. 

1409 

1410 * *make_axes* updates the position of the 

1411 parent. *make_axes_gridspec* replaces the grid_spec attribute 

1412 of the parent with a new one. 

1413 

1414 While this function is meant to be compatible with *make_axes*, 

1415 there could be some minor differences. 

1416 

1417 Keyword arguments may include the following (with defaults): 

1418 

1419 *orientation* 

1420 'vertical' or 'horizontal' 

1421 

1422 %s 

1423 

1424 All but the first of these are stripped from the input kw set. 

1425 

1426 Returns (cax, kw), the child axes and the reduced kw dictionary to be 

1427 passed when creating the colorbar instance. 

1428 ''' 

1429 

1430 orientation = kw.setdefault('orientation', 'vertical') 

1431 kw['ticklocation'] = 'auto' 

1432 

1433 x1 = 1 - fraction 

1434 

1435 # for shrinking 

1436 pad_s = (1 - shrink) * 0.5 

1437 wh_ratios = [pad_s, shrink, pad_s] 

1438 

1439 # we need to none the tree of layoutboxes because 

1440 # constrained_layout can't remove and replace the tree 

1441 # hierarchy w/o a seg fault. 

1442 gs = parent.get_subplotspec().get_gridspec() 

1443 layoutbox.nonetree(gs._layoutbox) 

1444 gs_from_subplotspec = gridspec.GridSpecFromSubplotSpec 

1445 if orientation == 'vertical': 

1446 pad = kw.pop('pad', 0.05) 

1447 wh_space = 2 * pad / (1 - pad) 

1448 gs = gs_from_subplotspec(1, 2, 

1449 subplot_spec=parent.get_subplotspec(), 

1450 wspace=wh_space, 

1451 width_ratios=[x1 - pad, fraction]) 

1452 gs2 = gs_from_subplotspec(3, 1, 

1453 subplot_spec=gs[1], 

1454 hspace=0., 

1455 height_ratios=wh_ratios) 

1456 anchor = (0.0, 0.5) 

1457 panchor = (1.0, 0.5) 

1458 else: 

1459 pad = kw.pop('pad', 0.15) 

1460 wh_space = 2 * pad / (1 - pad) 

1461 gs = gs_from_subplotspec(2, 1, 

1462 subplot_spec=parent.get_subplotspec(), 

1463 hspace=wh_space, 

1464 height_ratios=[x1 - pad, fraction]) 

1465 gs2 = gs_from_subplotspec(1, 3, 

1466 subplot_spec=gs[1], 

1467 wspace=0., 

1468 width_ratios=wh_ratios) 

1469 aspect = 1 / aspect 

1470 anchor = (0.5, 1.0) 

1471 panchor = (0.5, 0.0) 

1472 

1473 parent.set_subplotspec(gs[0]) 

1474 parent.update_params() 

1475 parent._set_position(parent.figbox) 

1476 parent.set_anchor(panchor) 

1477 

1478 fig = parent.get_figure() 

1479 cax = fig.add_subplot(gs2[1]) 

1480 cax.set_aspect(aspect, anchor=anchor, adjustable='box')

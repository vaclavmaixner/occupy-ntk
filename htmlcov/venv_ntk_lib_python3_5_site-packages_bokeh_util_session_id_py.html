55def generate_secret_key(): 

56 """ 

57 Generate a new securely-generated secret key appropriate 

58 for SHA-256 HMAC signatures. This key could be used to 

59 sign Bokeh server session IDs for example. 

60 """ 

61 return _get_random_string()

63def generate_session_id(secret_key=settings.secret_key_bytes(), signed=settings.sign_sessions()): 

64 """Generate a random session ID. 

65 

66 Typically, each browser tab connected to a Bokeh application 

67 has its own session ID. In production deployments of a Bokeh 

68 app, session IDs should be random and unguessable - otherwise 

69 users of the app could interfere with one another. 

70 

71 If session IDs are signed with a secret key, the server can 

72 verify that the generator of the session ID was "authorized" 

73 (the generator had to know the secret key). This can be used 

74 to have a separate process, such as another web application, 

75 which generates new sessions on a Bokeh server. This other 

76 process may require users to log in before redirecting them to 

77 the Bokeh server with a valid session ID, for example. 

78 

79 Args: 

80 secret_key (str, optional) : Secret key (default: value of 'BOKEH_SECRET_KEY' env var) 

81 signed (bool, optional) : Whether to sign the session ID (default: value of 

82 'BOKEH_SIGN_SESSIONS' env var) 

83 

84 """ 

85 secret_key = _ensure_bytes(secret_key) 

86 if signed: 

87 # note: '-' can also be in the base64 encoded signature 

88 base_id = _get_random_string(secret_key=secret_key) 

89 return base_id + '-' + _signature(base_id, secret_key) 

90 else: 

91 return _get_random_string(secret_key=secret_key)

93def check_session_id_signature(session_id, secret_key=settings.secret_key_bytes(), 

94 signed=settings.sign_sessions()): 

95 """Check the signature of a session ID, returning True if it's valid. 

96 

97 The server uses this function to check whether a session ID 

98 was generated with the correct secret key. If signed sessions are disabled, 

99 this function always returns True. 

100 

101 Args: 

102 session_id (str) : The session ID to check 

103 secret_key (str, optional) : Secret key (default: value of 'BOKEH_SECRET_KEY' env var) 

104 signed (bool, optional) : Whether to check anything (default: value of 

105 'BOKEH_SIGN_SESSIONS' env var) 

106 

107 """ 

108 secret_key = _ensure_bytes(secret_key) 

109 if signed: 

110 pieces = session_id.split('-', 1) 

111 if len(pieces) != 2: 

112 return False 

113 base_id = pieces[0] 

114 provided_signature = pieces[1] 

115 expected_signature = _signature(base_id, secret_key) 

116 # hmac.compare_digest() uses a string compare algorithm that doesn't 

117 # short-circuit so we don't allow timing analysis 

118 # encode_utf8 is used to ensure that strings have same encoding 

119 return hmac.compare_digest(encode_utf8(expected_signature), encode_utf8(provided_signature)) 

120 else: 

121 return True

131def _get_sysrandom(): 

132 # Use the system PRNG for session id generation (if possible) 

133 # NOTE: secure random string generation implementation is adapted 

134 # from the Django project. Reference: 

135 # https://github.com/django/django/blob/0ed7d155635da9f79d4dd67e4889087d3673c6da/django/utils/crypto.py 

136 import random 

137 try: 

138 random = random.SystemRandom() 

139 using_sysrandom = True 

140 except NotImplementedError: 

141 import warnings 

142 warnings.warn('A secure pseudo-random number generator is not available ' 

143 'on your system. Falling back to Mersenne Twister.') 

144 if settings.secret_key() is None: 

145 warnings.warn('A secure pseudo-random number generator is not available ' 

146 'and no BOKEH_SECRET_KEY has been set. ' 

147 'Setting a secret key will mitigate the lack of a secure ' 

148 'generator.') 

149 using_sysrandom = False 

150 return random, using_sysrandom

152def _ensure_bytes(secret_key): 

153 if secret_key is None: 

154 return None 

155 elif isinstance(secret_key, binary_type): 

156 return secret_key 

157 else: 

158 return codecs.encode(secret_key, 'utf-8') 

159 

160# this is broken out for unit testability 

161def _reseed_if_needed(using_sysrandom, secret_key): 

162 secret_key = _ensure_bytes(secret_key) 

163 if not using_sysrandom: 

164 # This is ugly, and a hack, but it makes things better than 

165 # the alternative of predictability. This re-seeds the PRNG 

166 # using a value that is hard for an attacker to predict, every 

167 # time a random string is required. This may change the 

168 # properties of the chosen random sequence slightly, but this 

169 # is better than absolute predictability. 

170 random.seed( 

171 hashlib.sha256( 

172 ("%s%s%s" % ( 

173 random.getstate(), 

174 time.time(), 

175 secret_key)).encode('utf-8') 

176 ).digest()) 

177 

178def _base64_encode(decoded): 

179 # base64 encode both takes and returns bytes, we want to work with strings. 

180 # If 'decoded' isn't bytes already, assume it's utf-8 

181 decoded_as_bytes = _ensure_bytes(decoded) 

182 encoded = codecs.decode(base64.urlsafe_b64encode(decoded_as_bytes), 'ascii') 

183 # remove padding '=' chars that cause trouble 

184 return str(encoded.rstrip('=')) 

185 

186def _signature(base_id, secret_key): 

187 secret_key = _ensure_bytes(secret_key) 

188 base_id = codecs.encode(base_id, "utf-8") 

189 signer = hmac.new(secret_key, base_id, hashlib.sha256) 

190 return _base64_encode(signer.digest())

192def _get_random_string(length=44, 

193 allowed_chars='abcdefghijklmnopqrstuvwxyz' 

194 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789', 

195 secret_key=settings.secret_key_bytes()): 

196 """ 

197 Return a securely generated random string. 

198 With the a-z, A-Z, 0-9 character set: 

199 Length 12 is a 71-bit value. log_2((26+26+10)^12) =~ 71 

200 Length 44 is a 261-bit value. log_2((26+26+10)^44) = 261 

201 """ 

202 secret_key = _ensure_bytes(secret_key) 

203 _reseed_if_needed(using_sysrandom, secret_key) 

204 return ''.join(random.choice(allowed_chars) for i in range(length))

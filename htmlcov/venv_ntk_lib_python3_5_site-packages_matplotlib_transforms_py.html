1""" 

2matplotlib includes a framework for arbitrary geometric 

3transformations that is used determine the final position of all 

4elements drawn on the canvas. 

5 

6Transforms are composed into trees of :class:`TransformNode` objects 

7whose actual value depends on their children. When the contents of 

8children change, their parents are automatically invalidated. The 

9next time an invalidated transform is accessed, it is recomputed to 

10reflect those changes. This invalidation/caching approach prevents 

11unnecessary recomputations of transforms, and contributes to better 

12interactive performance. 

13 

14For example, here is a graph of the transform tree used to plot data 

15to the graph: 

16 

17.. image:: ../_static/transforms.png 

18 

19The framework can be used for both affine and non-affine 

20transformations. However, for speed, we want use the backend 

21renderers to perform affine transformations whenever possible. 

22Therefore, it is possible to perform just the affine or non-affine 

23part of a transformation on a set of data. The affine is always 

24assumed to occur after the non-affine. For any transform:: 

25 

26 full transform == non-affine part + affine part 

27 

28The backends are not expected to handle non-affine transformations 

29themselves.

123 def invalidate(self): 

124 """ 

125 Invalidate this :class:`TransformNode` and triggers an 

126 invalidation of its ancestors. Should be called any 

127 time the transform changes. 

128 """ 

129 value = self.INVALID 

130 if self.is_affine: 

131 value = self.INVALID_AFFINE 

132 return self._invalidate_internal(value, invalidating_node=self)

134 def _invalidate_internal(self, value, invalidating_node): 

135 """ 

136 Called by :meth:`invalidate` and subsequently ascends the transform 

137 stack calling each TransformNode's _invalidate_internal method. 

138 """ 

139 # determine if this call will be an extension to the invalidation 

140 # status. If not, then a shortcut means that we needn't invoke an 

141 # invalidation up the transform stack as it will already have been 

142 # invalidated. 

143 

144 # N.B This makes the invalidation sticky, once a transform has been 

145 # invalidated as NON_AFFINE, then it will always be invalidated as 

146 # NON_AFFINE even when triggered with a AFFINE_ONLY invalidation. 

147 # In most cases this is not a problem (i.e. for interactive panning and 

148 # zooming) and the only side effect will be on performance. 

149 status_changed = self._invalid < value 

150 

151 if self.pass_through or status_changed: 

152 self._invalid = value 

153 

154 for parent in list(self._parents.values()): 

155 # Dereference the weak reference 

156 parent = parent() 

157 if parent is not None: 

158 parent._invalidate_internal( 

159 value=value, invalidating_node=self)

213 def recurse(root): 

214 if root in seen:

1363 A - B == a + b.inverted() 

1364 

1365 # sometimes, when A contains the tree B there is no need to 

1366 # descend all the way down to the base of A (via B), instead we 

1367 # can just stop at B. 

1368 

1369 (A + B) - (B)^-1 == A 

1370 

1371 # similarly, when B contains tree A, we can avoid decending A at 

1372 # all, basically: 

1373 A - (A + B) == ((B + A) - A).inverted() or B^-1 

1374 

1375 For clarity, the result of ``(A + B) - B + B == (A + B)``. 

1376 

1377 """ 

1378 # we only know how to do this operation if other is a Transform. 

1379 if not isinstance(other, Transform): 

1380 return NotImplemented 

1381 

1382 for remainder, sub_tree in self._iter_break_from_left_to_right(): 

1383 if sub_tree == other: 

1384 return remainder 

1385 

1386 for remainder, sub_tree in other._iter_break_from_left_to_right(): 

1387 if sub_tree == self: 

1388 if not remainder.has_inverse: 

1389 raise ValueError("The shortcut cannot be computed since " 

1390 "other's transform includes a non-invertable component.") 

1391 return remainder.inverted() 

1392 

1393 # if we have got this far, then there was no shortcut possible 

1394 if other.has_inverse: 

1395 return self + other.inverted() 

1396 else: 

1397 raise ValueError('It is not possible to compute transA - transB ' 

1398 'since transB cannot be inverted and there is no ' 

1399 'shortcut possible.')

1405 return self.get_affine().get_matrix()

1407 def transform(self, values): 

1408 """ 

1409 Performs the transformation on the given array of values. 

1410 

1411 Accepts a numpy array of shape (N x :attr:`input_dims`) and 

1412 returns a numpy array of shape (N x :attr:`output_dims`). 

1413 

1414 Alternatively, accepts a numpy array of length :attr:`input_dims` 

1415 and returns a numpy array of length :attr:`output_dims`. 

1416 """ 

1417 # Ensure that values is a 2d array (but remember whether 

1418 # we started with a 1d or 2d array). 

1419 values = np.asanyarray(values) 

1420 ndim = values.ndim 

1421 values = values.reshape((-1, self.input_dims)) 

1422 

1423 # Transform the values 

1424 res = self.transform_affine(self.transform_non_affine(values)) 

1425 

1426 # Convert the result back to the shape of the input values. 

1427 if ndim == 0: 

1428 assert not np.ma.is_masked(res) # just to be on the safe side 

1429 return res[0, 0] 

1430 if ndim == 1: 

1431 return res.reshape(-1) 

1432 elif ndim == 2: 

1433 return res 

1434 raise ValueError( 

1435 "Input values must have shape (N x {dims}) " 

1436 "or ({dims}).".format(dims=self.input_dims))

1551 def transform_angles(self, angles, pts, radians=False, pushoff=1e-5): 

1552 """ 

1553 Performs transformation on a set of angles anchored at 

1554 specific locations. 

1555 

1556 The *angles* must be a column vector (i.e., numpy array). 

1557 

1558 The *pts* must be a two-column numpy array of x,y positions 

1559 (angle transforms currently only work in 2D). This array must 

1560 have the same number of rows as *angles*. 

1561 

1562 *radians* indicates whether or not input angles are given in 

1563 radians (True) or degrees (False; the default). 

1564 

1565 *pushoff* is the distance to move away from *pts* for 

1566 determining transformed angles (see discussion of method 

1567 below). 

1568 

1569 The transformed angles are returned in an array with the same 

1570 size as *angles*. 

1571 

1572 The generic version of this method uses a very generic 

1573 algorithm that transforms *pts*, as well as locations very 

1574 close to *pts*, to find the angle in the transformed system. 

1575 """ 

1576 # Must be 2D 

1577 if self.input_dims != 2 or self.output_dims != 2: 

1578 raise NotImplementedError('Only defined in 2D') 

1579 

1580 if pts.shape[1] != 2: 

1581 raise ValueError("'pts' must be array with 2 columns for x,y") 

1582 

1583 if angles.ndim != 1 or angles.shape[0] != pts.shape[0]: 

1584 raise ValueError("'angles' must be a column vector and have same " 

1585 "number of rows as 'pts'") 

1586 

1587 # Convert to radians if desired 

1588 if not radians: 

1589 angles = angles / 180.0 * np.pi 

1590 

1591 # Move a short distance away 

1592 pts2 = pts + pushoff * np.c_[np.cos(angles), np.sin(angles)] 

1593 

1594 # Transform both sets of points 

1595 tpts = self.transform(pts) 

1596 tpts2 = self.transform(pts2) 

1597 

1598 # Calculate transformed angles 

1599 d = tpts2 - tpts 

1600 a = np.arctan2(d[:, 1], d[:, 0]) 

1601 

1602 # Convert back to degrees if desired 

1603 if not radians: 

1604 a = np.rad2deg(a) 

1605 

1606 return a

1613 temporary. An update to *self* does not cause a corresponding 

1614 update to its inverted copy. 

1615 

1616 ``x === self.inverted().transform(self.transform(x))`` 

1617 """ 

1618 raise NotImplementedError()

2861def nonsingular(vmin, vmax, expander=0.001, tiny=1e-15, increasing=True): 

2862 """ 

2863 Modify the endpoints of a range as needed to avoid singularities. 

2864 

2865 Parameters 

2866 ---------- 

2867 vmin, vmax : float 

2868 The initial endpoints. 

2869 expander : float, optional, default: 0.001 

2870 Fractional amount by which *vmin* and *vmax* are expanded if 

2871 the original interval is too small, based on *tiny*. 

2872 tiny : float, optional, default: 1e-15 

2873 Threshold for the ratio of the interval to the maximum absolute 

2874 value of its endpoints. If the interval is smaller than 

2875 this, it will be expanded. This value should be around 

2876 1e-15 or larger; otherwise the interval will be approaching 

2877 the double precision resolution limit. 

2878 increasing : bool, optional, default: True 

2879 If True, swap *vmin*, *vmax* if *vmin* > *vmax*. 

2880 

2881 Returns 

2882 ------- 

2883 vmin, vmax : float 

2884 Endpoints, expanded and/or swapped if necessary. 

2885 If either input is inf or NaN, or if both inputs are 0 or very 

2886 close to zero, it returns -*expander*, *expander*. 

2887 """ 

2888 

2889 if (not np.isfinite(vmin)) or (not np.isfinite(vmax)): 

2890 return -expander, expander 

2891 

2892 swapped = False 

2893 if vmax < vmin: 

2894 vmin, vmax = vmax, vmin 

2895 swapped = True 

2896 

2897 maxabsvalue = max(abs(vmin), abs(vmax)) 

2898 if maxabsvalue < (1e6 / tiny) * np.finfo(float).tiny: 

2899 vmin = -expander 

2900 vmax = expander 

2901 

2902 elif vmax - vmin <= maxabsvalue * tiny: 

2903 if vmax == 0 and vmin == 0: 

2904 vmin = -expander 

2905 vmax = expander 

2906 else: 

2907 vmin -= expander*abs(vmin) 

2908 vmax += expander*abs(vmax) 

2909 

2910 if swapped and not increasing: 

2911 vmin, vmax = vmax, vmin 

2912 return vmin, vmax

429def check_output(*popenargs, **kwargs): 

430 # shamelessly taken from Python 2.7 source 

431 r"""Run command with arguments and return its output as a byte string. 

432 

433 If the exit code was non-zero it raises a CalledProcessError. The 

434 CalledProcessError object will have the return code in the returncode 

435 attribute and output in the output attribute. 

436 

437 The arguments are the same as for the Popen constructor. Example: 

438 

439 >>> check_output(["ls", "-l", "/dev/null"]) 

440 'crw-rw-rw- 1 root root 1, 3 Oct 18 2007 /dev/null\n' 

441 

442 The stdout argument is not allowed as it is used internally. 

443 To capture standard error in the result, use stderr=STDOUT. 

444 

445 >>> check_output(["/bin/sh", "-c", 

446 ... "ls -l non_existent_file ; exit 0"], 

447 ... stderr=STDOUT) 

448 'ls: non_existent_file: No such file or directory\n' 

449 """ 

450 if 'stdout' in kwargs: 

451 raise ValueError('stdout argument not allowed, it will be overridden.') 

452 process = subprocess.Popen(stdout=subprocess.PIPE, stderr=subprocess.PIPE, 

453 *popenargs, **kwargs) 

454 output, unused_err = process.communicate() 

455 retcode = process.poll() 

456 if retcode: 

457 cmd = kwargs.get("args") 

458 if cmd is None: 

459 cmd = popenargs[0] 

460 raise subprocess.CalledProcessError(retcode, cmd, output=output) 

461 return output

766def assert_index_equal(left, right, exact='equiv', check_names=True,

769 """Check that left and right Index are equal. 

770 

771 Parameters 

772 ---------- 

773 left : Index 

774 right : Index 

775 exact : bool / string {'equiv'}, default 'equiv' 

776 Whether to check the Index class, dtype and inferred_type 

777 are identical. If 'equiv', then RangeIndex can be substituted for 

778 Int64Index as well. 

779 check_names : bool, default True 

780 Whether to check the names attribute. 

781 check_less_precise : bool or int, default False 

782 Specify comparison precision. Only used when check_exact is False. 

783 5 digits (False) or 3 digits (True) after decimal points are compared. 

784 If int, then specify the digits to compare 

785 check_exact : bool, default True 

786 Whether to compare number exactly. 

787 check_categorical : bool, default True 

788 Whether to compare internal Categorical exactly. 

789 obj : str, default 'Index' 

790 Specify object name being compared, internally used to show appropriate 

791 assertion message

795 def _check_types(l, r, obj='Index'): 

796 if exact: 

797 assert_class_equal(l, r, exact=exact, obj=obj) 

798 

799 # Skip exact dtype checking when `check_categorical` is False 

800 if check_categorical: 

801 assert_attr_equal('dtype', l, r, obj=obj) 

802 

803 # allow string-like to have different inferred_types 

804 if l.inferred_type in ('string', 'unicode'): 

805 assert r.inferred_type in ('string', 'unicode') 

806 else: 

807 assert_attr_equal('inferred_type', l, r, obj=obj)

809 def _get_ilevel_values(index, level): 

810 # accept level number only 

811 unique = index.levels[level] 

812 labels = index.codes[level] 

813 filled = take_1d(unique.values, labels, fill_value=unique._na_value) 

814 values = unique._shallow_copy(filled, name=index.names[level]) 

815 return values

817 # instance validation 

818 _check_isinstance(left, right, Index) 

819 

820 # class / dtype comparison 

821 _check_types(left, right, obj=obj) 

822 

823 # level comparison 

824 if left.nlevels != right.nlevels: 

825 msg1 = '{obj} levels are different'.format(obj=obj) 

826 msg2 = '{nlevels}, {left}'.format(nlevels=left.nlevels, left=left) 

827 msg3 = '{nlevels}, {right}'.format(nlevels=right.nlevels, right=right) 

828 raise_assert_detail(obj, msg1, msg2, msg3) 

829 

830 # length comparison 

831 if len(left) != len(right): 

832 msg1 = '{obj} length are different'.format(obj=obj) 

833 msg2 = '{length}, {left}'.format(length=len(left), left=left) 

834 msg3 = '{length}, {right}'.format(length=len(right), right=right) 

835 raise_assert_detail(obj, msg1, msg2, msg3) 

836 

837 # MultiIndex special comparison for little-friendly error messages 

838 if left.nlevels > 1: 

839 for level in range(left.nlevels): 

840 # cannot use get_level_values here because it can change dtype 

841 llevel = _get_ilevel_values(left, level) 

842 rlevel = _get_ilevel_values(right, level) 

843 

844 lobj = 'MultiIndex level [{level}]'.format(level=level) 

845 assert_index_equal(llevel, rlevel, 

846 exact=exact, check_names=check_names, 

847 check_less_precise=check_less_precise, 

848 check_exact=check_exact, obj=lobj) 

849 # get_level_values may change dtype 

850 _check_types(left.levels[level], right.levels[level], obj=obj) 

851 

852 # skip exact index checking when `check_categorical` is False 

853 if check_exact and check_categorical: 

854 if not left.equals(right): 

855 diff = np.sum((left.values != right.values) 

856 .astype(int)) * 100.0 / len(left) 

857 msg = '{obj} values are different ({pct} %)'.format( 

858 obj=obj, pct=np.round(diff, 5)) 

859 raise_assert_detail(obj, msg, left, right) 

860 else: 

861 _testing.assert_almost_equal(left.values, right.values, 

862 check_less_precise=check_less_precise, 

863 check_dtype=exact, 

864 obj=obj, lobj=left, robj=right) 

865 

866 # metadata comparison 

867 if check_names: 

868 assert_attr_equal('names', left, right, obj=obj) 

869 if isinstance(left, pd.PeriodIndex) or isinstance(right, pd.PeriodIndex): 

870 assert_attr_equal('freq', left, right, obj=obj) 

871 if (isinstance(left, pd.IntervalIndex) or 

872 isinstance(right, pd.IntervalIndex)): 

873 assert_interval_array_equal(left.values, right.values) 

874 

875 if check_categorical: 

876 if is_categorical_dtype(left) or is_categorical_dtype(right): 

877 assert_categorical_equal(left.values, right.values, 

878 obj='{obj} category'.format(obj=obj))

1097def assert_numpy_array_equal(left, right, strict_nan=False, 

1098 check_dtype=True, err_msg=None, 

1099 check_same=None, obj='numpy array'): 

1100 """ Checks that 'np.ndarray' is equivalent 

1101 

1102 Parameters 

1103 ---------- 

1104 left : np.ndarray or iterable 

1105 right : np.ndarray or iterable 

1106 strict_nan : bool, default False 

1107 If True, consider NaN and None to be different. 

1108 check_dtype: bool, default True 

1109 check dtype if both a and b are np.ndarray 

1110 err_msg : str, default None 

1111 If provided, used as assertion message 

1112 check_same : None|'copy'|'same', default None 

1113 Ensure left and right refer/do not refer to the same memory area 

1114 obj : str, default 'numpy array' 

1115 Specify object name being compared, internally used to show appropriate 

1116 assertion message 

1117 """ 

1118 __tracebackhide__ = True 

1119 

1120 # instance validation 

1121 # Show a detailed error message when classes are different 

1122 assert_class_equal(left, right, obj=obj) 

1123 # both classes must be an np.ndarray 

1124 _check_isinstance(left, right, np.ndarray) 

1125 

1126 def _get_base(obj): 

1127 return obj.base if getattr(obj, 'base', None) is not None else obj 

1128 

1129 left_base = _get_base(left) 

1130 right_base = _get_base(right) 

1131 

1132 if check_same == 'same': 

1133 if left_base is not right_base: 

1134 msg = "{left!r} is not {right!r}".format( 

1135 left=left_base, right=right_base) 

1136 raise AssertionError(msg) 

1137 elif check_same == 'copy': 

1138 if left_base is right_base: 

1139 msg = "{left!r} is {right!r}".format( 

1140 left=left_base, right=right_base) 

1141 raise AssertionError(msg) 

1142 

1143 def _raise(left, right, err_msg): 

1144 if err_msg is None: 

1145 if left.shape != right.shape: 

1146 raise_assert_detail(obj, '{obj} shapes are different' 

1147 .format(obj=obj), left.shape, right.shape) 

1148 

1149 diff = 0 

1150 for l, r in zip(left, right): 

1151 # count up differences 

1152 if not array_equivalent(l, r, strict_nan=strict_nan): 

1153 diff += 1 

1154 

1155 diff = diff * 100.0 / left.size 

1156 msg = '{obj} values are different ({pct} %)'.format( 

1157 obj=obj, pct=np.round(diff, 5)) 

1158 raise_assert_detail(obj, msg, left, right) 

1159 

1160 raise AssertionError(err_msg) 

1161 

1162 # compare shape and values 

1163 if not array_equivalent(left, right, strict_nan=strict_nan): 

1164 _raise(left, right, err_msg) 

1165 

1166 if check_dtype: 

1167 if isinstance(left, np.ndarray) and isinstance(right, np.ndarray): 

1168 assert_attr_equal('dtype', left, right, obj=obj) 

1169 

1170 return True

35def instances(query, cursor, context): 

36 """Return an ORM result as an iterator.""" 

37 

38 context.runid = _new_runid() 

39 context.post_load_paths = {} 

40 

41 filtered = query._has_mapper_entities 

42 

43 single_entity = query.is_single_entity 

44 

45 if filtered: 

46 if single_entity: 

47 filter_fn = id 

48 else: 

49 

50 def filter_fn(row): 

51 return tuple( 

52 id(item) if ent.use_id_for_hash else item 

53 for ent, item in zip(query._entities, row) 

54 ) 

55 

56 try: 

57 (process, labels) = list( 

58 zip( 

59 *[ 

60 query_entity.row_processor(query, context, cursor) 

61 for query_entity in query._entities 

62 ] 

63 ) 

64 ) 

65 

66 if not single_entity: 

67 keyed_tuple = util.lightweight_named_tuple("result", labels) 

68 

69 while True: 

70 context.partials = {} 

71 

72 if query._yield_per: 

73 fetch = cursor.fetchmany(query._yield_per) 

74 if not fetch: 

75 break 

76 else: 

77 fetch = cursor.fetchall() 

78 

79 if single_entity: 

80 proc = process[0] 

81 rows = [proc(row) for row in fetch] 

82 else: 

83 rows = [ 

84 keyed_tuple([proc(row) for proc in process]) 

85 for row in fetch 

86 ] 

87 

88 for path, post_load in context.post_load_paths.items(): 

89 post_load.invoke(context, path) 

90 

91 if filtered: 

92 rows = util.unique_list(rows, filter_fn) 

93 

94 for row in rows: 

95 yield row 

96 

97 if not query._yield_per: 

98 break 

99 except Exception as err: 

100 cursor.close() 

101 util.raise_from_cause(err)

105def merge_result(querylib, query, iterator, load=True): 

106 """Merge a result into this :class:`.Query` object's Session.""" 

107 

108 session = query.session 

109 if load: 

110 # flush current contents if we expect to load data 

111 session._autoflush() 

112 

113 autoflush = session.autoflush 

114 try: 

115 session.autoflush = False 

116 single_entity = len(query._entities) == 1 

117 if single_entity: 

118 if isinstance(query._entities[0], querylib._MapperEntity): 

119 result = [ 

120 session._merge( 

121 attributes.instance_state(instance), 

122 attributes.instance_dict(instance), 

123 load=load, 

124 _recursive={}, 

125 _resolve_conflict_map={}, 

126 ) 

127 for instance in iterator 

128 ] 

129 else: 

130 result = list(iterator) 

131 else: 

132 mapped_entities = [ 

133 i 

134 for i, e in enumerate(query._entities) 

135 if isinstance(e, querylib._MapperEntity) 

136 ] 

137 result = [] 

138 keys = [ent._label_name for ent in query._entities] 

139 keyed_tuple = util.lightweight_named_tuple("result", keys) 

140 for row in iterator: 

141 newrow = list(row) 

142 for i in mapped_entities: 

143 if newrow[i] is not None: 

144 newrow[i] = session._merge( 

145 attributes.instance_state(newrow[i]), 

146 attributes.instance_dict(newrow[i]), 

147 load=load, 

148 _recursive={}, 

149 _resolve_conflict_map={}, 

150 ) 

151 result.append(keyed_tuple(newrow)) 

152 

153 return iter(result) 

154 finally: 

155 session.autoflush = autoflush

211 refresh_state=None, 

212 with_for_update=None, 

213 only_load_props=None, 

214 identity_token=None,

216 

217 """Load the given primary key identity from the database.""" 

218

221 q._get_condition()

223 q = query._clone() 

224

339def _instance_processor( 

340 mapper, 

341 context, 

342 result, 

343 path, 

344 adapter, 

345 only_load_props=None, 

346 refresh_state=None, 

347 polymorphic_discriminator=None, 

348 _polymorphic_from=None, 

349): 

350 """Produce a mapper level row processor callable 

351 which processes rows into mapped instances.""" 

352 

353 # note that this method, most of which exists in a closure 

354 # called _instance(), resists being broken out, as 

355 # attempts to do so tend to add significant function 

356 # call overhead. _instance() is the most 

357 # performance-critical section in the whole ORM. 

358 

359 pk_cols = mapper.primary_key 

360 

361 if adapter: 

362 pk_cols = [adapter.columns[c] for c in pk_cols] 

363 

364 identity_class = mapper._identity_class 

365 

366 populators = collections.defaultdict(list) 

367 

368 props = mapper._prop_set 

369 if only_load_props is not None: 

370 props = props.intersection(mapper._props[k] for k in only_load_props) 

371 

372 quick_populators = path.get( 

373 context.attributes, "memoized_setups", _none_set 

374 ) 

375 

376 for prop in props: 

377 if prop in quick_populators: 

378 # this is an inlined path just for column-based attributes. 

379 col = quick_populators[prop] 

380 if col is _DEFER_FOR_STATE: 

381 populators["new"].append( 

382 (prop.key, prop._deferred_column_loader) 

383 ) 

384 elif col is _SET_DEFERRED_EXPIRED: 

385 # note that in this path, we are no longer 

386 # searching in the result to see if the column might 

387 # be present in some unexpected way. 

388 populators["expire"].append((prop.key, False)) 

389 else: 

390 getter = None 

391 # the "adapter" can be here via different paths, 

392 # e.g. via adapter present at setup_query or adapter 

393 # applied to the query afterwards via eager load subquery. 

394 # If the column here 

395 # were already a product of this adapter, sending it through 

396 # the adapter again can return a totally new expression that 

397 # won't be recognized in the result, and the ColumnAdapter 

398 # currently does not accommodate for this. OTOH, if the 

399 # column were never applied through this adapter, we may get 

400 # None back, in which case we still won't get our "getter". 

401 # so try both against result._getter(). See issue #4048 

402 if adapter: 

403 adapted_col = adapter.columns[col] 

404 if adapted_col is not None: 

405 getter = result._getter(adapted_col, False) 

406 if not getter: 

407 getter = result._getter(col, False) 

408 if getter: 

409 populators["quick"].append((prop.key, getter)) 

410 else: 

411 # fall back to the ColumnProperty itself, which 

412 # will iterate through all of its columns 

413 # to see if one fits 

414 prop.create_row_processor( 

415 context, path, mapper, result, adapter, populators 

416 ) 

417 else: 

418 prop.create_row_processor( 

419 context, path, mapper, result, adapter, populators 

420 ) 

421 

422 propagate_options = context.propagate_options 

423 load_path = ( 

424 context.query._current_path + path 

425 if context.query._current_path.path 

426 else path 

427 ) 

428 

429 session_identity_map = context.session.identity_map 

430 

431 populate_existing = context.populate_existing or mapper.always_refresh 

432 load_evt = bool(mapper.class_manager.dispatch.load) 

433 refresh_evt = bool(mapper.class_manager.dispatch.refresh) 

434 persistent_evt = bool(context.session.dispatch.loaded_as_persistent) 

435 if persistent_evt: 

436 loaded_as_persistent = context.session.dispatch.loaded_as_persistent 

437 instance_state = attributes.instance_state 

438 instance_dict = attributes.instance_dict 

439 session_id = context.session.hash_key 

440 version_check = context.version_check 

441 runid = context.runid 

442 identity_token = context.identity_token 

443 

444 if not refresh_state and _polymorphic_from is not None: 

445 key = ("loader", path.path) 

446 if key in context.attributes and context.attributes[key].strategy == ( 

447 ("selectinload_polymorphic", True), 

448 ): 

449 selectin_load_via = mapper._should_selectin_load( 

450 context.attributes[key].local_opts["entities"], 

451 _polymorphic_from, 

452 ) 

453 else: 

454 selectin_load_via = mapper._should_selectin_load( 

455 None, _polymorphic_from 

456 ) 

457 

458 if selectin_load_via and selectin_load_via is not _polymorphic_from: 

459 # only_load_props goes w/ refresh_state only, and in a refresh 

460 # we are a single row query for the exact entity; polymorphic 

461 # loading does not apply 

462 assert only_load_props is None 

463 

464 callable_ = _load_subclass_via_in(context, path, selectin_load_via) 

465 

466 PostLoad.callable_for_path( 

467 context, 

468 load_path, 

469 selectin_load_via.mapper, 

470 selectin_load_via, 

471 callable_, 

472 selectin_load_via, 

473 ) 

474 

475 post_load = PostLoad.for_context(context, load_path, only_load_props) 

476 

477 if refresh_state: 

478 refresh_identity_key = refresh_state.key 

479 if refresh_identity_key is None: 

480 # super-rare condition; a refresh is being called 

481 # on a non-instance-key instance; this is meant to only 

482 # occur within a flush() 

483 refresh_identity_key = mapper._identity_key_from_state( 

484 refresh_state 

485 ) 

486 else: 

487 refresh_identity_key = None 

488 

489 if mapper.allow_partial_pks: 

490 is_not_primary_key = _none_set.issuperset 

491 else: 

492 is_not_primary_key = _none_set.intersection

494 def _instance(row): 

495 

496 # determine the state that we'll be populating 

497 if refresh_identity_key: 

498 # fixed state that we're refreshing 

499 state = refresh_state 

500 instance = state.obj() 

501 dict_ = instance_dict(instance) 

502 isnew = state.runid != runid 

503 currentload = True 

504 loaded_instance = False 

505 else: 

506 # look at the row, see if that identity is in the 

507 # session, or we have to create a new one 

508 identitykey = ( 

509 identity_class, 

510 tuple([row[column] for column in pk_cols]), 

511 identity_token, 

512 ) 

513 

514 instance = session_identity_map.get(identitykey) 

515 

516 if instance is not None: 

517 # existing instance 

518 state = instance_state(instance) 

519 dict_ = instance_dict(instance) 

520 

521 isnew = state.runid != runid 

522 currentload = not isnew 

523 loaded_instance = False 

524 

525 if version_check and not currentload: 

526 _validate_version_id(mapper, state, dict_, row, adapter) 

527 

528 else: 

529 # create a new instance 

530 

531 # check for non-NULL values in the primary key columns, 

532 # else no entity is returned for the row 

533 if is_not_primary_key(identitykey[1]): 

534 return None 

535 

536 isnew = True 

537 currentload = True 

538 loaded_instance = True 

539 

540 instance = mapper.class_manager.new_instance() 

541 

542 dict_ = instance_dict(instance) 

543 state = instance_state(instance) 

544 state.key = identitykey 

545 state.identity_token = identity_token 

546 

547 # attach instance to session. 

548 state.session_id = session_id 

549 session_identity_map._add_unpresent(state, identitykey) 

550 

551 # populate. this looks at whether this state is new 

552 # for this load or was existing, and whether or not this 

553 # row is the first row with this identity. 

554 if currentload or populate_existing: 

555 # full population routines. Objects here are either 

556 # just created, or we are doing a populate_existing 

557 

558 # be conservative about setting load_path when populate_existing 

559 # is in effect; want to maintain options from the original 

560 # load. see test_expire->test_refresh_maintains_deferred_options 

561 if isnew and (propagate_options or not populate_existing): 

562 state.load_options = propagate_options 

563 state.load_path = load_path 

564 

565 _populate_full( 

566 context, 

567 row, 

568 state, 

569 dict_, 

570 isnew, 

571 load_path, 

572 loaded_instance, 

573 populate_existing, 

574 populators, 

575 ) 

576 

577 if isnew: 

578 if loaded_instance: 

579 if load_evt: 

580 state.manager.dispatch.load(state, context) 

581 if persistent_evt: 

582 loaded_as_persistent(context.session, state.obj()) 

583 elif refresh_evt: 

584 state.manager.dispatch.refresh( 

585 state, context, only_load_props 

586 ) 

587 

588 if populate_existing or state.modified: 

589 if refresh_state and only_load_props: 

590 state._commit(dict_, only_load_props) 

591 else: 

592 state._commit_all(dict_, session_identity_map) 

593 

594 if post_load: 

595 post_load.add_state(state, True) 

596 

597 else: 

598 # partial population routines, for objects that were already 

599 # in the Session, but a row matches them; apply eager loaders 

600 # on existing objects, etc. 

601 unloaded = state.unloaded 

602 isnew = state not in context.partials 

603 

604 if not isnew or unloaded or populators["eager"]: 

605 # state is having a partial set of its attributes 

606 # refreshed. Populate those attributes, 

607 # and add to the "context.partials" collection. 

608 

609 to_load = _populate_partial( 

610 context, 

611 row, 

612 state, 

613 dict_, 

614 isnew, 

615 load_path, 

616 unloaded, 

617 populators, 

618 ) 

619 

620 if isnew: 

621 if refresh_evt: 

622 state.manager.dispatch.refresh(state, context, to_load) 

623 

624 state._commit(dict_, to_load) 

625 

626 if post_load and context.invoke_all_eagers: 

627 post_load.add_state(state, False) 

628 

629 return instance 

630 

631 if mapper.polymorphic_map and not _polymorphic_from and not refresh_state: 

632 # if we are doing polymorphic, dispatch to a different _instance() 

633 # method specific to the subclass mapper 

634 _instance = _decorate_polymorphic_switch( 

635 _instance, 

636 context, 

637 mapper, 

638 result, 

639 path, 

640 polymorphic_discriminator, 

641 adapter, 

642 ) 

643 

644 return _instance

648 mapper = entity.mapper 

649

651 

652 if entity.is_aliased_class:

658 orig_query = context.query 

659

664 )

679def _populate_full( 

680 context, 

681 row, 

682 state, 

683 dict_, 

684 isnew, 

685 load_path, 

686 loaded_instance, 

687 populate_existing, 

688 populators, 

689): 

690 if isnew: 

691 # first time we are seeing a row with this identity. 

692 state.runid = context.runid 

693 

694 for key, getter in populators["quick"]: 

695 dict_[key] = getter(row) 

696 if populate_existing: 

697 for key, set_callable in populators["expire"]: 

698 dict_.pop(key, None) 

699 if set_callable: 

700 state.expired_attributes.add(key) 

701 else: 

702 for key, set_callable in populators["expire"]: 

703 if set_callable: 

704 state.expired_attributes.add(key) 

705 for key, populator in populators["new"]: 

706 populator(state, dict_, row) 

707 for key, populator in populators["delayed"]: 

708 populator(state, dict_, row) 

709 elif load_path != state.load_path: 

710 # new load path, e.g. object is present in more than one 

711 # column position in a series of rows 

712 state.load_path = load_path 

713 

714 # if we have data, and the data isn't in the dict, OK, let's put 

715 # it in. 

716 for key, getter in populators["quick"]: 

717 if key not in dict_: 

718 dict_[key] = getter(row) 

719 

720 # otherwise treat like an "already seen" row 

721 for key, populator in populators["existing"]: 

722 populator(state, dict_, row) 

723 # TODO: allow "existing" populator to know this is 

724 # a new path for the state: 

725 # populator(state, dict_, row, new_path=True) 

726 

727 else: 

728 # have already seen rows with this identity in this same path. 

729 for key, populator in populators["existing"]: 

730 populator(state, dict_, row)

736def _populate_partial( 

737 context, row, state, dict_, isnew, load_path, unloaded, populators 

738): 

739 

740 if not isnew: 

741 to_load = context.partials[state] 

742 for key, populator in populators["existing"]: 

743 if key in to_load: 

744 populator(state, dict_, row) 

745 else: 

746 to_load = unloaded 

747 context.partials[state] = to_load 

748 

749 for key, getter in populators["quick"]: 

750 if key in to_load: 

751 dict_[key] = getter(row) 

752 for key, set_callable in populators["expire"]: 

753 if key in to_load: 

754 dict_.pop(key, None) 

755 if set_callable: 

756 state.expired_attributes.add(key) 

757 for key, populator in populators["new"]: 

758 if key in to_load: 

759 populator(state, dict_, row) 

760 for key, populator in populators["delayed"]: 

761 if key in to_load: 

762 populator(state, dict_, row) 

763 for key, populator in populators["eager"]: 

764 if key not in unloaded: 

765 populator(state, dict_, row) 

766 

767 return to_load

849class PostLoad(object): 

850 """Track loaders and states for "post load" operations. 

851 

852 """ 

853 

854 __slots__ = "loaders", "states", "load_keys" 

855 

856 def __init__(self): 

857 self.loaders = {} 

858 self.states = util.OrderedDict() 

859 self.load_keys = None 

860 

861 def add_state(self, state, overwrite): 

862 # the states for a polymorphic load here are all shared 

863 # within a single PostLoad object among multiple subtypes. 

864 # Filtering of callables on a per-subclass basis needs to be done at 

865 # the invocation level 

866 self.states[state] = overwrite 

867 

868 def invoke(self, context, path): 

869 if not self.states: 

870 return 

871 path = path_registry.PathRegistry.coerce(path) 

872 for token, limit_to_mapper, loader, arg, kw in self.loaders.values(): 

873 states = [ 

874 (state, overwrite) 

875 for state, overwrite in self.states.items() 

876 if state.manager.mapper.isa(limit_to_mapper) 

877 ] 

878 if states: 

879 loader(context, path, states, self.load_keys, *arg, **kw) 

880 self.states.clear() 

881 

882 @classmethod 

883 def for_context(cls, context, path, only_load_props): 

884 pl = context.post_load_paths.get(path.path) 

885 if pl is not None and only_load_props: 

886 pl.load_keys = only_load_props 

887 return pl 

888 

889 @classmethod 

890 def path_exists(self, context, path, key): 

891 return ( 

892 path.path in context.post_load_paths 

893 and key in context.post_load_paths[path.path].loaders 

894 ) 

895 

896 @classmethod 

897 def callable_for_path( 

898 cls, context, path, limit_to_mapper, token, loader_callable, *arg, **kw 

899 ): 

900 if path.path in context.post_load_paths: 

901 pl = context.post_load_paths[path.path] 

902 else: 

903 pl = context.post_load_paths[path.path] = PostLoad() 

904 pl.loaders[token] = (token, limit_to_mapper, loader_callable, arg, kw)

907def load_scalar_attributes(mapper, state, attribute_names): 

908 """initiate a column-based attribute refresh operation.""" 

909 

910 # assert mapper is _state_mapper(state) 

911 session = state.session 

912 if not session: 

913 raise orm_exc.DetachedInstanceError( 

914 "Instance %s is not bound to a Session; " 

915 "attribute refresh operation cannot proceed" % (state_str(state)) 

916 ) 

917 

918 has_key = bool(state.key) 

919 

920 result = False 

921 

922 # in the case of inheritance, particularly concrete and abstract 

923 # concrete inheritance, the class manager might have some keys 

924 # of attributes on the superclass that we didn't actually map. 

925 # These could be mapped as "concrete, dont load" or could be completely 

926 # exluded from the mapping and we know nothing about them. Filter them 

927 # here to prevent them from coming through. 

928 if attribute_names: 

929 attribute_names = attribute_names.intersection(mapper.attrs.keys()) 

930 

931 if mapper.inherits and not mapper.concrete: 

932 # because we are using Core to produce a select() that we 

933 # pass to the Query, we aren't calling setup() for mapped 

934 # attributes; in 1.0 this means deferred attrs won't get loaded 

935 # by default 

936 statement = mapper._optimized_get_statement(state, attribute_names) 

937 if statement is not None: 

938 result = load_on_ident( 

939 session.query(mapper) 

940 .options(strategy_options.Load(mapper).undefer("*")) 

941 .from_statement(statement), 

942 None, 

943 only_load_props=attribute_names, 

944 refresh_state=state, 

945 ) 

946 

947 if result is False: 

948 if has_key: 

949 identity_key = state.key 

950 else: 

951 # this codepath is rare - only valid when inside a flush, and the 

952 # object is becoming persistent but hasn't yet been assigned 

953 # an identity_key. 

954 # check here to ensure we have the attrs we need. 

955 pk_attrs = [ 

956 mapper._columntoproperty[col].key for col in mapper.primary_key 

957 ] 

958 if state.expired_attributes.intersection(pk_attrs): 

959 raise sa_exc.InvalidRequestError( 

960 "Instance %s cannot be refreshed - it's not " 

961 " persistent and does not " 

962 "contain a full primary key." % state_str(state) 

963 ) 

964 identity_key = mapper._identity_key_from_state(state) 

965 

966 if ( 

967 _none_set.issubset(identity_key) and not mapper.allow_partial_pks 

968 ) or _none_set.issuperset(identity_key): 

969 util.warn_limited( 

970 "Instance %s to be refreshed doesn't " 

971 "contain a full primary key - can't be refreshed " 

972 "(and shouldn't be expired, either).", 

973 state_str(state), 

974 ) 

975 return 

976 

977 result = load_on_ident( 

978 session.query(mapper), 

979 identity_key, 

980 refresh_state=state, 

981 only_load_props=attribute_names, 

982 ) 

983 

984 # if instance is pending, a refresh operation 

985 # may not complete (even if PK attributes are assigned) 

986 if has_key and result is None: 

987 raise orm_exc.ObjectDeletedError(state)

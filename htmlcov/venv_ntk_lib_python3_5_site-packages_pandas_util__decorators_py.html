10def deprecate(name, alternative, version, alt_name=None, 

11 klass=None, stacklevel=2, msg=None): 

12 """Return a new function that emits a deprecation warning on use. 

13 

14 To use this method for a deprecated function, another function 

15 `alternative` with the same signature must exist. The deprecated 

16 function will emit a deprecation warning, and in the docstring 

17 it will contain the deprecation directive with the provided version 

18 so it can be detected for future removal. 

19 

20 Parameters 

21 ---------- 

22 name : str 

23 Name of function to deprecate. 

24 alternative : func 

25 Function to use instead. 

26 version : str 

27 Version of pandas in which the method has been deprecated. 

28 alt_name : str, optional 

29 Name to use in preference of alternative.__name__. 

30 klass : Warning, default FutureWarning 

31 stacklevel : int, default 2 

32 msg : str 

33 The message to display in the warning. 

34 Default is '{name} is deprecated. Use {alt_name} instead.' 

35 """ 

36 

37 alt_name = alt_name or alternative.__name__ 

38 klass = klass or FutureWarning 

39 warning_msg = msg or '{} is deprecated, use {} instead'.format(name, 

40 alt_name) 

41 

42 @wraps(alternative) 

43 def wrapper(*args, **kwargs): 

44 warnings.warn(warning_msg, klass, stacklevel=stacklevel) 

45 return alternative(*args, **kwargs) 

46 

47 # adding deprecated directive to the docstring 

48 msg = msg or 'Use `{alt_name}` instead.'.format(alt_name=alt_name) 

49 doc_error_msg = ('deprecate needs a correctly formatted docstring in ' 

50 'the target function (should have a one liner short ' 

51 'summary, and opening quotes should be in their own ' 

52 'line). Found:\n{}'.format(alternative.__doc__)) 

53 

54 # when python is running in optimized mode (i.e. `-OO`), docstrings are 

55 # removed, so we check that a docstring with correct formatting is used 

56 # but we allow empty docstrings 

57 if alternative.__doc__: 

58 if alternative.__doc__.count('\n') < 3: 

59 raise AssertionError(doc_error_msg) 

60 empty1, summary, empty2, doc = alternative.__doc__.split('\n', 3) 

61 if empty1 or empty2 and not summary: 

62 raise AssertionError(doc_error_msg) 

63 wrapper.__doc__ = dedent(""" 

64 {summary} 

65 

66 .. deprecated:: {depr_version} 

67 {depr_msg} 

68 

69 {rest_of_docstring}""").format(summary=summary.strip(), 

70 depr_version=version, 

71 depr_msg=msg, 

72 rest_of_docstring=dedent(doc)) 

73 

74 return wrapper

77def deprecate_kwarg(old_arg_name, new_arg_name, mapping=None, stacklevel=2): 

78 """ 

79 Decorator to deprecate a keyword argument of a function. 

80 

81 Parameters 

82 ---------- 

83 old_arg_name : str 

84 Name of argument in function to deprecate 

85 new_arg_name : str or None 

86 Name of preferred argument in function. Use None to raise warning that 

87 ``old_arg_name`` keyword is deprecated. 

88 mapping : dict or callable 

89 If mapping is present, use it to translate old arguments to 

90 new arguments. A callable must do its own value checking; 

91 values not found in a dict will be forwarded unchanged. 

92 

93 Examples 

94 -------- 

95 The following deprecates 'cols', using 'columns' instead 

96 

97 >>> @deprecate_kwarg(old_arg_name='cols', new_arg_name='columns') 

98 ... def f(columns=''): 

99 ... print(columns) 

100 ... 

101 >>> f(columns='should work ok') 

102 should work ok 

103 

104 >>> f(cols='should raise warning') 

105 FutureWarning: cols is deprecated, use columns instead 

106 warnings.warn(msg, FutureWarning) 

107 should raise warning 

108 

109 >>> f(cols='should error', columns="can\'t pass do both") 

110 TypeError: Can only specify 'cols' or 'columns', not both 

111 

112 >>> @deprecate_kwarg('old', 'new', {'yes': True, 'no': False}) 

113 ... def f(new=False): 

114 ... print('yes!' if new else 'no!') 

115 ... 

116 >>> f(old='yes') 

117 FutureWarning: old='yes' is deprecated, use new=True instead 

118 warnings.warn(msg, FutureWarning) 

119 yes! 

120 

121 To raise a warning that a keyword will be removed entirely in the future 

122 

123 >>> @deprecate_kwarg(old_arg_name='cols', new_arg_name=None) 

124 ... def f(cols='', another_param=''): 

125 ... print(cols) 

126 ... 

127 >>> f(cols='should raise warning') 

128 FutureWarning: the 'cols' keyword is deprecated and will be removed in a 

129 future version please takes steps to stop use of 'cols' 

130 should raise warning 

131 >>> f(another_param='should not raise warning') 

132 should not raise warning 

133 

134 >>> f(cols='should raise warning', another_param='') 

135 FutureWarning: the 'cols' keyword is deprecated and will be removed in a 

136 future version please takes steps to stop use of 'cols' 

137 should raise warning 

138 """ 

139 

140 if mapping is not None and not hasattr(mapping, 'get') and \ 

141 not callable(mapping): 

142 raise TypeError("mapping from old to new argument values " 

143 "must be dict or callable!") 

144 

145 def _deprecate_kwarg(func): 

146 @wraps(func) 

147 def wrapper(*args, **kwargs): 

148 old_arg_value = kwargs.pop(old_arg_name, None) 

149 

150 if new_arg_name is None and old_arg_value is not None: 

151 msg = ( 

152 "the '{old_name}' keyword is deprecated and will be " 

153 "removed in a future version. " 

154 "Please take steps to stop the use of '{old_name}'" 

155 ).format(old_name=old_arg_name) 

156 warnings.warn(msg, FutureWarning, stacklevel=stacklevel) 

157 kwargs[old_arg_name] = old_arg_value 

158 return func(*args, **kwargs) 

159 

160 if old_arg_value is not None: 

161 if mapping is not None: 

162 if hasattr(mapping, 'get'): 

163 new_arg_value = mapping.get(old_arg_value, 

164 old_arg_value) 

165 else: 

166 new_arg_value = mapping(old_arg_value) 

167 msg = ("the {old_name}={old_val!r} keyword is deprecated, " 

168 "use {new_name}={new_val!r} instead" 

169 ).format(old_name=old_arg_name, 

170 old_val=old_arg_value, 

171 new_name=new_arg_name, 

172 new_val=new_arg_value) 

173 else: 

174 new_arg_value = old_arg_value 

175 msg = ("the '{old_name}' keyword is deprecated, " 

176 "use '{new_name}' instead" 

177 ).format(old_name=old_arg_name, 

178 new_name=new_arg_name) 

179 

180 warnings.warn(msg, FutureWarning, stacklevel=stacklevel) 

181 if kwargs.get(new_arg_name, None) is not None: 

182 msg = ("Can only specify '{old_name}' or '{new_name}', " 

183 "not both").format(old_name=old_arg_name, 

184 new_name=new_arg_name) 

185 raise TypeError(msg) 

186 else: 

187 kwargs[new_arg_name] = new_arg_value 

188 return func(*args, **kwargs) 

189 return wrapper 

190 return _deprecate_kwarg

222class Substitution(object): 

223 """ 

224 A decorator to take a function's docstring and perform string 

225 substitution on it. 

226 

227 This decorator should be robust even if func.__doc__ is None 

228 (for example, if -OO was passed to the interpreter) 

229 

230 Usage: construct a docstring.Substitution with a sequence or 

231 dictionary suitable for performing substitution; then 

232 decorate a suitable function with the constructed object. e.g. 

233 

234 sub_author_name = Substitution(author='Jason') 

235 

236 @sub_author_name 

237 def some_function(x): 

238 "%(author)s wrote this function" 

239 

240 # note that some_function.__doc__ is now "Jason wrote this function" 

241 

242 One can also use positional arguments. 

243 

244 sub_first_last_names = Substitution('Edgar Allen', 'Poe') 

245 

246 @sub_first_last_names 

247 def some_function(x): 

248 "%s %s wrote the Raven" 

249 """ 

250 

251 def __init__(self, *args, **kwargs): 

252 if (args and kwargs): 

253 raise AssertionError("Only positional or keyword args are allowed") 

254 

255 self.params = args or kwargs 

256 

257 def __call__(self, func): 

258 func.__doc__ = func.__doc__ and func.__doc__ % self.params 

259 return func

283class Appender(object): 

284 """ 

285 A function decorator that will append an addendum to the docstring 

286 of the target function. 

287 

288 This decorator should be robust even if func.__doc__ is None 

289 (for example, if -OO was passed to the interpreter). 

290 

291 Usage: construct a docstring.Appender with a string to be joined to 

292 the original docstring. An optional 'join' parameter may be supplied 

293 which will be used to join the docstring and addendum. e.g. 

294 

295 add_copyright = Appender("Copyright (c) 2009", join='\n') 

296 

297 @add_copyright 

298 def my_dog(has='fleas'): 

299 "This docstring will have a copyright below" 

300 pass 

301 """ 

302 

303 def __init__(self, addendum, join='', indents=0): 

304 if indents > 0: 

305 self.addendum = indent(addendum, indents=indents) 

306 else: 

307 self.addendum = addendum 

308 self.join = join 

309 

310 def __call__(self, func): 

311 func.__doc__ = func.__doc__ if func.__doc__ else '' 

312 self.addendum = self.addendum if self.addendum else '' 

313 docitems = [func.__doc__, self.addendum] 

314 func.__doc__ = dedent(self.join.join(docitems)) 

315 return func

325def make_signature(func): 

326 """ 

327 Returns a tuple containing the paramenter list with defaults 

328 and parameter list. 

329 

330 Examples 

331 -------- 

332 >>> def f(a, b, c=2): 

333 >>> return a * b * c 

334 >>> print(make_signature(f)) 

335 (['a', 'b', 'c=2'], ['a', 'b', 'c']) 

336 """ 

337 

338 spec = signature(func) 

339 if spec.defaults is None: 

340 n_wo_defaults = len(spec.args) 

341 defaults = ('',) * n_wo_defaults 

342 else: 

343 n_wo_defaults = len(spec.args) - len(spec.defaults) 

344 defaults = ('',) * n_wo_defaults + tuple(spec.defaults) 

345 args = [] 

346 for var, default in zip(spec.args, defaults): 

347 args.append(var if default == '' else var + '=' + repr(default)) 

348 if spec.varargs: 

349 args.append('*' + spec.varargs) 

350 if spec.keywords: 

351 args.append('**' + spec.keywords) 

352 return args, spec.args

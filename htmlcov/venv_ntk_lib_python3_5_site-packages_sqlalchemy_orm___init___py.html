63def create_session(bind=None, **kwargs): 

64 r"""Create a new :class:`.Session` 

65 with no automation enabled by default. 

66 

67 This function is used primarily for testing. The usual 

68 route to :class:`.Session` creation is via its constructor 

69 or the :func:`.sessionmaker` function. 

70 

71 :param bind: optional, a single Connectable to use for all 

72 database access in the created 

73 :class:`~sqlalchemy.orm.session.Session`. 

74 

75 :param \*\*kwargs: optional, passed through to the 

76 :class:`.Session` constructor. 

77 

78 :returns: an :class:`~sqlalchemy.orm.session.Session` instance 

79 

80 The defaults of create_session() are the opposite of that of 

81 :func:`sessionmaker`; ``autoflush`` and ``expire_on_commit`` are 

82 False, ``autocommit`` is True. In this sense the session acts 

83 more like the "classic" SQLAlchemy 0.3 session with these. 

84 

85 Usage:: 

86 

87 >>> from sqlalchemy.orm import create_session 

88 >>> session = create_session() 

89 

90 It is recommended to use :func:`sessionmaker` instead of 

91 create_session(). 

92 

93 """ 

94 kwargs.setdefault("autoflush", False) 

95 kwargs.setdefault("autocommit", True) 

96 kwargs.setdefault("expire_on_commit", False) 

97 return Session(bind=bind, **kwargs)

208def clear_mappers(): 

209 """Remove all mappers from all classes. 

210 

211 This function removes all instrumentation from classes and disposes 

212 of their associated mappers. Once called, the classes are unmapped 

213 and can be later re-mapped with new mappers. 

214 

215 :func:`.clear_mappers` is *not* for normal use, as there is literally no 

216 valid usage for it outside of very specific testing scenarios. Normally, 

217 mappers are permanent structural components of user-defined classes, and 

218 are never discarded independently of their class. If a mapped class 

219 itself is garbage collected, its mapper is automatically disposed of as 

220 well. As such, :func:`.clear_mappers` is only for usage in test suites 

221 that re-use the same classes with different mappings, which is itself an 

222 extremely rare use case - the only such use case is in fact SQLAlchemy's 

223 own test suite, and possibly the test suites of other ORM extension 

224 libraries which intend to test various combinations of mapper construction 

225 upon a fixed set of classes. 

226 

227 """ 

228 mapperlib._CONFIGURE_MUTEX.acquire() 

229 try: 

230 while _mapper_registry: 

231 try: 

232 # can't even reliably call list(weakdict) in jython 

233 mapper, b = _mapper_registry.popitem() 

234 mapper.dispose() 

235 except KeyError: 

236 pass 

237 finally: 

238 mapperlib._CONFIGURE_MUTEX.release()
